עדות כפולה: הכנסייה
בכתבי יוחנן המטביל
23 במרץ 2024

הַמְתִּינוּ לִי עוֹד מְעַט: לֹא אֶתְמַהְמֵהַּ

הַמְתִּינוּ לִי עוֹד מְעַט, יְלָדַי, וְלֹא אֶתְמַהְמֵהַּ. לֹא אַעַזְבְכֶם וְלֹא אֶטָּשְׁכֶם, כִּי אַתֶּם בָּבַת עֵינוֹ שֶׁל אָבִי וּבָבַת עֵינִי שֶׁלִּי. אַתֶּם כַּלָּתִי, וְאֵיזֶה חָתָן יַעֲזֹב אֶת כַּלָּתוֹ לְמִשְׁפַּט הַחוֹטְאִים?

דְּעוּ כִּי אֲנִי, הָאָדוֹן יֵשׁוּעַ הַמָּשִׁיחַ, אוֹשִׁיעַ אֶת עַמִּי. הַאִם לֹא קְרָאתֶם בְּכִתְבֵי הַקֹּדֶשׁ? הַאִם לֹא לָקַח אָבִי אֶת חֲנוֹךְ הַשָּׁמַיְמָה? הַאִם לֹא לָקַח אָבִי אֶת אֵלִיָּהוּ הַשָּׁמַיְמָה? הַאִם לֹא חִלֵּץ אָבִי אֶת בְּנֵי יִשְׂרָאֵל מִמִּצְרַיִם? הַאִם לֹא חִלֵּץ אֶת לוֹט מִסְּדוֹם בְּטֶרֶם יָרְדָה אֵשׁ עַל הֶעָרִים הָהֵן וְהִשְׁמִידָה אֶת כֻּלָּן? עַל אַחַת כַּמָּה וְכַמָּה אֲחַלֵּץ אֶת כַּלָּתִי, אֲשֶׁר קָנִיתִי בְּדָמִי שֶׁלִּי, מִידֵי הַמִּשְׁפָּט הֶעָתִיד לָבוֹא עַל הָעוֹלָם הַזֶּה. עַד מָתַי תְּיַחֲסוּ לִי אִוֶּלֶת, לִי אֲדוֹנְכֶם, וּלְאָבִי, אֱלֹהֵיכֶם?

נָתַתִּי לָכֶם חַיֵּי עוֹלָם, וְשָׂכָר גָּדוֹל צָפוּן לַנֶּאֱמָנִים בַּשָּׁמַיִם, לָכֵן אַל תִּירְאוּ, כִּי נַחֲלַתְכֶם קְרוֹבָה לְהִתְמַמֵּשׁ. אֲנִי הָרוֹעֶה הַצַּדִּיק הַמּוֹשִׁיעַ אֶת הָעֵדֶר מִפְּנֵי הַזְּאֵבִים הַטּוֹרְפִים. אוֹתָם זְאֵבִים יִשְׁמְעוּ בְּקָרוֹב: "סוּרוּ מִמֶּנִּי, פֹּעֲלֵי אָוֶן; מֵעוֹלָם לֹא יָדַעְתִּי אֶתְכֶם, אַף שֶׁאַתֶּם אוֹמְרִים, אָדוֹן, אָדוֹן!" אֲנִי הָרוֹעֶה הַנֶּאֱמָן אֲשֶׁר לֹא אִבֵּד אִישׁ מִכָּל אֵלֶּה שֶׁנָּתַן לִי הָאָב, מִלְּבַד בֶּן הָאֲבַדּוֹן אֲשֶׁר בָּגַד בִּי בִּשְׁלֹשִׁים כֶּסֶף, וְהַכָּתוּב אָכֵן הִתְקַיֵּם בִּפְשָׁעוֹ וּבְסֵרוּבוֹ. אֵילוּ לְקָחִים יֵשׁ בָּזֶה עֲבוּרְכֶם, יְלָדַי, אֶלָּא שֶׁלֹּא תּוּכְלוּ לֵאָבֵד מִיָּדִי וְאֵין אִישׁ יָכוֹל לָחְטֹף אֶתְכֶם מִמֶּנִּי?

אַל תִּירְאוּ מִפְּנֵי מִשְׁפַּט עוֹשֵׂי הָרֶשַׁע, נֶאֱמָנַי, וּדְעוּ כִּי חֲרוֹן אַפּוֹ שֶׁל אֲבִיכֶם לֹא יִגַּע בָּכֶם. מִי מִכֶּם הָאָבוֹת יִמְסֹר וְלוּ אֶחָד מִילָדָיו לִידֵי הָאוֹיֵב? אַף שֶׁזֶּהוּ חֵטְא נָפוֹץ בְּקֶרֶב בְּנֵי הָאָדָם, אֵין צֵל שֶׁל שִׁנּוּי בַּאֲבִי הָאוֹרוֹת! לָכֵן אַל תִּירְאוּ וִהְיוּ בְּשָׁלוֹם. חִדְלוּ מִמַּעֲשֵׂיכֶם וְנוּחוּ בִּי. רִדְפוּ אַחַר מַעֲשַׂי וְהִתְעַנְּגוּ בִּמְנוּחָה בִּי, בָּאָדוֹן יֵשׁוּעַ הַמָּשִׁיחַ, אֲשֶׁר נָתַן אֶת עַצְמוֹ קָרְבָּן עַל חֲטָאֵי כֹּל, פַּעַם אַחַת וּלְתָמִיד.

הוֹי, אַתֶּם יוֹשְׁבֵי תֵבֵל, חֲטָאֵיכֶם לְפָנַי, וְצָבַרְתֶּם לְעַצְמְכֶם זַעַם רַב וּמִשְׁפָּט! בִּמְקוֹם לִצְבֹּר נַחֲלַת בְּרָכוֹת, צְבַרְתֶּם זַעַם עַל זַעַם לַיּוֹם הַגָּדוֹל הַזֶּה הָעוֹמֵד לְהַשִּׂיגְכֶם. הוֹי, דִּבְרֵי הַהֶבֶל שֶׁדִּבַּרְתֶּם, בְּאָמְרְכֶם, "אַיֵּה הָאָדוֹן; אֵינוֹ אֶלָּא מִיתוֹס שֶׁלֹּא הָיָה וְלֹא נִבְרָא, וּמָחָר יִהְיֶה כְּהַיּוֹם וּכְאֶתְמוֹל." מִשְׁפַּטְכֶם יַשִּׂיגְכֶם פִּתְאוֹם, וְהַמְּאֹרָעוֹת יִתְרַחֲשׁוּ מַהֵר כָּל כָּךְ עַד שֶׁתֹּאבְדוּ בְּסֵבֶל. דִּבְרֵיכֶם הָרָעִים וּדְרָכֵיכֶם הָרָעוֹת יָבִיאוּ עֲלֵיכֶם חֻרְבָּן גָּדוֹל כְּהֶרֶף עַיִן. כָּל הַיָּקָר לָכֶם יִלָּקַח מִכֶּם, וְכָל אוֹצְרוֹתֵיכֶם יֵעָלְמוּ מִידֵיכֶם, כִּי נְתַתִּים בִּידֵי שׂוֹנְאֵיכֶם, כִּי שְׂנֵאתֶם אוֹתִי חִנָּם.

זַעַם הַשֶּׂה עוֹמֵד לָבוֹא עֲלֵיכֶם, וְחֶבְלֵי הַלֵּדָה מִתְקָרְבִים זֶה לָזֶה, כֹּה קְרוֹבִים עַד שֶׁאַתֶּם טוֹעִים בָּהֶם וְחוֹשְׁבִים אוֹתָם לְזַעַם הַשֶּׂה. דַּע, הוֹי אֱנוֹשׁ כְּסִיל, כִּי בְּאֹרֶךְ רוּחִי אֲנִי מַזְהִירְךָ מִפְּנֵי אֲבַדּוֹנְךָ, וְאַתָּה עוֹדְךָ לוֹעֵג בְּחֹסֶר אֱמוּנָה. בְּקָרוֹב מְאֹד אָסִיר אֶת הַמּוֹנֵעַ מִקִּרְבְּכֶם, וְאָסִיר אֶת כַּלָּתִי מִקִּרְבְּכֶם, וְחֻרְבָּן גָּמוּר יָבוֹא עֲלֵיכֶם פִּתְאוֹם. הַשָּׂטָן הֵזִיז אֶת צִבְאוֹתָיו וְאֶת כֵּלָיו לִמְקוֹמָם, מוּכָן לְשַׁחְרֵר עֲלֵיכֶם אֶת מְלוֹא כַּעֲסוֹ וְזַעֲמוֹ, וַאֲנִי נָתַתִּי אֶתְכֶם בְּיָדָיו. אֻמַּתְכֶם הַגְּדוֹלָה נִמְחֲקָה וְסֻמְּנָה לְחֻרְבָּן גָּמוּר; הוֹי, לוּ הֲבַנְתֶּם אֶת הַמִּלִּים, סוֹד בָּבֶל. אַךְ עֲדַיִן יֵשׁ זְמַן לַחֲזֹר בִּתְשׁוּבָה, אֲבָל רֻבְּכֶם לֹא תַּעֲשׂוּ זֹאת בְּדַרְכֵיכֶם הָעַקְשָׁנִיּוֹת, קְשׁוֹת הָעֹרֶף, שֶׁל מֶרֶד וּכְפִירָה.

מִשְׁפַּטְכֶם יָבוֹא עֲלֵיכֶם בִּשְׁתֵּי דְּרָכִים עִקָּרִיּוֹת: מִבִּפְנִים וּמִבַּחוּץ. הַזּוֹנָה הַגְּדוֹלָה תִּקָּרַע לִגְזָרִים; הִיא תְּכֻלֶּה בַּמַּיִם וְתִשָּׂרֵף בָּאֵשׁ וּתְכֻסֶּה בְּמַגֵּפָה וּבְרָעָב. אָכֵן, הוֹי בֶּן אֱנוֹשׁ אַרְצִי, מִי יַצִּילְךָ מִיָּדִי? אִישׁ לֹא יָבוֹא לְהוֹשִׁיעֲךָ. חֻרְבָּנְךָ קָרוֹב לָבוֹא. הַצְּבָאוֹת הָאֲדֻמִּים נִכְנְסוּ לְאֻמַּתְכֶם מַמָּשׁ מִתַּחַת לְאַפְּכֶם, וְלֹא הָיָה לָכֶם אִכְפַּת כְּלָל כָּל עוֹד הָיִיתֶם מְבֻדָּרִים וּשְׂבֵעִים מֵעֹשֶׁר זְמַנִּי שֶׁמֵּעוֹלָם לֹא הָיָה שַׁיָּךְ לָכֶם, כִּי לַיהוָה הָאָרֶץ וּמְלוֹאָהּ. לֹא רַק שֶׁחֲרַצְתֶּם אֶת דִּין חֻרְבַּנְכֶם, אֶלָּא שֶׁמְּכַרְתֶּם אֶת יַלְדֵיכֶם וְאֶת נִכְדֵיכֶם לְחֻרְבָּן גָּמוּר, וַעֲדַיִן אַתֶּם לוֹעֲגִים. צֶאֱצָאֵיכֶם אֲבוּדִים. לֹא תִּלְעֲגוּ עוֹד זְמַן רַב, כִּי כָּל דַּרְכֵיכֶם הָרָעוֹת וְהַמֻּשְׁחָתוֹת יַעַטְפוּ אֶתְכֶם בְּיָגוֹן, בְּעִנּוּיִים וּבְחֻרְבָּן, אַף שֶׁהָמוֹן רַב יְטֹהַר דֶּרֶךְ הַצָּרָה הַגְּדוֹלָה.

כָּל אֵלֶּה הַפּוֹנִים אֵלַי וְקוֹרְאִים בִּשְׁמִי וְאֵינָם אוֹהֲבִים אֶת חַיֵּיהֶם עַד קֵץ, אֶגְמֹל לָהֶם לָנֶצַח! לֹא יִסְבְּלוּ עוֹד לְעוֹלָם, אַךְ סִבְלְכֶם לֹא יִגָּמֵר לְעוֹלָם, הוֹי רְשָׁעִים. אֲבָל אַתֶּם, הוֹי אֱנוֹשׁ טִבְעִי הַדּוֹחֶה אֶת אֱלֹהִים, אֲשֶׁר הִכְרַזְתֶּם מִלְחָמָה נֶגֶד בּוֹרַאֲכֶם וְחָשַׁקְתֶּם בְּמֶרֶד נֶגְדִּי וְנֶגֶד דְּרָכַי. חֻרְבַּנְכֶם יָבוֹא פִּתְאוֹם וּלְלֹא מַרְגּוֹעַ. הֲפַכְתֶּם לְאֻמָּה שֶׁל זוֹנוֹת, וְכַמָּה הוֹלֵם שֶׁאֻמּוֹתֵיכֶם מְכֻנּוֹת זוֹנָה בְּכִתְבֵי הַקֹּדֶשׁ. הַצְּבָאוֹת יְכַלּוּ אֶת מַה שֶּׁנּוֹתַר מֵאֻמַּתְכֶם שֶׁהָיְתָה פַּעַם גְּדוֹלָה וְאַדִּירָה, וְטִהוּר גָּדוֹל יָבוֹא עַל הָאָרֶץ, וְעֹנֶשׁ כָּבֵד יְכַלֶּה אֶתְכֶם בִּגְלַל דַּרְכֵיכֶם הָרָעוֹת.

לָקַחְתִּי מִכֶּם הַכֹּל בִּגְלַל דַּרְכֵיכֶם הַמַּרְדָּנִיּוֹת וּקְשׁוֹת הָעֹרֶף; וְאַף שֶׁשִּׁלַּמְתִּי עַל חֲטָאֵיכֶם, דְּחִיתֶם אֶת הַצָּעָתִי, וּבָחַרְתֶּם בִּמְקוֹם זֹאת לְשַׁלֵּם עֲלֵיהֶם בְּעַצְמְכֶם, וְתִתְחָרְטוּ עַל כָּךְ לָנֶצַח נְצָחִים. כַּאֲשֶׁר יֵעָקֵר הַמּוֹנֵעַ, חֻרְבָּן מִיָּדִי יִהְיֶה בֶּן לְוָיַתְכֶם; וְהִקְרַבְתֶּם אֶת יַלְדֵיכֶם לְאוֹתוֹ גּוֹרָל כְּמוֹכֶם, אַף שֶׁרַבִּים יִוָּשְׁעוּ. אֲפִלּוּ כַּמָּה מִיַּלְדֵיכֶם יִפְנוּ אֵלַי, וַאֲנִי אֲקַבְּלֵם לִכְבוֹדִי, אַף שֶׁמְּנַעְתֶּם אוֹתָם מִמֶּנִּי, וְאַתֶּם תֵּלְכוּ לָאֲבַדּוֹן בָּאֲגַם הַגָּדוֹל הַהוּא הַמְּצַפֶּה לַשָּׂטָן וּלְמַלְאָכָיו.

יְלָדַי הַקְּטַנִּים, אַל תִּירְאוּ. הַמְשִׁיכוּ לַעֲמֹד בְּנֶאֱמָנוּת לְמַעֲנִי כָּאן בָּרֶגַע הָאַחֲרוֹן הַזֶּה, אָכֵן, בָּרֶגַע הָאַחֲרוֹן שֶׁלָּכֶם בָּעוֹלָם הַגּוֹוֵעַ הַזֶּה. קָרָאתִי לָכֶם, בָּחַרְתִּי בָּכֶם, בָּרַרְתִּי אֶתְכֶם בְּיָדַי בִּזְכוּת אַהֲבַתְכֶם אֵלַי וְנֶאֱמָנוּתְכֶם, וְאוֹצָרוֹת גְּדוֹלִים מְצַפִּים לָכֶם בְּבֵית אָבִי. תִּלָּקְחוּ לְמָקוֹם בָּטוּחַ וְלִגְמוּל גָּדוֹל; אַף שֶׁכַּמָּה יִסְבְּלוּ הֶפְסֵד, כָּל בְּחִירַי יִוָּשְׁעוּ בִּמְלוֹאָם וְעַד תַּכְלִית.

הִתְמַקְּדוּ בִּי, הִתְחַבְּרוּ עִמִּי; קָרוֹב אֲנִי יוֹתֵר מִנְּשִׁימַתְכֶם הַבָּאָה. דְּעוּ, בְּחִירַי, כִּי אָכֵן נִלְחָם אֲנִי לְמַעַנְכֶם; וְאַף שֶׁרְדִיפוֹת וְצָרוֹת רַבּוֹת, תִּנָּצְלוּ מִן הַזַּעַם. לָכֵן, הִטַּהֲרוּ בְּצֵאתְכֶם מִן הָעוֹלָם הַזֶּה וּבְדַחְיַת דְּרָכָיו הָרָעוֹת. רַחֲצוּ אֶת יְדֵיכֶם מֵחֲטָאָיו וְסוּרוּ מֵעֲוֹנוֹתָיו, כִּי כָּל הַשּׁוֹכֵב בְּמִטַּת הַזּוֹנָה יִסְבֹּל מְצוּקָה. הִטַּהֲרוּ בְּדָמִי הַמְכַפֵּר וְרַחֲצוּ עַצְמְכֶם כָּלִיל. הִנֵּה חָתַמְתִּי אֶתְכֶם בְּחוֹתָמוֹ שֶׁל אָבִי, וְאֶגְאַל וְאֶבְרָא אֶתְכֶם מֵחָדָשׁ בְּצַלְמִי שֶׁלִּי. אָכֵן, תִּהְיוּ אָדָם חָדָשׁ וּבְרִיאָה חֲדָשָׁה אַתֶּם בִּי, וּבְקָרוֹב יִהְיוּ שָׁמַיִם חֲדָשִׁים וְאֶרֶץ חֲדָשָׁה, כִּי אָבִי עוֹשֶׂה הַכֹּל חָדָשׁ. לָכֵן עִמְדוּ אֵיתָן וּדְעוּ כִּי אַהֲבָתִי אֵשׁ לוֹהֶטֶת הִיא, וְאַתֶּם יְקָרִים וַאֲהוּבִים עָלַי מְאֹד. בְּקָרוֹב תִּהְיוּ עִמִּי.

אֲנִי אֲדוֹנְכֶם הַחַי וּמוֹשִׁיעֲכֶם, יֵשׁוּעַ הַמָּשִׁיחַ.

אֲנִי הַשּׁוֹפֵט הַצַּדִּיק הַשּׁוֹקֵל הַכֹּל בַּמֹּאזְנַיִם.

אֲנִי שֹׁרֶשׁ יִשַׁי אֲשֶׁר אֵשֵׁב עַל כִּסֵּא דָוִד בְּמַלְכוּתִי הַקְּרֵבָה.

אֲנִי בֶּן הָאֱלֹהִים אֲשֶׁר בָּא בַּבָּשָׂר וְנִצְלַב עַל חֲטָאֵיכֶם, בְּתָפְשִׂי אֶת מְקוֹמְכֶם בַּמִּשְׁפָּט.

אֲנִי בֶּן הָאָדָם שֶׁנֻּבָּא בִּימֵי קֶדֶם.

אֲנִי הוּא הַמִּתְהַלֵּךְ בְּתוֹךְ שֶׁבַע הַמְּנוֹרוֹת.

אֲנִי אוֹר הָעוֹלָם.

אֲנִי לֶחֶם הַחַיִּים.

אֲנִי הוּא אֲשֶׁר מִבִּטְנוֹ נוֹבְעִים נַהֲרוֹת מַיִם חַיִּים.

אֲנִי הוּא אֲשֶׁר עָשָׂה אֶתְכֶם חֲדָשִׁים וּבָרָא אֶתְכֶם מֵחָדָשׁ.

הִשָּׁאֲרוּ אֵיתָנִים, נֶאֱמָנַי, כִּי גְאֻלַּתְכֶם קְרוֹבָה מִכְּפִי שֶׁתְּתָאֲרוּ.
