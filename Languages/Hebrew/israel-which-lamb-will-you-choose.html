עֵד כָּפוּל: יִשְׂרָאֵל
הַמַּר וְהַמָּתוֹק
13 בנובמבר 2024

הוֹ יִשְׂרָאֵל! בְּאֵיזֶה שֶׂה תִּבְחָרוּ?

שִׁמְעוּ אֶת דִּבְרֵי יהוה הָאֲנִי הַגָּדוֹל וְרוּצוּ לְקוֹל דְּבָרַי לְמַעַן תְּקַבְּלוּ חָכְמָה, וְלְמַעַן יֶחְכַּם הֶחָכָם שֶׁבָּכֶם עוֹד יוֹתֵר.

הִנֵּה! בָּאתִי לְהַגִּיד לָכֶם עַל הֶעָתִיד לָבוֹא, לְמַעַן תֵּדְעוּ אֶת וַדָּאוּת דְּבָרַי וּלְמַעַן תַּגִּיעוּ לִישׁוּעַתְכֶם הַשְּׁלֵמָה בְּיֵשׁוּעַ הַמָּשִׁיחַ, אֲשֶׁר הוּא בֶּאֱמֶת מְשִׁיחֲכֶם הָאֲמִתִּי, מַלְכְּכֶם, אֲדוֹנְכֶם וְכֹהַנְכֶם הַגָּדוֹל לְעוֹלָם עַל-דִּבְרָתִי מַלְכִּי-צֶדֶק.

עַל כֵּן, אֲנִי מְצַוֶּה עֲלֵיכֶם לִשְׁמֹעַ בּוֹ בְּכָל יֵשׁוּתְכֶם וּלָמוּל אֶת לְבַבְכֶם בְּצַיְתָנוּת! כִּי יֵשׁוּעַ הַמָּשִׁיחַ הוּא שֶׁנִּתַּן לֹא רַק לִישׁוּעַת יִשְׂרָאֵל, אֶלָּא גַּם לִישׁוּעַת כָּל גֶּזַע אָדָם, כִּי כָּל הַקּוֹרֵא בְּשֵׁם אֲדֹנָי עַד קֵץ כָּל הַדּוֹרוֹת יִוָּשַׁע, בֵּין אִם יִשְׂרְאֵלִי הוּא וּבֵין אִם מִן הַגּוֹיִם, וְיֵשׁוּעַ הַמָּשִׁיחַ יוֹשִׁיעַ בַּחַיִּים שְׁאֵרִית מִיִּשְׂרָאֵל, אַף מִגֶּזַע אָדָם, אֲשֶׁר תַּמְשִׁיךְ אֶל מַלְכוּת הָאֶלֶף.

וְכֵן הוּא, שֶׁאֲנִי סִפַּקְתִּי אֶת מַה שֶׁהָיָה בִּלְתִּי אֶפְשָׁרִי לְגֶזַע אָדָם לְסַפֵּק לִישׁוּעָתוֹ שֶׁלּוֹ, כִּי גֶּזַע אָדָם יָכוֹל הָיָה לִמְצֹא גְּאֻלָּה שְׁלֵמָה רַק בְּדַם הַשֶּׂה. כִּי אָנֹכִי, יהוה, הָאֲנִי הַגָּדוֹל, הוּא שֶׁסִּפַּקְתִּי אֶת פִּסְחֲכֶם הַמֻּשְׁלָם, פַּעַם אַחַת, עַתָּה וּלְעוֹלְמֵי עוֹלָמִים, וּלְכָל הַזְּמַנִּים קָרְבָּנוֹ מְקֻבָּל. לָכֵן, אֵין עוֹד צֹרֶךְ בְּקָרְבָּן, הוֹ יִשְׂרָאֵל.

כַּכָּתוּב בָּאִגֶּרֶת אֶל הָעִבְרִים, כֵּן אוֹמֵר אֲנִי שׁוּב לָכֶם עַתָּה: כִּי אִם דַּם פָּרִים וּשְׂעִירִים וְאֵפֶר פָּרָה הַמַּזֶּה עַל הַטְּמֵאִים מְקַדֵּשׁ לְטָהֳרַת הַבָּשָׂר, עַל אַחַת כַּמָּה וְכַמָּה דָּמוֹ שֶׁל הַמָּשִׁיחַ, אֲשֶׁר הִקְרִיב עַצְמוֹ לֵאלֹהִים בְּרוּחַ עוֹלָמִים תָּמִים בְּלִי מוּם, יְטַהֵר אֶת מַצְפּוּנְכֶם מִמַּעֲשֵׂי מָוֶת לַעֲבֹד אֶת אֱלֹהִים חַיִּים? וְעַל כֵּן אֲנִי יוֹעֵץ לָכֶם, הוֹ יִשְׂרָאֵל, קַבְּלוּ אֶת קָרְבָּנִי, כִּי לֹא נוֹתַר עוֹד קָרְבָּן עַל חֵטְא!

לָכֵן הָבִינוּ, יִשְׂרָאֵל, שֶׁאֵין אֶפְשָׁרוּת לְאִישׁ לָגֶשֶׁת עוֹד אֶל אֱלֹהֵי הַצֶּדֶק הַנִּצְחִי דֶּרֶךְ מֹשֶׁה, כִּי שַׂמְתִּי אֶת יֵשׁוּעַ הַמָּשִׁיחַ כַּדֶּרֶךְ, הָאֱמֶת וְהַחַיִּים, וְאֵין אִישׁ בָּא אֶל יהוה בִּלְתִּי דֶּרֶךְ יֵשׁוּעַ הַמָּשִׁיחַ!

וְצִוִּיתִי עַל צַיְתָנוּתְכֶם לְקוֹלוֹ, אַף דֶּרֶךְ הַתּוֹרָה, שָׁם קוֹלוֹ שֶׁל מֹשֶׁה עַצְמוֹ זוֹעֵק עַד הַיּוֹם כְּעֵד נֶגְדְּכֶם, וּבְדַרְכֵיכֶם הַבַּעֲרוֹת וַחֲסֵרוֹת הָאֱמוּנָה, תַּחַת צַיְתָנוּת, אַתֶּם מְבַקְשִׁים בְּבַעֲרוּת לְהָקִים מִחָדָשׁ מִקְדָּשׁ מַעֲשֵׂה יְדֵי אָדָם, בְּעוֹדְכֶם מוּלָכִים שׁוֹלָל עַל יְדֵי עֵשָׂו! כִּי דְּחִיתֶם אֶת הַמִּקְדָּשׁ שֶׁבָּנָה בּוֹרַאֲכֶם, מִקְדָּשׁ שֶׁלֹּא נַעֲשָׂה בְּיָדַיִם, הוּא הַמִּקְדָּשׁ שֶׁעָשָׂה יהוה עַצְמוֹ לְיִשְׂרָאֵל שֶׁל אֱלֹהִים, אֲשֶׁר יֵשׁוּעַ הַמָּשִׁיחַ הוּא אֶבֶן הַפִּנָּה שֶׁלּוֹ. לָכֵן, חִדְלוּ מִמַּעֲשֵׂיכֶם שֶׁלָּכֶם וְנוּחוּ בְּמַעֲשָׂיו שֶׁל מְשִׁיחֲכֶם הָאֲמִתִּי, וַאֲנִי אֶתֵּן לָכֶם שַׁבָּת.

לָכֵן אֲנִי מְצַוֶּה עֲלֵיכֶם, הוֹ יִשְׂרָאֵל, וַאֲנִי הוּא שֶׁנָּתַתִּי אֶת עֶצֶם הַדְּבָרִים שֶׁדִּבֵּר מֹשֶׁה לַאֲבוֹתֵיכֶם בְּסִינַי, וּלְדְבָרִים אֵלֶּה אֲנִי מְצַוֶּה עַל צַיְתָנוּתְכֶם תָּמִיד וּלְעוֹלְמֵי עוֹלָמִים, אַף בְּאִי-צַיְתָנוּתְכֶם הַזְּדוֹנִית וְקַשַּׁת הָעֹרֶף. וְכֹה אָמַר מֹשֶׁה לְמַעַן צַיְתָנוּת כֻּלָּם: נָבִיא מִקִּרְבְּךָ מֵאַחֶיךָ כָּמֹנִי יָקִים לְךָ יהוה אֱלֹהֶיךָ, אֵלָיו תִּשְׁמָעוּן!

בְּעוֹד שֶׁמֹּשֶׁה הָיָה הַמְתַוֵּךְ שֶׁל הַבְּרִית הָרִאשׁוֹנָה (הַחֲרוּתָה בְּאֶצְבָּעִי בְּסִינַי), שֶׁנִּתְּנָה לְהַרְשִׁיעַ אֶת הַחֵטְא בַּבָּשָׂר, וְאוֹתָהּ בְּרִית זוֹרַחַת כָּעֵת בְּפָנָיו שֶׁל יֵשׁוּעַ הַמָּשִׁיחַ, הוּא אֲדוֹנְכֶם וּמְשִׁיחֲכֶם הָאֲמִתִּי! וְאַף עַל פִּי כֵן, תּוֹרַת מֹשֶׁה הִיא צֵל הַדְּבָרִים הַבָּאִים, דְּבָרִים אֲשֶׁר הִתְגַּשְּׁמוּ בַּמָּשִׁיחַ יֵשׁוּעַ, וַאֲנִי עֲשִׂיתִיו לְרוּחַ מְחַיָּה, וְאֵין עוֹד מַכִּירִים אֶת יֵשׁוּעַ הַנָּצְרִי לְפִי הַבָּשָׂר!

וּכְפִי שֶׁהֻשְׂלַם בְּיָדִי הַחֲזָקָה, בְּעֶצֶם אֶצְבַּע יהוה הָאֲנִי הַגָּדוֹל, חָרַתְתִּי אֶת חֻקּוֹתַי בְּלוּחוֹת אֶבֶן בְּסִינַי, שָׁם נָתַתִּי לְמֹשֶׁה אֶת חֻקּוֹתַי לְמַעַן יִחְיֶה יִשְׂרָאֵל לְפָנַי. אַךְ מַה שֶׁנִּתַּן לְיִשְׂרָאֵל בְּסִינַי הָיָה לְמַעֲשֶׂה הַרְשָׁעַת הַחֵטְא בַּבָּשָׂר, כְּדֵי שֶׁיִּהְיֶה לְיִשְׂרָאֵל אוֹמֵן שֶׁיּוֹבִילֵם אֶל מְשִׁיחָם הָאֲמִתִּי, הוּא יֵשׁוּעַ הַמָּשִׁיחַ אֲשֶׁר שָׁלַחְתִּי לִישׁוּעַתְכֶם הַשְּׁלֵמָה.

וְכֵן הוּא, תּוֹרַת מֹשֶׁה טוֹבָה הִיא, וִישָׁרָה, וּמֻשְׁלֶמֶת, וּקְדוֹשָׁה, וְאִם יְקַלֵּל אִישׁ אֶת תּוֹרַת מֹשֶׁה, יִשָּׁפֵט נָא בָּהּ בְּכָל הַחֻמְרָה הַקְּבוּעָה תַּחַת קִלְלַת וְעֹנֶשׁ תּוֹרַת מֹשֶׁה. וְאַף עַל פִּי כֵן, מַה שֶּׁטּוֹב וְיָשָׁר וּמֻשְׁלָם וְקָדוֹשׁ פָּעַל מָוֶת לְכָל הַנִּדּוֹנִים בּוֹ בְּחֵטְא הַבָּשָׂר, כִּי בְּאֵין תּוֹרָה אֵין חֵטְא נֶחְשָׁב, וְעַל כֵּן נִתְּנָה תּוֹרַת מֹשֶׁה לְגַלּוֹת וּלְהַרְשִׁיעַ אֶת הַחֵטְא בַּבָּשָׂר.

וְאַף עַל פִּי כֵן, דַּוְקָא בְּרוּחַ הַתּוֹרָה וְלֹא בְּאוֹתָהּ הַכְּתוּבָה מוֹצֵאת הָאֱנוֹשׁוּת רַחֲמִים בְּאַהֲבָתִי אֲלֵיהֶם, כִּי אוֹת הַתּוֹרָה מְבִיאָה חֻמְרָה (הַאִם לֹא קְרָאתֶם 'חֶסֶד חָפַצְתִּי וְלֹא-זָבַח'?), וְלָכֵן בְּאַהֲבָתִי לִבְנֵי יִשְׂרָאֵל אֲשֶׁר בְּכַוָּנָתִי עוֹד לְהוֹשִׁיעַ אִם תִּהְיוּ צַיְתָנִים לְיֵשׁוּעַ הַמָּשִׁיחַ, קָבַעְתִּי קָרְבָּן, כִּי בְּלִי שְׁפִיכַת דָּם אֵין מְחִילַת חֲטָאִים. וְכָךְ בְּיָדוֹ הַמַּשְׁגִּיחָה שֶׁל מֹשֶׁה נִשְׁמַר עַמִּי עַד בּוֹא הַמָּשִׁיחַ, אַף שֶׁרַבִּים מִכֶּם פְּנִיתֶם לֶאֱלִילִים וְאוֹהֲבִים אֶת קֻבִּיּוֹתֵיכֶם הַשְּׁחֹרוֹת, דְּבָרִים אֲשֶׁר הֵם תּוֹעֵבָה גְּמוּרָה בְּעֵינַי, הוֹ יִשְׂרָאֵל, לָכֵן סוּרוּ מֵהֶם מִיָּד!

לָכֵן הָבִינוּ אֶת כַּעֲסִי עֲלֵיכֶם, הוֹ יִשְׂרָאֵל, כִּי אָנֹכִי יהוה, הָאֲנִי הַגָּדוֹל, הוּא אֵל שַׁדַּי, אֱלֹהֵי הַצֶּדֶק הַנִּצְחִי וְהַכֹּל יָכוֹל אֲשֶׁר לֹא יִיעַף וְאֵין דָּבָר קָשֶׁה מִדַּי לְיָדִי הַחֲזָקָה לְהוֹצִיא לַפּוֹעַל. וַאֲנִי הוּא אֵל שַׁדַּי, הָאֵל הַכֹּל יָכוֹל אֲשֶׁר יָצַר אֶת הַבְּרִיאָה בְּחָכְמָתוֹ הַנִּצְחִית, וּבְכֹחַ רוּחוֹ הָאַדִּיר כָּל הַדְּבָרִים מִתְקַיְּמִים יַחְדָּו בְּצַיְתָנוּת לְקוֹלִי.

אָנֹכִי אֱלֹהֵיכֶם, אֲשֶׁר רִבְבוֹת רְבָבוֹת סוֹבְבִים אֶת כִּסְאִי, צְבָא מַלְאָכִים אַדִּיר שֶׁל רוּחוֹת מְשָׁרְתוֹת אֲשֶׁר שָׁרָיו לַהֶבֶת אֵשׁ, הַמִּצְטַיְּנִים בְּחָכְמָה וּבְרוּאִים בִּבְרִילְיַנְטִיּוּת וּבִשְׁלֵמוּת, כְּמוֹ שֶׁבָּרָאתִי אוֹתָם בְּאַהֲבָתִי אֲלֵיהֶם! וּבְאַהֲבָתָם אֵלַי, הֵם צַיְתָנִים בִּשְׁלֵמוּת לִרְצוֹנִי הַנִּצְחִי, וְנָתַתִּי לְיֵשׁוּעַ הַמָּשִׁיחַ שִׁלְטוֹן עַל כֻּלָּם.

אַךְ אָנֹכִי, יהוה הָאֲנִי הַגָּדוֹל, הַמֻּקָּף בִּצְבָא מַלְאָכִים שֶׁאֵין לוֹ מִסְפָּר, אֵינֶנִּי זָקוּק לְיוֹעֵץ בְּחָכְמָתִי הַנִּצְחִית, וְעַל בְּרִיאָתִי אֲנִי רִבּוֹן וּפוֹעֵל אֶת הַכֹּל עַל-פִּי עֲצַת רְצוֹנִי, וּמִכָּל חַכְמֵי כָּל הַנְּצָחִים אֵין חָכָם מֵהָאֲנִי הַגָּדוֹל! לָכֵן שִׁמְעוּ בְּקוֹלִי וְהַחֲרִישׁוּ לְשׁוֹנוֹתֵיכֶם אִם תִּבְחֲרוּ בְּחַיֵּי נֶצַח עַל פְּנֵי לֶהָבוֹת עֹנֶשׁ שֶׁאֵינָן כָּבוֹת, הוֹ יִשְׂרָאֵל, כִּי זֹאת הַבְּחִירָה שֶׁכָּל אָדָם חַיָּב לַעֲשׂוֹת.

וּבְחָכְמָתִי הַנִּצְחִית שָׁלַחְתִּי אֶת יֵשׁוּעַ הַנָּצְרִי לְיִשְׂרָאֵל, בְּעֵת שֶׁיְּהוּדָה נִמְשְׁלָה עַל יְדֵי פּוֹנְטִיּוּס פִּילָטוֹס, בְּעֵת שֶׁהַכֹּהֲנִים הַגְּדוֹלִים וְהַסַּנְהֶדְרִין הֶאֱשִׁימוּהוּ בְּאִי-צֶדֶק, וּבִידֵי רְשָׁעִים נִשְׁחַט וְנִצְלַב שֵׂה הָאֱלֹהִים, כְּדֵי שֶׁגּוּף הַחֵטְא הַכּוֹבֵל אֶת אָדָם יָשְׁמַד בְּכָל אֵלֶּה הַחַיִּים בְּצַיְתָנוּת דֶּרֶךְ אֱמוּנָה, כִּי צַדִּיק בֶּאֱמוּנָתוֹ יִחְיֶה! וְאָנֹכִי הוּא, אֵל שַׁדַּי (הָאֵל הַכֹּל יָכוֹל) אֲשֶׁר שַׂמְתִּי אֶת כָּל אֲשֶׁר בַּשָּׁמַיִם וּבָאָרֶץ בְּיָדָיו הָאַדִּירוֹת שֶׁל הַמָּשִׁיחַ יֵשׁוּעַ, הוּא יֵשׁוּעַ הַמָּשִׁיחַ וְהוּא הָאַרְיֵה מִשֵּׁבֶט יְהוּדָה אֲשֶׁר אָנֹכִי שׁוֹלֵחַ לִשְׁפֹּט וְלַעֲשׂוֹת מִלְחָמָה וְלִישׁוּעַת הַנֶּאֱמָנִים וְהַצַּיְתָנִים. לָכֵן, שִׁמְעוּ בִּדְבָרַי בְּצַיְתָנוּת לִמְשִׁיחֲכֶם הָאֲמִתִּי, כִּי אֵין אַחֵר!

לָכֵן הָבִינוּ אֶת כַּעֲסִי עֲלֵיכֶם, הוֹ יִשְׂרָאֵל, שֶׁבְּעִוָּרוֹנְכֶם הַחֶלְקִי שֶׁיְּרַשְׁתֶּם מֵאֲבוֹתֵיכֶם קְשֵׁי הָעֹרֶף, שֶׁאֶת קוֹלָם הַתַּלְמוּדִי אַתֶּם מוֹקִירִים יוֹתֵר מֵאֱלֹהֵיכֶם. כֵּן, אַתֶּם מַחֲשִׁיבִים אֶת מָסֹרֶת חַכְמֵיכֶם כִּבְיָכוֹל יוֹתֵר מִדִּבְרֵי הָאֲנִי הַגָּדוֹל, הוּא בּוֹרֵא הָעוֹלָם הַנִּצְחִי וְהַקַּיָּם שֶׁל כָּל אֲשֶׁר נִבְרָא! כִּי עַל-יְדֵי יהוה נִבְרָא הַכֹּל, אֲשֶׁר בַּשָּׁמַיִם וַאֲשֶׁר בָּאָרֶץ, הַנִּרְאֶה וְהַבִּלְתִּי נִרְאֶה: אִם כִּסְאוֹת וְאִם מֶמְשָׁלוֹת, אִם שִׂרְרוֹת וְאִם רָשֻׁיּוֹת, הַכֹּל נִבְרָא עַל-יָדִי וּלְמַעֲנִי, וַאֲנִי קֹדֶם לַכֹּל, וּבְיהוה הַכֹּל מִתְקַיֵּם.

אַךְ אַתֶּם חוֹשְׁבִים שֶׁאַתֶּם יוֹעֲצַי וּבְטִפְּשׁוּת מְכַנִּים עַצְמְכֶם עַמִּי הַנִּבְחָר, כָּל זֹאת בְּעוֹדְכֶם מְסָרְבִים לִדְבָרַי בְּאִי-צַיְתָנוּת חַסְרַת אֱמוּנָה? הוֹ יִשְׂרָאֵל, סַרְתֶּם אַחֲרֵי אֱלִילֵי לֵב אִלְּמִים וּדְבָרָיו הַשִּׁקְרִיִּים שֶׁל עֵשָׂו, שֶׁאֶת עֲצָתוֹ אַתֶּם מַעֲדִיפִים עַל פְּנֵי אֱלֹהֵי הַצֶּדֶק הַנִּצְחִי, הוּא אֱלֹהֵיכֶם! וְאַתֶּם מוֹעֲדִים בְּעִוָּרוֹן חֶלְקִי כִּי בְּמֹשֶׁה לִבּוֹתֵיכֶם מְכֻסִּים. אַף עַל פִּי כֵן, כַּאֲשֶׁר יִפְנֶה לְבַבְכֶם אֶל מְשִׁיחֲכֶם הָאֲמִתִּי, יָסוּר הַמַּסְוֶה וּבִמְקוֹם עִוָּרוֹן חֶלְקִי, תִּרְאוּ!

מִי אַתֶּם, הוֹ יִשְׂרָאֵל, שֶׁתְּמַנּוּ עַצְמְכֶם לְיוֹעֲצוֹ שֶׁל יהוה? אוּלָם, בִּמְקוֹם דִּבְרֵי יהוה הַנִּתָּנִים בְּפָנָיו שֶׁל יֵשׁוּעַ הַמָּשִׁיחַ, הֶאֱמַנְתֶּם לִשְׁקָרִים שֶׁעָבְרוּ בְּמָסֹרֶת זִקְנֵיכֶם שֶׁהָיוּ בְּשִׁעְבּוּד לְעֵשָׂו, שֶׁבִּכְתָבָיו כְּפַתֶּם עַצְמְכֶם לְעֵשָׂו כְּמוֹתָם וּלְעֵשָׂו אַתֶּם כְּבוּלִים בְּשַׁלְשְׁלָאוֹת! וַאֲנִי כָּתַבְתִּי בְּבֵרוּר לְמֹשֶׁה לָתֵת לַאֲבוֹתֵיכֶם בְּסִינַי: אֵת כָּל-הַדָּבָר אֲשֶׁר אָנֹכִי מְצַוֶּה אֶתְכֶם אֹתוֹ תִשְׁמְרוּ לַעֲשׂוֹת, לֹא-תֹסֵף עָלָיו וְלֹא תִגְרַע מִמֶּנּוּ. וְשׁוּב כָּתַבְתִּי בְּחָכְמַת שְׁלֹמֹה, חָכְמָה הָעוֹמֶדֶת לָנֶצַח: כָּל-אִמְרַת אֱלוֹהַּ צְרוּפָה, מָגֵן הוּא לַחֹסִים בּוֹ. אַל-תּוֹסְףְ עַל-דְּבָרָיו פֶּן-יוֹכִיחַ בְּךָ וְנִכְזָבְתָּ.

הוֹ יִשְׂרָאֵל, הַלְוַאי שֶׁתַּרְכִּינוּ אֶת לִבּוֹתֵיכֶם בְּצַיְתָנוּת לַדְּבָרִים שֶׁנִּתְּנוּ לִישׁוּעַתְכֶם וּלְשִׂמְחַתְכֶם הַנִּצְחִית, אִלּוּ רַק תְּצַיְּתוּ שׁוּב לְדִבְרֵי בּוֹרַאֲכֶם! וְאָנֹכִי הוּא שֶׁיְּשַׁקֵּם אֶתְכֶם וְיָבִיא אֶתְכֶם אֶל תּוֹךְ הַצֹּאן הַנִּצְחִי הַנִּמְצָא תַּחַת יָדוֹ הָאַדִּירָה שֶׁל רוֹעִי הָרָאשִׁי, לוֹ נָתַתִּי הַכֹּל! הוּא יֵשׁוּעַ הַמָּשִׁיחַ, הָרוֹעֶה אֶת יִשְׂרָאֵל הָאֲמִתִּי שֶׁל אֱלֹהִים בִּשְׁלֵמוּת בְּאַהֲבָתוֹ הַצַּיְתָנִית לְקוֹלִי, דֻּגְמָה שֶׁאַתֶּם בָּזִים לָהּ. כִּי יֵשׁוּעַ הַמָּשִׁיחַ הוּא הָאָדָם הַיָּחִיד אֵי פַּעַם שֶׁמְּצַיֵּת בִּשְׁלֵמוּת לִרְצוֹן יהוה, הָאֲנִי הַגָּדוֹל, אֲשֶׁר בּוֹ חָפַצְתִּי תָּמִיד, לָכֵן עֲשִׂיתִיו לְרוּחַ מְחַיָּה, אִם כֵּן שִׁמְעוּ בּוֹ, בַּאֲשֶׁר שָׁלַחְתִּי לְמַעַנְכֶם, כִּי אֵין אַחֵר.

וְהִקְשֵׁיתֶם אֶת לִבּוֹתֵיכֶם כִּקְשִׁי אֶבֶן הַשְּׁתִיָּה הַשְּׁבוּיָה בִּידֵי בֵּית הַכְּנֶסֶת שֶׁל הַשָּׂטָן, כִּי דְּחִיתֶם אֶת דִּבְרֵי עֲצָתִי הַנִּתָּנִים בְּפָנָיו שֶׁל הַמָּשִׁיחַ יֵשׁוּעַ, אוֹתוֹ אַתֶּם דּוֹחִים בְּבַעֲרוּת, וּבְכָךְ דּוֹחִים אֶת תִּקְוַתְכֶם הַיְחִידָה לְשִׁקּוּם וְלִישׁוּעָה נִצְחִית וּלְעוֹלְמֵי עוֹלָמִים. וְעַל כֵּן אַתֶּם עוֹמְדִים עִוְרִים וַעֲרֻמִּים וּבְבוּשָׁה בְּחַטַּאתְכֶם, חוֹשְׁבִים בְּבַעֲרוּת שֶׁהֲפַכְתֶּם לְיוֹעֲצוֹ שֶׁל יהוה, הָאֲנִי הַגָּדוֹל!

לָכֵן אֲנִי נוֹתֵן לָכֶם לְהָבִין בְּעִוָּרוֹנְכֶם הַחֶלְקִי, שֶׁכְּשַׁלְתֶּם בְּאֶבֶן הַפִּנָּה שֶׁלִּי שֶׁהַבּוֹנִים שֶׁלָּכֶם דָּחוּ, בְּאֶבֶן הַנֶּגֶף שֶׁלִּי. כִּי יֵשׁוּעַ הַמָּשִׁיחַ הוּא אֶבֶן הַפִּנָּה, הִיא אֶבֶן הַשְּׁתִיָּה הָאֲמִתִּית, הוּא יֵשׁוּעַ הַנָּצְרִי אוֹתוֹ אַתֶּם דּוֹחִים בְּאִי-צַיְתָנוּת זְדוֹנִית, אַף שֶׁבָּחַרְתִּי אֶתְכֶם לְעַצְמִי לְעַם סְגֻלָּה וְלִבִּי נִשְׁבָּר.

שִׁמְעוּ, שִׁמְעוּ, הוֹ יִשְׂרָאֵל, וְהָבִינוּ כִּי אָנֹכִי הוּא הַחוֹרֵת אֶת תּוֹרָתִי הַנִּצְחִית, הַבָּאָה לִידֵי בִּטּוּי בִּרְצוֹנִי הַנִּצְחִי בְּתָכְנִית הַגְּאֻלָּה הַקַּיֶּמֶת שֶׁלִּי שֶׁעָשִׂיתִי עַל-פִּי עֲצַת רְצוֹנִי הַנִּצְחִי וְהַמֻּשְׁלָם, בְּלוּחוֹת אֶבֶן לְמַעַן צַיְתָנוּת יִשְׂרָאֵל בְּסִינַי! לָכֵן, אָנֹכִי הוּא שֶׁאֶכְתֹּב אֶת תּוֹרָתִי עַל לוּחוֹת הַבָּשָׂר שֶׁל לִבְּכֶם, כִּי בְּאַהֲבָתִי אֲלֵיכֶם, דַּי בִּקְצָת מִכֹּחִי הָאַדִּיר לַחֲרֹת אֶת תּוֹרָתִי עַל לוּחוֹתֵיכֶם הָאֲבָנִים, וּבְחַסְדִּי וּבְרַחֲמַי הַבָּאִים לִידֵי בִּטּוּי בְּפָנָיו שֶׁל יֵשׁוּעַ הַמָּשִׁיחַ, עוֹד אֲנַפֵּץ אֶת לוּחוֹת הָאֶבֶן שֶׁלָּכֶם לְתוֹךְ חֶרֶס הַיּוֹצֵר, לְמַעַן תִּבָּרְאוּ מֵחָדָשׁ לָנֶצַח בִּצְדָקָה וּבִקְדֻשָּׁה אֲמִתִּית, כְּמוֹ שֶׁנִּשְׁבַּעְתִּי בִּבְרִיתוֹתַי לַאֲבוֹתֵיכֶם שֶׁאֶעֱשֶׂה.

לָכֵן, הוֹ יִשְׂרָאֵל, אֲנִי מְצַוֶּה עֲלֵיכֶם לִשְׁמֹעַ בְּקוֹלוֹ שֶׁל יֵשׁוּעַ הַמָּשִׁיחַ, כִּי אֵין יְשׁוּעָה בְּשׁוּם שֵׁם אַחֵר שֶׁנִּתַּן מִלְּבַד יֵשׁוּעַ הַמָּשִׁיחַ הַצַּדִּיק, אוֹתוֹ אַתֶּם מְכַנִּים בְּבַעֲרוּת אֱלִיל הַגּוֹיִם, כִּי אַתֶּם מְשֻׁעְבָּדִים לְעֵשָׂו!

אַךְ כָּךְ כָּתוּב וְכָךְ הָיָה: מֵאַבְרָהָם עַד יִצְחָק, כְּשֶׁנּוֹלְדוּ לוֹ יַעֲקֹב וְעֵשָׂו, יַד יַעֲקֹב אָחֲזָה תְּחִלָּה בַּעֲקֵב עֵשָׂו. כִּי עֵשָׂו הוּא קֵץ הָעוֹלָם, וְיַעֲקֹב הוּא תְּחִלַּת הַבָּא אַחֲרָיו. יַד הָאָדָם הִיא בֵּין הֶעָקֵב לַיָּד. אַךְ כֵּן הוּא שֶׁכָּעֵת אֶפְעַל לְהַפְרִיד בֵּין עֵשָׂו לְיַעֲקֹב, כִּי עֵשָׂו הוּא בֵּית הַכְּנֶסֶת שֶׁל הַשָּׂטָן, הֵם זִזָּנִים רָעִים אֲשֶׁר נָגְעוּ בְּאֶרֶץ הַהַבְטָחָה פַּעַם נוֹסֶפֶת, וְיִשְׂרָאֵל שׁוּב נָגוּעַ בִּכְנַעֲנִים.

וְכַכָּתוּב כֵּן יֵעָשֶׂה, אוֹמֵר הָאֲנִי הַגָּדוֹל: כִּי-הִנֵּה הַיּוֹם בָּא, בֹּעֵר כַּתַּנּוּר, וְהָיוּ כָל-זֵדִים וְכָל-עֹשֵׂה רִשְׁעָה קַשׁ, וְלִהַט אֹתָם הַיּוֹם הַבָּא, אָמַר יהוה צְבָאוֹת, אֲשֶׁר לֹא-יַעֲזֹב לָהֶם שֹׁרֶשׁ וְעָנָף. וְזָרְחָה לָכֶם יִרְאֵי שְׁמִי שֶׁמֶשׁ צְדָקָה וּמַרְפֵּא בִּכְנָפֶיהָ, וִיצָאתֶם וּפִשְׁתֶּם כְּעֶגְלֵי מַרְבֵּק.

וּבְעוֹד יֵשׁוּעַ הַמָּשִׁיחַ מְאַסֵּף אֶת הַנֶּאֱמָנִים בַּקְּהִלָּה, הִיא גּוּפוֹ וְכַלָּתוֹ, אָנִיעַ כָּעֵת אֶת יֵשׁוּעַ הַמָּשִׁיחַ לַהַפְרָדָה הַסּוֹפִית בֵּין עֵשָׂו לְיַעֲקֹב, וְיַעֲקֹב עוֹד יִוָּשַׁע אַף דֶּרֶךְ נִסְיוֹנוֹת הַשָּׁבוּעַ הַשִּׁבְעִים שֶׁל דָּנִיֵּאל. לָכֵן, פֶּן-יִהְיֶה אִישׁ זוֹנֶה אוֹ-חָנֵף כְּעֵשָׂו אֲשֶׁר בְּתַבְשִׁיל אֶחָד מָכַר אֶת-בְּכֹרָתוֹ. כִּי יוֹדְעִים אַתֶּם כִּי-גַם אַחֲרֵי-כֵן כַּאֲשֶׁר רָצָה לָרֶשֶׁת אֶת-הַבְּרָכָה נִמְאַס, כִּי לֹא-מָצָא מָקוֹם לִתְשׁוּבָה, אַף כִּי-בִקֵּשׁ אֶת הַבְּרָכָה בִּדְמָעוֹת.

וְכֵן הָאֱנוֹשׁוּת נִמְצֵאת בֵּין יַד יַעֲקֹב לַעֲקֵב עֵשָׂו, אֶלָּא שֶׁעֵשָׂו מְבַקֵּשׁ לְהַנְחִית עַל יַעֲקֹב פְּצִיעַת רֹאשׁ סוֹפִית, דָּבָר שֶׁלֹּא אַרְשֶׁה. וְכָךְ יִהְיֶה, שֶׁכָּל יוֹשְׁבֵי תֵּבֵל שֶׁנּוֹתְרוּ לִצְרַת יַעֲקֹב לַעֲבֹר אֶת אֵשׁ הַמְּצָרֵף, אוֹ שֶׁיִּהְיוּ מִיַּעֲקֹב דֶּרֶךְ יֵשׁוּעַ הַנָּצְרִי, הֵם בְּנֵי הַיְרֻשָּׁה, אוֹ שֶׁיִּדָּחוּ בֵּין אַנְשֵׁי עֵשָׂו שֶׁמָּכַר אֶת בְּכֹרָתוֹ בִּנְזִיד עֲדָשִׁים. לָכֵן, הָאָדָם הַטִּבְעִי אוֹ שֶׁיִּקְרָא בְּשֵׁם יֵשׁוּעַ הַנָּצְרִי עַד קִצּוֹ כְּדֵי שֶׁיָּקוּם לִתְחִיָּה לְתוֹךְ יִשְׂרָאֵל שֶׁל אֱלֹהִים, אוֹ שֶׁגַּם הוּא יִמְכֹּר אֶת יְרֻשָּׁתוֹ הַפּוֹטֶנְצִיאָלִית וְהַנִּצְחִית תְּמוּרַת פַּת לֶחֶם שֶׁתִּנָּתֵן מִיָּדוֹ שֶׁל בַּר-אַבָּא, כִּי בַּר-אַבָּא יוֹבִיל אֶת בֵּית הַכְּנֶסֶת שֶׁל הַשָּׂטָן, וְהוּא שֵׂה הַתַּנִּין.

וְכָךְ תֻּצַּב הַמַּלְכֹּדֶת בָּאָרֶץ לִלְכֹּד אֶת יַעֲקֹב לְצָמִיתוּת, וּבַר-אַבָּא מֵבִיא רַמָּאוּת גְּדוֹלָה עִם שִׁלְטוֹנוֹ וְרֹעַ רַב יַשְׁלִים גַּם בַּיַּבָּשָׁה וְגַם בַּיָּם. אַךְ כָּךְ יִהְיֶה, שֶׁכַּאֲשֶׁר תִּשְׁתּוֹלֵלְנָה וְתִבְעַרְנָה לֶהָבוֹת הַמִּלְחָמָה, יַעֲלֶה בַּר-אַבָּא וְאַף יִטְעַן כִּי קָם לִתְחִיָּה לְהַגְשִׁים אֶת מַטְּרוֹת אֱלֹהִים, וְדִבְרֵי גִּדּוּף וְרֶשַׁע רַבִּים יִטְעַן. וְהוּא יָבִיא אֶת עֲנַף הַזַּיִת שֶׁלּוֹ וְיִטְעַן כִּי הוּא נֹחַ, שֶׁקָּם לִתְחִיָּה לְהוֹשִׁיעַ אֶת הָעוֹלָם מִן הַדִּין הַמְיֹעָד לָרְשָׁעִים.

כִּי הַחַיָּה מִן הָאֲדָמָה, שֶׁהִיא גַּם נְבִיא הַשֶּׁקֶר שֶׁלָּהֶם, תָּבִיא אֶת פִּתְרוֹנָהּ, כִּי הִיא שֶׂה עִם שְׁתֵּי קְרָנַיִם הַמְּדַבֵּר כְּתַנִּין. וְשִׁלְטוֹנוֹ יִהְיֶה בַּעַל טֶבַע כָּפוּל, כְּשֶׁיְּבַקֵּשׁ לְאַחֵד אֶת כָּל הָאֱנוֹשׁוּת תַּחַת שִׁלְטוֹן הַתַּנִּין, הוּא הַשָּׂטָן הַקַּדְמוֹן עַצְמוֹ. כִּי בַּר-אַבָּא אֵינוֹ בָּא לִמְשֹׁל לְבַדּוֹ, אַף שֶׁיִּטְעַן כִּי הוּא עַצְמוֹ אֱלֹהִים, אֶלָּא הוּא מֵבִיא בְּרִית שֶׁל פִּלּוּג שֶׁבָּהּ שְׁתֵּי קְרָנַיִם תִּשְׁלֹטְנָה גַּם בַּיָּם וְגַם בָּאָרֶץ, וּמִשְּׁתֵּיהֶן אַחַת הִיא הַקֶּרֶן הַקְּטַנָּה שֶׁל דָּנִיֵּאל. אַךְ אֲנִי הִבְטַחְתִּי שֶׁיִּשְׂרָאֵל לֹא יִהְיֶה חָצוּי עוֹד, וּמִשְּׁתֵּי מְדִינוֹתָיו לֹא יִהְיֶה לְעוֹלָם פִּתְרוֹן, אֶלָּא יֵשׁוּעַ הַמָּשִׁיחַ יְאַחֵד אֶת יִשְׂרָאֵל בִּשְׁלֵמוּת וּלְעוֹלְמֵי עוֹלָמִים.

(אַךְ כָּךְ יִהְיֶה, אֲהוּבַי, לָכֵן הָבִינוּ מַה שֶּׁעוֹד אַשְׁלִים דֶּרֶךְ יֵשׁוּעַ הַמָּשִׁיחַ: וְלָקַחְתִּי אֶת-בְּנֵי יִשְׂרָאֵל מִבֵּין הַגּוֹיִם אֲשֶׁר הָלְכוּ-שָׁם, וְקִבַּצְתִּי אֹתָם מִסָּבִיב, וְהֵבֵאתִי אוֹתָם אֶל-אַדְמָתָם. וְעָשִׂיתִי אֹתָם לְגוֹי אֶחָד בָּאָרֶץ בְּהָרֵי יִשְׂרָאֵל, וּמֶלֶךְ אֶחָד יִהְיֶה לְכֻלָּם לְמֶלֶךְ, וְלֹא יִהְיוּ עוֹד לִשְׁנֵי גוֹיִם, וְלֹא יֵחָצוּ עוֹד לִשְׁתֵּי מַמְלָכוֹת עוֹד. וְלֹא יִטַּמְּאוּ עוֹד בְּגִלּוּלֵיהֶם וּבְשִׁקּוּצֵיהֶם וּבְכֹל פִּשְׁעֵיהֶם, וְהוֹשַׁעְתִּי אֹתָם מִכֹּל מוֹשְׁבֹתֵיהֶם אֲשֶׁר חָטְאוּ בָהֶם, וְטִהַרְתִּי אֹתָם, וְהָיוּ-לִי לְעָם, וַאֲנִי אֶהְיֶה לָהֶם לֵאלֹהִים.)

הָאַחֶרֶת הִיא הַחַיָּה מִן הַיָּם, שֶׁעָלֶיהָ כָּתוּב: וְעַל-חֶמְדַּת נָשִׁים לֹא יָבִין וְעַל כָּל-אֱלוֹהַּ לֹא יָבִין, כִּי עַל-כֹּל יִתְגַּדָּל. אֶשְׁאַל אֶתְכֶם כָּעֵת וּרְאוּ אִם-זָכָר יָלֵד? אַךְ אִשְׁתּוֹ נִרְאֵית כְּאִשָּׁה עִם יְלָדִים, אַף שֶׁאֵינָם יְלָדָיו. וְהַחַיָּה מִן הַיָּם תַּעֲלֶה מְאֻחֶדֶת לְגַמְרֵי בְּמַטְּרָתָהּ עִם הַחַיָּה מִן הָאֲדָמָה, וְהֵן מַעֲמִידוֹת פָּנִים כְּאוֹיְבוֹת הַנִּרְאוֹת חֲלוּקוֹת בְּדֵעוֹת וּבְדָת.

אַךְ הֵן מְשַׁקְּרוֹת וּמַעֲמִידוֹת פָּנִים, וַאֲבִיהֶן הוּא אוֹתוֹ תַּנִּין שֶׁיָּצַר אֶת שְׁתֵּי הַחַיּוֹת הָרָעוֹת הַלָּלוּ לְמַטְּרוֹתָיו הַזְּדוֹנִיּוֹת, לְהַשְׁמִיד אֶת אָדָם כְּדֵי שֶׁיּוּכַל לִגְזֹל אֶת הָאָרֶץ לְעוֹלָם לְעַצְמוֹ, וְאֵין זֶה אֶפְשָׁרִי שֶׁיַּשְׁלִים אֶת תָּכְנִיּוֹתָיו הָרְשָׁעוֹת, כִּי לַיהוָה הָאָרֶץ וּמְלוֹאָהּ, כֹּה נְאֻם יהוה, הָאֲנִי הַגָּדוֹל!

כי בר-אבא מביא שתי מדינות אשר יתאחדו תחת הדרקון, ושלום כזב יהיה להם בעודם מבטיחים חיי נצח, בריאות ושלום, אם רק תיקחו את חותמם ותצטרפו אליהם, בעודם מערבבים ברזל וחומר, אך לא ידבקו זה בזה. הבינו, כל אלה מכם השוקלים לקשור עצמכם לנצח לנפולים באמצעות חותמם, כי הם יספרו שקרים רבים באומרם כי זוהי תרופת פלא שתעניק חיים ובריאות, ומושלמת להפליא בתכנונה!

אך כל הנוטלים את רעלם ישנו את עצמם לעד ולעולמי עולמים, שינוי אשר אין ממנו החלמה. כי לחלל את צלם יהוה אשר בו נברא אדם, פירושו להצטרף לחטאי הנפולים המתיימרים להיות אלים ואינם כאלה, ולכל העושים כן אין סליחה ואין להם שלום, ואל אותו אגם אש שאינו כבה ילכו לעולמי עד, כי כך בחרו.

אולם, כפי שהשלושה הטמאים יקומו, כך יקומו גם שני עדיי, והם יתנבאו 1260 ימים לבושים שק, ועדויות רבות יתנו. הם יבואו לישראל ויעידו על אלוהי הצדק הנצחי, ויעידו בעוצמה על ישוע המשיח, הוא ישוע הנצרתי, אשר בלעדיו אין איש יכול לבוא אל אל שדי! אולם, השלושה הטמאים יכריזו על ישוע המשיח כפורע חוק, ויוציאו מחוץ לחוק את הכתבים הקדושים אשר ישוע הוא הדבר שנהיה בשר. אך אותו אחד שהם מבקשים להוציא מחוץ לחוק ירמוס את ראשיהם עד תבוסה נצחית.

לכן, בימי עדותם, אם ירצה איש להרע להם, אש תצא מפיהם לכלות את אויביהם: כי אם ירצה איש להרע להם, כך יורע להם, ואם ירצה איש להרוג אותם, כך ייהרגו הם. להם הכוח לסגור את השמיים, שלא ירד גשם בימי נבואתם, וכוח על המים להפכם לדם, ולהכות את הארץ בכל מיני מכות, ככל אשר ירצו. וכך יהיו ימי עדותם.

והם ידברו על דברים רבים ויבשרו את הבשורה הנצחית, אותה אחת אשר נגלתה בעתה להתגשמות כל הדברים בפני ישוע המשיח. ואת עדויותיהם שלהם יתנו, ודברים רבים יגלו בימי עדותם, והכול יירשם לעולמי עד. ובסוף ימיהם, עליהם להיהרג, וגופותיהם תונחנה ברחוב העיר הגדולה, אשר ברוחניות נקראת סדום ומצרים, שם גם נצלב אדונכם, והם ישמחו מאוד ויחליפו מתנות על מותם. ובסוף שלושה ימים וחצי, ישמעו קול אדיר האומר "עלו הנה", והם יקומו לתחייה חיים לנצח ויילקחו שוב לשמיים, ובכך תושלמנה עדויותיהם.

על כן הבינו את הרמייה שבאה עד פתח דלתכם, ישראל, והבינו את הבחירה שעליכם לעשות. או שתקבלו את בר-אבא כמשיחכם הכוזב ותיקחו את חותמו, ובכך תצטרפו לעשו בעודכם מוכרים את בכורתכם בנזיד עדשים לנצח נצחים, או שתכניעו את לבבכם ותהיו צייתנים לקול יהוה, האל הגדול והנצחי, הוא "אהיה אשר אהיה" הגדול בפני ישוע המשיח, הוא המשיח ישוע אשר הקמתיו מן המתים והעליתיו השמיימה, והאדרתיו בתפארת נצח, ולשלטונו ומלכותו לא יהיה קץ, ואין איש יכול לגשת אל יהוה בשום שם אחר אשר ניתן, בין אם בשמיים, בארץ, או מתחת לארץ, לכל הזמנים ולעולמי עולמים. כי במשיח הכול יחיו וייעשו כאדם חדש אחד לנצח, כי המשיח ישוע הוא אדם האחרון.

והם יילחמו בקדושים ויגברו עליהם, כי ניתן להם כוח לעשות כן. אך הם לא יגברו על ישראל, לא, שארית ישראל עתידה להיוושע כאשר ישובו אל ישוע המשיח בכל לבבם.

לכן שמעו את דבריי, הוי ישראל, למען תחיו לפניי, ואקבץ אתכם אליי תחת סמכותו האדירה של משיחכם האמיתי, הוא ישוע המשיח. וכתוב: "אך לא נאמר כי נפל דבר אלוהים, כי לא כל אשר מישראל, ישראל המה. ולא מפני שהם זרע אברהם, כולם בנים, אלא: ביצחק יקרא לך זרע. כלומר, לא בני הבשר הם בני האלוהים, אלא בני ההבטחה נחשבים לזרע".

אנוכי יהוה, אהיה אשר אהיה הגדול, ואנוכי קורא לכם, ישראל, למען תבוא שארית נאמנה אל ישוע המשיח, כדי שיושיע אתכם מן המלכודת שבר-אבא טומן בארץ להשמדתו הסופית של יעקב, השמדה אשר לא אתן לה לקרות.

אנוכי הוא השולח דברים אלה בקול קורא במדבר, אשר נגאל מבין הגויים על ידי המשיח ישוע והובדל למטרותינו הנצחיות, ואת ישוע המשיח הוא משרת את יהוה, אהיה אשר אהיה הגדול, כי אין איש יכול לבוא אליי בשום שם אחר, אף לא בשם אהובי משה לבדו, כי משה דורש ציות לישוע הנצרתי!

אנוכי הוא אשר יסיר את העיוורון החלקי מישראל, העיוורים לברית החדשה בשל מסווה משה אשר עודנו פרוש על לבבם בעיוורון חלקי. וכל ישראל אשר יקראו בשם ישוע המשיח ישיבו את מאור עיניהם ויראו במלואו, אף בפני ישוע המשיח, משיחכם הנצחי!

אנוכי אלוהי אברהם, יצחק ויעקב, ואני בא לקיים את כל הבטחותיי פי כמה וכמה, ואעשה זאת במלואן על ידי מעשיו הנצחיים של בני אהובי, אשר לו אנוכי מצווה את מלוא נאמנותכם, אמונתכם, אהבתكم וצייתנותكم, הוי ישראל, אלא אם כן בכוונתכם למכור את בכורתכם ולהצטרף לעשו כברירת מחדל בלהבות עונש נצחיות וגמול הצדק.

הבינו, הוי ישראל, כי הארץ תדע שיש אלוהים ביעקב, וכי אלוהי יעקב הוא לעולמי עולמים ונצח נצחים יהוה, הוא אהיה אשר אהיה הגדול, ובמשיח ישוע ייאסף ישראל של אלוהים, ואקבצם אליי בדיוק כפי שהובטח. בין אם הם קרובים או רחוקים, ובין אם הם יהודים או גויים, למען ייאספו כולם במשיח, יורשי חיי הנצח והם בני ההבטחה האמיתיים, כי בני ההבטחה הם הצדיקים החיים באמונה.

וכך יהיה, ויהוה הכריז כן, והכול יהיה כן ואמן ואמן ואמן!

אנוכי אלוהים חיים, יהוה צבאות אלוהיכם.
