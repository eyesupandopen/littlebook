עדות כפולה: הכנסייה
בכתבי יוחנן המטביל
7 במרץ, 2024

הזמנתי עודנה פתוחה

דבר אדוני ירד עליי, ואני מציג אותו מבלי להוסיף או לגרוע דבר.

הִנֵה! הַזְמָנָתִי עוֹדֶנָּה פְּתוּחָה! בּוֹאוּ כֻּלְּכֶם, וְיָבוֹא כָּל הֶחָפֵץ, כִּי אֲנִי, הָאָדוֹן יֵשׁוּעַ הַמָּשִׁיחַ, נִתַּנְתִּי לָכֶם לְסֶלַע מָגֵן וּלְמִגְדַּל עֹז. אֵין שֵׁם אַחֵר אֲשֶׁר נִתַּן תַּחַת הַשָּׁמַיִם וּבוֹ יוּכְלוּ בְּנֵי אָדָם לְהִוָּשַׁע, מִלְּבַד שְׁמִי, יֵשׁוּעַ הַמָּשִׁיחַ. אֲנִי הוּא אֶבֶן הַפִּנָּה שֶׁמָּאֲסוּ, בְּחָשְׁבָם כִּי בְּנֵי חוֹרִין הֵם. חֵרוּת אֵינָהּ נִמְצֵאת בְּשׁוּם מָקוֹם זוּלָתִי, בַּמָּשִׁיחַ יֵשׁוּעַ. מַדּוּעַ תַּרְשׁוּ לַשְּׂכִירִים לְהַתְעוֹת אֶתְכֶם? מַדּוּעַ תַּנִּיחוּ לִרְפֵי הַיָּדַיִם לְהַשְׁחִית אֶתְכֶם מִן הַתְּמִימוּת הַנִּמְצֵאת בִּי? דְּרָכִים רַבּוֹת מוֹבִילוֹת לַאֲבַדּוֹן. אַךְ אֲנִי הַדֶּרֶךְ, הָאֱמֶת וְהַחַיִּים, הַדֶּרֶךְ הָאֲמִתִּית אֶל הָאָב הַמְּבִיאָה יְשׁוּעָה וְחֵרוּת: יְשׁוּעָה, כִּי אֲנִי מַצִּיעַ מְחִילַת חֲטָאִים בְּדָמִי הַגּוֹאֵל, וְחֵרוּת, כִּי הַבֵּן יְשַׁחְרֵר אֶתְכֶם לְחֵרוּת אֲמִתִּית.

מַדּוּעַ עֲיֵפִים אַתֶּם, עַמִּי? מַדּוּעַ תִּתְעַלְּפוּ? הַאִם לֹא הִמְצֵאתִי לָכֶם מְנוּחָה? הַאִם לֹא גָּבַרְתִּי עַל הַכֹּל, בִּסְלֹל דֶּרֶךְ לִפְנֵיכֶם לָלֶכֶת בְּעִקְּבוֹת נִצְחוֹנִי וּלְהִשְׁתַּתֵּף בְּסִבְלוֹתַי? רַק בְּקַבָּלַת קָרְבָּנִי עַצְמִי תּוּכְלוּ לְהִגָּאֵל מֵחֻרְבָּן וַדָּאִי. רַק בְּאֶמְצָעוּת דָּמִי תִּזְכּוּ לִמְחִילַת חֲטָאִים, וּבְאֶמְצָעוּת גּוּפִי הַשָּׁבוּר נִתְּנָה מַרְפֵּא, וְגוּף חָדָשׁ אֶתֵּן לָכֶם בַּיּוֹם הַהוּא, אַף לְנֶאֱמָנַי וּבְחִירַי. כִּי כָךְ דָּאַג לָכֶם אֲדוֹנְכֶם. סָלַלְתִּי נָתִיב בִּמְקוֹם שֶׁלֹּא הָיָה נָתִיב: שְׁבִיל בַּמִּדְבָּר שֶׁסּוֹפוֹ בְּהִתְפָּאֲרוּתְכֶם, אִם רַק תֶּחְדְּלוּ מֵחֵטְא.

עוֹלָם זֶה מֻשְׁחָת, יְדִיעוֹתָיו מֻשְׁחָתוֹת, מַדָּעָיו מְזֻיָּפִים, וְרֶשֶׁת שֶׁל מִרְמָה לוֹכֶדֶת אֶת עַמִּי, וּמוֹצִיאָה אוֹתָם מִן הַדֶּרֶךְ שֶׁבָּחַרְתִּי עֲבוּרָם. הַאִם אֵינְכֶם מְבִינִים כִּי קָרָאתִי לָכֶם? כִּי בָּחַרְתִּי בָּכֶם? הַאִם אֵינְכֶם מְבִינִים כִּי עֲלֵיכֶם לָמוּת לָעַצְמִי כְּדֵי לִמְצֹא חַיִּים בִּי, שֶׁכֵּן חַיֵּיכֶם הָאֲמִתִּיִּים טְמוּנִים בִּי, הָאָדוֹן הַנֶּאֱמָן? כַּאֲשֶׁר אוֹפִיעַ, גַּם אַתֶּם תּוֹפִיעוּ בְּכָבוֹד; לָכֵן, בְּלֵב תָּמִים, לְכוּ אַחֲרַי. בְּדַעַת צְלוּלָה, בּוֹאוּ אֵלַי בֶּאֱמוּנָה עֲנָוָה. דַּי בֶּאֱמוּנָה כְּגַרְגֵּר חַרְדָּל, וְהָעֵץ יִגְדַּל וְיִהְיֶה גָּדוֹל מִכֻּלָּם.

עִתּוֹת הַגּוֹיִים קְרֵבוֹת לְקִצָּן, אַךְ יַעֲקֹב יִוָּשַׁע, וְלוּ גַּם כְּמוֹ מִתּוֹךְ אֵשׁ. בְּאֵשׁ אוֹכֶלֶת כֹּל יִשָּׁפֵט עוֹלָם זֶה. אָכֵן, שַׂר הָעוֹלָם הַזֶּה נִשְׁפַּט וְהוּא אוֹיֵב מֻבָס. דִּין נֶחְרַץ עָלָיו, עַל מַלְכוּתוֹ, עַל חֲסִידָיו וְעַל הָעוֹלָם שֶׁנִּסָּה לִגְזֹל מֵאָבִי. אֱמֶת, הוּא חוֹפֵץ לִשְׁדֹּד מִכֶּם אֶת נַחֲלַתְכֶם, שֶׁכֵּן סָלַלְתִּי לָכֶם דֶּרֶךְ לְהִשְׁתַּתֵּף בִּכְבוֹד כִּסְאִי כְּשֵׁם שֶׁאֲנִי מִשְׁתַּתֵּף בִּכְבוֹד כֵּס אָבִי. הָלַכְתִּי לְהָכִין לָכֶם מָקוֹם. הַאִם אֵינְכֶם מְבִינִים כִּי מְעוֹנוֹת רַבִּים בְּבֵית אָבִי? הֲיִפָּלֵא מִמֶּנִּי דָּבָר? הַאִם הַבְטָחוֹתַי אִכְזְבוּ אֶתְכֶם? אוֹ שֶׁמָּא הוּטְעֵיתֶם מִן הַכַּוָּנָה הַטְּהוֹרָה שֶׁנָּתַתִּי לָכֶם? מַדּוּעַ עוֹדְכֶם בּוֹחֲרִים בִּדְרָכֵיכֶם שֶׁלָּכֶם?

שׁוּבוּ בִּתְשׁוּבָה, הוֹי בָּנַי! הִתְוַדּוּ עַל חֲטָאֵיכֶם וְאַף עַל חֲטָאֵי אֲבוֹתֵיכֶם, כִּי אֶשְׁמַע אֶתְכֶם בָּעֵת הַקְּבוּעָה. אֲפָאֵר אֶתְכֶם; אֶתֵּן לָכֶם חַדְוַת חַיִּים בִּי. בּוֹאוּ אֵלַי וְקַבְּלוּ בִּגְדֵי שֵׁשׁ! בּוֹאוּ אֵלַי וְקַבְּלוּ שֵׁם חָדָשׁ שֶׁאֶתֵּן לָכֶם עַל אֶבֶן שֶׁהֵכַנְתִּי עֲבוּרְכֶם. אַל יַשִּׂיא אִישׁ אֶתְכֶם מִגְּמוּלְכֶם. הֶרֶס גָּדוֹל וּפוּרְעָנוּת עוֹמְדִים בְּשַׁעַרְכֶם. הָעוֹלָם עוֹמֵד לִגְלֹשׁ לְרֹעַ חֲסַר רֶסֶן. הַתְּהוֹם תִּפָּתַח, וּמַלְכוּתוֹ הַמְּלֵאָה שֶׁל הַשָּׂטָן תְּשֻׁלַּח לַחָפְשִׁי. אַרְעִישׁ שָׁמַיִם, אַרְעִישׁ יָם, רְעָדָה גְּדוֹלָה תָּבוֹא עַל הָאָרֶץ. בֶּן-הָאֲבַדּוֹן מִתְהַלֵּךְ בֵּינֵיכֶם כִּזְאֵב בְּעוֹר כֶּבֶשׂ, מְחַכֶּה לְמַלְכוּתוֹ הַשְּׂטָנִית שֶׁתִּתְגַּלֶּה בִּמְלוֹאָהּ לְמַעַן יִגָּלֶה בְּעִתּוֹ. אַל תֵּלְכוּ אַחֲרָיו!

רוּצוּ אֵלַי, בָּנַי, כִּי זַעַם גָּדוֹל וְהֶרֶס רַב יָבוֹא עַל אֻמַּתְכֶם, וְאֻמּוֹת רַבּוֹת תִּזְדַּעְזַעְנָה עַד הַיְסוֹד בְּקָרוֹב. הוֹ, לוּ תָּשׁוּבוּ אֵלַי וּתְבִינוּ אֶת עִתּוֹתֵיכֶם. מִתּוֹךְ הַחֻרְבָּן הַקָּרֵב יָקוּם אִישׁ הַשֶּׁקֶר, בְּפַתּוֹתוֹ רַבִּים בִּלְשׁוֹן רְמִיָּה וְיַרְאֶה מוֹפְתֵי שֶׁקֶר רַבִּים וְאוֹתוֹת כָּזָב. כְּבָר עַתָּה פָּלְשׁוּ צִבְאוֹתָיו לַעֲרֵיכֶם וּמִלְּאוּ אֶת רְחוֹבוֹתֵיכֶם. אַל תִּירְאוּ מֵהֶם. קְצָתָם יַשִּׂיגוּ יְשׁוּעָה. הַמֹּץ יִנָּתֵן לָאֵשׁ. צָרָה גְּדוֹלָה תָּבוֹא עַל עוֹלָם זֶה פִּתְאֹם. פִּתְאֹם תִּפּוֹל עֲלֵיהֶם פּוּרְעָנוּתָם, וְלֹא יִמָּלְטוּ. הַשָּׁעָה הִיא הָאַחֲרוֹנָה. אַל תִּהְיוּ שֻׁתָּפִים לְמַכּוֹתֶיהָ; צְאוּ מִתּוֹכָהּ, עַמִּי, כִּי הַזּוֹנָה הַגְּדוֹלָה נִשְׁפְּטָה, וּפוּרְעָנוּתָהּ תָּבוֹא עָלֶיהָ בְּשָׁעָה אֶחָת. סוֹחֲרֶיהָ הַגְּדוֹלִים שֶׁסָּחֲרוּ בִּכְשָׁפֶיהָ יִפְּלוּ לִמְצוּקָה גְּדוֹלָה, וַעֲשַׁן שְׂרֵפָתָהּ יִהְיֶה לְמַרְאֵה כֹּל. סוֹחֲרֵי הָעוֹלָם יְקוֹנְנוּ עַל פּוּרְעָנוּתָהּ מֵרָחוֹק.

אַךְ לִי תָּכְנִיּוֹת עֲבוּרְכֶם, עַמִּי. חַיֵּיכֶם צְפוּנִים בִּי. אַף כִּי בְּחַנְתִּי אֶתְכֶם לְעוֹנָה, צְרַפְתִּי אֶתְכֶם כְּזָהָב בְּכוּר הַצּוֹרֵף. צְאוּ מִתּוֹכָהּ! עוּרוּ לִצְדָקָה וְאַל תֶּחֱטָאוּ. הִטַּהֲרוּ בְּדָמִי הַמְכַפֵּר. בְּכוּ וְהֵילִילוּ עַל פִּשְׁעֵיכֶם; שׁוּבוּ מֵחַטֹּאתֵיכֶם בְּכָל לְבַבְכֶם. אֲנִי הוּא הַבּוֹחֵן לִבּוֹת אֱנוֹשׁ; לֹא תּוּכְלוּ לְהַסְתִּיר מִמֶּנִּי דָּבָר. בּוֹאוּ אֵלַי בְּלִי מִרְמָה. בּוֹאוּ אֵלַי בְּכֵנוּת וּבִגְלוּי לֵב, וְאֶסְעַד עִמָּכֶם וְאָשִׂים לָכֶם מָקוֹם בְּשֻׁלְחָנִי. בְּקָרוֹב תֵּאָסֵף הַכַּלָּה, מַהֵר מִכְּפִי שֶׁתְּשַׁעֲרוּ.

כַּאֲשֶׁר תִּלָּקַח כַּלָּתִי בָּאֲוִיר, יֻתַּר הָרֶסֶן, וְשׁוּם דָּבָר רַע לֹא יֵעָצֵר מִפְּנֵי הָאָדָם הַטִּבְעִי. וְהֵם יֵלְכוּ בְּגָלוּי אַחַר תַּאֲווֹת וְרֹעַ לִבָּם; עַל כֵּן, דִּין גָּדוֹל יָבוֹא עֲלֵיהֶם, וְלֹא תִּהְיֶה לוֹ תְּרוּפָה. שׁוּבוּ אֵלַי, יֵשׁוּעַ הַמָּשִׁיחַ! קִרְאוּ בִּשְׁמִי בְּהִמָּצְאִי! אַף בְּיוֹם הַחֻרְבָּן הַגָּדוֹל, שְׁמִי יָבִיא יְשׁוּעָה לַקּוֹרְאִים בּוֹ. בּוֹאוּ אֵלַי עַתָּה וּצְאוּ מִתּוֹכָהּ, כִּי הַזְּמַן כִּמְעַט תַּם. צִבְאוֹת הַשָּׁמַיִם מוּכָנִים, מְחַכִּים לַשּׁוֹפָר.

אֲנִי הוּא אֶבֶן הַפִּנָּה שֶׁמָּאֲסוּ, בְּחָשְׁבָם כִּי בְּנֵי חוֹרִין הֵם.

אֲנִי הַדֶּרֶךְ, הָאֱמֶת וְהַחַיִּים: הַדֶּרֶךְ הָאֲמִתִּית אֶל הָאָב הַמְּבִיאָה יְשׁוּעָה וְחֵרוּת.

אֲנִי הוּא הַבּוֹחֵן לִבּוֹת אֱנוֹשׁ; לֹא תּוּכְלוּ לְהַסְתִּיר מִמֶּנִּי דָּבָר.

מעשי השליחים ב׳: 16–21

אֶלָּא שֶׁזֶּהוּ הַדָּבָר שֶׁנֶּאֱמַר בְּפִי יוֹאֵל הַנָּבִיא;

וְהָיָה בְּאַחֲרִית הַיָּמִים, יֹאמַר אֱלֹהִים, אֶשְׁפּוֹךְ מֵרוּחִי עַל-כָּל-בָּשָׂר, וְנִבְּאוּ בְּנֵיכֶם וּבְנוֹתֵיכֶם, וּבַחוּרֵיכֶם חֶזְיֹנוֹת יִרְאוּ, וְזִקְנֵיכֶם חֲלֹמוֹת יַחֲלֹמוּן:

וְגַם עַל-עֲבָדַי וְעַל-שִׁפְחוֹתַי בַּיָּמִים הָהֵמָּה אֶשְׁפּוֹךְ מֵרוּחִי, וְנִבָּאוּ: וְנָתַתִּי מוֹפְתִים בַּשָּׁמַיִם מִמַּעַל, וְאֹתוֹת בָּאָרֶץ מִתָּחַת: דָּם וָאֵשׁ וְתִימֲרוֹת עָשָׁן: הַשֶּׁמֶשׁ יֵהָפֵךְ לְחֹשֶׁךְ, וְהַיָּרֵחַ לְדָם, לִפְנֵי בוֹא יוֹם יְהוָה הַגָּדוֹל וְהַנּוֹרָא. וְהָיָה, כֹּל אֲשֶׁר-יִקְרָא בְּשֵׁם יְהוָה יִמָּלֵט.
