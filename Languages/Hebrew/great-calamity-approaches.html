עֵד כָּפוּל: הַקְּהִילָה
בְּמַעֲשֵׂי יוֹחָנָן הַמַּטְבִּיל
11 במרץ 2024

פּוּרְעָנוּת גְּדוֹלָה מִתְקָרֶבֶת

דְּבַר יהוה הָיָה אֵלַי; אָנָּא שִׁמְעוּ.

שִׁמְעוּ קוֹלִי, נְאֻם יהוה צְבָאוֹת! עַמִּי, שׁוּבוּ אֵלַי, כִּי פּוּרְעָנוּת גְּדוֹלָה עֲתִידָה לָבוֹא עַל אֻמַּתְכֶם. פּוּרְעָנוּת גְּדוֹלָה בְּפִתְחֲכֶם, וּפוּרְעָנוּת גְּדוֹלָה תָּבוֹא אֶל חוֹפֵיכֶם. שׁוּבוּ אֵלַי וִירְאוּ אוֹתִי, נְאֻם יהוה צְבָאוֹת. אֱלֹהִים חָרַץ אֶת דִּינָהּ שֶׁל אָמֶרִיקָה בִּשְׁתֵּי חֲזִיתוֹת. יִהְיֶה דִּין מִבִּפְנִים וְיִהְיֶה דִּין מִבַּחוּץ. מִבִּפְנִים יִהְיֶה זַעֲזוּעַ גָּדוֹל, זַעֲזוּעַ גָּדוֹל, וְאָמֶרִיקָה תִּפּוֹל נְפִילָה אַדִּירָה בְּזַעֲזוּעַ גָּדוֹל. מִבַּחוּץ, צִבְאוֹתָיו בָּאוּ לִגְנֹב וְלַהֲרֹג וּלְהַשְׁמִיד אֶת כָּל אֲשֶׁר בְּדַרְכָּם. אֵין רַחֲמִים בְּלִבָּם. דֵּעָה אַחַת לָהֶם. וְהֵם יִקְחוּ מִכֶּם אֶת שִׁפְעֲכֶם; יִקְחוּ מִכֶּם אֶת טוּבְכֶם; יִתְפְּסוּ אֶת צֶאֱצָאֵיכֶם; וְרָעָב גָּדוֹל יָבוֹא עַל אַרְצְכֶם. אֲשֶׁר לֹא יֵהָרֵג בַּמִּלְחָמָה יֵהָרֵג בָּאָסוֹן; אֲשֶׁר לֹא יֵהָרֵג בָּאָסוֹן יֵהָרֵג בְּחַיַּת הַשָּׂדֶה; אֲשֶׁר לֹא יֵהָרֵג בְּחַיַּת הַשָּׂדֶה יִגְוַע בָּרָעָב. אַךְ גַּם בְּפוּרְעֲנוּתְכֶם, כָּל הַקּוֹרֵא בְּשֵׁם יהוה יִמָּלֵט.

וְאוֹשִׁיעַ אֶת עַמִּי, אַף בְּצָרָה גְדוֹלָה וּבְאֵשׁ הַמְצָרֵף. בּוֹאוּ אֵלַי, עַמִּי, בְּעוֹד מוֹעֵד. בּוֹאוּ אֵלַי, אֶל הָאָדוֹן יֵשׁוּעַ הַמָּשִׁיחַ, כִּי אֲנִי הָאָדוֹן הַצַּדִּיק. אֲנִי מֶלֶךְ הַמְּלָכִים. אֲנִי הָרוֹעֶה הַטּוֹב, וְאִישׁ לֹא יוּכַל לַחְטֹף אֶת צֹאנִי מִיָּדִי. בּוֹאוּ אֵלַי! בּוֹאוּ אֵלַי בְּעוֹד יֵשׁ זְמַן, וְהִתְוַדּוּ עַל חֲטָאֵיכֶם; הִתְוַדּוּ עַל שְׁמִי הָאַדִּיר כָּאָדוֹן, כִּי אֵין יְשׁוּעָה בְּשֵׁם אַחֵר. כִּי אֲנִי, יֵשׁוּעַ הַמָּשִׁיחַ, נִתַּנְתִּי לִישׁוּעַתְכֶם אֲשֶׁר קָנִיתִי בְּדָמִי שֶׁלִּי. בּוֹאוּ אֵלַי, כֵּן, בּוֹאוּ אֵלַי בְּאַהֲבָתִי אֲשֶׁר לִי אֲלֵיכֶם. אַל תִּתְנַגְּדוּ לִי, אֶלָּא בּוֹאוּ אֵלַי, לְמַעַן עוֹד אוֹשִׁיעֲכֶם וְאֶתֵּן לָכֶם חַיֵּי עוֹלָם, כְּדֵי שֶׁאוּכַל לְבָרֵךְ אֶת צֶאֱצָאֵיכֶם. בּוֹאוּ אֵלַי, עַמִּי, וְאַל תַּקְשׁוּ אֶת לְבַבְכֶם, כְּמוֹ הָאָבוֹת בַּמִּדְבָּר. כִּי יהוה אֱלֹהֵיכֶם חָרַץ מִשְׁפָּט גָּדוֹל עַל אֻמַּתְכֶם, כִּי רַבִּים וְנוֹרָאִים חֲטָאֶיהָ.

כֵּן, שִׁלַּמְתִּי עַל הַחֲטָאִים הָהֵם וְעַל כָּל חֲטָאֵי הָאֱנוֹשׁוּת, אַךְ הַצָּעָתִי וּזְרוֹעוֹתַי לֹא תִּהְיֶינָה פְּרוּשׂוֹת לָעַד. בְּיוֹם הַפּוּרְעָנוּת, תֻּשַּׂג הַיְשׁוּעָה דֶּרֶךְ מְצוּקָה גְּדוֹלָה וְצָרָה גְּדוֹלָה שֶׁאָבִיא עַל הָעוֹלָם לִבְחֹן אֶת לִבּוֹת בְּנֵי הָאָדָם, וְהַמַּחֲזִיק מַעֲמָד עַד הַקֵּץ יִוָּשַׁע. אַל תֵּלְכוּ אַחֲרֵי אִישׁ הַחֵטְא. אַל תִּקְחוּ אֶת הַתָּו שֶׁלּוֹ וְאַל תִּקְחוּ אֶת תְּרוּפָתוֹ, כִּי נְשִׁיכַת נָחָשׁ בָּהּ, וְלֹא תֵּרָפְאוּ מֵאֶרֶסָהּ. כָּל הַלּוֹקֵחַ אֶת תָּו הַחַיָּה, חֶלְקוֹ יִהְיֶה בַּאֲגַם הָאֵשׁ עִם הַחַיָּה וּנְבִיא הַשֶּׁקֶר. הַאִם לֹא כָּתוּב בִּדְבָרִי הַקָּדוֹשׁ? הַאִם לֹא קְרָאתֶם? הַאִם לֹא שְׁמַעְתֶּם? שׁוּבוּ אֶל אַהֲבָתִי וּבוֹאוּ אֵלַי, יְלָדַי, כִּי קוֹרֵא אֲנִי לָכֶם בְּשָׁעָה אַחֲרוֹנָה זוֹ, כִּי פּוּרְעָנוּת גְּדוֹלָה בְּפִתְחֲכֶם.

שִׁמְעוּנִי, נְאֻם יהוה צְבָאוֹת:

כִּי אֲנִי הוּא, בֶּן-הָאָדָם אֲשֶׁר נִצְלַב תַּחַת פּוֹנְטִיּוּס פִּילָטוֹס, אֲשֶׁר מֵת עַל חַטֹּאתֵיכֶם וְקָם בַּיּוֹם הַשְּׁלִישִׁי. קַמְתִּי וְנִתְפָּאַרְתִּי.

אֲנִי אֲדוֹן הָאֲדוֹנִים וּמֶלֶךְ הַמְּלָכִים.

אֲנִי בֶּן-הָאָדָם, הָאָדוֹן הַצַּדִּיק, הָאָדוֹן הָאֲמִתִּי אֲשֶׁר נֶאֱמָן וְצַדִּיק, וְאַף אֶחָד מִכֶּם לֹא יֵחָטֵף מִיָּדַי.

אֲנִי הָרוֹעֶה הַטּוֹב, וְאִישׁ לֹא יוּכַל לַחְטֹף אֶת צֹאנִי מִיָּדִי.

אֶת כָּל אֲשֶׁר נָתַן לִי הָאָב, אַצִּיג לְפָנָיו, וּרְצוֹנִי לְהַצִּיגְכֶם לִפְנֵי הָאָב לְלֹא דֹּפִי, כֶּתֶם אוֹ אַשְׁמָה, לְבוּשִׁים בְּפִשְׁתָּן לָבָן טָבוּל בְּדָמִי. הַדֶּרֶךְ הַיְחִידָה שֶׁבָּהּ אָדָם יָכוֹל לַעֲמֹד לְפָנַי הִיא לְקַבֵּל אֶת קָרְבָּנִי עַל חֲטָאֵיכֶם. אַל תִּתְמַהְמְהוּ עִם הַחוֹטְאִים! אַל תִּתְמַהְמְהוּ; אַל תִּתְמַהְמְהוּ! צְאוּ מִתּוֹכָם וְהִבָּדְלוּ מֵחַטֹּאתָם. הִתְוַדּוּ לְפָנַי עַל חַטֹּאתֵיכֶם וְהִתְוַדּוּ לְפָנַי עַל חַטֹּאת אֲבוֹתֵיכֶם, כִּי הֵם עוֹבְרִים אֲלֵיכֶם. חֲיוּ אֶת הַחַיִּים שֶׁנָּתַתִּי לָכֶם, וּלְכוּ בַּנָּתִיב שֶׁהִנְחַלְתִּי לָכֶם, כִּי קָרָאתִי לָכֶם בְּשֵׁם. אַל תִּפְנוּ שְׂמֹאלָה וְאַל תִּפְנוּ יָמִינָה, אֶלָּא הַקְדִּישׁוּ תְּשׂוּמֶת לֵב וּדְבֵקוּת מַטָּרָה לִדְבָרַי. וְעִם זֹאת, רַבִּים יְסָרְבוּ לְהַזְמָנָתִי, וְרַבִּים יֹאבְדוּ. הַאֲמִינוּ בִּי, כִּי אַהֲבַת אֱלֹהִים תּוֹשִׁיעַ אֶת כָּל הַקּוֹרֵא בְּשֵׁם יהוה. הַאֲמִינוּ בִּי, וּנַהֲרֵי מַיִם חַיִּים יִזְרְמוּ מִבִּטְנְכֶם כַּאֲשֶׁר חַטֹּאתֵיכֶם יִרָּחֲצוּ וְרוּחֲכֶם תִּהְיֶה שְׁלֵמָה. בּוֹאוּ אֵלַי בְּעוֹד מוֹעֵד; הִתְוַדּוּ עַל שְׁמִי; עֲשׂוּנִי לְאָדוֹן, כִּי אֵין יְשׁוּעָה בְּשֵׁם אַחֵר שֶׁנִּתַּן תַּחַת הַשָּׁמַיִם אֲשֶׁר בּוֹ יוּכְלוּ בְּנֵי אָדָם לְהִוָּשַׁע.
