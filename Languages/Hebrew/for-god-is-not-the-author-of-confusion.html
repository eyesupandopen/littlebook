עֵדוּת כְּפוּלָה: הַקְּהִלָּה
בְּמַעֲשֵׂי פֶּטְרוֹס
20 באפריל 2024

כִּי לֹא אֱלֹהֵי מְהוּמָה הוּא, כִּי אִם-אֱלֹהֵי שָׁלוֹם!

כִּי לֹא אֱלֹהֵי מְהוּמָה הוּא, כִּי אִם-אֱלֹהֵי שָׁלוֹם, כְּמוֹ בְּכָל-קְהִלּוֹת הַקְּדוֹשִׁים! קִרְאוּ דְּבָרִים אֵלֶּה וְהַאֲזִינוּ לְקוֹלִי. הַטּוּ אָזְנְכֶם וּשְׁמְעוּ דִּבְרֵי חֲכָמִים, וְלִבְּכֶם תָּשִׁיתוּ לְדַעְתִּי. אוּלַי רָאוּי לָכֶם לָלֶכֶת בְּעִקְּבוֹת דֻּגְמָתִי מִן הַגַּן, כַּאֲשֶׁר הִתְפַּלַּלְתִּי בְּיִסּוּרִים וְנָפַלְתִּי אַרְצָה לִפְנֵי אָבִי בִּתְפִלָּה וּבַעֲנָוָה, וְהוּא חִזְּקַנִי לְקַבֵּל אֶת כּוֹס הַזַּעַם עַל עֲוֹנוֹתֵיכֶם שֶׁלָּכֶם, אֲשֶׁר נָשָׂאתִי עַל עַצְמִי לְמַעַנְכֶם, כִּי יָכֹלְתִּי לְשֵׂאתָם. אֲנִי מְדַבֵּר אֶל הַגֵּאִים שֶׁבִּקְהִלָּתִי, וַאֲנִי מְדַבֵּר אֶל הַפִּלּוּגִים הָרַבִּים שֶׁבֵּינֵיכֶם. וְאַתָּה, לָמָּה תָּדִין אֶת-אָחִיךָ? אוֹ לָמָּה תִּבְזֶה אֶת-אָחִיךָ? הֲלֹא כֻלָּנוּ נַעֲמֹד לִפְנֵי כִּסֵּא הַמִּשְׁפָּט שֶׁל הַמָּשִׁיחַ. הַאִם לֹא כָּתוּב בָּאִגֶּרֶת שֶׁשָּׁלַחְתִּי לְהוֹלְכֵי הַדֶּרֶךְ בְּרוֹמָא? וּמָה אוֹמֵר זְכַרְיָה בָּעִנְיָן: וַיְמָאֲנוּ לְהַקְשִׁיב, וַיִּתְּנוּ כָתֵף סֹרֶרֶת, וְאָזְנֵיהֶם הִכְבִּידוּ מִשְּׁמוֹעַ. הַאִם לֹא קְרָאתֶם אֶת יְשַׁעְיָהוּ, קְטַנַּי? כִּי מָה אוֹמְרִים נְבִיאֵי קֶדֶם? כִּי שׁוּב אוֹמֵר יְשַׁעְיָהוּ, שִׁמְעוּ שָׁמוֹעַ וְאַל-תָּבִינוּ, וּרְאוּ רָאוֹ וְאַל-תֵּדָעוּ.

יְלָדַי, הֶרְאֵיתִי לָכֶם דֶּרֶךְ מְעֻלָּה יוֹתֵר, אַךְ רַבִּים מִכֶּם נָפְלוּ מִן הַחֶסֶד וְגַאֲוַתְכֶם הָעַזָּה גָּרְמָה לָכֶם לְהִתְרוֹמֵם מֵעַל לַמִּדָּה. וְעִם זֹאת, דְּבָרַי שֶׁנָּתַתִּי לַשָּׁלִיחַ פּוֹלוֹס עֲבוּר גָּלַטְיָה אוֹמְרִים בְּבֵרוּר: כִּי אִם-יַחְשֹׁב אִישׁ לִהְיוֹת דָּבָר וְהוּא אָיִן, מַתְעֶה הוּא אֶת-נַפְשׁוֹ. יְלָדַי הַקְּטַנִּים, יוֹדֵעַ אֲנִי כִּי לִבְּכֶם נִלְהָב לְשָׁרְתֵנִי, וְיוֹדֵעַ אֲנִי כִּי אַתֶּם אוֹהֲבִים אוֹתִי, אַךְ לְעִתִּים לֹא עַל-פִּי דַעַת. הַאִם אֵינְכֶם מְבִינִים כִּי אֱלֹהִים לַגֵּאִים יִתְקַוָּם וְלַעֲנָוִים יִתֶּן-חֵן? לָכֵן, הַשְׁפִּילוּ עַצְמְכֶם תַּחַת יַד אֱלֹהִים הַחֲזָקָה, לְמַעַן יְרוֹמֵם אֶתְכֶם בְּעִתּוֹ. כְּמוֹ כֵן, אַתֶּם הַצְּעִירִים, הִכָּנְעוּ לִפְנֵי זִקְנֵיכֶם. כֵּן, כֻּלְּכֶם הֱיוּ נִכְנָעִים אִישׁ לְרֵעֵהוּ, וְהִתְעַטְּפוּ בַּעֲנָוָה. אֱלֹהִים לַגֵּאִים יִתְקַוָּם וְלַעֲנָוִים יִתֶּן-חֵן. אַל תַּחְשְׁבוּ אֶת דְּבָרַי לְחוֹזְרִים עַל עַצְמָם, כִּי עַל-פִּי שְׁנַיִם עֵדִים אוֹ עַל-פִּי שְׁלֹשָׁה עֵדִים יָקוּם כָּל-דָּבָר. הָבִינוּ, יְלָדַי, אֵין בִּיכָלְתְּכֶם לְשָׁרְתֵנִי בִּבְשַׂרְכֶם: כִּי כָּל-אֲשֶׁר בָּעוֹלָם, תַּאֲוַת הַבָּשָׂר וְתַאֲוַת הָעֵינַיִם וּגְאוֹן הַחַיִּים, לֹא מִן-הָאָב הוּא כִּי אִם מִן-הָעוֹלָם. קְהִלָּתִי, הִכָּנְעוּ אִישׁ לְרֵעֵהוּ בְּדַעַת הָעֲטוּפָה בַּעֲנָוָה, כִּי יִתָּכֵן וַאֲנִי הוּא הַנּוֹתֵן אֶת הַדְּבָרִים לֵאמֹר וְאֶת הַהַשְׁרָאָה לְלַמֶּדְכֶם, אַף שֶׁרַבִּים מִכֶּם יָשִׂימוּ הַצִּדָּה אֶת קוֹל הֶעָנָו לְטוֹבַת שְׁמוֹ הַטּוֹב וּמַעֲמָדוֹ שֶׁל מִישֶׁהוּ. הֱיוּ הוֹלְכִים בְּעִקְּבוֹת אֱלֹהִים, יְלָדַי, כִּי בֶאֱמֶת אֱלֹהִים אֵינוֹ נוֹשֵׂא פָּנִים.

הִנֵּה! כְּבוֹד אֱלֹהִים הַסְתֵּר דָּבָר; וּכְבוֹד מְלָכִים חֲקֹר דָּבָר. הָבִינוּ כִּי אֶעֱשֶׂה אֶתְכֶם מְלָכִים וְכֹהֲנִים לֵאלֹהֵינוּ וְאָבִינוּ, לָכֵן הָבִינוּ אֶת אִמְרֵי הֶחָכָם שֶׁבָּכֶם: הַאִם לֹא נָתַן לָכֶם אֱלֹהִים שְׁתֵּי אָזְנַיִם וּפֶה אֶחָד? הַאִם לֹא קְרָאתֶם? לְכָל-אֶחָד מִכֶּם יֵשׁ מִזְמוֹר, יֵשׁ תּוֹרָה, יֵשׁ לָשׁוֹן, יֵשׁ הִתְגַּלּוּת, יֵשׁ פֵּרוּשׁ: יֵעָשׂוּ נָא כָל-הַדְּבָרִים לִבְנִיָּה! אַךְ לֹא, אַתֶּם נוֹשְׁכִים וּבוֹלְעִים אִישׁ אֶת רֵעֵהוּ בְּשַׁחְצָנוּתְכֶם; בְּגַאֲוַתְכֶם אַתֶּם תּוֹקְפִים אֶת הַמְנַסִּים לְלַמֶּדְכֶם, וְנַעֲשֵׂיתֶם בִּלְתִּי-נִתָּנִים לְלִמּוּד. אַתֶּם מִתְוַכְּחִים, רָבִים, דָּנִים וְנִלְחָמִים אִישׁ בְּרֵעֵהוּ תַּחַת שִׁקּוּל עָנָו שֶׁל דִּבְרֵי הָאַחֵר. מֵעֵבֶר לָזֶה, הַאִם כֹּה גֵּאִים אַתֶּם לַחְשֹׁב שֶׁכֵּיוָן שֶׁהַדְּבָרִים יוֹצְאִים מִפִּיכֶם, הֵם אֱמֶת מִפְּנֵי שֶׁאַתֶּם אֲמַרְתֶּם אוֹתָם? הִכָּנְעוּ אִישׁ לְרֵעֵהוּ. חִשְׂכוּ אֶת לְשׁוֹנְכֶם וּבַחֲנוּ אֶת דִּבְרֵי אַחֵיכֶם לְפִי הַכְּתוּבִים, לְמַעַן תִּבְחֲנוּ כָּל דָּבָר וְתֹאחֲזוּ בַּטּוֹב. פִּתְחוּ אָזְנֵיכֶם וְהַקְשִׁיבוּ בְּכָל-לֵב וְשִׁקְלוּ, כִּי בְּעָשְׂרוֹתְכֶם כֵּן אֶעֱשֶׂה אֶתְכֶם חֲכָמִים בַּעֲנָוָתְכֶם, לָכֵן אַל תִּהְיוּ חֲכָמִים בְּעֵינֵיכֶם. אֲנִי יוֹעֵץ לְכָל אֶחָד מִכֶּם, שִׁמְעוּ דְּבָרַי. וּמָה אוֹמְרִים אוֹתָם דְּבָרִים? כִּי אוֹמֵר אֲנִי, עַל-פִּי הַחֶסֶד הַנִּתָּן לִי, לְכָל-אִישׁ אֲשֶׁר בָּכֶם, לְבִלְתִּי חֲשֹׁב עַל-עַצְמוֹ גְּבֹהָה מִמַּה-שֶּׁרָאוּי לוֹ לַחְשֹׁב, כִּי אִם-לַחְשֹׁב בְּצִנְעָה, כְּפִי שֶׁחָלַק אֱלֹהִים לְכָל-אִישׁ מִדַּת אֱמוּנָה.

לֹא, עִצְרוּ, שִׁקְלוּ אֶת הַדְּבָרִים שֶׁזֶּה עַתָּה נָתַתִּי לָכֶם, פִּתְחוּ אֶת סִפְרֵי הַתָּנָ"ךְ שֶׁלָּכֶם וּקְרְאוּ דְּבָרִים אֵלֶּה שׁוּב. שִׁמְעוּ אוֹתָם וְהַאֲמִינוּ בָּהֶם, כִּי חוֹבַתְכֶם כְּלַפֵּי עַצְמְכֶם הִיא לַחְשֹׁב טוֹבוֹת עַל עַצְמְכֶם, כִּי הֲקִימוֹתִי אֶתְכֶם וְהוֹשַׁבְתִּי אֶתְכֶם יַחַד בַּשָּׁמַיִם בִּי, הַמָּשִׁיחַ יֵשׁוּעַ. אַךְ בְּאוֹתָהּ עֵת, עֲלֵיכֶם לַחְשֹׁב בְּצִנְעָה, כִּי אֱלֹהִים נָתַן לְכָל-אִישׁ בָּכֶם אוֹתָהּ מִדַּת אֱמוּנָה, לָכֵן הָרוּחַ שֶׁנִּתְּנָה אֵינָהּ גְּדוֹלָה אוֹ קְטַנָּה יוֹתֵר בְּאַחֵיכֶם, אֶלָּא לְכָל-אֶחָד מִכֶּם נִתַּן חֶסֶד, לְפִי מִדַּת מַתְּנַת הַמָּשִׁיחַ. עַל-כֵּן הוּא אוֹמֵר: כַּאֲשֶׁר עָלָה לַמָּרוֹם, שָׁבָה שֶׁבִי וְנָתַן מַתָּנוֹת לִבְנֵי אָדָם. נָתַתִּי מֵהֶם שְׁלִיחִים, וּמֵהֶם נְבִיאִים, וּמֵהֶם מְבַשְּׂרִים, וּמֵהֶם רוֹעִים וּמוֹרִים: לְשִׁכְלוּל הַקְּדוֹשִׁים, לְמַעֲשֵׂה הַשֵּׁרוּת, לִבְנִיַּת גּוּף הַמָּשִׁיחַ. הִכָּנְעוּ אִישׁ לְרֵעֵהוּ וְהַשְׁלִיכוּ אֶת גַּאֲוַתְכֶם חַסְרַת הָעֵרֶךְ, כִּי הִיא גּוֹרֶמֶת לְכִתּוֹת בֵּינֵיכֶם. מַדּוּעַ אֵינְכֶם מְבִינִים שֶׁלֹּא צְרִיכִים לִהְיוֹת פִּלּוּגִים בֵּינֵיכֶם, יְלָדַי? הַאִם גּוּף הַמָּשִׁיחַ מְחֻלָּק? מָה אָמַרְתִּי לַקּוֹרִינְתִּים?

וַאֲנִי מַפְצִיר בָּכֶם, אַחַי, בְּשֵׁם אֲדוֹנֵנוּ יֵשׁוּעַ הַמָּשִׁיחַ, שֶׁתְּדַבְּרוּ כֻּלְּכֶם דָּבָר אֶחָד, וְשֶׁלֹּא יִהְיוּ בָּכֶם פִּלּוּגִים; אֶלָּא שֶׁתִּהְיוּ מְאֻחָדִים לְגַמְרֵי בְּדֵעָה אַחַת וּבְמִשְׁפָּט אֶחָד. הַאִם אֵינְכֶם כֻּלְּכֶם אֶחָד בִּי, הָאָדוֹן יֵשׁוּעַ הַמָּשִׁיחַ? וְאַף עַל פִּי כֵן יֵשׁ בָּכֶם כִּתּוֹת רַבּוֹת כָּל כָּךְ, וְאַתֶּם נוֹשְׁכִים וּבוֹלְعִים אִישׁ אֶת רֵעֵהוּ עַל אִי-הַסְכָּמוֹת זְעִירוֹת בְּיוֹתֵר. דְּעוּ כִּי בְּכִתּוֹתֵיכֶם וּבִלְשׁוֹן הָרַע שֶׁלָּכֶם אַתֶּם מְשַׁעַשְׁעִים אֶת הַשָּׂטָן, וְהוּא מִתְפָּאֵר בְּחֹסֶר הַתּוֹעֶלֶת הַמֻּחְלָט שֶׁל מַאֲמַצֵּיכֶם הַמֻּשְׁקָעִים בַּמַּאֲבָק בֵּינֵיכֶם, כִּי רַבִּים מִכֶּם חֲסֵרִים עֲנָוָה. וְאָזְנֵיכֶם אֲטוּמוֹת וְאֵינְכֶם יְכוֹלִים לִשְׁמֹעַ אִישׁ אֶת קוֹל רֵעֵהוּ, וְעַצַמְתֶּם אֶת עֵינֵיכֶם, וְהֵן דְּבוּקוֹת בְּגַאֲוָה וּבְאִי-אֱמוּנָה, כִּי כֹּה רַבִּים מִכֶּם חוֹשְׁבִים שֶׁבִּרְשׁוּתָם כָּל הָאֱמֶת. וְאַתֶּם מְזַלְזְלִים בְּדִבְרֵי אַחֵיכֶם, וְאַתֶּם מְזַלְזְלִים בְּעֶצֶם הַדְּבָרִים שֶׁנָּתַתִּי לָהֶם עֲבוּרְכֶם, וּבְגַאֲוַתְכֶם הָעַזָּה אַתֶּם חֲסֵרִים אֶת הָעֲנָוָה הַבְּסִיסִית לִשְׁמֹעַ וּלְשַׁקֵּל. כִּי אִם-אָדָם לֹא יִשְׁמַע דָּבָר, אִוֶּלֶת וּכְלִמָּה לוֹ. וְשׁוּב, מַדּוּעַ אַתֶּם מְזַלְזְלִים בְּאַחֵיכֶם, הֲלֹא כֻּלָּנוּ חַיָּבִים לַעֲמֹד לִפְנֵי כִּסֵּא הַמִּשְׁפָּט שֶׁל הַמָּשִׁיחַ? אֲנִי יוֹעֵץ לָכֶם, קְהִלָּתִי, לְהִתְכַּחֵשׁ לְגַאֲוַתְכֶם וּלְדַרְכֵיכֶם הַשַּׁחְצָנִיּוֹת וְלָבוֹא אֵלַי כְּתַלְמִידֵי דְּבַר הַחַיִּים, כִּי אֲנִי דְּבַר אֱלֹהִים שֶׁנַּעֲשָׂה בָּשָׂר.

לָכֵן הַקְשִׁיבוּ בְּכָל-נַפְשְׁכֶם וְשִׂימוּ לֵב לְצִלְלֵי אָזְנֵיכֶם וְהָבִינוּ אֶת דְּבָרַי. לָכֵן הָבִינוּ אֶת סֵדֶר הָאֵרוּעִים הַנָּתוּן בִּדְבָרִי, כִּי רַבִּים מִכֶּם מְרֻמִּים עַל-יְדֵי זְאֵבִים טוֹרְפִים שֶׁאַתֶּם הוֹלְכִים אַחֲרֵיהֶם בְּבַעֲרוּת, כִּי הִקְשַׁבְתֶּם לָרוֹעִים הַסּוֹרְרִים. מִי אַתֶּם שֶׁתַּטִּיפוּ כִּי אֵין הִתְאַסְּפוּת שֶׁל קְהִלָּתִי אֵלַי, שֶׁלֹּא אֶפְגֹּשׁ אֶת קְהִלָּתִי בָּאֲוִיר כַּאֲשֶׁר אָבוֹא בִּתְרוּעָה גְּדוֹלָה וּבְקוֹל שַׂר-הַמַּלְאָכִים וּבְשׁוֹפַר אֱלֹהִים? כִּי הַשּׁוֹפָר יִתָּקַע וְהַמֵּתִים בַּמָּשִׁיחַ יָקוּמוּ רִאשׁוֹנָה וְאַתֶּם הַחַיִּים הַנּוֹתָרִים לֹא תַּקְדִּימוּ אוֹתָם, וְכֻלְּכֶם תֵּחָטְפוּ בֶּעָנָנִים לִפְגֹּשׁ אוֹתִי וּלְעוֹלְמֵי עוֹלָמִים תִּהְיוּ עִמִּי. הַאִם דְּבָרִים אֵלֶּה טוֹבִים מִדַּי לְאַמִּתָּתָם? אוֹ הַאִם תַּאֲמִינוּ לְדִבְרֵי הָרוֹעִים הַסּוֹרְרִים הַמִּתְיַהֲדִים עַל פְּנֵי קוֹל אֲדוֹנְכֶם וּמוֹשִׁיעֲכֶם, יֵשׁוּעַ הַמָּשִׁיחַ? מַדּוּעַ אֵין לָכֶם אֱמוּנָה בִּדְבָרִי אוֹ בְּעֶצֶם הַמִּלִּים הַמַּרְכִּיבוֹת אֶת דְּבָרִי? הַאִם אֵינְכֶם מְבִינִים מַהִי תַּעֲלוּמָה מִקְרָאִית אוֹ מַטְּרָתָהּ?

אֹמַר לָכֶם: תַּעֲלוּמָה הִיא דָּבָר הַנִּשְׁמָר בְּסוֹד אַךְ נוּבָּא מֵרֹאשׁ, הַנִּגְלֶה בַּזְּמַן הַנָּכוֹן. וּמַדּוּעַ יַעֲשֶׂה הָאָב דָּבָר כָּזֶה, תִּשְׁאֲלוּ? הַאִם לֹא קְרָאתֶם אֶת הַדּוּ"חַ שֶׁלִּי? אֶלָּא אֲנַחְנוּ מְדַבְּרִים אֶת חָכְמַת אֱלֹהִים בְּתַעֲלוּמָה, הִיא הַחָכְמָה הַנִּסְתֶּרֶת אֲשֶׁר קָבַע אֱלֹהִים מִלִּפְנֵי עוֹלָם, לִכְבוֹדֵנוּ. אֲשֶׁר אִישׁ מִשָּׂרֵי הָעוֹלָם הַזֶּה לֹא יָדַע: כִּי לוּ יָדְעוּ, לֹא הָיוּ צוֹלְבִים אֶת אֲדוֹן הַכָּבוֹד. מִי אַתֶּם שֶׁתְּזַלְזְלוּ בְּתָכְנִיּוֹתָיו וּבְדַרְכָיו שֶׁל אָבִי? כִּי אֵינוֹ זָקוּק לְיוֹעֵץ, אַף אִם רְבָבָה רִבְבוֹת מַקִּיפִים אֶת כִּסְאוֹ.

לָכֵן נַחֲמוּ אִישׁ אֶת רֵעֵהוּ בִּדְבָרַי, כִּי נוֹשַׁעְתֶּם מִזַּעַם בִּי. אָז, בְּדַרְכֵיכֶם חַסְרוֹת הָאֱמוּנָה, אַתֶּם מִתְוַכְּחִים עַל עִתּוּי מַאֲרָע זֶה. אַף שֶׁחֵלֶק מִכֶּם אוֹמְרִים בְּטָעוּת שֶׁאֵין הִתְאַסְּפוּת, רַבִּים מִכֶּם דָּנִים בְּעִתּוּי הִתְרַחֲשׁוּתוֹ בְּיַחַס לִצְרַת יַעֲקֹב. אַבְהִיר זֹאת לִפְנֵיכֶם אִם תַּקְשִׁיבוּ, אַף שֶׁרַבִּים מִכֶּם יִכְפּוּ בְּעַקְשָׁנוּת עַל מַאֲרָע זֶה, עַד כְּדֵי הִתְכַּחֲשׁוּת לִי וַחֲתִימַת גּוֹרַלְכֶם בְּהֶחְלֵט לִתְפֹּס אֶת מְקוֹמְכֶם בֵּין הָעֲלָמוֹת הַכְּסִילוֹת. כִּי מָה אוֹמֵר דְּבָרִי?

וַאֲנַחְנוּ מְבַקְּשִׁים מִכֶּם, אַחַי, בְּבוֹא אֲדוֹנֵנוּ יֵשׁוּעַ הַמָּשִׁיחַ, וּבְהִתְאַסְּפֵנוּ אֵלָיו, שֶׁלֹּא תִּטַּלְטְלוּ פִּתְאֹם מִדַּעְתְּכֶם, וְלֹא תִּבָּהֲלוּ, לֹא מֵרוּחַ, לֹא מִדָּבָר, וְלֹא מֵאִגֶּרֶת, כְּאִלּוּ מֵאִתָּנוּ, כְּאִלּוּ יוֹם הַמָּשִׁיחַ קָרֵב. אִישׁ אַל-יַטְעֶה אֶתְכֶם בְּשׁוּם אֹפֶן: כִּי לֹא יָבוֹא הַיּוֹם הַהוּא, אֶלָּא אִם-כֵּן תָּבוֹא הָעֲזִיבָה תְּחִלָּה, וְיִגָּלֶה אִישׁ הַחַטָּאת, הוּא בֶּן-הָאֲבַדּוֹן. (הַבִּיטוּ אֶל הַבִּיזַנְטִי, יְלָדַי, כִּי סִפְרֵי תָּנָ"ךְ מֻשְׁחָתִים רַבִּים בָּאוּ מִטֶּקְסְטִים גְּנוֹסְטִיִּים, טֶקְסְטִים מְזֻיָּפִים שֶׁנִּתְּנוּ עַל-יְדֵי אֲנָשִׁים חוֹטְאִים. הַאִם אֵינְכֶם מְבִינִים אֶת הַהֶבְדֵּל בֵּין טֶקְסְט מִקְרָאִי לְגִרְסַת תָּנָ"ךְ?) תִּהְיֶה נְטִישַׁת הָאֱמוּנָה בִּכְפִירָה, וְאֵלֶּה הַמְּסָרְבִים לַעֲזֹב אֶת הָאֱמוּנָה יַעַזְבוּ אֶת הָאָרֶץ לִפְגֹּשׁ אוֹתִי בָּאֲוִיר.

כִּי כָּאן נִמְצָא מְשַׁל עֶשֶׂר הָעֲלָמוֹת. חָמֵשׁ עֲלָמוֹת חֲכָמוֹת קָנוּ שֶׁמֶן וְטִיְּבוּ אֶת פְּתִילוֹתֵיהֶן וְהָיוּ מוּכָנוֹת כְּשֶׁהִגִּיעַ הֶחָתָן, וְחָמֵשׁ לֹא הָיוּ מוּכָנוֹת וְהִזְנִיחוּ לִקְנוֹת שֶׁמֶן, וְנִסּוּ תַּחַת זֹאת לָקַחַת מִצָּרְכֵי אֲחֵרִים. חָמֵשׁ הָעֲלָמוֹת הַמּוּכָנוֹת הִמְשִׁיכוּ עִם הֶחָתָן, וְחָמֵשׁ הַכְּסִילוֹת בִּקְּשׁוּ לְהִכָּנֵס לְאַחַר מִכֵּן וְנֶאֱמַר לָהֶן, סוּרוּ מִמֶּנִּי, מֵעוֹלָם לֹא יְדַעְתִּיכֶן. אַל תְּכַבּוּ אֶת רוּחִי שֶׁבָּכֶם אֲשֶׁר תְּאַמֵּת דְּבָרִים אֵלֶּה. הָבִינוּ שֶׁבֶּן-הָאֲבַדּוֹן אֵינוֹ יָכוֹל לַעֲלוֹת לַשִּׁלְטוֹן וּמַלְכוּתוֹ הַמְּלֵאָה שֶׁל הַשָּׂטָן אֵינָהּ יְכוֹלָה לַעֲלוֹת מִן הַתְּהוֹם עַד שֶׁיּוּסַר הַמְּעַכֵּב, וּמִכֵּיוָן שֶׁנּוֹשַׁעְתֶּם מִן הַזַּעַם, הַמְּעַכֵּב יוּסַר כַּאֲשֶׁר אֲקַבֵּץ אֶת כַּלָּתִי אֵלַי בֶּעָנָנִים כְּשֶׁתִּפְגֹּשׁ אוֹתִי בָּאֲוִיר.

אָז, בְּסוֹף שֶׁבַע שְׁנוֹת צָרַת יַעֲקֹב, תָּבוֹא תְּחִיַּת הַצַּדִּיקִים. הִנֵּה! וְרָאִיתִי כִסְאוֹת, וַיֵּשְׁבוּ עֲלֵיהֶם, וּמִשְׁפָּט נִתַּן לָהֶם: וְרָאִיתִי אֶת-נַפְשׁוֹת הָעֲרוּפִים עַל-עֵדוּת יֵשׁוּעַ וְעַל-דְּבַר אֱלֹהִים, וַאֲשֶׁר לֹא הִשְׁתַּחֲווּ לַחַיָּה וְלֹא לְצַלְמָהּ, וְלֹא קִבְּלוּ אֶת-הַתָּו עַל-מִצְחָם אוֹ בְיָדָם; וַיִּחְיוּ וַיִּמְלְכוּ עִם הַמָּשִׁיחַ אֶלֶף שָׁנִים. וּשְׁאָר הַמֵּתִים לֹא חָיוּ עוֹד עַד תֹּם אֶלֶף הַשָּׁנִים. זֹאת הַתְּחִיָּה הָרִאשׁוֹנָה. בָּרוּךְ וְקָדוֹשׁ הוּא אֲשֶׁר לוֹ חֵלֶק בַּתְּחִיָּה הָרִאשׁוֹנָה: עַל-אֵלֶּה אֵין לַמָּוֶת הַשֵּׁנִי שִׁלְטוֹן, אֶלָּא יִהְיוּ כֹּהֲנִים לֵאלֹהִים וְלַמָּשִׁיחַ, וְיִמְלְכוּ עִמּוֹ אֶלֶף שָׁנִים.

הַאִם עֲדַיִן אֵינְכֶם מַאֲמִינִים לִדְבָרַי? אָכֵן, רַבִּים מִכֶּם יִדְבְּקוּ בְּעַקְשָׁנוּת בְּטָעוּתָם וּבְדִבְרֵי הָרוֹעִים הַסּוֹרְרִים, וְרַבִּים מִכֶּם שֶׁיָּכְלוּ לְהִמָּלֵט מִזַּעַם אֱלֹהִים הַבָּא לְנַסּוֹת אֶת הָאָרֶץ בִּצְרַת יַעֲקֹב, יִבְחֲרוּ תַּחַת זֹאת לִסְבֹּל זַעַם גָּדוֹל כַּאֲשֶׁר כְּבָר הִכַּחְתִּי אֶתְכֶם וְנִסִּיתִי אֶתְכֶם, אֶלָּא שֶׁתְּסָרְבוּ בְּעַקְשָׁנוּת לְהוֹפָעָתִי, וְתִדְבְּקוּ בְּזַעַם שֶׁאֵינוֹ מְיֹעָד לָכֶם. הַאִם לֹא אָמַרְתִּי לָכֶם לָצֵאת מִתּוֹכָהּ פֶּן תִּשָּׁתְפוּ בְּמַכּוֹתֶיהָ?

וְעִם זֹאת, תִּהְיֶה עוֹד תְּחִיָּה אַחַת. וְרָאִיתִי כִּסֵּא לָבָן גָּדוֹל, וְאֶת-הַיּוֹשֵׁב עָלָיו, אֲשֶׁר מִפָּנָיו נָסוּ אֶרֶץ וְשָׁמַיִם, וְלֹא-נִמְצָא לָהֶם מָקוֹם. וְרָאִיתִי אֶת-הַמֵּתִים, קְטַנִּים וּגְדוֹלִים, עֹמְדִים לִפְנֵי אֱלֹהִים: וַיִּפָּתְחוּ סְפָרִים: וַיִּפָּתַח סֵפֶר אַחֵר, וְהוּא סֵפֶר הַחַיִּים: וַיִּשָּׁפְטוּ הַמֵּתִים מִן-הַדְּבָרִים הַכְּתוּבִים בַּסְּפָרִים, לְפִי מַעֲשֵׂיהֶם. וַיִּתֵּן הַיָּם אֶת-הַמֵּתִים אֲשֶׁר בּוֹ: וְהַמָּוֶת וְהַשְּׁאוֹל נָתְנוּ אֶת-הַמֵּתִים אֲשֶׁר בָּהֶם: וַיִּשָּׁפְטוּ אִישׁ אִישׁ לְפִי מַעֲשֵׂיהֶם, כִּי זֹאת הִיא תְּחִיַּת הָרְשָׁעִים עַמִּי.

מַדּוּעַ אֵינְכֶם מְבִינִים כִּי הִתְאַסְּפוּת קְהִלָּתִי לִפְגֹּשׁ אוֹתִי בֶּעָנָנִים בְּהוֹפָעָתִי שׁוֹנָה מִשִּׁיבָתִי עִם כַּלָּתִי וְצִבְאוֹת הַשָּׁמַיִם? מַדּוּעַ אֵינְכֶם מְבִינִים שֶׁהִתְאַסְּפוּת קְהִלָּתִי הִיא תַּעֲלוּמָה שֶׁנִּגְלְתָה עַל-יְדֵי הַשָּׁלִיחַ פּוֹלוֹס בְּהֶקְשֵׁר הַתַּעֲלוּמָה הַגְּדוֹלָה שֶׁל אֱלֹהִים. לָכֵן זוֹהִי תַּעֲלוּמָה בְּתוֹךְ תַּעֲלוּמָה, וְהִיא לֹא נֶחְשְׂפָה בִּמְלוֹאָהּ לִבְנֵי הָאָדָם עַד שֶׁנִּגְלְתָה לְפּוֹלוֹס שֶׁכָּתַב אוֹתָהּ, וַאֲנִי שִׁמַּרְתִּי אוֹתָהּ לְמַעַנְכֶם.

וְעִם זֹאת, יֵשׁ דֻּגְמָאוֹת מִסְפָּר שֶׁל אֱלֹהִים הַלּוֹקֵחַ אֲנָשִׁים לַשָּׁמַיִם לִמְטָרוֹת מְסֻיָּמוֹת. חֲנוֹךְ נֶאֱסַף. אֵלִיָּהוּ נֶאֱסַף. אֲנִי נֶאֱסַפְתִּי לַשָּׁמַיִם בְּעוֹד שְׁלִיחַי צוֹפִים בְּיִרְאָה. פִילִיפּוֹס נֶאֱסַף מִמָּקוֹם אֶחָד לְאַחֵר כַּאֲשֶׁר הָאָב לְקָחוֹ לִמְטָרוֹתַי. וְכֵן תֵּאָסֵף כַּלָּתִי אֵלַי בֶּעָנָנִים לְהִלָּקַח לַשָּׁמַיִם לְהִנָּשֵׂא לִי וּלְשָׁרֵת אֶת מַטְּרוֹת הַגְּאֻלָּה שֶׁלִּי.

הַאִם אֵינְכֶם מְבִינִים שֶׁכַּלָּתִי תִּהְיֶה מְלָכִים וְכֹהֲנִים לֵאלֹהַי וְאָבִי וְתִמְלֹךְ עִמִּי בְּמַלְכוּתִי הַבָּאָה וּתְהִי עִמִּי לְעוֹלָם? מַדּוּעַ אַתֶּם כֹּה אִטִּיִּים לְהַאֲמִין לִדְבָרַי? הַאִם הֻקְסַמְתֶּם וְהוּטְעֵיתֶם עַל-יְדֵי הָרוֹעִים הַסּוֹרְרִים? וַעֲדַיִן, רַבִּים מִכֶּם יִהְיוּ לִנְבוּאָה הַמַּגְשִׁימָה אֶת עַצְמָהּ וְיַחְתְּמוּ אֶת עַצְמָם לִנְפִילָה עַל-יְדֵי כְּפִירָה בְּמִצְוֹתַי, בִּמְיֻחָד בְּתוֹרַת הִתְאַסְּפוּת קְהִלָּתִי אֵלַי. בּוּשָׁה לָכֶם הַמְּשַׁיְּכִים תּוֹרָה זוֹ לִנְבִיאֵי שֶׁקֶר בַּהִיסְטוֹרְיָה הָאַחֲרוֹנָה, וּמַכְרִיזִים עַל הִלָּקְחוּת שִׁקְרִית וַחֲשָׁאִית הַקְּשׁוּרָה לְתוֹרוֹת שֶׁקֶר אֲחֵרוֹת שֶׁל דִּסְפֶּנְסַצְיוֹת.

הָבִינוּ שֶׁיֵּשׁ עִדָּנִים בְּצִיר הַזְּמַן שֶׁל אֱלֹהִים וְלֹא דִּסְפֶּנְסַצְיוֹת, וְאֵין הֵם אוֹתוֹ מוּשָׂג. לִמְדוּ הֵיטֵב לְהַצְדִּיק עַצְמְכֶם, וְחִדְלוּ מִלִּהְיוֹת כֹּה עַצְלָנִים שִׂכְלִית. חִדְלוּ מִלְּהִסְתַּמֵּךְ וּמִלִּקְרֹא אֶת פֵּרוּשֵׁי כָּל הָאֲחֵרִים לַכָּתוּב, וִחְיוּ עַל-פִּי הַכָּתוּב: כִּי אֵין לָכֶם צֹרֶךְ שֶׁאִישׁ יְלַמֵּד אֶתְכֶם. וְעִם זֹאת, רַבִּים מִכֶּם יִלְעֲגוּ וִיסָרְבוּ לַעֲצָתִי. אֶלָּא שֶׁדָּבָר זֶה לֹא יֵעָשֶׂה בַּחֲשַׁאי וְהַשָּׂטָן יָשִׁיק הַטְעָיָה גְּדוֹלָה כְּדֵי לְכַסּוֹת עַל אֵרוּעַ זֶה, כִּי הוּא יִהְיֶה יָדוּעַ בְּכָל הָעוֹלָם. רְעִידָה גְּדוֹלָה תִּתְרַחֵשׁ, וְהָעוֹלָם כֻּלּוֹ יִשְׁתַּנֶּה מִן הָעוֹלָם שֶׁהִכַּרְתֶּם, וְאוֹתוֹ אִישׁ הַחַטָּאת יִתְגַּלֶּה וְיַעֲלֶה מִן הָאֵפֶר שֶׁהוֹתִירוּ אַרְבַּעַת הַפָּרָשִׁים. הִגִּיעַ הַזְּמַן לְהִתְכּוֹנֵן, כַּלָּתִי, וַעֲשׂוּ זֹאת בִּמְהִירוּת, כִּי הִנֵּה הוֹפָעָתִי קְרוֹבָה!

אֲנִי הוּא בֶּן-הָאָדָם שֶׁנֻּבָּא עַל-יְדֵי חֲנוֹךְ, אֲשֶׁר לְקָחוֹ אֱלֹהִים לַשָּׁמַיִם.

אֲנִי הוּא שֶׁעָבַר הִשְׁתַּנוּת וְדִבֵּר עִם מֹשֶׁה וְאֵלִיָּהוּ עַל עֲזִיבָתִי, וֶאֱלֹהִים לָקַח אֶת-אֵלִיָּהוּ לַשָּׁמַיִם בְּעִתּוֹ.

אֲנִי הוּא שֶׁנִּלְקַח בְּעָנָן כַּאֲשֶׁר נֶאֱסַפְתִּי לַשָּׁמַיִם עַד עִתּוֹת הַשִּׁיבָה.

אֲנִי הוּא שֶׁשָּׁלַח אֶת-פִילִיפּוֹס לַסָּרִיס הָאֶתְיוֹפִּי לִפְנֵי שֶׁלְּקָחוֹ אֱלֹהִים לְמָקוֹם אַחֵר עֲלֵי אֲדָמוֹת.

אֲנִי הוּא שֶׁחָטַף אֶת-פּוֹלוֹס לַשָּׁמַיִם הַשְּׁלִישִׁיִּים וְהֶרְאָה לוֹ דְּבָרִים שֶׁאֵין לְתָאֲרָם אֲשֶׁר לֹא הָיָה חָפְשִׁי לְחַלֵּק.

אֲנִי הוּא שֶׁיֵּרֵד מִן הַשָּׁמַיִם בִּתְרוּעָה גְּדוֹלָה לֶאֱסֹף אֶת-כַּלָּתִי לִפְגֹּשׁ אוֹתִי בָּאֲוִיר.

אֲנִי הוּא שֶׁיָּקִים אֶת-שְׁנֵי עֵדַי לִתְחִיָּה לְאַחַר שֶׁיִּהְיוּ מֵתִים שְׁלֹשָׁה יָמִים וָחֵצִי, וְאֶקַּח אוֹתָם לַשָּׁמַיִם.

אֲנִי הוּא הָאָלֶף וְהַתָּו, כִּי אֲנִי הָרִאשׁוֹן וְהָאַחֲרוֹן, וְהִנֵּה! אֲנִי בָּא מַהֵר מִשֶּׁאַתֶּם חוֹשְׁבִים, לָכֵן הֱיוּ מוּכָנִים לִפְגֹּשׁ אוֹתִי אוֹ לִנְפֹּל.

כָּל-הַקּוֹרֵא בְּשֵׁם יהוה עַד קֵץ יִוָּשַׁע.
