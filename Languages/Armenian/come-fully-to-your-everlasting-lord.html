Երկակի Վկա՝ Իսրայել
Դառը և Քաղցրը
Հոկտեմբերի 8, 2024

Լիովին եկե՛ք դեպի ձեր Հավիտենական Տերը։

Լսեցե՛ք, լսեցե՛ք։ Լսեցե՛ք ձեր Տիրոջ ձայնը և լսեցե՛ք Եշուա ՀաՄաշիախի խոսքերը և խոնարհեցրե՛ք ձեր սրտերը ձեր Տիրոջ կամքին հնազանդվելով և թո՛ւյլ տվեք, որ իմ խոսքերը մխիթարեն ձեր հոգիները։ Ուստի ձեզ հարց եմ տալիս։ Արդյո՞ք լսում եք։ Երբ Մարդու Որդին գա, արդյոք հավատք կգտնի՞ երկրի վրա։ Ճշմարիտ, ճշմարիտ եմ ասում, ես հավատք կգտնեմ իմ հավատարիմների մեջ, որոնց իրենց հավատարմության համար կտանեմ երկնքի ապահովությունը, բայց երկրի վրա գտնվողների գերակշիռ մեծամասնությունն անհավատ է։ Եվ ուստի ես շարունակում եմ զգուշացնել ձեզ, իմ սիրելինե՛ր, որ հեռանաք մեղավորների մեղքերից, դուրս գաք նրանց ճանապարհներից և առանձնանաք, որպեսզի չմասնակցեք նրանց դատաստաններին։

Սակայն այնպես է, որ շատ մարդիկ կեղծավորաբար մոտենում են ինձ իրենց շուրթերով, մինչդեռ նրանց սրտերը հեռու են ինձնից։ Այդ օրը շատերը կասեն ինձ՝ Տե՛ր, Տե՛ր, և ես նրանց կասեմ՝ հեռացե՛ք ինձնից, ես ձեզ երբեք չեմ ճանաչել։ Հասկացե՛ք, որ բավարար չէ ինձ Տեր հռչակել անհավատ սրտով, այլ նրանք են ինձ իսկապես Տեր դարձնում, ովքեր ապաշխարում են իրենց մեղքերից, դադարում են դրանք գործելուց և ձգտում են արդարությամբ ապրել իմ առաջ։ Մի՞թե չգիտեք, որ ես՝ Հիսուս Նազովրեցին, կնախընտրեի ձեր ամբողջ սիրտը և նվիրվածությունը, քան ձեր շուրթերի զոհաբերությունները։ Այո՛, ես ողորմություն կկամենայի, ոչ թե զոհ, քանզի մասամբ իմ ողորմության՝ արժանի ցասման զսպման միջոցով է, որ իմ սերն է դրսևորվում, նույնիսկ Յահվեի՝ մեծ հավիտենական Աստծո սերը։ Սակայն իմ նախազգուշական խոսքերին ականջալուր են նրանք, ովքեր սիրում են ինձ և ապաշխարությամբ են պատասխանել իրենց հանդեպ իմ սիրուն, քանի որ ես ինձ հայտնել եմ նրանց՝ իրենց փրկության համար։ Եվ այսպես, իմ սերն ընդունվում է նրանց կողմից, ում սրտերը փափուկ են իմ ձայնի հանդեպ, և, իսկապես, բոլոր նրանց կողմից, ովքեր սիրում են ինձ և ում ես հրահանգել եմ սովորել ինձնից, նույնիսկ նրանց, ում ես առաջնորդում եմ իմ փրկության ճանապարհներով։

Ինչո՞ւ եք թույլ տալիս, որ ձեր սրտերը տագնապեն, իմ փոքրիկնե՛ր։ Մի՞թե իմ առաքյալներն ու մարգարեները չեն նախազգուշացրել այս օրվա մասին դարաշրջանի արշալույսից ի վեր։ Մի՞թե հին մարգարեները չեն կանխագուշակել Դանիելի յոթանասուներորդ շաբաթվա գալուստը՝ մեծ դատաստանով երկրի վրա։ Ուստի, հասկացե՛ք, դուք ապրում եք այն օրերին, որոնք նախորդում են Հակոբի նեղությանը, որը լիովին կտարածվի ամբողջ երկրով մեկ, և այս օրերի մասին բազմաթիվ աղետներ, նշաններ, հրաշքներ, զարմանալի գործեր և նույնիսկ դատաստաններ են կանխագուշակվել և այժմ սանձազերծված են, ուստի ինչո՞ւ եք զարմանում։ Սրանք վերջին նախազգուշացումներն են ավելի փոքր մասշտաբով այն բանի, ինչը պիտի գա ամբողջ երկրի վրա, երբ իմ հավատարիմները երկրից կհեռանան դեպի ինձ՝ երկինք, երբ ես նրանց կհավաքեմ ինչպես գոմի հորթերի, և նրանք կգտնեն բացարձակ անվտանգություն և լիակատարություն, երբ կանգնեն իրենց առաջին պահերին՝ վերստին ծնված հավիտենության մեջ, որտեղ նրանք երբեք չեն կորչի, չեն տառապի, ոչ էլ հիվանդություն կտանեն, այլ կունենան կատարյալ ուրախություն և կատարյալ ներդաշնակություն Յահվեի, և իսկապես իմ՝ Եշուա ՀաՄաշիախի հետ, և մեծ վարձատրություն ու ժառանգություն կտամ նրանց, քանզի նրանց վարձը իմ հզոր ձեռքում է։

Իմ հավատարիմներից շատերը խմորվել են բարգավաճման ավետարանի կեղծ վարդապետություններով, որը սխալմամբ քարոզվում է կեղծ մարգարեների և ուսուցիչների կողմից, ովքեր ստում են իմ ժողովրդին՝ ասելով, թե աղետը երբեք չի դիպչի նրանց, քանի դեռ նրանք երջանիկ մտքեր ունեն։ Այս անջուր ջրհորները մեկ այլ Հիսուս են քարոզում, և նրանք չեն ճանաչում ինձ՝ Եշուա ՀաՄաշիախին, և ոչ էլ ես եմ նրանց ուղարկել։ Միշտ համեմատե՛ք նրանց խոսքերը, ովքեր պնդում են, թե խոսում են իմ անունից, սուրբ գրքի հետ, և որտեղ նրանք հակասում են սուրբ գրքին, այնտեղ նրանք խոսում են իրենց սեփական սրտերից՝ իրենց ագահ շահի համար։

Ճշմարիտ, ճշմարիտ եմ ասում, կան բազմաթիվ կեղծ ծառայողներ, ովքեր ստեր են սերմանել իմ ժողովների մեջ, և կեղծ բարգավաճման ավետարանը տանում է դեպի կործանման լայն ճանապարհը, որտեղ դժոխքի դարպասները լայն կբացվեն նրանց առջև, ովքեր հետևում են դեպի այն հավերժ այրվող կրակի լիճը, որն այրվում է կոբալտի կապույտի ամենաթեժ բոցերով, պատրաստված ընկածների և նրանց հետևողների համար, ուստի մի՛ ընկեք նրանց ստերի մեջ։ Ես չեմ խոսում իմ հավատարիմ մարգարեների մասին, ովքեր գուցե սխալ են արտասանում մի խոսք իմ անունից, ոչ էլ նրանց մասին, ովքեր անկեղծ սխալ են գործում և ուղղում իրենց։ Հասկացե՛ք, որ իմ մարգարեներից ոչ ոք լիովին չի մարգարեանում, այլ իմ բոլոր մարգարեները մասամբ են մարգարեանում, և նրանք երբեք չեն հակասի սուրբ գրքին, երբ իսկապես խոսում են իմ՝ Քրիստոս Հիսուսի անունից, որովհետև ես եմ մարմին դարձած Աստծո խոսքը, որ բնակվում է ձեր մեջ, թեև մարմնով այլևս հայտնի չեմ։

Իմ փոքրիկնե՛ր, մի՛ խաբվեք, որովհետև այս աշխարհում միշտ կա հալածանք և տառապանք, հատկապես հիմա՝ Սատանայի թագավորության վերջում, երբ ընկածները դատաստանի են ենթարկվում, և երբ նրանք պատրաստվում են անել այն, ինչ թույլատրված է նրանց իրենց դատաստանի մեջ, որովհետև նույնիսկ Յահվեի թշնամիներն են հնազանդվում նրա ձայնին։ Մի՞թե չեք կարդացել Հռոմեացիներում, որտեղ ես Պողոսին հրահանգեցի իմ ճանապարհներով։ Ո՞վ մեզ կբաժանի Քրիստոսի սիրուց։ Նեղությո՞ւնը, թե՞ անձկությունը, թե՞ հալածանքը, թե՞ սովը, թե՞ մերկությունը, թե՞ վտանգը, թե՞ սուրը։ (Ինչպես գրված է. «Քո համար ամբողջ օրը մեզ սպանում են, մորթվելիք ոչխարների պես ենք համարվում»)։ Ո՛չ, այս բոլոր բաներում մենք ավելի քան հաղթողներ ենք նրա միջոցով, ով սիրեց մեզ։ Քանզի համոզված եմ, որ ո՛չ մահը, ո՛չ կյանքը, ո՛չ հրեշտակները, ո՛չ իշխանությունները, ո՛չ զորությունները, ո՛չ ներկա բաները, ո՛չ գալիք բաները, ո՛չ բարձրությունը, ո՛չ խորությունը, ո՛չ էլ որևէ այլ արարած չի կարող մեզ բաժանել Աստծո սիրուց, որը Քրիստոս Հիսուս մեր Տիրոջ մեջ է։

Եվ այսպես, իմ ընտրյալները միշտ շատ բաներ են տուժել իմ անվան համար, և իմ շնորհի դարաշրջանում եղել է նահատակների հսկայական բազմություն, ովքեր իրենց կյանքն են տվել ծառայության մեջ՝ փրկության ավետարանի առաջխաղացման համար, հենց այն ավետարանի, որը փրկել է ձեր հոգիները Հիսուս Նազովրեցու հանդեպ հավատքով, քանի որ մասամբ իմ նահատակների գործի միջոցով էր, որ փրկության ավետարանը հագեցրեց երկիրը։ Սակայն ձեզանից շատերը խմորվել են «անվանիր և պահանջիր» տիպի տարբեր կեղծ վարդապետությունների խմորով, որոնք հագեցրել են իմ Լաոդիկյան Եկեղեցին։

Այս չար վարդապետություններն ինձնից չեն, և ձեր մտքերը ոչինչ չեն իրականացնում ձեզնից դուրս, եթե դուք չեք խոսում իմ խոսքերը հավատքի շուրթերով, երբ ես նշաններ, հրաշքներ և զարմանալի գործեր եմ շնորհում, որ կատարվեն ձեր ձեռքերով։ Այնպես որ, դուք չեք կարող ձեզ պատկերացնելով դուրս բերել նեղություններից և հալածանքներից, ոչ էլ իմ կանչված, ընտրյալ և հավատարիմները զերծ են աղետից։ Որտե՞ղ է Աստծո Խոսքում ասվում, որ ով ինձ հավատում է, երբեք չի տառապի նեղությունից, կամ անձկությունից, կամ հալածանքից, կամ սովից, կամ մերկությունից, կամ վտանգից, կամ սրից։ Ընդհակառակը, դրանցից մի քանիսը կամ նույնիսկ բոլորը երաշխավորված են բոլոր նրանց, ովքեր հավատարմորեն ապրում են ինձ համար, և ուստի ես ձեզ խորհուրդ եմ տալիս հերքել նրանց խոսքերը, որովհետև նրանց խոսքերը խորամանկորեն հորինված առասպելներ են, ուղարկված ընկածների կողմից՝ տապալելու հենց ընտրյալների հավատքը, եթե այդպիսի բան հնարավոր լիներ։ Սակայն, հասկացե՛ք, որ իմ բոլոր հավատարիմները, ովքեր հետևում են ինձ իրենց ամբողջ սրտով, փրկված են Գառան ցասումից և Աստծո ցասումից, և նրանք կտարվեն Յահվեի գահասենյակի ապահովությունը, նախքան դատաստանը լիովին կիջնի երկրի վրա։

Եվ այսպես, իմ զավակնե՛ր, ձեզանից շատերը կտուժեն տարբեր աղետներից և հալածանքներից, որոնք բերվել են ընկածների կողմից, և շատերը կկորցնեն այս աշխարհի շահը այս վերջնական փուլում, նախքան ես կհավաքեմ իմ հավատարիմներին։ Ամեն անցնող վայրկյանի հետ իմ հայտնությունը այդքան ավելի մոտ է։ Ուստի ես քաջալերում եմ ձեզ, իմ սիրելինե՛ր, կանգնել ինձ համար և ամուր պահել իմ սերը՝ որպես ձեր դրդող ուժ՝ տանելու այն, ինչ անհրաժեշտ է, որպեսզի դուք փրկություն բերեք կորուսյալներին, երբ նրանց առաջնորդում եք դեպի Տապանը, նախքան աղետը կհարվածի։

Հասկացե՛ք, թե ինչ տվեցի Պողոսին գրելու Կողոսացիներին. «Այժմ ես ուրախանում եմ իմ չարչարանքներով, որոնք կրում եմ ձեզ համար, և լրացնում եմ այն, ինչ պակասում է Քրիստոսի կրքերից իմ մարմնում՝ նրա մարմնի՝ ժողովի համար, որի ծառայողը ես դարձա՝ ըստ Աստծո հրամանի, որը տրվեց ինձ ձեզ համար՝ լրացնելու Աստծո խոսքը»։ Այսպես է, իմ սիրելինե՛ր, որ երբ դուք տառապում եք իմ անվան համար, ապա դուք շարունակում եք և մասնակցում եք իմ տառապանքներին, որոնք ես կրեցի Գողգոթայում։ Թեև ես ոչինչ անավարտ չթողեցի ձեր փրկության մեջ, և ես եմ բոլոր ժամանակների համար կատարյալ և լիակատար Զատիկը, ես կանչել եմ ձեզանից յուրաքանչյուրին՝ շատ բաներ տանելու իմ անվան համար, ուստի այդ ամենը ուրախություն համարե՛ք։ Եվ իմացե՛ք, իմ զավակնե՛ր, որ աչքը չի տեսել, ականջը չի լսել, ոչ էլ մարդու սիրտն է մտել այն բաները, որ Աստված պատրաստել է նրանց համար, ովքեր սիրում են իրեն։ Ուստի վազե՛ք, վազե՛ք, որպեսզի հասնեք։ Քանզի Աստված անիրավ չէ, որ մոռանա ձեր գործը և սիրո աշխատանքը, որը դուք ցույց տվեցիք նրա Անվան հանդեպ՝ ծառայելով Սրբերին և դեռ ծառայելով։

Ձեզանից շատերն այնքան են մոլորվել իրենց սպասումների մեջ, որ այլևս չեն գիտակցում, որ շատ անգամ իմ կանչված, ընտրյալ և հավատարիմները կամովին են տանում դժվարություններ, նվաստացում, կորուստ և շատ ավելին, և նրանց տոկունության վերջնական արդյունքը Քրիստոսի տառապանքների մեջ փրկությունն է կորուսյալների համար և մեծ վարձատրություն ու հատուցում նրանց համար, ովքեր իրենց զոհաբերում են այդ կերպ։ Մի՞թե չեք կարդացել Գործքերում, որտեղ Պողոսն ու Սիղան շատ դժվարություններ կրեցին Մակեդոնիայում՝ դաժանորեն ծեծվելով և անարդարացիորեն բանտարկվելով ու Փիլիպպեում կոճղերի մեջ ամրացվելով։

Սակայն ես երկրաշարժ ուղարկեցի՝ նրանց կապանքներից ազատելու համար, և այդպես Փիլիպպեի բանտապետն ու նրա տունը փրկություն ձեռք բերեցին Պողոսի և Սիղայի տոկունության միջոցով։ Եվ ինձ հետևողներից շատերի համար նրանք կտանեն դժվարություններ, հալածանքներ և նեղություններ։ Բայց այս բոլոր բաներում դուք ավելի քան հաղթողներ եք, և ես իսկապես պայքարում եմ ձեզ համար, իմ զավակնե՛ր, եթե միայն հնազանդ մնաք իմ ձայնին և հետևեք իմ կամքին ձեր ամբողջ սրտով, և ես կստիպեմ ձեզ հաղթահարել ամեն ինչ, եթե միայն հավատարիմ մնաք իմ ձայնին և թույլ տաք, որ ես ուղղորդեմ ձեր քայլերը, քանզի ձեր փրկությունը ձեր հավիտենական փրկչին հնազանդվելու մեջ է։

Եվ ուստի ես ձեզ խորհուրդ եմ տալիս, իմ սիրելինե՛ր, կանգնել ինձ համար և իմ սերը տալ մահացող և դատապարտված աշխարհին, որը կանգնած է շղթաների մեջ, երբ դատաստանը պատրաստվում է լիովին իջնել։ Թույլ կտա՞ք, որ ես օգնեմ ձեզ հաղթահարել հուսահատությունը, որպեսզի ձեր հույսը պահեք ձեր մտքում։ Մոռացե՞լ եք իմ խոսքերը, իմ սիրելինե՛ր։ Եվ այժմ մնում են հավատքը, հույսը և սերը, բայց դրանցից ամենամեծը սերն է։ Եվ ուստի ես ձեզ ուղարկում եմ սիրելու կորուսյալներին, որովհետև շատ անգամ Աստծո բարությունն է, որ մարդկանց տանում է դեպի ապաշխարություն։

Ուստի, ծառայե՛ք ինձ ձեր համայնքներում, աղոթե՛ք նրանց համար, ովքեր կորցրել են ամեն ինչ, որովհետև քանի դեռ նրանք շունչ են քաշում, հնարավոր է, որ այս աշխարհի կորուստը ստիպի նրանց շահել ամբողջ հավիտենությունը։ Քարոզե՛ք մեղավորներին և փրկություն բերե՛ք կորուսյալներին, և հիշե՛ք, որ դուք իմ դեսպաններն եք, իմ սիրելինե՛ր, և կատարե՛ք ձեր կոչումն իմ մեջ և հաստատուն մնացե՛ք այս վերջին ժամանակներում։ Այսպիսով, ես ձեզ ուղարկում եմ սիրելու ձեր հարևաններին և ապրուստ ու օգնություն տրամադրելու կարիքավորներին։ Նույնիսկ մի բաժակ ջուր կամ մի ափսե ուտելիք, տվե՛ք այն ամենը, ինչ կարող եք՝ օգնելու նրանց, ովքեր տառապում են՝ ոչնչի մեջ չկասկածելով։ Եվ եթե պատահի, որ տառապեք կորուսյալների հետ միասին, ապա լիովին սիրե՛ք նրանց և ցույց տվե՛ք իմ վկայությունն ու զորությունը և խոսե՛ք իմ անունից հավատացող շուրթերով՝ հավատքով, հույսով և սիրով լցված սրտից։

Եվ ուստի ես ձեզ հարցնում եմ, վախենո՞ւմ եք օգնել մեղավորներին։ Մի՞թե չեք հասկանում, որ երբ ես ուղարկված էի միայն Իսրայելի տան կորած ոչխարների մոտ, ես իմ ժամանակն անցկացնում էի մաքսավորների և մեղավորների հետ։ Մոռացե՞լ եք, որ դուք էլ մի ժամանակ սովորական մեղքի կյանքով էիք ապրում։ Մոռացե՞լ եք, որ ես միակ անմեղ մարդն եմ, նույնիսկ Աստծո գառը։ Ուստի թող Տիրոջ փրկվածներն ասեն, որոնց ես փրկագնել եմ թշնամու ձեռքից։ Սիրե՛ք, սիրե՛ք, սիրե՛ք մինչև ձեր վերջին շունչը, քանզի ավելի լավ է մեռնել՝ ծառայելով Քրիստոսի սիրով, քան վախից կծկվել թույլ և աղքատ տարրերի խղճուկ ջանքերից, որոնց ես լիովին հաղթել եմ և հաղթանակ եմ տվել բոլոր նրանց, ովքեր ինձ իրենց հավիտենական Տերն են դարձնում։ Ուստի մի՛ սիրեք ձեր կյանքը մինչև վերջ և հետևե՛ք իմ անվախ ծառայության օրինակին՝ չարի և մահվան դեմ հանդիման, որոնք լիովին հաղթահարվել են մեծ հավիտենական Աստծո՝ նույնիսկ Էլ Շադդայի հավիտենական սիրով։

Մի՛ վախեցիր, որովհետև ես քեզ հետ եմ, մի՛ զարհուրիր, որովհետև ես քո Աստվածն եմ։ Ես քեզ կզորացնեմ, կօգնեմ քեզ և կպահպանեմ քեզ իմ արդարության աջով։ Ուրեմն եկե՛ք ինձ մոտ, իմ զավակնե՛ր, և հանգստացե՛ք իմ սիրո մեջ ձեր հանդեպ, և ես կմխիթարեմ ձեր հոգիները և կբուժեմ ձեր վերքերը, և ես կզորացնեմ ձեզ ձեր կոտրվածության մեջ, որպեսզի դուք կարողանաք տալ նրանց, ովքեր ձեր կարիքն ունեն, քանզի դուք այդպես եք ուղարկված։ Որովհետև դուք եք իմ դեսպանները, որոնք ոչ միայն ընտրված և կանչված եք իմ Թագավորության փառքին, այլև՝ իմ անվան համար տառապելու։

Ճշմարիտ, ճշմարիտ եմ ասում ձեզ, դուք իմ չարչարանքների մասնակիցներն եք, միայն հետևե՛ք իմ օրինակին և մի՛ նվաղեք ձեր հուսահատության ժամին։ Քանզի ես ձեզ հետ եմ ամեն քայլափոխի, և ես՝ Տեր Հիսուս Քրիստոսս, մաքրում եմ ձեր հոգիները իմ զտող կրակի մեջ, թեև դուք ընդամենը հողեղեն անոթներ եք բրուտի ձեռքում, զտող կրակները ձեր դեմքը կայծքարի պես կդարձնեն, որպեսզի դուք անխորտակելի լինեք իմ զորությամբ և միևնույն ժամանակ՝ ձեր մարմնի տկարության մեջ։ Սակայն այդպես է, որ մեղավոր մարմնի տկարության մեջ իմ սերը կատարելագործվում է ձեր մեջ, երբ ես ձեզ ավելի մոտ եմ ձգում ինձ և հետագայում պատրաստում ձեր հավիտենական նպատակներին։ Ուստի, արդար է, որ փրկվածները տառապեն իմ անվան համար, որովհետև ես իմ ամբողջ էությունը տվեցի տանջալից զոհաբերությամբ, որպեսզի դուք կարողանաք ապրել Յահվեի՝ մեծ ամենակարող Աստծո առաջ։ Հասկացե՛ք, որ ձեր փորձությունների և տառապանքների մեջ դուք ոչ միայն իմ փրկարար շնորհն եք բերում կարիքավորներին, այլև գործում եք ձեր սեփական փրկությունը վախով և դողով, որպեսզի ես ձեզ հասցնեմ ձեր լիակատար վարձքին, և այդպես ձեր հույսը կիրականանա։

Թերևս լսել եք Հոբի համբերության մասին։ Թե ինչպես Հոբը հավատարիմ մարդ էր, որը դժվար ժամանակների միջով անցավ և անարդարացիորեն հարձակման ենթարկվեց Սատանայի կողմից։ Թերևս լսել եք Հոբի վախճանի մասին, որովհետև Հոբը անիրավ մարդ չէր, որը պարտվեց չարին, այլ արդար մարդ, որը ողորմություն գտավ, որը չարին հաղթեց բարիով՝ հետևելով Յահվեի խոսքերին և մնալով արդարության մեջ նույնիսկ չար հալածանքի դեմ հանդիման։ Եվ այսպես Յահվեն կրկնակի վերադարձրեց Հոբին այն ամենը, ինչ անարդարացիորեն խլվել էր նրանից, և Յահվեն հատուցեց Հոբին իր դժվարությունների և հալածանքի համար։ Ահա, մենք երանելի ենք համարում նրանց, ովքեր համբերում են։ Դուք լսել եք Հոբի համբերության մասին և իմացել, թե ինչ վախճան տվեց Տերը։ Որովհետև Տերը շատ գթառատ է և ողորմած։

Եվ այսպես մնում են հավատքը, հույսը և սերը, և դրանցից ամենամեծը սերն է։ Ուրեմն հասկացե՛ք, իմ սիրելինե՛ր, որ հավատարիմների համար հույսը շուտով լիովին կիրականանա, երբ դուք կհափշտակվեք դեպի երկնքի անվտանգությունը՝ գալիք դատաստաններից։ Լսե՛ք իմ ձայնը և հասկացե՛ք իմ սերը ձեր հանդեպ, իմ կանչվածնե՛ր, հավատարիմնե՛ր և ընտրյալնե՛ր, իմ մարմի՛ն և իմ հարս։ Ուստի, հանգստացե՛ք իմ սիրո մեջ, որովհետև նույնիսկ ամենամեծ դժբախտության մեջ իմ սերը մնում է ուժեղ և անհաղթահարելի, քանզի ես մեկ եմ միասնության և նպատակի մեջ իմ Հոր հետ, և Յահվեի հավիտենական սերն է, որ ես տալիս եմ բոլոր նրանց, ովքեր հավիտենական կյանքի ժառանգներն են, և բոլոր նրանց, ովքեր գալիս են Հիսուս Նազովրեցու մոտ փրկության համար։

Հասկացե՛ք, որ ոչ մի մարդ չի կարող գալ Յահվեի մոտ, եթե ոչ իմ՝ Քրիստոս Հիսուսի միջոցով։ Եվ այսպես է, որ Յահվեի սերն անհաղթահարելի է, որովհետև Յահվեն անհաղթահարելի է, և հետևաբար ես անհաղթահարելի եմ, քանզի ես հաղթել եմ։ Եվ ես փրկության հավիտենական Տերն եմ, նա, ում Յահվեն դարձրել է Տերերի Տեր և Թագավորների Թագավոր, և նա ինձ տվել է ամեն անունից մեծ անուն, որպեսզի Հիսուս Քրիստոսի անվան առաջ ծնկի գա ամեն ծունկ երկնքում, երկրի վրա և երկրի տակ։ Հասկացե՛ք, որ ընկածները վախենում են ձեզանից, իմ սիրելինե՛ր, և մեծ վախ է պատել նրանց սրտերը՝ իրենց մոտալուտ դատաստանի պատճառով, որը չի ուշանում։

Ես պատվիրում և քաջալերում եմ ձեզ, իմ զավակնե՛ր, որ այս վերջին ժամանակներում, մինչև աղետը լիովին կհասնի, իմ սերը տանեք այս հիվանդ, մահացող և դատապարտված աշխարհին։ Ասում եմ ձեզ՝ բուժե՛ք հիվանդներին, հարություն տվե՛ք մեռելներին, մաքրե՛ք բորոտներին, դևեր հանե՛ք, քանզի նա, ով հավատում է ինձ, այն գործերը, որ ես արեցի երկրի վրա, նա ևս կանի, և ավելի մեծ գործեր էլ, որովհետև ես գնացել եմ մեր Հոր մոտ։ Քայլե՛ք մեծ սիրով և խոնարհությամբ, և հանդիպե՛ք կարիքավորներին Քրիստոսի զորությամբ, որը ես հզորապես կգործադրեմ ձեր մեջ։ Քանզի ձեր՝ իմ հողեղեն անոթների մեջ է բնակվում Աստծո գերազանց զորությունը Քրիստոսի միջոցով։ Հավատքով խոսե՛ք այն խնդիրների հետ, որոնց բախվում եք։ Հավատք ունեցե՛ք Յահվեի հանդեպ և հավատք՝ Եշուա ՀաՄաշիախի հանդեպ, և ասացե՛ք այդ սարին՝ ՎԵ՛Ր ԿԱՑ ԵՎ ԾՈ՛ՎՆ ընկիր։ Եվ երբ դուք դա անեք հավատարմորեն, ես կշնորհեմ, որ բազում նշաններ, հրաշքներ և զարմանալի գործեր կատարվեն հավատարիմների ձեռքով։

Եղե՛ք համարձակ և քաջ, քանզի ես Հուդայի ցեղի Առյուծն եմ, որը մարտնչում է ձեզ համար։ Ես աղաչում եմ ձեզ և հորդորում Աստծո սիրով, որը ձեր Տեր Քրիստոս Հիսուսի մեջ է, որ կամովին համբերեք իմ անվան համար և մոլորվածներին տուն բերեք մեր Հոր մոտ նույնիսկ այս վերջին ժամանակներում։ Ես աղաչում եմ ձեզ, իմ ընտրյալնե՛ր, օրինակ ծառայե՛ք, քանզի դուք իմ հավատարիմ և ընտրյալներն եք, որոնց ես կոփել եմ նեղության մեջ, որպեսզի ձեր մարմնի տկարության և կոտրվածության մեջ դուք կարողանաք դրսևորել մեծ հավիտենական Աստծո՝ հենց Յահվեի զորությունը, երբ ծառայության և մեծ իշխանությամբ գնում եք նրանց մոտ, ում ես ուղարկում եմ ձեզ ծառայելու։

Ես ձեզ հետ եմ ամեն քայլափոխի։

Ես մեկ եմ միասնության և նպատակի մեջ իմ Հոր հետ։

Ես փրկության հավիտենական Տերն եմ։

Ես նա եմ, ում Յահվեն դարձրել է Տերերի Տեր և Թագավորների Թագավոր։

Ես Հուդայի ցեղի Առյուծն եմ, որը մարտնչում է ձեզ համար։

Ես՝ Եշուա ՀաՄաշիախս, այս խոսքերն ուղարկել եմ իմ ծառայի միջոցով, և իմ ծառայի միջոցով իմ խոսքերն ուղարկվել են բոլոր նրանց հնազանդության և քաջալերության համար, ում այս խոսքերը կհասնեն։ Եվ իմ Խոսքերը կգտնեն նրանց աչքերը, ականջները և սրտերը, ում ուղարկված են։ Հաստա՛տ կանգնեք, իմ սիրելինե՛ր, քանզի ձեր վարձքը ձեր շեմին է, միայն թե մի՛ նվաղեք։ Հանգստացե՛ք իմ սիրո մեջ և հանգստացե՛ք իմ գործերի մեջ՝ դադարելով ձեր սեփականներից, և հիմա ձեր փրկագնման ժամանակն է, և հիմա ձեր լիակատար փրկության ժամանակն է, և ես գալիս եմ ձեզ համար, իմ սիրելինե՛ր։ Լսե՛ք ինձ և լսե՛ք իմ խոսքերը, ես ձեզ չեմ թողնի և չեմ լքի, իմ հավատարիմնե՛ր Քրիստոսի մեջ, բոլոր նրանց, ովքեր հնազանդ են իմ ձայնին։ Նայե՛ք երկնքին, քանզի իմ հայտնությունը շուտով կբացվի երկնակամարում մի ժամի և պահի, երբ ամենաքիչն եք սպասում։ Ես գալիս եմ ձեզ համար, իմ հավատարիմնե՛ր, պատրա՞ստ եք։
