اپنے ابدی خُداوند کے پاس پوری طرح سے آ جاؤ!

سُنو، سُنو! اپنے خُداوند کی آواز سُنو اور یسوع مسیح کے کلمات سُنو اور اپنے دِلوں کو اپنے خُداوند کی مرضی کے تابع جھکاؤ اور میرے کلمات کو اپنی رُوحوں کو تسلی دینے دو۔ تو میں تم سے ایک سوال کرتا ہوں؟ کیا تم سُن رہے ہو؟ جب ابنِ آدم آئے گا، تو کیا میں زمین پر ایمان پاؤں گا؟ سچ، سچ، میں اپنے وفاداروں میں ایمان پاؤں گا جنہیں میں اُن کی وفاداری کے سبب آسمانوں کی حفاظت میں اُٹھا لے جاؤں گا، لیکن زمین پر رہنے والوں کی بڑی اکثریت بے ایمان ہے۔ اور اِس لیے میں تمہیں، میرے عزیزو، خبردار کرتا رہتا ہوں کہ گنہگاروں کے گناہوں سے الگ ہو جاؤ، اور اُن کے طریقوں سے نکل آؤ اور الگ رہو، کہیں ایسا نہ ہو کہ تم اُن کی سزاؤں میں شریک ہو جاؤ۔

پھر بھی، ایسا ہی ہے، کہ بہت سے لوگ بے دِلی سے اپنے ہونٹوں سے میرے قریب آتے ہیں جبکہ اُن کے دِل مجھ سے دُور ہیں۔ اُس دِن بہت سے لوگ مجھ سے کہیں گے خُداوند، خُداوند، اور میں اُن سے کہوں گا، مجھ سے دُور ہو جاؤ میں نے تمہیں کبھی نہیں جانا! سمجھو کہ بے ایمان دِل سے مجھے خُداوند کہنا کافی نہیں، بلکہ وہ جو سچائی سے مجھے خُداوند مانتے ہیں وہ ہیں جو اپنے گناہوں سے توبہ کرتے اور باز آتے ہیں اور میرے حضور راستبازی سے زندگی گزارنے کی کوشش کرتے ہیں۔ کیا تم نہیں جانتے کہ میں، یسوع ناصری، تمہارے ہونٹوں کی قربانیوں پر تمہارے پورے دِل کو محبت اور وابستگی میں زیادہ پسند کروں گا؟ ہاں، میں رحم پسند کروں گا نہ کہ قربانی، کیونکہ جزوی طور پر میرے رحم کے ذریعے، یعنی واجب غضب کو روکنے سے، میری محبت ظاہر ہوتی ہے، یہاں تک کہ یہوواہ عظیم ابدی خُدا کی محبت! پھر بھی میرے انتباہی کلمات وہ سُنتے ہیں جو مجھ سے محبت کرتے ہیں اور جنہوں نے میری محبت کے جواب میں توبہ کی ہے جیسا کہ میں نے اُن کی نجات کے لیے خود کو اُن پر ظاہر کیا ہے۔ اور اِس طرح، میری محبت اُن کے درمیان قبول کی جاتی ہے جن کے دِل میری آواز کے لیے نرم ہیں اور، درحقیقت، اُن سب کے لیے جو مجھ سے محبت کرتے ہیں اور جنہیں میں نے مجھ سے سیکھنے کی ہدایت کی ہے، یہاں تک کہ وہ بھی جن کی میں اپنی نجات کی راہوں میں رہنمائی کرتا ہوں۔

تم اپنے دِلوں کو کیوں پریشان ہونے دیتے ہو، میرے بچو؟ کیا میرے رسولوں اور نبیوں نے اِس زمانے کے آغاز سے اِس دِن کے خلاف خبردار نہیں کیا؟ کیا قدیم نبیوں نے دانیال کے سترویں ہفتے کے زمین پر بڑی عدالت کے ساتھ آنے کی پیشین گوئی نہیں کی تھی؟ اِس لیے، سمجھو، تم اُن دِنوں میں جی رہے ہو جو یعقوب کی مصیبت کے پوری زمین پر پھیلنے سے ٹھیک پہلے کے ہیں اور اِن دِنوں کی بہت سی آفتوں، نشانوں، معجزوں، عجائب، اور یہاں تک کہ سزاؤں کی پیشین گوئی کی گئی ہے اور اب وہ نازل ہو رہی ہیں، تو تم حیران کیوں ہوتے ہو؟ یہ آنے والے وقت کے چھوٹے پیمانے پر آخری انتباہات ہیں جو پوری زمین پر آئیں گے جب میرے وفادار زمین سے میرے پاس آسمانوں میں چلے جائیں گے جب میں اُنہیں باڑے کے بچھڑوں کی طرح اکٹھا کروں گا، اور وہ مکمل حفاظت اور تکمیل پائیں گے جب وہ اپنی ابدیت میں نئے سرے سے پیدا ہونے کے پہلے لمحات میں کھڑے ہوں گے، جہاں وہ کبھی ہلاک نہ ہوں گے، نہ دُکھ اُٹھائیں گے، نہ بیماری برداشت کریں گے، بلکہ اُنہیں کامل خوشی اور یہوواہ کے ساتھ کامل ہم آہنگی حاصل ہو گی، اور درحقیقت میرے، یسوع مسیح کے ساتھ، اور میں اُنہیں بڑا اجر اور میراث دوں گا، کیونکہ اُن کا اجر میرے قوی ہاتھ میں ہے۔

میرے بہت سے وفادار خوشحالی کی انجیل کی جھوٹی تعلیمات سے خمیر آلود ہو گئے ہیں جو جھوٹے نبیوں اور اُستادوں کی طرف سے غلط طور پر پھیلائی جاتی ہیں جو میرے لوگوں سے جھوٹ بولتے ہیں، اُنہیں بتاتے ہیں کہ جب تک وہ خوشگوار خیالات سوچتے رہیں گے، آفت اُنہیں کبھی نہیں چھوئے گی۔ یہ بے آب کنویں ایک اَور یسوع کی منادی کرتے ہیں اور وہ مجھے، یسوع مسیح کو، نہیں جانتے، اور نہ ہی میں نے اُنہیں بھیجا ہے۔ ہمیشہ اُن لوگوں کے کلمات کا موازنہ کلامِ مقدس سے کرو جو میرے لیے بولنے کا دعویٰ کرتے ہیں اور جہاں وہ کلامِ مقدس سے متصادم ہوں وہاں وہ اپنے لالچی فائدے کے لیے اپنے دِلوں سے بولتے ہیں۔

سچ، سچ، بہت سے جھوٹے خادم ہیں جنہوں نے میری کلیسیاؤں میں جھوٹ بویا ہے اور جھوٹی خوشحالی کی انجیل تباہی کے چوڑے راستے کی طرف لے جاتی ہے جہاں جہنم کے دروازے اُن لوگوں کے لیے کھل جائیں گے جو اُس ہمیشہ جلنے والی آگ کی جھیل کی پیروی کرتے ہیں جو کوبالٹ نیلے رنگ کے گرم ترین شعلوں سے جل رہی ہے، جو گِرے ہوئے فرشتوں اور اُن کے پیروکاروں کے لیے تیار کی گئی ہے، لہٰذا اُن کے جھوٹ کے فریب میں نہ آؤ! میں اپنے اُن وفادار نبیوں کی بات نہیں کر رہا جو شاید میرے لیے کوئی لفظ غلط بول دیں، نہ ہی اُن کی جو ایک دیانتدارانہ غلطی کرتے ہیں اور خود کو درست کر لیتے ہیں۔ سمجھو کہ میرا کوئی بھی نبی مکمل طور پر نبوت نہیں کرتا، بلکہ میرے تمام نبی جزوی طور پر نبوت کرتے ہیں اور جب وہ سچائی سے میرے، مسیح یسوع کے لیے، بولتے ہیں تو وہ کبھی کلامِ مقدس سے متصادم نہیں ہوں گے، کیونکہ میں خُدا کا کلام ہوں جو مجسم ہوا اور تمہارے درمیان سکونت کرتا ہے، اگرچہ اب میں جسم کے اعتبار سے نہیں پہچانا جاتا۔

میرے بچو، دھوکہ نہ کھاؤ کیونکہ اِس دُنیا میں ہمیشہ ظلم و ستم اور اذیت ہوتی ہے، خاص طور پر اب شیطان کی بادشاہی کے خاتمے پر جب گِرے ہوئے فرشتے عدالت میں آتے ہیں اور جب وہ وہ کرنے کی تیاری کرتے ہیں جو اُنہیں اپنی عدالت میں کرنے کی اجازت ہے کیونکہ یہوواہ کے دشمن بھی اُس کی آواز مانتے ہیں۔ کیا تم نے رومیوں میں نہیں پڑھا جہاں میں نے پولس کو اپنے طریقوں کی ہدایت کی؟ کون ہمیں مسیح کی محبت سے جُدا کرے گا؟ کیا مصیبت، یا تنگی، یا ظلم، یا کال، یا ننگا پن، یا خطرہ، یا تلوار؟ (جیسا کہ لکھا ہے، تیری خاطر ہم دِن بھر قتل کیے جاتے ہیں، ہم ذبح ہونے والی بھیڑوں کے برابر گِنے جاتے ہیں۔) نہیں، اِن تمام چیزوں میں ہم اُس کے وسیلہ سے جو ہم سے محبت کرتا تھا، غالب سے بھی بڑھ کر ہیں۔ کیونکہ مجھے یقین ہے کہ نہ موت، نہ زندگی، نہ فرشتے، نہ حکومتیں، نہ قدرتیں، نہ حال کی چیزیں، نہ مستقبل کی، نہ بلندی، نہ گہرائی، اور نہ کوئی اَور مخلوق، ہمیں خُدا کی اُس محبت سے جُدا کر سکے گی، جو ہمارے خُداوند مسیح یسوع میں ہے۔

اور اِس طرح میرے چُنے ہوئے لوگوں نے ہمیشہ میرے نام کی خاطر بہت سی تکلیفیں اُٹھائی ہیں اور میرے فضل کے دور میں شہیدوں کی ایک بہت بڑی تعداد رہی ہے جنہوں نے نجات کی خوشخبری کو آگے بڑھانے کی خدمت میں اپنی جانیں دیں، وہی خوشخبری جس نے یسوع ناصری پر ایمان کے ذریعے تمہاری رُوحوں کو بچایا ہے کیونکہ جزوی طور پر میرے شہیدوں کے کام کے ذریعے ہی نجات کی خوشخبری نے زمین کو سیراب کیا۔ پھر بھی، تم میں سے بہت سے لوگ مختلف "نام لو اور حاصل کرو" کی جھوٹی تعلیمات کے خمیر سے خمیر آلود ہو گئے ہیں جنہوں نے میری لاؤدیکیہ کی کلیسیا کو سیراب کر دیا ہے۔

یہ بُری تعلیمات میری طرف سے نہیں ہیں اور تمہارے خیالات تمہارے اپنے باہر کچھ بھی پورا نہیں کرتے جب تک کہ تم ایمان کے ہونٹوں سے میرے کلمات نہ بولو جب میں تمہارے ہاتھوں سے نشانیاں، معجزے اور عجائب ہونے کی اجازت دیتا ہوں۔ لہٰذا تم اپنے آپ کو مصیبتوں اور ظلم و ستم سے تصور کر کے باہر نہیں نکال سکتے، نہ ہی میرے بلائے ہوئے، چُنے ہوئے اور وفادار لوگ آفت سے مستثنیٰ ہیں۔ کلامِ خُدا میں کہاں لکھا ہے کہ جو کوئی مجھ پر ایمان لائے گا وہ کبھی مصیبت، یا تنگی، یا ظلم، یا کال، یا ننگا پن، یا خطرہ، یا تلوار کا سامنا نہیں کرے گا؟ اِس کے برعکس، اِن میں سے کچھ یا سبھی اُن تمام لوگوں کے لیے یقینی ہیں جو میرے لیے وفاداری سے جیتے ہیں، اور اِس لیے میں تمہیں صلاح دیتا ہوں کہ اُن کے کلمات کو رد کر دو کیونکہ اُن کے کلمات چالاکی سے گھڑی ہوئی کہانیاں ہیں جو گِرے ہوئے فرشتوں کی طرف سے بھیجی گئی ہیں تاکہ برگزیدہ لوگوں کے ایمان کو بھی اُلٹ دیں اگر ایسی کوئی بات ممکن ہو۔ تاہم، سمجھو کہ میرے تمام وفادار جو پورے دِل سے میری پیروی کرتے ہیں، برّہ کے غضب اور خُدا کے غضب سے بچائے گئے ہیں اور اُنہیں یہوواہ کے تخت کے کمرے میں حفاظت سے لے جایا جائے گا اِس سے پہلے کہ عدالت پوری طرح زمین پر نازل ہو۔

اور اِس طرح میرے بچو، تم میں سے بہت سے لوگ گِرے ہوئے فرشتوں کی لائی ہوئی مختلف آفتوں اور ظلم و ستم کا شکار ہوں گے اور بہت سے لوگ اِس آخری وقت میں اِس دُنیا کا فائدہ کھو دیں گے اِس سے پہلے کہ میں اپنے وفاداروں کو جمع کروں۔ ہر گزرتے لمحے کے ساتھ، میرا ظہور اُتنا ہی قریب ہوتا جا رہا ہے۔ لہٰذا میں تمہیں، میرے عزیزو، حوصلہ دیتا ہوں کہ میرے لیے کھڑے رہو اور میری محبت کو اپنی محرک قوت کے طور پر مضبوطی سے تھامے رہو تاکہ وہ برداشت کر سکو جو تمہارے لیے کھوئے ہوؤں کو نجات دلانے کے لیے ضروری ہے جب تم اُنہیں آفت آنے سے پہلے کشتی پر لے آؤ۔

سمجھو جو میں نے پولس کو کلسیوں کو لکھنے کے لیے دیا: اب میں اپنی اُن تکلیفوں میں خوش ہوں جو میں تمہارے لیے سہتا ہوں، اور مسیح کے دکھوں کی کمی کو اپنے جسم میں اُس کے بدن یعنی کلیسیا کی خاطر پورا کرتا ہوں، جس کا میں خُدا کے اُس انتظام کے موافق خادم بنا جو تمہارے لیے مجھے سونپا گیا تاکہ میں خُدا کے کلام کو پورا کروں۔ تو اَے میرے عزیزو، ایسا ہی ہے کہ جب تم میرے نام کی خاطر دُکھ اُٹھاتے ہو، تو تم میرے اُن دُکھوں میں جاری رہتے اور شریک ہوتے ہو جو میں نے کلوری پر برداشت کیے۔ اگرچہ، میں نے تمہاری نجات میں کچھ بھی ادھورا نہیں چھوڑا، اور میں ہر زمانے کے لیے کامل اور مکمل فسح ہوں، میں نے تم میں سے ہر ایک کو میرے نام کی خاطر بہت سی چیزیں برداشت کرنے کے لیے بلایا ہے، لہٰذا اِسے پوری خوشی جانو! اور جان لو میرے بچو کہ نہ آنکھ نے دیکھا، نہ کان نے سُنا، نہ ہی انسان کے دِل میں یہ بات آئی کہ خُدا نے اپنے محبت کرنے والوں کے لیے کیا تیار کیا ہے۔ لہٰذا دوڑو، دوڑو تاکہ تم حاصل کر سکو! کیونکہ خُدا بے انصاف نہیں کہ تمہارے کام، اور محبت کی محنت کو بھول جائے، جو تم نے اُس کے نام کی خاطر دکھائی، کہ تم نے مقدسوں کی خدمت کی ہے، اور اب بھی کرتے ہو۔

تم میں سے بہت سے لوگ اپنی توقعات میں اِس قدر گمراہ ہو گئے ہیں کہ تم اب یہ محسوس نہیں کرتے کہ کئی بار میرے بلائے ہوئے، چُنے ہوئے، اور وفادار لوگ خوشی سے مشکلات، ذلت، نقصان، اور بہت کچھ برداشت کرتے ہیں اور مسیح کے دُکھوں میں اُن کی برداشت کا آخری نتیجہ کھوئے ہوؤں کے لیے نجات اور اُن لوگوں کے لیے بڑا اجر اور معاوضہ ہے جو اِس طرح خود کو قربان کرتے ہیں۔ کیا تم نے اعمال میں نہیں پڑھا جہاں پولس اور سیلاس نے مکدونیہ میں بہت سی مشکلات برداشت کیں، شدید مار کھائی اور غیر منصفانہ طور پر قید میں ڈالے گئے اور فلپی میں کاٹھ میں جکڑے گئے؟

پھر بھی میں نے اُنہیں اُن کے بندھنوں سے آزاد کرنے کے لیے ایک زلزلہ بھیجا، اور اِس طرح فلپی کے داروغہ اور اُس کے گھرانے نے پولس اور سیلاس کی برداشت کے ذریعے نجات حاصل کی۔ اور میرے بہت سے پیروکاروں کے لیے، وہ مشکلات اور ظلم و ستم اور مصیبت برداشت کریں گے۔ لیکن اِن تمام چیزوں میں تم غالب سے بھی بڑھ کر ہو اور میں تمہارے لیے لڑتا ہوں میرے بچو، اگر تم صرف میری آواز کے فرمانبردار رہو اور پورے دِل سے میری مرضی کی پیروی کرو اور میں تمہیں تمام چیزوں پر غالب آنے دوں گا اگر تم صرف میری آواز کے وفادار رہو اور مجھے اپنے قدموں کی رہنمائی کرنے دو، کیونکہ تمہاری نجات تمہارے ابدی نجات دہندہ کی فرمانبرداری میں ہے۔

اور اِس لیے میں تمہیں، میرے عزیزو، صلاح دیتا ہوں کہ میرے لیے کھڑے رہو اور میری محبت ایک مرتی ہوئی اور مجرم دُنیا کو دو جو زنجیروں میں کھڑی ہے جب کہ عدالت پوری طرح نازل ہونے والی ہے۔ کیا تم مجھے مایوسی پر قابو پانے میں تمہاری مدد کرنے دو گے تاکہ تم اپنی اُمید کو ذہن میں رکھو؟ کیا تم میرے کلمات بھول گئے ہو، میرے عزیزو؟ اور اب ایمان، اُمید اور محبت باقی ہیں، لیکن اِن میں سب سے بڑی محبت ہے۔ اور اِس لیے میں تمہیں کھوئے ہوؤں سے محبت کرنے کے لیے بھیجتا ہوں، کیونکہ کئی بار یہ خُدا کی بھلائی ہے جو لوگوں کو توبہ کی طرف راغب کرتی ہے۔

لہٰذا، اپنی برادریوں میں میری خدمت کرو، اُن لوگوں کے لیے دُعا کرو جنہوں نے سب کچھ کھو دیا ہے کیونکہ جب تک وہ سانس لے رہے ہیں، ہو سکتا ہے کہ اِس دُنیا کا نقصان اُنہیں پوری ابدیت حاصل کرنے کا سبب بنے۔ گنہگاروں کو منادی کرو اور کھوئے ہوؤں کو نجات دلاؤ اور یاد رکھو تم میرے سفیر ہو میرے عزیزو اور مجھ میں اپنی بلاہٹ کو پورا کرو اور اِس آخری زمانے میں ثابت قدم رہو۔ لہٰذا میں تمہیں اپنے پڑوسیوں سے محبت کرنے اور ضرورت مندوں کو رزق اور مدد فراہم کرنے کے لیے بھیجتا ہوں۔ یہاں تک کہ ایک گلاس پانی یا کھانے کی ایک پلیٹ، جو کچھ بھی تم دے سکتے ہو اُن کی مدد کے لیے دو جو بغیر کسی شک کے دُکھ اُٹھا رہے ہیں۔ اور اگر تم کھوئے ہوؤں کے ساتھ دُکھ اُٹھاتے ہو تو اُن سے پوری طرح محبت کرو اور میری گواہی اور قدرت دکھاؤ اور ایمان سے بھرے دِل سے یقین کے ہونٹوں پر میرے لیے بولو، اُمید، اور محبت!

اور اِس لیے میں تم سے پوچھتا ہوں، کیا تم گنہگاروں کی مدد کرنے سے ڈرتے ہو؟ کیا تم نہیں سمجھتے کہ جب مجھے صرف اسرائیل کے گھرانے کی کھوئی ہوئی بھیڑوں کے پاس بھیجا گیا تھا تو میں نے اپنا وقت محصول لینے والوں اور گنہگاروں کے ساتھ گزارا؟ کیا تم بھول گئے ہو کہ تم بھی کبھی عادی گناہ کی زندگی گزار رہے تھے؟ کیا تم بھول گئے ہو کہ میں ہی واحد بے گناہ انسان ہوں، یہاں تک کہ خُدا کا برّہ؟ تو خُداوند کے چھڑائے ہوئے ایسا ہی کہیں جنہیں میں نے دشمن کے ہاتھ سے فدیہ دے کر چھڑایا ہے۔ محبت کرو، محبت کرو، محبت کرو اپنی آخری سانس تک کیونکہ مسیح کی محبت میں خدمت کرتے ہوئے مرنا کمزور اور حقیر عناصر کی قابلِ رحم کوششوں سے ڈر کر پیچھے ہٹنے سے بہتر ہے، جن پر میں نے پوری طرح قابو پا لیا ہے اور اُن سب کو فتح دی ہے جو مجھے اپنا ابدی خُداوند بناتے ہیں۔ لہٰذا اپنی جانوں کو آخر تک عزیز نہ رکھو اور بدی اور موت کے سامنے بے خوف خدمت کی میری مثال پر چلو جن پر عظیم ابدی خُدا، یہاں تک کہ ایل شدائی کی ابدی محبت نے پوری طرح قابو پا لیا ہے!

خوف نہ کھا، کیونکہ میں تیرے ساتھ ہوں: مت ڈر، کیونکہ میں تیرا خُدا ہوں: میں تجھے مضبوط کروں گا، اور تیری مدد کروں گا، اور اپنی عدالت کے داہنے ہاتھ سے تجھے سنبھالوں گا۔ لہٰذا میرے پاس آؤ میرے بچو اور میری محبت میں آرام کرو اور میں تمہاری رُوحوں کو سکون دوں گا اور تمہارے زخموں کو بھروں گا اور میں تمہیں تمہاری شکستگی میں مضبوط کروں گا تاکہ تم اُنہیں دے سکو جنہیں تمہاری ضرورت ہے کیونکہ تم اِسی لیے بھیجے گئے ہو۔ کیونکہ یہ تم ہی ہو میرے سفیر جو نہ صرف میری بادشاہی کے جلال کے لیے چُنے اور بلائے گئے ہو، بلکہ میرے نام کی خاطر دُکھ اُٹھانے کے لیے بھی۔

سچ، سچ، تم میرے دُکھوں میں شریک ہو، صرف میری مثال پر چلو اور اپنی مایوسی کی گھڑی میں ہمت نہ ہارو۔ کیونکہ میں ہر قدم پر تمہارے ساتھ ہوں، اور یہ میں ہی ہوں، خُداوند یسوع مسیح جو تمہاری رُوحوں کو اپنی صاف کرنے والی آگ میں پاک کرتا ہوں، اگرچہ تم کمہار کے ہاتھوں میں مٹی کے برتن ہو، پاک کرنے والی آگ تمہارے چہرے کو چقماق کی طرح سخت کر دے گی تاکہ تم میری طاقت میں اٹوٹ ہو جاؤ اور پھر بھی اپنے جسم کی کمزوری میں رہو۔ پھر بھی، ایسا ہی ہے کہ گناہ آلود جسم کی کمزوری میں میری محبت تم میں کامل ہوتی ہے جب میں تمہیں اپنے قریب لاتا ہوں اور تمہیں تمہارے ابدی مقاصد کے لیے مزید تیار کرتا ہوں۔ لہٰذا، چھڑائے ہوؤں کے لیے میرے نام کی خاطر دُکھ اُٹھانا ایک منصفانہ بات ہے کیونکہ میں نے اپنی پوری ہستی کو اذیت ناک قربانی میں دے دیا تاکہ تم یہوواہ، یہاں تک کہ عظیم قادرِ مطلق خُدا کے سامنے زندہ رہ سکو۔ سمجھو کہ تمہاری آزمائشوں اور دُکھوں میں تم نہ صرف میری نجات بخش فضل کو ضرورت مندوں تک پہنچا رہے ہو بلکہ خوف اور کپکپی کے ساتھ اپنی نجات کو بھی عمل میں لا رہے ہو تاکہ میں تمہیں تمہارے پورے اجر تک پہنچا سکوں اور اِس طرح تمہاری اُمید پوری ہونے والی ہے!

شاید تم نے ایوب کے صبر کے بارے میں سُنا ہو؟ کہ ایوب ایک وفادار آدمی تھا جو مشکل وقت میں آیا اور جس پر شیطان نے غیر منصفانہ طور پر حملہ کیا۔ شاید تم نے ایوب کے انجام کے بارے میں سُنا ہو کیونکہ ایوب کوئی ناراست آدمی نہیں تھا جو بدی سے مغلوب ہو گیا ہو، بلکہ ایک راستباز آدمی تھا جس نے رحم پایا جس نے یہوواہ کے کلمات پر عمل کرتے ہوئے اور بدی کے ظلم کے سامنے بھی راستبازی میں قائم رہتے ہوئے بدی کو نیکی سے مغلوب کیا۔ اور اِس طرح یہوواہ نے ایوب کو اُس سب کا دُگنا واپس دیا جو اُس سے غیر منصفانہ طور پر چھین لیا گیا تھا، اور یہوواہ نے ایوب کو اُس کی پریشانیوں اور ظلم کا معاوضہ دیا۔ دیکھو، ہم اُنہیں مبارک سمجھتے ہیں جو برداشت کرتے ہیں۔ تم نے ایوب کے صبر کے بارے میں سُنا ہے، اور جانا ہے کہ خُداوند نے کیا انجام دیا۔ کیونکہ خُداوند بہت رحم دل اور مہربان ہے۔

اور اِس طرح ایمان، اُمید، اور محبت باقی ہیں، اور اِن میں سب سے بڑی محبت ہے۔ تو سمجھو میرے عزیزو کہ وفاداروں کے لیے اُمید پوری طرح سے پوری ہونے والی ہے جب تم آنے والی سزاؤں سے آسمانوں کی حفاظت میں اُٹھا لیے جاؤ گے۔ میری آواز سُنو اور میری محبت کو سمجھو جو میرے بلائے ہوئے، وفادار، اور چُنے ہوئے، یہاں تک کہ میرے بدن اور یہاں تک کہ میری دُلہن کے لیے ہے! لہٰذا، میری محبت میں آرام کرو کیونکہ سب سے بڑی مصیبت میں بھی میری محبت مضبوط اور ناقابلِ تسخیر رہتی ہے کیونکہ میں اپنے باپ کے ساتھ وحدت اور مقصد میں ایک ہوں، اور یہ یہوواہ کی ابدی محبت ہے جو میں اُن سب کو دیتا ہوں جو ابدی زندگی کے وارث ہیں اور وہ سب جو نجات کے لیے یسوع ناصری کے پاس آتے ہیں۔

سمجھو کہ کوئی بھی آدمی میرے، مسیح یسوع کے، وسیلہ کے بغیر یہوواہ کے پاس نہیں آ سکتا۔ اور اِس طرح یہ ہے کہ یہوواہ کی محبت ناقابلِ تسخیر ہے کیونکہ یہوواہ ناقابلِ تسخیر ہے، اور اِس لیے میں ناقابلِ تسخیر ہوں، کیونکہ میں نے فتح پا لی ہے۔ اور میں نجات کا ابدی خُداوند ہوں، وہی جسے یہوواہ نے خُداوندوں کا خُداوند اور بادشاہوں کا بادشاہ بنایا ہے، اور اُس نے مجھے ہر نام سے بڑا نام دیا ہے تاکہ یسوع مسیح کے نام پر ہر گھٹنا آسمان اور زمین اور زمین کے نیچے جھکے۔ سمجھو کہ گِرے ہوئے فرشتے تم سے ڈرتے ہیں میرے عزیزو اور اُن کے دِلوں پر اُن کی آنے والی سزا کی وجہ سے بڑا خوف طاری ہے جو دیر نہیں لگائے گی۔

میں تمہیں، میرے بچو، حکم دیتا اور حوصلہ دیتا ہوں کہ اِس آخری وقت میں اِس بیمار اور مرتی ہوئی اور سزا یافتہ دُنیا میں میری محبت لے جاؤ اِس سے پہلے کہ آفت پوری طرح نازل ہو۔ میں تم سے کہتا ہوں بیماروں کو شفا دو، مُردوں کو زندہ کرو، کوڑھیوں کو پاک کرو، بد رُوحوں کو نکالو کیونکہ جو مجھ پر ایمان رکھتا ہے وہ کام جو میں نے زمین پر کیے وہ بھی کرے گا اور درحقیقت اِس سے بھی بڑے کام کرے گا کیونکہ میں اپنے باپ کے پاس چلا گیا ہوں۔ بڑی محبت اور فروتنی میں چلو، اور اُن سے ملو جو مسیح کی اُس قدرت کے ساتھ ضرورت مند ہیں جسے میں تم میں بڑی قوت سے متحرک کروں گا۔ کیونکہ یہ تم ہی ہو، میرے مٹی کے برتن، جن میں مسیح میں خُدا کی فضیلت کی قدرت بستی ہے۔ اُن مسائل سے ایمان میں بات کرو جن کا تم سامنا کرتے ہو۔ یہوواہ پر ایمان رکھو، اور یسوع مسیح پر ایمان رکھو اور اُس پہاڑ سے کہو ہٹ جا اور سمندر میں جا پڑ! اور جب تم وفاداری سے ایسا کرو گے، تو میں وفاداروں کے ہاتھوں سے بہت سی نشانیاں، معجزے اور عجائب ہونے دوں گا۔

دلیر اور بہادر بنو، کیونکہ میں یہوداہ کے قبیلے کا شیر ببر ہوں جو تمہارے لیے لڑتا ہے۔ میں تم سے التجا کرتا ہوں اور تمہیں خُدا کی اُس محبت سے جو تمہارے خُداوند مسیح یسوع میں ہے، مجبور کرتا ہوں کہ میرے نام کی خاطر خوشی سے برداشت کرو اور کھوئے ہوؤں کو اِس آخری وقت میں بھی ہمارے باپ کے گھر واپس لاؤ۔ میں تم سے التجا کرتا ہوں میرے چُنے ہوئے لوگو، مثال قائم کرو کیونکہ تم میرے وفادار اور چُنے ہوئے ہو جنہیں میں نے مصیبت میں ڈھالا ہے تاکہ تمہارے جسم کی کمزوری اور شکستگی میں تم عظیم ابدی خُدا، یہاں تک کہ خود یہوواہ کی قدرت کو ظاہر کر سکو جب تم خدمت اور بڑے اختیار کے ساتھ اُن کے پاس جاؤ جن کی خدمت کے لیے میں تمہیں بھیجتا ہوں۔
