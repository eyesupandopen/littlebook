اے میرے بچو، اپنے نجات دہندہ، خداوند اور بادشاہ کے کلام کو سنو اور اپنے کان اپنے خداوند، یعنی یسوع مسیح راستباز کے کلام کی طرف لگاؤ، کیونکہ میں راستبازی کا بیٹا ہوں اور میں اپنے جلال میں طلوع ہوتا ہوں، اور تم نسبتاً امن و سلامتی میں باڑے کے بچھڑوں کی طرح پلے بڑھے ہو اور میں نے تمہیں ایک آسمانی مقصد کے لیے پروان چڑھایا ہے جو تم میں، میرے وفادار اور چنے ہوئے لوگوں میں، پوری طرح ظاہر ہونے والا ہے۔ دیکھو! میں ہی وہ ہوں جو انسانوں کے دلوں کو آزماتا ہوں تاکہ تم یقینی طور پر جان سکو کہ آیا تم واقعی ایمان میں ہو، یا تم دھوکہ کھا کر اپنے گناہوں اور اس موجودہ شدید گمراہی میں بہہ گئے ہو جس نے دنیا کو مسحور کر رکھا ہے۔

کیونکہ میں وہ ہوں جس کے ساتھ یہوواہ نے اپنا تخت بانٹا ہے، یعنی مسیح یسوع، جی اٹھا اور بااختیار خداوند، اور میں خداوندوں کا خداوند اور بادشاہوں کا بادشاہ ہوں اور دیکھو یہوداہ کے قبیلے کا شیر بادلوں میں آ رہا ہے تاکہ میرے وفاداروں کو اپنے پاس لے جائے تاکہ وہ برّہ کی شادی کے لیے آسمانوں پر اٹھائے جائیں، کیونکہ میری دلہن نے خود کو تیار کر لیا ہے۔ اور اس کے چراغ بھرے ہوئے ہیں، اور اس نے اپنی بتیاں درست کر لی ہیں، اور وہ ایک عقلمند کنواری کی طرح میرے ظہور کا انتظار کر رہی ہے، یعنی وہ لوگ جنہیں میں نے پوری طرح پاک کر دیا ہے اور جو خدا کے تخت کے سامنے بے عیب ہیں۔ یعنی وہ لوگ جنہیں برابا نے مکاری سے نہیں پکڑا، کیونکہ برابا پرتشدد بغاوت، جنگ اور آفت لاتا ہے، کیونکہ وہ فی الحال کئی سروں میں سے ایک سر ہے جو ملی بھگت سے کام کرتے ہیں، لیکن وہ ایک جھوٹے موشیاخ کے طور پر ابھرے گا، اور اسے اس سرزمین کے لوگوں نے چنا ہے تاکہ وہ شیطان کے عبادت خانے پر حکومت کرے۔ کیونکہ وہ اس کا وارث ہے جسے ان کے باپ دادا نے چنا تھا، اور وہ اسے اپنے بادشاہ کے طور پر قبول کرنے کے لیے خود کو تیار کر رہے ہیں۔

لیکن اے لودیکیہ کی کلیسیا، اب تمہاری ویرانی اور عدالت کا وقت آ گیا ہے۔ کیونکہ تم نے اپنے لیے ایک دوسرے نجات دہندہ کو اپنا بادشاہ چن لیا ہے، یعنی وہ جس کی آمد جھوٹے نشانوں اور عجائبات سے ہوتی ہے اور جس نے فی الحال تمہیں جادو اور ٹونے سے گمراہ کیا ہے۔ میں یشوعا ہاماشیاخ ہوں اور میں نے تمہیں اپنے ہی کفارہ دینے والے، بے گناہ خون سے خریدا ہے، اور میں ہی وہ ہوں جو تمہیں مکمل کرے گا، اور تمہیں اپنی روح میں نئے سرے سے پیدا کرے گا، کیونکہ مجھے زندگی بخشنے والی روح بنایا گیا ہے کیونکہ یہوواہ نے اپنے بیٹے کو تمام برتری عطا کی ہے، یعنی وہ جو خدا کے برّہ اور یہوداہ کے قبیلے کے شیر کا کمال ہے! اور میں راستبازی سے عدالت کروں گا اور جنگ کروں گا کیونکہ خدا کا برّہ اور یہوداہ کے قبیلے کا شیر مجھ میں، مسیح یسوع میں، یعنی یشوعا ہاماشیاخ میں مکمل ہو گئے ہیں، کیونکہ میں وہی ہوں!

اے لودیکیہ کی کلیسیا، میں ابھی تمہاری نجات مکمل کروں گا، لیکن تم اس وقت میری بادشاہی میں داخل ہونے کے لیے تیار نہیں ہو کیونکہ تم مسیح میں اپنے یکسو مقصد سے بھٹک گئے ہو! میں تمہیں تمہاری غلطی دکھاؤں گا، اور تمہارے پاس تمام گناہوں سے توبہ کرنے کا موقع ضرور ہے، لیکن تمہارے پاس اپنے خداوند کی آواز سے بحث کرنے کا کوئی وقت باقی نہیں ہے، کیونکہ یہ سنگین گناہ ان بہت سے گناہوں میں سے ایک ہے جسے شیطان نے تمہیں توبہ میں پوری طرح میرے پاس آنے سے روکنے کے لیے استعمال کیا ہے تاکہ میں تمہیں اپنے کفارہ دینے والے خون سے تمہارے گناہوں سے پوری طرح پاک کر سکوں۔

دیکھو! غیر قوموں کا زمانہ ختم ہو رہا ہے اور وہ جنگلی زیتون کا درخت جو ان کے درمیان پیوند کیا گیا تھا، اور ان کے ساتھ زیتون کے درخت کی جڑ اور چکنائی میں شریک تھا، اب کاٹ دیا جائے گا۔ کیونکہ تم نے مسیح یسوع کے کلام کو نظر انداز کیا ہے، کیونکہ میں نے تمہیں صاف صاف کہا تھا کہ ان شاخوں کے خلاف فخر نہ کرو جو بے ایمانی میں گر گئیں: لیکن اگر تم فخر کرتے ہو، تو تم جڑ کو نہیں اٹھاتے، بلکہ جڑ تمہیں اٹھاتی ہے۔ میں یسی کی جڑ ہوں، اور میں ہی وہ ہوں جو ہر زمانے کے لیے امید کی کرن بن کر کھڑا ہے، یعنی وہ جو زمین پر نجات اور یہوواہ کے طریقے لاتا ہے اور میں اپنی میراث پر داؤد کے تخت سے راستبازی کے ساتھ حکومت کروں گا اور میں نے زمین کو اپنے بے گناہ خون سے خریدا ہے اور میں جو کچھ حق کے ساتھ میرا ہے اسے غاصبوں، یعنی گرے ہوئے فرشتوں کے ہاتھوں سے زبردستی چھین لوں گا۔

میں ہی وہ ہوں جس نے زمین پر فضل اور رحم کیا ہے، کیونکہ یہوواہ کی طرف سے مجھے غیر قوموں کی نجات کے لیے فضل کا ایک پورا زمانہ دیا گیا تھا۔ کیونکہ اسرائیل میں یہوواہ کے مقاصد ہمیشہ سے یہی تھے کہ غیر قومیں برکت پائیں اور اسرائیل کی فرمانبرداری کے ذریعے نجات حاصل کریں۔ لیکن وقت گزرنے کے ساتھ وہ مغرور اور شیخی خور ہو گئے، اور بے ایمانی کی وجہ سے قدرتی زیتون کے درخت سے کاٹ دیے گئے۔ لیکن تم، میری لودیکیہ کی کلیسیا، تم بھی اسی قسم کی تکبر، بے ایمانی اور اندھے پن کا مظہر ہو جو ان قدرتی شاخوں نے دکھایا تھا جو بے ایمانی میں کاٹ دی گئی تھیں، اور اب تم قدرتی زیتون کے درخت سے کاٹ دیے جاؤ گے۔ اور اب میں قدرتی زیتون کے درخت، یعنی اسرائیل، میرے رشتہ دار اور لوگ، یعنی وہ تمام لوگ جنہیں میں نے مجھے اپنے خداوند کے طور پر قبول کرنے کے لیے چنا ہے، پر اپنا کام مکمل کرنے کے لیے آگے بڑھوں گا تاکہ میں انہیں ایک نئے انسان کے طور پر دوبارہ تخلیق کر سکوں اور خدا کا بھید پورا ہو، یعنی میرے کفارہ دینے والے نجات کے کاموں کا کمال۔

اور تم، میری لودیکیہ کی کلیسیا، نے اپنی پہلی محبت چھوڑ دی ہے اور ایک دھوکے باز کی بانہوں میں ہو جس سے تم نے وفاداری کا حلف اٹھایا ہے۔ تم نے اپنے لیے ایک نئے برابا کو اپنا بادشاہ چن لیا ہے اور یسوع ناصری کو اپنے خداوند کے طور پر رد کر دیا ہے اور ایک دوسرے جھوٹے نجات دہندہ کی بانہوں میں ہو۔ یعنی وہ جو اسرارِ بابل کے نجات دہندہ کا روپ دھارتا ہے، جو تم میں سے ایک کی طرح بات کرتا اور کام کرتا ہے، لیکن اژدہے کی طرح بولتا ہے۔ کیونکہ اسرارِ بابل عیش و عشرت اور بڑی مادی فراوانی، لذت اور دولت میں رہتی ہے۔

اور اسرارِ بابل نے پوری دنیا کو اپنے سنہری پیالے میں موجود حرام کاری کی غلاظت سے مست کر دیا ہے، لیکن گرے ہوئے فرشتوں کے طریقے بھی ایسے ہی ہیں، جو بدترین غلاظت پیش کرتے ہیں، اور تم اسے خوشی سے اور بڑی لذت سے پیتے ہو کیونکہ تم سنہری پیالے کی وجہ سے اس انتہائی غلاظت سے اندھے ہو گئے ہو جسے تم غلیظ طریقے سے کھاتے ہو۔ یہاں تک کہ میری لودیکیہ کی کلیسیا بھی اب میرے مقدس صحیفوں کے گرد جمع نہیں ہوتی بلکہ برابا کی ریلیوں میں، جس پر تم بھروسہ کرتے ہو۔ اور اس طرح تم ایک نیا نجات دہندہ چنتے ہو کیونکہ تم دنیا اور دنیا کی چیزوں سے محبت کرتے ہو اور اس خیال کا تصور بھی نہیں کر سکتے کہ تم اپنی قیمتی لذتیں کھو دو گے، سوائے اس کے کہ میں اپنے آنے والے غضب میں تم سے تمام خوشیاں چھین لوں گا، اور اب بھی تمہارے نقصانات بڑھ رہے ہیں!

کیونکہ تم اپنے تکبر میں اندھے اور فریب خوردہ ہو کہ یہوواہ میرے، یسوع ناصری کے علاوہ کوئی اور نجات دہندہ فراہم کرے گا، کیونکہ میں عورت کی موعودہ نسل ہوں اور دان شیر کا بچہ ہے جس سے کوئی مسیحا پیدا نہیں ہوگا۔ اے لودیکیہ کی کلیسیا، میری بات سنو اور اچھی طرح سنو، کیونکہ اب یا تو تمہارے دل کے جرم سے توبہ کا وقت ہے یا اب وہ وقت ہے جب میں تمہیں صاف کرنے والی آگ کے حوالے کر دوں گا اور جو کوئی آخر تک خداوند کا نام لے گا وہ نجات پائے گا اور بہت سے جنہوں نے بے ایمانی سے میرے نام کا اقرار کیا وہ گمراہی میں پڑ جائیں گے، انہوں نے اپنا موقع دینداری کی ایک ایسی شکل پر ضائع کر دیا جس نے اس کی قدرت کا انکار کیا: میرے لودیکیہ کے لوگو، میں ہی وہ قدرت ہوں اور مجھ میں یہوواہ نے تمام برتری رکھی ہے، اور تم نے مجھے، اپنے خداوند اور نجات دہندہ کا انکار کیا ہے، اور میں تمہیں اپنے منہ سے اگل دوں گا جب تک کہ تم ابھی توبہ نہ کرو!

میرے فضل کے زمانے میں، نجات قوموں میں پھیلی، اور ایک کے بعد ایک قوم نے قدرتی زیتون کے درخت کی جڑ میں حصہ لیا اور شریک ہوئی۔ اور اب، قومیں مزید نجات کی کرن نہیں رہیں گی اور بیٹے میں ان کا وقت میرے رشتہ داروں اور لوگوں کے انکار سے آیا کیونکہ انہوں نے بے ایمانی میں چلایا کہ ہمارے لیے برابا کو چھوڑ دو! لیکن جیسے جیسے یہوواہ کے بھید میرے مقدس رسولوں اور نبیوں کے ذریعے ظاہر ہوئے، میں نے برابا کو روکے رکھا اور اسے پوری طرح نہیں چھوڑا، بلکہ صرف شرارت کے بھید کو آگے بڑھنے دیا۔ اور اب وہ وقت آ گیا ہے جب میں برابا کو پوری طرح چھوڑ دوں گا جب میرا روکنے والا ہٹا دیا جائے گا اور جب میرے وفادار، بلائے گئے اور چنے ہوئے لوگ آسمانوں پر اٹھا لیے جائیں گے جہاں میں نے ان کے مستقل ٹھکانے تیار کیے ہیں، اور انہیں بے داغ باریک کتان پہنایا جائے گا جو کبھی میلا یا داغدار نہیں ہو سکتا۔

اے لودیکیہ کی کلیسیا، کیا تم اپنی دنیا سے اتنے اندھے ہو، اور سننے سے اتنے بہرے ہو کہ اب تم میری آواز نہیں سنتے؟ تم نے شرارت کے بھید سے لائے گئے جھوٹ پر یقین کر لیا ہے، اور تم نے اپنی بگڑی ہوئی بائبلوں میں زہریلا خمیر آنے دیا ہے جس نے تمہاری کلیسیاؤں میں مکمل بگاڑ پیدا کر دیا ہے جس نے تمہارے اپنے دلوں کو غلط بے ایمانی میں کہانیوں سے آلودہ کر دیا ہے۔ تم نے زمانہ پرستی کے جادوگروں کے جھوٹ کو تمہیں جھوٹے متن اور جھوٹے الفاظ سے مسحور کرنے دیا۔ تم اب کلام کو نہیں سمجھتے، اور تم نے اپنے لیے بڑی مصیبتیں جمع کر لی ہیں جو تم پر اچانک آ پڑیں گی۔ خدا کا ہر کلام پاک ہے اور اس کا مقصد ایمان کی فرمانبرداری اور وصول کنندگان کی اصلاح ہے۔ میرے کلام کے الفاظ کو سمجھو اور اس زمانے کو سمجھو جس میں تم رہتے ہو۔ کیونکہ میں نے یوحنا عارف سے لکھوایا: اور میں نے ایک حیوان کو سمندر سے نکلتے دیکھا، جس کے سات سر اور دس سینگ تھے، اور اس کے سینگوں پر دس تاج تھے، اور اس کے سروں پر کفر کا نام تھا۔
