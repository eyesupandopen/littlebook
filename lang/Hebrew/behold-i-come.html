עדות כפולה: ישראל
נספח
14 במרץ, 2025

הִנֵּה אָנֹכִי בָא: אִגֶּרֶת אַהֲבָה וְנֶחָמָה לַנֶּאֱמָנִים בַּמָּשִׁיחַ

שִׁמְעוּ אֶת דִּבְרֵי יֵשׁוּעַ הַמָּשִׁיחַ וְקַבְּלוּ אֶת דִּבְרֵי נֶחָמָתִי וְשִׂמְחָתִי, כִּי גְּאֻלַּתְכֶם הַשְּׁלֵמָה בָּאָה וְקָרְבָה. אָמֵן אָמֵן, אָנֹכִי הוּא הַבָּא לִקְהִלָּתִי, וַחֲתַנְכֶם אֵינוֹ מִתְמַהְמֵהַּ, כַּלָּתִי; עִמְדוּ אֵיתָן בֶּאֱמוּנָה רַבָּה וְהַנִּיחוּ לִדְבָרַי וּלְכֹחִי וְלִגְבוּרָתִי לִהְיוֹת יַצִּיבוּת עִתּוֹתֵיכֶם וְעֹז יְשׁוּעַתְכֶם, וְיִרְאַת יהוה הִיא אוֹצַרְכֶם.

אָכֵן, הָעִתִּים בָּאוּ עֲלֵיכֶם, אֲהוּבַי, וְהִגִּיעַ זְמַן הִתְגַּלּוּתִי הַשְּׁלֵמָה. וְאֶתְגַּלֶּה לְכָל אֶחָד וְאֶחָד בְּתוֹרוֹ. תְּחִלָּה אֶתְגַּלֶּה בִּמְלוֹאִי לַנֶּאֱמָנִים בַּמָּשִׁיחַ בַּשָּׁמַיִם, פִּתְאוֹם וּבְבִלְתִּי צָפוּי עַתָּה, בִּקְרִיאָתוֹ הָאַדִּירָה שֶׁל שַׂר הַמַּלְאָכִים, וְאוֹתוֹ שׁוֹפָר אַדִּיר יִתְקַע, כִּי נִתַּן בְּיָדוֹ לִתְקֹעַ בִּפְקֻדָּתִי בְּאוֹתוֹ רֶגַע מֻשְׁלָם בְּדִיּוּק שֶׁנִּבְחַר עַל יְדֵי יָהְוֶה עַצְמוֹ בְּנִצְחֵי עָבָר! אָכֵן, בְּתָכְנִיּוֹתָיו הַנִּצְחִיּוֹת וְהָעוֹלָמִיּוֹת לִישׁוּעַת גֶּזַע אָדָם, וְקוֹלוֹ יְמַלֵּא אֶת הָאָרֶץ, וְהַנֶּאֱמָנִים בַּמָּשִׁיחַ יִשְׁמְעוּ "עֲלוּ הֵנָּה!" – כֹּה פִּתְאוֹם וְכֹה בִּלְתִּי צָפוּי עַתָּה.

אַךְ הַבּוֹגְדִים וְעַקְשָׁנִים יִוָּתְרוּ מֵאָחוֹר עִם הַנְּפוּלִים וַחֲסֵרֵי הָאֱמוּנָה לִקְרֹא בִּשְׁמִי עַד קִצָּם, וְכָל הָעוֹשִׂים כֵּן יִלָּקְחוּ לִישׁוּעַת עוֹלָמִים וּלְכָבוֹד, אָכֵן גְּמוּל נֶצַח אֲשֶׁר עֶרְכּוֹ וּמִשְׁקָלוֹ בֶּאֱמֶת אֵינָם רְאוּיִים לְהַשְׁוָאָה לְסִבְלוֹת וּלְנִסְיוֹנוֹת הָאָרֶץ הַזֹּאת, הַזְּעִירִים וַחֲסֵרֵי הַחֲשִׁיבוּת בְּהַשְׁוָאָה. כֹּה אָמַר זֶה אֲשֶׁר סָבַל אֶת מְלוֹא עֹנֶשׁ חֲטָאֵיכֶם בִּבְשָׂרִי חֲסַר הַחֵטְא, שׁוֹפֵךְ בְּיִסּוּרִים גְּדוֹלִים אֶת דָּמִי חֲסַר הַחֵטְא עַד הַטִּפָּה הָאַחֲרוֹנָה, וְלֹא עָצַרְתִּי דָּבָר בַּעַדְכֶם, אֲהוּבַי. לָכֵן, הַבִּיטוּ אֵלַי, יֵשׁוּעַ הַנָּצְרִי, אָכֵן שֵׂה הָאֱלֹהִים אֲשֶׁר הוּא מְכוֹנֵן אֱמוּנַתְכֶם וּמַשְׁלִימָהּ, אֲשֶׁר בַּעֲבוּר הַשִּׂמְחָה הָעֲרוּכָה לְפָנַי סָבַלְתִּי אֶת הַצְּלָב, בּוֹזֵז לַחֶרְפָּה, וְהִנְנִי יוֹשֵׁב לִימִין כִּסֵּא הָאֱלֹהִים. סֶלָה!

וְכֵן אַתֶּם, הַנֶּאֱמָנִים בַּמָּשִׁיחַ, כֻּלְּכֶם תִּלָּקְחוּ לִשְׁמֵי הַשָּׁמַיִם, שָׁם אַצִּיג אֶתְכֶם לִפְנֵי יָהְוֶה עַצְמוֹ! וְתִתְפַּלְּאוּ עַל הֶהָדָר שֶׁבּוֹ תִּהְיוּ לְבוּשִׁים, וְתֵדְעוּ כְּפִי שֶׁנּוֹדַעְתֶּם, וְתִרְאוּ כְּפִי שֶׁנִּרְאֵיתֶם, וְתִשְׁמְעוּ כְּפִי שֶׁנִּשְׁמַעְתֶּם עַל יְדֵי אֱלֹהֵי הַצֶּדֶק הָעוֹלָמִי, אָכֵן יָהְוֶה עַצְמוֹ הַזּוֹכֵר כָּל מִלָּה בְּכָל תְּפִלָּה שֶׁאֵי פַּעַם הֲגִיתֶם, בֵּין אִם בְּקוֹל רָם בְּאֶמְצָעוּת כְּלֵי הַתִּקְשֹׁרֶת שֶׁלָּכֶם, וּבֵין אִם בְּשֶׁקֶט בְּתוֹךְ לִבְּכֶם, כִּי יָהְוֶה הוּא חוֹקֵר לְבָבוֹת, וּמָה הוּא אֲשֶׁר יָכוֹל לְהִסָּתֵר מֵעֵינָיו? אָמֵן אָמֵן, מָה הוּא אֲשֶׁר יָכוֹל לְהִסָּתֵר מֵעֵינַי? לָכֵן, הָבִינוּ כִּי אָבִי וְאָנֹכִי אֶחָד אָנוּ בְּאַחְדּוּת וּבְמַטָּרָה, וְכָל אֲשֶׁר לָאָב, נָתַן בְּיָדִי לִישׁוּעַת אָדָם, וּבֶאֱמֶת לִישׁוּעַת וּלְקִדּוּשׁ הָאָרֶץ עַצְמָהּ, כִּי לִי הָאָרֶץ וּבְצֶדֶק אָבֹא לִשְׁפֹּט וְלַעֲשׂוֹת מִלְחָמָה בַּנְּפוּלִים אֲשֶׁר עִתּוֹתֵיהֶם בָּאוּ, וּמִשְׁפָּטָם הַמָּלֵא אֵינוֹ מִתְמַהְמֵהַּ וְהֵם לֹא יִמָּלְטוּ.

עַל כֵּן, אֲנִי נוֹתֵן לָכֶם לְהָבִין, אֲהוּבַי, לַעֲמֹד אֵיתָן לְמַעֲנִי וּלְהַמְשִׁיךְ לְמַלֵּא אֶת מְנוֹרוֹתֵיכֶם בְּשֶׁמֶן רוּחִי שֶׁסִּפַּקְתִּי לָכֶם, וְהִתְהַלְּכוּ בְּרוּחַ הַצֶּדֶק שֶׁלִּי וּבְתוֹכָהּ כְּשֶׁאַתֶּם מִתְהַלְּכִים בֶּאֱמוּנָה, כִּי צַדִּיק בֶּאֱמוּנָתוֹ יִחְיֶה, סֶלָה! וְזֹאת כְּדֵי שֶׁלֹּא תִּתְפַּתּוּ בְּשָׁעָה מְאֻחֶרֶת זוֹ לָצֵאת בְּכֹחַ עַצְמְכֶם לְמַלֵּא אֶת מְנוֹרוֹתֵיכֶם בְּשֶׁמֶן טָמֵא הַמְּבֻסָּס עַל מַעֲשֵׂיכֶם, אֶלָּא מַלְּאוּ אֶת מְנוֹרוֹתֵיכֶם בְּשֶׁמֶן טָהוֹר שֶׁל רוּחִי לְמַעַן יָאִירוּ דַּרְכֵיכֶם, כִּי אָמֵן אָמֵן, דְּבָרִי נֵר לְרַגְלְכֶם. וּמִלֵּאתִי אֶתְכֶם כֻּלְּכֶם עַד תֹּם בְּרוּחַ הַבָּנוּת שֶׁלִּי וְשָׁפַכְתִּי אֶת רוּחִי בָּאַחֲרִית הַיָּמִים הַלָּלוּ, כַּמֻּבְטָח עַל יְדֵי יוֹאֵל בְּכִתְבֵי קֶדֶם שֶׁנִּתְּנוּ לַאֲבוֹתַי, אֲשֶׁר הֵם אֲבוֹתֵיכֶם בְּנַחֲלַת הַבְטָחָה, כִּי אַתֶּם בְּנֵי הַהַבְטָחָה, אָכֵן בְּנֵי אַבְרָהָם הַנֶּאֱמָן, סֶלָה!

הַנִּיחוּ לִי לְהַדְרִיךְ אֶתְכֶם כֵּיצַד לְהֵיטִיב אֶת פְּתִילוֹתֵיכֶם, כִּי אַתֶּם אוֹר הָעוֹלָם אֲשֶׁר לֹא יָכוֹל לְהִסָּתֵר! דֶּרֶךְ דְּבָרַי וְרוּחִי אַתֶּם מְקַבְּלִים הוֹרָאוֹת בְּצֶדֶק, וּבְרוּחַ הַבָּנוּת שֶׁלִּי אַתֶּם מֻרְשָׁעִים בְּצֶדֶק, כְּדֵי שֶׁבֶּאֱמֶת וּבְתָמִים תִּחְיוּ בְּצֶדֶק בְּכָל דָּבָר. כִּי בְּרוּחַ הַבָּנוּת שֶׁלִּי מוּעֲנֶקֶת בְּתוֹכְכֶם צִדְקָתִי הָאֲמִתִּית, וּבֶאֱמֶת וּבְתָמִים יֵשׁ לְכֻלְּכֶם בְּחִירָה לְמִי תְּצַיְּתוּ, אִם לַחֵטְא עַד מָוֶת, אוֹ לַצִּיּוּת לַצֶּדֶק עַד חַיֵּי עוֹלָם וּגְמוּל מָלֵא. לָכֵן, צַיְּתוּ לְהוֹרָאוֹתַי הַמְּפֹרָטוֹת לְחַיֵּיכֶם וְהֱיוּ עֲנָוִים לְקוֹלִי וּלְקוֹלוֹת אֵלֶּה שֶׁאֲנִי שׁוֹלֵחַ וְהַמְּדַבְּרִים בִּשְׁמִי, וּבְפִרְיָם תַּכִּירוּם, וּבִדְבָרֵיהֶם שֶׁאֲנִי נוֹתֵן לָהֶם לִישׁוּעַתְכֶם וּלְטוֹבַתְכֶם. וַאֲנִי מְצַוֶּה אֶתְכֶם לִפְעֹל אֶת יְשׁוּעַתְכֶם בְּיִרְאָה וּבִרְעָדָה, וַאֲנִי מְצַוֶּה אֶתְכֶם שֶׁלֹּא תִּמְנְעוּ טוֹב כַּאֲשֶׁר יֵשׁ בְּכֹחַ יֶדְכֶם לְהַשִּׂיגוֹ.

לָכֵן, הָבִינוּ שֶׁהֶעֱצַמְתִּי וְאִפְשַׁרְתִּי לְרַבִּים מִכֶּם לִפְעֹל לְמַטְּרוֹת יְשׁוּעָתִי, וְכִי עֲלֵיכֶם לִתְמֹךְ זֶה בָּזֶה בְּאַהֲבָה. עַל כֵּן, אַל תֶּאֶגְרוּ אֶת הַטּוּב הַחָמְרִי וְהַכַּסְפִּי שֶׁנָּתַתִּי לָכֶם, אַל תִּמְנְעוּ אוֹתוֹ מֵאֲחֵרִים בְּדַרְכֵי הַשָּׂטָן הַמְּבַקֵּשׁ לָשִׂים מִשְׁקָלִים וּמִדּוֹת עַל טוּבוֹ שֶׁל יָהְוֶה, שֶׁהוּא בֶּאֱמֶת בִּלְתִּי נִתָּן לִמְדִידָה, כִּי לֹא קַיָּם מֹאזְנַיִם בַּשָּׁמַיִם אוֹ בָּאָרֶץ הַמַּסְפִּיקִים לִמְדֹּד אֶת שֶׁפַע יָהְוֶה שֶׁהֵכִין בִּפְנֵי יֵשׁוּעַ הַמָּשִׁיחַ לָאוֹהֲבִים אוֹתוֹ.

לָכֵן, צְאוּ מִתּוֹכָם לְגַמְרֵי וְאַל תִּקְחוּ חֵלֶק בְּחַטֹּאתָם, אַל תֶּאֶגְרוּ לְעַצְמְכֶם אֶת שִׁפְעוֹ הַכַּסְפִּי וְהַחָמְרִי שֶׁל יָהְוֶה, הַנִּתָּן בִּפְנֵי יֵשׁוּעַ הַמָּשִׁיחַ, כְּאִלּוּ בְּכֹחַ יֶדְכֶם הִשַּׂגְתֶּם אֶת הַשֶּׁפַע הַזֶּה כְּדֵי לְהַחְבִּיאוֹ וְלֶאֱגֹר אוֹתוֹ לְתַאֲווֹתֵיכֶם. אָכֵן, לְמַעַן תִּמְצְאוּ נֶחָמָה בְּכָךְ שֶׁדְּאַגְתֶּם לְעַצְמְכֶם בְּגַאֲוָה וּבְחֵטְא, וְתִלְעֲגוּ לַחֲסֵרֵי-כֹּל, וְאַף תִּתְעַנְּגוּ בְּבוּשָׁה רַבָּה עַל הַמִּסְפָּרִים בְּחֶשְׁבּוֹנוֹתֵיכֶם.

לָכֵן, הָבִינוּ וְהִזָּהֲרוּ כִּי כָּל הָעוֹשִׂים דְּבָרִים כָּאֵלֶּה, מֵהַגָּדוֹל וְעַד הַקָּטָן, יִסְבְּלוּ אָבְדַן גָּדוֹל שֶׁל גְּמוּל וּבֶאֱמֶת יִקְחוּ לְעַצְמָם אֶת הַכִּנּוּי הַנָּמוּךְ בְּיוֹתֵר בְּמַלְכוּת שָׁמַיִם! וְזֹאת מִשּׁוּם שֶׁבִּגְלַל חַמְדָנוּתְכֶם שֶׁלָּכֶם יִשָּׂרְפוּ מַעֲשֵׂיכֶם לְאֵפֶר לִפְנֵי כֵּס מִשְׁפָּטִי, שָׁם תַּעַמְדוּ כֻּלְּכֶם וְתִתְּנוּ דִּין וְחֶשְׁבּוֹן. עַל כֵּן, חַלְּקוּ כָּל דָּבָר לְפִי צָרְכֵי הַקְּדוֹשִׁים בְּאַהֲבַת אִישׁ אֶת רֵעֵהוּ וְאַל תִּמְנְעוּ כָּל דָּבָר טוֹב, כִּי אָמֵן אָמֵן, הַשֶּׁפַע הַחָמְרִי וְהַכַּסְפִּי שֶׁלָּכֶם נִתַּן בְּיָדִי לְמַטְּרוֹתַי מֵאִוְצְרוֹתָיו שֶׁל יָהְוֶה, לָכֵן, אַל תֶּאֶגְרוּ אֶת טוּבוֹ שֶׁל הָאָדוֹן אֶלָּא הַשְּׁלִימוּ אֶת אוֹתָם דְּבָרִים שֶׁבָּהֶם אֲנַהֵל אֶת צַעֲדֵיכֶם בְּשֵׁרוּת וּבְאַהֲבָה לַקְּדוֹשִׁים.

צְאוּ מֵחַטֹּאתֵיכֶם, עַמִּי, פֶּן תִּקְחוּ חֵלֶק בְּמִשְׁפָּטֶיהָ, כִּי אֵלֶּה הֵן דַּרְכֵי בָּבֶל הַסּוֹדִית וְעִתּוֹת מִשְׁפָּטָהּ בָּאוּ, וְאַדִּיר הוּא אֲדֹנָי יָהְוֶה הַשּׁוֹפֵט אוֹתָהּ בְּמִשְׁפַּט כִּפְלַיִם, לָכֵן סוּרוּ מִדְּרָכֶיהָ וּמֵחֲטָאֶיהָ שֶׁאֵינָם מְקֻבָּלִים וּרְשָׁעִים בְּעֵינַי לְגַמְרֵי! סֶלָה!

כֵּן, בֶּאֱמֶת, אֲנִי יוֹעֵץ לְכָל אֶחָד וְאֶחָד לְהַמְשִׁיךְ וְלָצֵאת מֵחַטֹּאתֵיכֶם וּלָבוֹא בִּמְלוֹאֲכֶם אֶל צִדְקָתִי, וּלְהַנִּיחַ לִי לִהְיוֹת דַּיְכֶם בְּכָל דָּבָר, כִּי אָמֵן אָמֵן, אָנֹכִי הוּא.

אָנֹכִי הוּא הַבָּא בֶּעָנָנִים לַנֶּאֱמָנִים בַּמָּשִׁיחַ כֹּה פִּתְאוֹם וּבְבִלְתִּי צָפוּי עַתָּה, כִּי הָעוֹלָם, בְּחֹסֶר אֱמוּנָתוֹ, אֵינוֹ מְצַפֶּה שֶׁיֵּשׁוּעַ הַמָּשִׁיחַ יֵרָאֶה בֶּעָנָנִים כַּמֻּבְטָח בַּכְּתוּבִים. אַךְ, הָרֶגַע הַזֶּה לֹא יַפְתִּיעַ אֶתְכֶם, אֲהוּבַי, כְּגַנָּב בַּלַּיְלָה, אִם תִּשָּׁאֲרוּ עֲנָוִים לְהוֹרְאוֹתַי וְתִשְׁמְרוּ אֶת פְּרָטֵי מִצְוֹתַי בְּכָל לְבַבְכֶם, בִּידִיעָה שֶׁעֲבוּר חָרִיצוּתְכֶם וְנֶאֱמָנוּתְכֶם יָבוֹא הַגְּמוּל הַמָּלֵא שֶׁל נַחֲלַת הַנֶּאֱמָנִים בַּמָּשִׁיחַ, שֶׁבֶּאֱמֶת, עוֹד לֹא נִכְנַס בִּמְלוֹאוֹ לְלִבּוֹת בְּנֵי אָדָם בָּעֵת הַזֹּאת. לָכֵן, אַל תַּנִּיחוּ לַשָּׂטָן לִקְנוֹת אֶת גְּמוּלֵי נַחֲלַתְכֶם בְּדַרְכֵי עֵשָׂו, שֶׁהָיָה אִישׁ חֹל וְנוֹאֵף שֶׁמָּכַר אֶת בְּכוֹרָתוֹ בַּעֲבוּר נְזִיד עֲדָשִׁים שֶׁהֻשְׁלַךְ בַּבַּצֹּרֶת. סֶלָה!

לָכֵן, הָבִינוּ אֶת קִרְבַת זְמַן הוֹפָעָתִי וְכַמָּה קְרוֹבָה גְּאֻלַּתְכֶם הַשְּׁלֵמָה, אֲהוּבַי. לָכֵן אֲנִי נוֹתֵן לָכֶם לְהָבִין שֶׁצְּבָא הַשָּׁמַיִם תָּפַס אֶת מְקוֹמוֹתָיו בְּסִדְרֵיהֶם וּבְמַעַרְכֵיהֶם, וְהֵם הִתְיַצְּבוּ לְשֵׁרוּת, וְהֵם עוֹמְדִים עַל הַמִּשְׁמָר. הֵם תָּפְסוּ אֶת מְקוֹמוֹתֵיהֶם, וְהֵם מוּכָנִים! לָכֵן, הָבִינוּ, שֶׁאֲנִי אוֹחֵז בְּאוֹתוֹ שׁוֹפָר אַדִּיר בְּיָדִי, שֶׁבִּפְקֻדָּתִי הַמֵּתִים בַּמָּשִׁיחַ יָקוּמוּ בְּאֵין כִּלָּיוֹן, וְאַתֶּם הַחַיִּים וְהַנּוֹתָרִים תִּלָּקְחוּ פִּתְאוֹם לִפְגֹּשׁ אוֹתִי בֶּעָנָנִים, וְכָךְ כֻּלְּכֶם תִּהְיוּ לְעוֹלָם כְּמוֹ אֲדוֹנְכֶם, וְאֵלֶּה שֶׁהִתְגַּבְּרוּ בִּמְלוֹאָם יִהְיוּ תָּמִיד עִמִּי, כִּי אֶקַּח אֶתְכֶם לְעַצְמִי בִּבְרִית כְּשֶׁלִּי, וּמַעֲשִׂים אַדִּירִים רַבִּים תַּעֲשׂוּ, כִּי כַלָּתִי הִיא צָבָא אַדִּיר שֶׁל נְבִיאִים, מְלָכִים וְכֹהֲנִים שֶׁיֻּעצְמוּ בְּהַעצָמָה גְּדוֹלָה כְּכַלָּתִי הַלּוֹחֶמֶת, וּמִי הוּא אֲשֶׁר יוּכַל לַעֲמֹד בִּפְנֵי כַּלָּתִי שֶׁל בְּנֵי אַלְמָוֶת מֻעצָמִים הַנִּגְאָלִים מִן הָאָרֶץ בְּכֹחַ וּבְצֶדֶק רַב, סֶלָה!

לָכֵן, הָבִינוּ שֶׁהַזְּמַן הַמְּדֻיָּק שֶׁבָּחַר יָהְוֶה בִּידִיעָתוֹ מֵרֹאשׁ הָיָה תָּמִיד יוֹם, זְמַן וְשָׁעָה מְדֻיָּקִים, אֶלָּא שֶׁהוּא הָיָה חָבוּי בְּלִבּוֹ שֶׁל יָהְוֶה בִּלְבַד עַד שֶׁהִגִּיעוּ הָעִתִּים הַנְּכוֹנוֹת לְהִתְגַּלּוּתוֹ בַּשָּׁמַיִם בִּלְבַד, כִּי זְמַן זֶה אֵינוֹ נִגְלֶה עַל הָאָרֶץ, אַף שֶׁהַשָּׂטָן עוֹמֵד עַל הַמִּשְׁמָר לִטְרֹף אֶת כָּל הַנּוֹתָרִים וְהַנִּשְׁאָרִים עַל הָאָרֶץ; לָכֵן, כָּל הַקּוֹרְאִים בְּשֵׁם יֵשׁוּעַ עַד קִצָּם יִוָּשֵׁעוּ. הָבִינוּ שֶׁזְּמַן מְדֻיָּק זֶה לֹא נִתַּן גַּם לַשָּׂטָן, אַךְ עָלָיו לַעֲמֹד עַל הַמִּשְׁמָר עַד שֶׁיְּשֻׁחְרַר לְשִׁלְטוֹנוֹ הַקָּצָר, שֶׁלְּאַחֲרָיו יִכָּבֵל בְּשַׁרְשֶׁרֶת גְּדוֹלָה בִּתְהוֹם בְּעִנּוּיִים וּבְיִסּוּרִים, שָׁם יָבִין בִּמְלוֹאוֹ אֶת נַחֲלָתוֹ הַבָּאָה.

וְכָךְ הוּא שֶׁאֲנִי בָּא לָתֵת לָכֶם הֲבָנָה שֶׁל הָעִתִּים בָּהֶן אַתֶּם חַיִּים, וַאֲנִי עוֹשֶׂה זֹאת בְּכָךְ שֶׁאֲנִי נוֹתֵן לָכֶם שׁוּב אֶת אוֹתָן מִלִּים שֶׁנָּתַתִּי לְפֶטְרוֹס כְּדֵי לִפְנוֹת אֶל הֶהָמוֹן בְּאוֹתוֹ יוֹם פֶּנְטֵקוֹסְט גָּדוֹל לִפְנֵי יוֹבְלוֹת רַבִּים כָּעֵת, שֶׁאַף עַל פִּי כֵן, דְּבָרָיו מְהַדְהֲדִים אֱמֶת לָנֶצַח בְּהִתְגַּשְׁמוּת נִצְחִית וּלְעוֹלָם. לָכֵן אֲנִי אוֹמֵר לָכֶם שֶׁזֶּה הוּא אֲשֶׁר דֻּבַּר עַל יְדֵי הַנָּבִיא יוֹאֵל: וְהָיָה בְּאַחֲרִית הַיָּמִים, יֹאמַר אֱלֹהִים, אֶשְׁפֹּךְ מֵרוּחִי עַל כָּל בָּשָׂר, וְנִבְּאוּ בְּנֵיכֶם וּבְנוֹתֵיכֶם, וּבַחוּרֵיכֶם חֶזְיֹנוֹת יִרְאוּ, וְזִקְנֵיכֶם חֲלֹמוֹת יַחֲלֹמוּן. וְגַם עַל עֲבָדַי וְעַל שִׁפְחוֹתַי בַּיָּמִים הָהֵם אֶשְׁפּוֹךְ מֵרוּחִי, וְנִבָּאוּ. וְנָתַתִּי מוֹפְתִים בַּשָּׁמַיִם מִמַּעַל, וְאֹתוֹת בָּאָרֶץ מִתַּחַת, דָּם וָאֵשׁ וְתִימְרוֹת עָשָׁן. הַשֶּׁמֶשׁ יֵהָפֵךְ לְחֹשֶׁךְ, וְהַיָּרֵחַ לְדָם, לִפְנֵי בוֹא יוֹם יהוה הַגָּדוֹל וְהַנּוֹרָא. וְהָיָה, כֹּל אֲשֶׁר יִקְרָא בְּשֵׁם יהוה יִמָּלֵט.

לָכֵן, אֲנִי בָּא לְהַכְרִיז עַל דְּבָרִים אֵלֶּה כְּמִתְגַּשְּׁמִים בְּאָזְנֵיכֶם בְּעֶצֶם הַיּוֹם הַזֶּה, וְכִי פִּתְאוֹם וּבְבִלְתִּי צָפוּי עַתָּה הַנֶּאֱמָנִים בַּמָּשִׁיחַ יִלָּקְחוּ וְיִגָּאֲלוּ וְיִשְׁתַּנּוּ וְיִפָּאֲרוּ וְיִתְגַּמְּלוּ לְעוֹלְמֵי עוֹלָמִים. לָכֵן הָבִינוּ אֶת הַהִתְגַּשְׁמוּת הַפִּתְאוֹמִית שֶׁעוֹמֶדֶת לָבוֹא כֹּה בְּפִתְאוֹמִיּוּת. אַל תַּקְשִׁיבוּ לְקוֹבְעֵי הַתַּאֲרִיכִים וְלַמִּתְיַהֲדִים שֶׁאֵינָם מְבִינִים לֹא אֶת מַה שֶּׁהֵם מְלַמְדִים, וְלֹא אֶת הָאֱמֶת הָאֲמִתִּית שֶׁל הַכְּתוּבִים שֶׁהֵם מְסַלְּפִים לְרָעָתָם.

לָכֵן, אֲנִי נוֹתֵן לָכֶם לְהָבִין שֶׁהִתְכַּנְּסוּת הַנֶּאֱמָנִים בַּמָּשִׁיחַ אֵינָהּ קְשׁוּרָה בְּאֹפֶן סְפֵּצִיפִי לְשׁוּם יוֹם קֹדֶשׁ, וְלֹא לְמוֹעֲדִים, וְלֹא לְשׁוּם תַּאֲרִיךְ אוֹ אֵרוּעַ אַחֵר שֶׁנִּתַּן לְיִשְׂרָאֵל הָעַתִּיקָה בְּתוֹרַת מֹשֶׁה, אֲשֶׁר תּוֹרָה זוֹ הִתְגַּשְּׁמָה בִּמְלוֹאָהּ בְּיֵשׁוּעַ הַמָּשִׁיחַ אֲדוֹנְכֶם הָעוֹלָמִי שֶׁל הַצֶּדֶק. כְּמוֹ שֶׁאָמַרְתִּי לָכֶם בֶּעָבָר אֲנִי אוֹמֵר לָכֶם שׁוּב, אֲהוּבַי: לָכֵן, אִישׁ אַל יָדִין אֶתְכֶם עַל מַאֲכָל וּמַשְׁקֶה, אוֹ בְּעִנְיַן מוֹעֵד, אוֹ רֹאשׁ חֹדֶשׁ, אוֹ שַׁבָּתוֹת, אֲשֶׁר הֵם רַק צֵל הָעֲתִידוֹת לָבוֹא: אֲבָל הַגּוּף הוּא בַּמָּשִׁיחַ.

אַךְ, הָבִינוּ שֶׁאֵרוּעַ זֶה יָכוֹל לְהִתְרַחֵשׁ בְּכָל שְׁנִיָּה וּבְכָל יוֹם, וְיִתָּכֵן מְאֹד שֶׁיִּקְרֶה בְּיוֹם, זְמַן אוֹ שָׁעָה מַשְׁמָעוּתִיִּים, אוֹ אֲפִלּוּ בִּימֵי קֹדֶשׁ, אוֹ מוֹעֲדִים, אוֹ בַּשַׁבָּתוֹת, שֶׁהֵם צֵל הָעֲתִידוֹת לָבוֹא! לָכֵן, חַכּוּ וְצַפּוּ וְיַחֲלוּ בְּצִפִּיָּה עֲצוּמָה לְהוֹפָעָתִי, כִּי בָּאוּ עִתּוֹת גְּאֻלַּתְכֶם הַשְּׁלֵמָה. וְלָכֵן אֲנִי נוֹתֵן לָכֶם לְהָבִין וְגַם לְצַיֵּת, שֶׁתְּחַכּוּ וְתִצְפּוּ בְּכָל יוֹם (וְלֹא רַק בִּזְמַנֵּי הַמִּשְׁמָר הַגָּבֹהַּ שֶׁקְּבַעְתֶּם לְעַצְמְכֶם) וְשֶׁתִּתְּנוּ שׂוּמֶת לֵב רַבָּה יוֹתֵר לִדְבָרִים אֵלֶּה בְּצִפִּיָּה מְלֵאָה לְהִתְגַּשְּׁמוּת, כִּי אֵין עוֹד עִכּוּב. סֶלָה!

אֲנִי יֵשׁוּעַ הַמָּשִׁיחַ וַאֲנִי הוּא שֶׁשָּׁלַחְתִּי אֶת הַדְּבָרִים הָאֵלֶּה עַל יְדֵי שְׁלִיחִי וְעַבְדִּי, וְעַל יְדֵי שְׁלִיחִי וְעַבְדִּי נִשְׁלְחוּ הַדְּבָרִים הָאֵלֶּה. אָכֵן כְּעִדּוּד כְּדֵי שֶׁקְּהִלָּתִי, גּוּפִי, וְאַף כַּלָּתִי יִתְנַחֲמוּ בְּנֶחָמָה גְּדוֹלָה, וְכְדֵי שֶׁתָּבִינוּ בִּמְלוּאוֹ אֶת הַקִּרְבָה הַמְּדֻיֶּקֶת בִּזְמַן שֶׁל אֵרוּעַ זֶה, כְּדֵי שֶׁלֹּא תֻּפְתְּעוּ מֵאֵרוּעַ זֶה שֶׁבְּוַדָּאוּת יִקְרֶה כֹּה פִּתְאוֹם עַתָּה.

אֲנִי יֵשׁוּעַ הַמָּשִׁיחַ שֶׁכָּעֵת אֶפְעַל לִשְׁלֹחַ אֶת שְׁנֵי עֵדַי וְאֶת שְׁתֵּי מְנוֹרוֹתַי, אָכֵן כַּמֻּכְרָז בִּבְשׂוֹרַת הָעוֹלָמִים, כִּי יָהְוֶה מֵעוֹלָם לֹא הִשְׁאִיר עַצְמוֹ בְּלִי עֵדוּת, וּשְׁנֵי עֵדַי יָעִידוּ עַל כָּל הַזְּמַנִּים וּמִפֶּרְסְפֶּקְטִיבָה אַרְצִית וּשְׁמֵימִית, וַחֲנוֹךְ הַסּוֹפֵר מֵבִיא אֶת כִּתְבֵי קֶדֶם בְּצִקְלוֹנוֹ, וְעֵדוּיוֹתֵיהֶם יִוָּסְפוּ לִבְשׂוֹרַת הָעוֹלָמִים שֶׁתַּעֲמֹד לָעַד יַחַד עִם כָּל דִּבְרֵי יָהְוֶה שֶׁיֻּחְזְרוּ לָאָרֶץ בְּשִׁלְמוּת מְדֻיֶּקֶת כְּפִי שֶׁנִּרְשְׁמוּ בַּשָּׁמַיִם, אָכֵן כְּדֵי שֶׁמַּעֲשֵׂי הַשָּׂטָן יְבֻטְּלוּ לְגַמְרֵי וְכִי דַּרְכֵי וְדִבְרֵי יָהְוֶה יָצִיפוּ אֶת הָאָרֶץ בְּצֶדֶק לְכָל הַזְּמַנִּים וּלְעוֹלְמֵי עוֹלָמִים! סֶלָה.

הִנֵּה! אָנֹכִי הוּא הַבָּא בֶּעָנָנִים וְעִתּוֹת הִתְגַּלּוּתִי קְרֵבוֹת.

הַאִם אַתֶּם צוֹפִים, אֲהוּבַי? הַאִם צִפִּיַּתְכֶם עַזָּה וְחַיָּה?

אֲנִי מְצַוֶּה אֶתְכֶם לְקַבֵּל אֶת דְּבָרַי בִּמְלוֹאָם וְלִצְפּוֹת, וּלְחַכּוֹת, וְלִהְיוֹת מוּכָנִים.

עַד שֶׁפִּתְאוֹם... עֲלֵי אֲדָמוֹת... אֵינְכֶם.

הִנֵּה! הַשְּׁנִיָּה הַמְּדֻיֶּקֶת שֶׁל הַגְּאֻלָּה מִתְקָרֶבֶת וְנִרְאֵית בָּאֹפֶק.

שִׁקְלוּ אֶת דְּבָרַי וְאֶת בִּרְכוֹת הַבֵּנֵדִיקְצְיָה שֶׁלִּי, שֶׁבְּוַדַּאי יַעַמְדוּ בְּנֵצַח הַנְּצָחִים לְעוֹלְמֵי עוֹלָמִים. סֶלָה!

קַבְּלוּ אֶת אִגֶּרֶת הָאַהֲבָה שֶׁלִּי, כַּלָּתִי, כִּי מָסַרְתִּי אוֹתָהּ לְנֶחָמַתְכֶם וּלְצִפִּיַּתְכֶם.

בְּאַהֲבַת יָהְוֶה שֶׁנִּתְּנָה לִי אֲלֵיכֶם,

יֵשׁוּעַ הַמָּשִׁיחַ: אֲדוֹן הַצֶּדֶק הָעוֹלָמִי.
