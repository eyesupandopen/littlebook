וְעַתָּה אָנֹכִי בָּא כְּאַרְיֵה מִשֵּׁבֶט יְהוּדָה כִּי שׁוּב נִדְחֵיתִי עַל יְדֵי אֵלֶּה אֲשֶׁר אֲלֵיהֶם פָּנִיתִי. וּכְשֵׁם שֶׁנִּדְחֵיתִי עַל יְדֵי יִשְׂרָאֵל, עַמִּי שֶׁלִי אֲשֶׁר אֵלָיו נִשְׁלַחְתִּי, וְכַתּוֹצָאָה מִכָּךְ פָּנִיתִי אֶל הַגּוֹיִים, כָּךְ עַתָּה הַגּוֹיִים דָּחוּ אוֹתִי, וְשׁוּב אָשׁוּב לְיִשְׂרָאֵל, חֲזָרָה לְאֶרֶץ הַבְטָחָתִי, כִּי עוֹד אוֹשִׁיעַ שְׁאֵרִית מִיַּעֲקֹב לְמַעַן אוֹשִׁיעַ מִקְּצָת הַכֹּל. וּמִכֵּס דָּוִד תִּמָּשֵׁל הָאָרֶץ מֵאֶרֶץ הַהַבְטָחָה, שֶׁהֻבְטְחָה מִכְּבָר עַל יְדֵי יְהוָה, וּמַה שֶּׁיְהוָה הִבְטִיחַ יִתְקַיֵּם עַד תֹּם! וַאֲנִי הוּא הַנִּתָּן לְהַגְשָׁמַת כָּל הַדְּבָרִים.

הִנֵּה! הַזְּמַן הוּא עַתָּה, הַזְּמַן הוּא עַתָּה, כַּאֲשֶׁר הַלֵּדָה הַחֲדָשָׁה תַּגִּיעַ לִשְׁלֵמוּת וּלְהַגְשָׁמָה מְלֵאָה בִּי, הָאָדוֹן יֵשׁוּעַ הַמָּשִׁיחַ, וַאֲנִי אַגְשִׁים אֶת מַטְּרוֹתָיו הַנִּצְחִיּוֹת שֶׁל יְהוָה מִקֶּדֶם, אֲשֶׁר הֵכִין בְּעִתּוֹת עָבָר (אַף מֵעִתִּים כֹּה עַתִּיקוֹת שֶׁאֵינְכֶם יְכוֹלִים לְהָבִין בְּעֵת זוֹ) כִּי תָּמִיד הֱיִיתֶם חֲבוּיִים בְּלִבּוֹ שֶׁל יְהוָה. וְאַף עַל פִּי כֵן, מַטְּרוֹתָיו הַנִּצְחִיּוֹת וְהָעוֹלָמִיּוֹת שֶׁל יְהוָה עַתָּה מוּכָנוֹת לְהִתְגַּשֵּׁם, וּקְצִיר הַבִּכּוּרִים בַּפֶּתַח, וְהַשָּׁעָה הִגִּיעָה. כִּי הָאָרֶץ הָפְכָה עַתָּה כְּאִשָּׁה יוֹלֶדֶת, הַמּוּכָנָה לָלֶדֶת, וְצִירֶיהָ וְכַאֲבֵי לֵדָתָהּ אֵינָם פוסקים עתה, וּגְאֻלָּתָהּ קְרֵבָה, וְשַׁעַת הַתְּחִיָּה וְהַהִתְכַּנְּסוּת כָּאן.

כִּי פִּתְאֹם יִקְרֶה הַדָּבָר כַּאֲשֶׁר אֲנִי, הַמָּשִׁיחַ יֵשׁוּעַ, אֵרֵד מִן הַשָּׁמַיִם בִּתְרוּעָה, בְּקוֹל רַב-מַלְאָךְ, וּבְשׁוֹפַר אֱלֹהִים, וְאוֹתוֹ שׁוֹפָר אַחֲרוֹן יִתָּקַע, וּבְאוֹתוֹ קוֹל תִּפְרֹץ הוֹפָעָתִי בַּשְׁחָקִים בְּזֹהַר רַב (כְּבוֹדִי הוּא בְּהִירוּת נֶצַח), וְהָעוֹלָם יֵדַע כִּי יֵשׁ אֱלֹהִים בַּשָּׁמַיִם וְהָעוֹלָם יָבִין כִּי יוֹם יהוה בָּא עֲלֵיהֶם. וְכָךְ תּוֹלִיד הָאָרֶץ אֶת כָּל אֵלֶּה אֲשֶׁר נִרְדְּמוּ בַּמָּשִׁיחַ וְשָׁבוּ לַעֲפַר הָאָרֶץ, וְרוּחָם שָׁבָה לֵאלֹהִים, וְנַפְשָׁם הָיְתָה בִּמְנוּחָה מְתוּקָה מֵהַיּוֹם שֶׁבּוֹ עָצְמוּ עֵינֵיהֶם לִישׁוֹן בָּאֲדָמָה הָאֲדֻמָּה. כִּי הָאָדָם הַיָּשָׁן מֵאֲדָמָה אֲדֻמָּה יִפְרַץ בְּחַיִּים חֲדָשִׁים, וְהָאָרֶץ תַּסְגִּיר עַל פִּי פְּקֻדָּתִי אֶת הַנִּרְדָּמִים בַּמָּשִׁיחַ, וְאַתֶּם הַחַיִּים וְהַנּוֹתָרִים תֵּחָטְפוּ עִמָּם בֶּעָנָנִים לִפְגֹּשׁ אוֹתִי בָּאֲוִיר. וְהַדָּבָר יִקְרֶה פִּתְאוֹם, וְכֻלְּכֶם תִּוָּלְדוּ לְתוֹךְ פְּאֵר נִצְחִי וְתִבָּרְאוּ מֵחָדָשׁ בִּמְלוֹאֲכֶם כִּבְרִיאָה חֲדָשָׁה אֲשֶׁר כֹּחַ וְהָדָר נִצְחִי שְׁזוּרִים בְּמַלְבּוּשֵׁיכֶם הַצְּחוֹרִים אֲשֶׁר בָּהֶם תִּמְצְאוּ עַצְמְכֶם לְבוּשִׁים פִּתְאוֹם, וְהַזְּמַן הוּא עַתָּה.

אוֹי לְיוֹשְׁבֵי תֵבֵל. אוֹי, אוֹי, אוֹי רַב לְיוֹשְׁבֵי תֵבֵל. אוֹי לְךָ, הָאָדָם הַטִּבְעִי, כִּי סֵרַבְתָּ לִקְרִיאָתִי. הִזְהַרְתִּי וְקָרָאתִי בְּאַזְעֲקַת חֵרוּם 911, וְעַתָּה בָּא עָלֶיךָ הַיּוֹם, וְאַתָּה עוֹדְךָ בְּבַעֲרוּת לֹא מוּדָע לְחֻרְבָּנְךָ הַמִּתְקָרֵב. אַתָּה, הָאָדָם הַטִּבְעִי, הִתְעַלַּמְתָּ מִקּוֹלוֹתֵיהֶם שֶׁל אֵלֶּה שֶׁשָּׁלַחְתִּי, וְקוֹלוֹת רַבִּים שָׁלַחְתִּי אֵלֶיךָ, וְדָחִיתָ אֶת שְׁלִיחַי וּנְבִיאַי וְשַׁגְרִירַי; וְאַף אֶת הַיְלָדִים דָּחִיתָ, כִּי מִפִּי עוֹלָלִים יִסַּדְתִּי עֹז, וְהַקְּטַנִּים חָלְמוּ חֲלוֹמוֹת וְנִתְּנוּ לָהֶם חֶזְיוֹנוֹת, וְהֵם מַכְרִיזִים בֶּאֱמוּנָה עַל הוֹפָעָתִי! וְכֵן יִהְיֶה שֶׁחֻרְבַּן פִּתְאוֹמִי יָבוֹא וְהָאַזְהָרוֹת תִּפָּסַקְנָה. כִּי זֹאת אֵינָהּ אַזְהָרָה אֶלָּא הַצְהָרָה, כִּי עִתּוֹת הָאַזְהָרָה עֲתִידוֹת לִפְסֹק פִּתְאוֹם, וּבִמְקוֹם אַזְהָרָה תַּעֲמֹד הַגְשָׁמָה, וְחֻרְבַּן פִּתְאוֹמִי יָבוֹא פִּתְאֹם, וְלֹא תִּמָּלְטוּ, הוֹי יוֹשְׁבֵי תֵבֵל הַדּוֹחִים אֶת בּוֹרַאֲכֶם.

הָאָדָם הַטִּבְעִי, אַתָּה הוֹלֵךְ מֵחַטָּאת לְחַטָּאת בִּדְרָכֶיךָ הַטְּמֵאוֹת וְהַחוֹטְאוֹת, וְאַף כִּי הִמְטַרְתִּי עַל הָאָרֶץ מֵחַסְדִּי וּמֵרַחֲמַי לְמַעַן תִּמָּלֵט מִמִּשְׁפָּטְךָ הַמִּתְקָרֵב, אַתָּה בָּחַרְתָּ תַּחַת זֹאת לָשֵׂאת אֶת מִשְׁפָּטְךָ וְאֶת אֲסוֹנְךָ, וְכֵן יִהְיֶה. וְכָעֵת מַתְחִיל הַמִּשְׁפָּט בְּבֵית הָאֱלֹהִים, וְכֵן יִהְיֶה. בָּבֶל הַסּוֹדִית, אַף אוֹתָהּ אֻמָּה גְדוֹלָה שֶׁל אַרְצוֹת הַבְּרִית, עוֹמֶדֶת עַתָּה כְּבוּלָה, עֲרֻמָּה וַחֲסַרַת כֹּל וּמְחַכָּה לְחֻרְבָּנָהּ הַסּוֹפִי.

בִּגְלַל חֲטָאֶיהָ, זְנוּנֶיהָ, נִאוּפֶיהָ, תּוֹעֲבוֹתֶיהָ, וְטֻמְאָתָהּ וְשַׁחֲתוּתָהּ הַגְּמוּרוֹת שֶׁבָּהֶן הִשְׁחִיתָה אֶת הָאָרֶץ, הִיא תַּעֲמֹד עַתָּה בִּפְנֵי חֻרְבָּנָהּ. וְהַזּוֹנָה הַגְּדוֹלָה נִתְּנָה עַתָּה בִּמְלוֹאָהּ בִּידֵי אוֹיְבֶיהָ, אַךְ הַמְּעַכֵּב מְעַכֵּב עַד שֶׁלֹּא יִהְיֶה עוֹד מְעַכֵּב לְעַכֵּב, וּפִתְאֹם יְסֻלַּק הַמְּעַכֵּב בַּהִתְאַסְּפוּת הַגְּדוֹלָה יַחַד שֶׁל קְדוֹשַׁי! וּבְעֵת שֶׁיְּסֻלַּק הַמְּעַכֵּב, הֵם יְבַצְּעוּ אֶת מִתְקֶפֶת הַסֵּתֶר שֶׁתִּכְנְנוּ וְתִאֲמוּ שָׁנִים רַבּוֹת עַתָּה, כִּי כֵן נִתַּן בִּידֵיהֶם לְבַצֵּעַ, וּבָבֶל הַסּוֹדִית תָּחֳרַב כָּלִיל בְּשָׁעָה אַחַת.

וַעֲשֶׂרֶת הַקְּרָנַיִם תַּשְׁחֵתְנָה אֶת הַזּוֹנָה הַגְּדוֹלָה כִּי שׂוֹנְאוֹת הֵן אֶת הַזּוֹנָה, וְתַעֲשֶׂינָה אוֹתָהּ שׁוֹמֵמָה וַעֲרֻמָּה וְתֹאכַלְנָה אֶת בְּשָׂרָהּ וְתִשְׂרֹפְנָה אוֹתָהּ בָּאֵשׁ. וּרְעָדָה גְדוֹלָה תָּבוֹא עַל הָאָרֶץ, וְהִיא תִּבָּקַע, וְהַמַּיִם יְכַסּוּ אֶת חוֹפֶיהָ, וְהַמַּיִם יְכַסּוּ אֶת לִבָּהּ, וְהִיא תִּתְחַלֵּק לִשְׁנַיִם. כִּי כֵן כָּתוּב: חֲטָאֶיהָ הִגִּיעוּ עַד הַשָּׁמַיִם, וֵאלֹהִים זָכַר אֶת עֲווֹנוֹתֶיהָ. גִּמְלוּ לָהּ כְּגְמוּלָהּ לָכֶם, וְתֵנוּ לָהּ כֶּפֶל כְּמַעֲשֶׂיהָ: וּבַכּוֹס שֶׁמִּלְאָה לָכֶם, מַלְּאוּ לָהּ כִּפְלַיִם. כְּכָל שֶׁהִלְּלָה עַצְמָהּ וְחָיְתָה בְּתַעֲנוּגוֹת, כָּךְ תְּנוּ לָהּ עִנּוּיִים וְיָגוֹן: כִּי הִיא אוֹמֶרֶת בְּלִבָּהּ, יוֹשֶׁבֶת אֲנִי כְּמַלְכָּה, וְאֵינֶנִּי אַלְמָנָה, וְלֹא אֶרְאֶה אֵבֶל. לָכֵן, יָבוֹאוּ מַכּוֹתֶיהָ בְּיוֹם אֶחָד, מָוֶת, וְיָגוֹן, וְרָעָב, וְהִיא תִּשָּׂרֵף בָּאֵשׁ: כִּי הָאֵל הַדָּן אוֹתָהּ הוּא אָדוֹן חָזָק. אָז נָשָׂא מַלְאָךְ אַדִּיר אֶבֶן כְּאֶבֶן רֵחַיִם גְּדוֹלָה, וְהִשְׁלִיכָהּ לַיָּם, וְאָמַר, בְּכָזֹאת אֲלִימוּת תֻּשְׁלַךְ הָעִיר הַגְּדוֹלָה בָּבֶל, וְלֹא תִּמָּצֵא עוֹד.

וְכֵן יִהְיֶה שֶׁהָאֻמָּה הַגְּדוֹלָה הַזֹּאת שֶׁפָּעַם הָיְתָה, אֲשֶׁר מִמֶּנָּה גָּאַלְתִּי רַבִּים, תִּלָּקַח עַתָּה בְּמִשְׁפָּט, בְּעוֹד שֶׁקְּהִלָּתִי וְכַלָּתִי תִּלָּקַחְנָה לַשָּׁמַיִם, אֶל מָקוֹם שֶׁהֵכַנְתִּי לָהֶן, מְקוֹם בִּטָּחוֹן וּמִקְלָט שָׁם תְּקַבֵּלְנָה אֶת נַחֲלָתָן. וְלֹא תִּהְיֶה עוֹד בָּבֶל הַסּוֹדִית לְעוֹלָם, וְעִירָהּ הַגְּדוֹלָה בְּיוֹתֵר הַיּוֹשֶׁבֶת בְּיַהֲרוּת כְּאִלּוּ דָּבָר לֹא יִשְׁתַּנֶּה לְעוֹלָם, תִּפֹּל כָּלִיל עִמָּהּ, וְלֹא תָּקוּם עוֹד.

אָמְנָם וְאָמְנָם, כָּל עָרֶיהָ הַחוֹטְאוֹת וְהַמַּבְחִילוֹת תִּפֹּלְנָה, וּמִגְדָּלֶיהָ יִתְמוֹטְטוּ, וּשְׂרֵפוֹת גְּדוֹלוֹת וְחֻרְבָּן יַהַפְכוּ אֶת דְּרָכֶיהָ לְבִלְתִּי עֲבִירוֹת. אָז יְשֻׁלְּחוּ הַצְּבָאוֹת הָאֲדֻמִּים, אַף צִבְאוֹת הַחַיָּה שֶׁיִּבֵּא לְחֻרְבָּנָהּ, עַל מַה שֶּׁנּוֹתַר, וְהֵם יֹאכְלוּ אֶת בְּשָׂרָהּ. וְכֵן יִהְיֶה שֶׁכָּל הַקּוֹרֵא בְּשֵׁם יהוה יִנָּצֵל. וְאַף עַל פִּי כֵן, הַנּוֹתָרִים נִמְכָּרִים לְעַבְדוּת בְּנֶשֶׁךְ, כִּי מַנְהִיגֵי בָּבֶל הַסּוֹדִית בָּגְדוּ בְּאֶזְרָחֶיהָ וְנָתְנוּ אוֹתָם כְּעֵרָבוֹן, וְהַתַּשְׁלוּם נִדְרָשׁ עַתָּה.

הָבִינוּ מָה קָרָה, עַמִּי, וְזִכְרוּ אֶת דְּבָרַי. הָרוֹעִים הַמְּצֹרָעִים עָבְדוּ יַחַד לְחֻרְבָּנָהּ; הָרוֹעֶה הַמְּצֹרָע הָאֶחָד הַשּׁוֹלֵט בַּסֵּתֶר מוֹשֵׁךְ בְּחוּטֵי הַשּׁוֹטֶה הַזָּקֵן, בְּעוֹד שֶׁהָרוֹעֶה הַמְּצֹרָע הָאַחֵר, הַנִּרְאֶה נִרְדָּף עַל יְדֵי הַשּׁוֹטֶה הַזָּקֵן, פּוֹעֵל יָד בְּיָד עִם הָרוֹעֶה הַמְּצֹרָע הַשּׁוֹלֵט בַּסֵּתֶר. כִּי הֵם מַעֲמִידִים פָּנִים שֶׁהֵם אוֹיְבִים בָּעֵת הַזֹּאת, אַךְ כֻּלָּם מֵאוֹתוֹ עֵדֶר מְצֹרָע, מַגְשִׁימִים אֶת מַטְּרוֹתָיו שֶׁל הַתַּנִּין, וְכֵן אָמַרְתִּי לִקְהִלָּתִי, גּוּפִי, וְאַף לְכַלָּתִי! וְכָךְ הִסְגִּירוּ הָרוֹעִים הַמְּצֹרָעִים אֶת בָּבֶל הַסּוֹדִית לִידֵי אוֹיְבֶיהָ, כִּי לְכָךְ נִשְׁלְחוּ.

הֵם הִפְשִׁיטוּ אֶת הַזּוֹנָה, כִּי כֵן נִתַּן לָהֶם לְבַצֵּעַ, כִּי אַף אוֹיְבֵי יהוה מְצַיְּתִים לְקוֹלוֹ. הָבִינוּ שֶׁהֵם הִשְׁאִירוּ בְּכַוָּנָה אֶת כֹּחָהּ הַצְּבָאִי בְּאֲרָצוֹת זָרוֹת וְנָתְנוּ אֶת סוֹדוֹתֶיהָ לַעֲשֶׂרֶת הַקְּרָנַיִם כְּדֵי שֶׁיִּשְׁלְטוּ בְּעָצְמָה. עֲשֶׂרֶת הַקְּרָנַיִם הֵן אֵלֶּה שֶׁיִּתְּנוּ אֶת כֹּחָן לַחַיָּה, כִּי כֵן נִתַּן לָהֶן לְבַצֵּעַ.

הַזּוֹנָה הַגְּדוֹלָה נִלְחֲמָה מִלְחָמוֹת אֵין-סוֹפִיּוֹת וַעֲקָרוֹת שֶׁנּוֹעֲדוּ לְשַׁחֵק אֶת כֹּחָהּ. בָּבֶל הַסּוֹדִית, בְּכַוָּנָה, הִסִּיתָה לְמִלְחָמָה עִם הַדֹּב כְּדֵי לִשְׁחֹק אֶת כֹּחָהּ וְעָצְמָתָהּ כָּךְ שֶׁעַתָּה הִיא נוֹתְרָה עֲרֻמָּה וַחֲסַרַת הֲגַנָּה, וּלְיָדָיו שֶׁל הַתַּנִּין הָאָדֹם הִיא נִתְּנָה עַתָּה כְּשָׁלָל. וְכָל פָּן בְּאִימְפֶּרְיָה שֶׁלָּהּ הַיּוֹשֶׁבֶת עַל מַיִם רַבִּים יִנָּתֵן כְּמוֹ כֵן כְּשָׁלָל לִידֵיהֶם. הָבִינוּ שֶׁהֵם מוּכָנִים וְיִפְעֲלוּ פִּתְאוֹם לְבַצֵּעַ פְּעֻלָּה מְשֻׁתֶּפֶת שֶׁל מִתְקָפָה עוֹלָמִית, וּבְכָל הַחֲזִיתוֹת יִתְקְפוּ בְּהֶרֶס מָהִיר, וְנִצָּחוֹן קָצָר מְאוֹד יִהְיֶה לָהֶם לִפְנֵי שֶׁיִּפּוֹל מִשְׁפָּט בְּזַעַם גָּדוֹל עַל הָאָרֶץ.

הָבִינוּ שֶׁהָאָרֶץ נִמְשֶׁלֶת עַתָּה עַל יְדֵי הַתַּנִּין, וְהוּא כָּבוּל עַתָּה לָאָרֶץ כְּשֶׁהוּא מְחַכֶּה לְמִשְׁפָּטוֹ, וְכָךְ תָּחֳרַב וְתֵעָנֵשׁ מַלְכוּתוֹ כָּלִיל (כִּי אֵינוֹ מֶלֶךְ), וְהַשָּׂטָן יִכָּבֵל בְּעֹנֶשׁ בְּשַׁרְשֶׁרֶת גְּדוֹלָה בַּתְּהוֹם לַזְּמַן שֶׁנִּקְבַּע, כִּי אֶלֶף שָׁנָה נִתְּנוּ לְיוֹם מְנוּחָה לַגְּאוּלִים, וְשַׁבָּתוֹת רַבּוֹת תִּהְיֶינָה לָנוּ. וְכָךְ נִמְתָּחוֹת זְרוֹעוֹתָיו וּמְכַסּוֹת אֶת מַלְכֵי הָאָרֶץ כְּמוֹ שֶׁתַּמְנוּן עוֹטֵף אֶת טַרְפּוֹ, וּבְאוֹתָהּ דֶּרֶךְ שֶׁעַכָּבִישׁ טוֹוֶה אֶת קוּרָיו; כָּךְ הוּא לוֹקֵחַ אֶת טַרְפּוֹ, וְכָל הַנִּלְקָחִים נִשְׁמָרִים לַאֲגַם הָאֵשׁ הַגָּדוֹל הַהוּא.

כִּי הָעַכָּבִישׁ יוֹשֵׁב בְּמַאֲרָבוֹ, וְהָעַכָּבִישׁ הִסְתַּתֵּר בְּתוֹךְ אֶרֶץ הַהַבְטָחָה. וְכָאֵלֶּה דַּרְכֵי הַשָּׂטָן, כִּי דַּרְכֵי הַשָּׂטָן דַּרְכֵי מוֹרֶךְ לֵב. הוּא זָחַל עַל גְּחוֹנוֹ וְנִסָּה לְהִסְתַּתֵּר בְּגַן יַהְוֶה, אַךְ דָּבָר אֵינוֹ נִסְתָּר מֵעֵינֵי יַהְוֶה, וְדָבָר אֵינוֹ נִסְתָּר מֵעֵינַי, כִּי אֲנִי יֵשׁוּעַ הַמָּשִׁיחַ, זֶרַע הָאִשָּׁה הַמֻּבְטָח. וַאֲנִי אָשׁוּף רֹאשׁ נָחָשׁ, וּפֶצַע עוֹלָמִים יִהְיֶה לוֹ, אֲשֶׁר מִמֶּנּוּ לֹא יִתְאוֹשֵׁשׁ לְעוֹלָם.

אַךְ כֵּן הוּא שֶׁזִּיזָנָיו נוֹשְׂאִים אֶת טִבְעוֹ הָרַע, וְלָכֵן טָוָה הָעַכָּבִישׁ אֶת הָעוֹלָם בְּקוּרָיו בְּרֹב רֶשַׁע, וְהָעַכָּבִישׁ יוֹשֵׁב כָּעֵת מוּכָן, כִּי בָּאָה שְׁעָתוֹ. וְעַתָּה מֵנִיעַ הָעַכָּבִישׁ בְּרַגְלָיו הָרַבּוֹת אֶת הַכֵּלִים עַל הַלּוּחַ, לְמַעַן יִהְיוּ מוּכָנִים. הוּא פִּתָּה אֶת הָעוֹלָם אֶל תּוֹךְ קוּרָיו, בְּהַבְטָחַת שָׁוְא לְיוֹם חָדָשׁ; וּתְחִיַּת תּוֹר הַזָּהָב שֶׁלָּהֶם הֵם מְבַקְשִׁים, אַךְ לְעוֹלָם לֹא יִזְכּוּ בְּתוֹר הַזָּהָב שֶׁלָּהֶם, כִּי אִם אֲגַם אֵשׁ גָּדוֹל הַבּוֹעֵר בְּלֶהָבוֹת כְּחֻלּוֹת וְחַמּוֹת בְּיוֹתֵר הוּא נַחֲלָתָם.

וְכָךְ מֵנִיעַ אוֹתָם הָעַכָּבִישׁ עַל פְּנֵי קוּרָיו וּמַצִּיבָם בְּמוּכָנוּת. הָעַכָּבִישׁ הוּא כְּנֶסֶת הַשָּׂטָן, אֲשֶׁר פָּלַשׁ לְאַרְצוֹת נַחֲלָתִי וְהַבְטָחָתִי, וְהֵם יֵעָקְרוּ וְיֻשְׁמְדוּ כָּלִיל, וְלֹא יִהְיֶה לָהֶם לֹא שֹׁרֶשׁ וְלֹא צֶאֱצָא. וְעִם זֹאת יְשָׁרְתוּ אֶת מַטְּרוֹתָיו הַנִּצְחִיּוֹת שֶׁל יַהְוֶה לְעוֹנָה קְצָרָה, כִּי רִבְבוֹת רְבָבוֹת יִוָּשְׁעוּ מֵחָרְבָּן בְּמַהֲלַךְ הַשָּׁבוּעַ הַשִּׁבְעִים לְדָנִיֵּאל. כָּל הַקּוֹרֵא בְּשֵׁם אֲדֹנָי יִמָּלֵט, אוֹמֵר אֲדֹנָי הַצַּדִּיק, הוּא יֵשׁוּעַ הַמָּשִׁיחַ, אֲשֶׁר הוּא ג'יזס כרייסט בַּלָּשׁוֹן הָאַנְגְלִית; וּבְכָל לָשׁוֹן אֲשֶׁר יִקְרְאוּ, יִשָּׁמְעוּ וְיִוָּשְׁעוּ וְיָקוּמוּ לִחְדוּת חַיִּים, וְאֶל מַלְכוּת צִדְקִי יָבֹאוּ בְּגָמוּל נֶצַח וּבְהָדָר נֶצַח.

וְכֵן הוּא, חָרְבָּנָהּ שֶׁל בָּבֶל הַסּוֹדִית קָרֵב וּבָא, וְנֶאֱמָנַי עֲתִידִים לְהִלָּקַח. וַאֲנִי, הַמָּשִׁיחַ יֵשׁוּעַ, שׁוּב אֵינֶנִּי בָּא אֶל יוֹשְׁבֵי תֵבֵל בְּאַזְהָרָה, כִּי הִזְהַרְתִּים הֵיטֵב וּבִדְרָכִים רַבּוֹת, רַק כְּדֵי שֶׁיִּדְחוּ אוֹתִי בְּכָל הַנְּקֻדּוֹת; וְכֵן הוּא, כִּי כֵן כָּתוּב. אַף עַל פִּי כֵן, חֵרֶף יְדִיעָתוֹ הַמֻּקְדֶּמֶת שֶׁל יַהְוֶה, יַהְוֶה מַזְהִיר אֶת הָעֲתִידִים לְהִלָּקַח בְּמִשְׁפָּט מִפְּנֵי גּוֹרָלָם הַמִּתְקָרֵב, וּבְאַהֲבָתוֹ שְׁלָחַנִי לִישׁוּעָתָם, וְרַבִּים נוֹשְׁעוּ.

אַךְ כָּעֵת דָּחוּ אוֹתִי לְטוֹבַת מוֹשִׁיעָם הַשִּׁקְרִי, כִּי עַתָּה הוֹלְכִים הֵם אַחַר בַּר-אַבָּא, וְעַתָּה בָּחֲרוּ בַּתַּנִּין לֵאלֹהֵיהֶם. וְעַל כֵּן שׁוּב אֵינֶנִּי בָּא לְהַזְהִיר אֶת הָאָדָם הַטִּבְעִי אֶלָּא לְהוֹרוֹת וּלְהָכִין אֶת עַמִּי לְמַעַן יִהְיוּ מוּכָנִים לַעֲזִיבָתָם, כִּי לֹא נוֹתַר עוֹד זְמַן. יַהְוֶה, חֵרֶף יְדִיעָתוֹ אֶת כָּל אֵלֶּה בְּנִצְחֵי עַד, וּבְצִדְקוֹ וּבְאַהֲבָתוֹ, הִזְהִיר בְּכָל זֹאת אֶת אֵלֶּה שֶׁאֵינָם חוֹזְרִים בִּתְשׁוּבָה, אַף שֶׁיָּדַע כִּי רַבִּים לֹא יַחְזְרוּ בִּתְשׁוּבָה בָּעֵת הַזֹּאת, וְעִם זֹאת רִבְבוֹת רְבָבוֹת יִוָּשְׁעוּ דֶּרֶךְ כּוּר הַמְּצָרֵף שֶׁל צָרַת יַעֲקֹב. וְכָל הַקּוֹרֵא בְּשֵׁם אֲדֹנָי עַד קֵץ יִוָּשַׁע!

אַךְ רַבִּים פָּנוּ אֵלַי, וַאֲלֵיכֶם, כַּלָּתִי וּקְהִלָּתִי וְגוּפִי, אֲנִי בָּא כָּעֵת לְקַבֵּץ וּלְהָכִין אֶתְכֶם לַעֲזִיבַתְכֶם אֶל הַשָּׁמַיִם, כִּי עַתָּה הָעִתִּים וְהָעִתִּים הֵן עַתָּה. הָבִינוּ, יְלָדַי, שֶׁאֵין זֶה עוֹד הָעוֹלָם שֶׁאַתֶּם רְגִילִים לִחְיוֹת בּוֹ, וִימֵי קֶדֶם חָלְפוּ עַתָּה, וְיוֹם אֲדֹנָי הִגִּיעַ. לָכֵן אֲנִי יוֹעֵץ וּמוֹרֶה לְכָל אֲשֶׁר בִּקְהִלָּתִי, לְכָל אֲשֶׁר יִשְׁמַע אֶת קוֹלִי, לְמַלֵּא מְנוֹרוֹתֵיהֶם בַּשֶּׁמֶן אֲשֶׁר סִפַּקְתִּי וּלְהֵיטִיב פְּתִילוֹתֵיכֶם כְּפִי שֶׁהוֹרֵיתִי, כִּי בָּא יוֹם גְּאֻלַּתְכֶם. אָמֵן, אָמֵן, אֵינֶנִּי מַזְהִיר עוֹד אֶלָּא הִנֵּנִי כָּאן לְהַכְרִיז בְּשֵׁם יַהְוֶה, הָאֵל הַנִּצְחִי הַגָּדוֹל, כִּי בָּאוּ הָעִתִּים; וְאֵת אֲשֶׁר יַכְרִיז יַהְוֶה אֵין אִישׁ יָכוֹל לְשַׁנּוֹת, וְאַף לֹא לְשַׁנּוֹת בְּמַשֶּׁהוּ קַל, וְהַכֹּל יִתְגַּשֵּׁם כַּכָּתוּב.

וְעַל כֵּן, יְלָדַי, הָבִינוּ אֶת הָעִתִּים בָּהֶן אַתֶּם חַיִּים, כִּי שׁוּב אֵינְכֶם יְכוֹלִים לִחְיוֹת חַיִּים פְּרוּצִים. הָאֵרוּעִים כְּבָר מִתְרַחֲשִׁים וְהַזְּמַנִּים מְסֻכָּנִים. הִסָּתְרוּ, קְטַנַּי, מִפְּנֵי הַמִּשְׁפָּטִים שֶׁאֵינָם מְיֹעָדִים לָכֶם, וְהָבִינוּ כִּי אַף הַמַּיִם מַתְחִילִים לְהִהָפֵךְ מָרִים. אֲנִי מְבַקֵּשׁ מִכֶּם, יְלָדַי, לְהִשָּׁאֵר בַּמִּקְלָט שֶׁסִּפַּקְתִּי לָכֶם וּלְצֵאת מִתּוֹכָם עַתָּה וּלְהִבָּדֵל כָּלִיל, פֶּן תִּשְׁתַּתְּפוּ בְּמַגֵּפוֹתֵיהֶם. אֵרוּעִים אֵלֶּה שֶׁאֲנִי מְדַבֵּר עֲלֵיהֶם אֵינָם מְיֹעָדִים לָכֶם, יְלָדַי, אֶלָּא נִתָּנִים כְּדֵי לְהָעִיר אֶת אַחֲרוֹנֵי הַבָּאִים לִישׁוּעָה לִפְנֵי שֶׁיַּהְוֶה יִסְגֹּר אֶת דֶּלֶת הַתֵּבָה בְּעֵת אִסּוּף קְהִלָּתִי בַּשּׁוֹפָר הָאַחֲרוֹן, וּמְעַטִּים מְאֹד הֵם אֵלֶּה שֶׁנּוֹתְרוּ.

לָכֵן הַאֲזִינוּ לְקוֹלִי, יְלָדַי, וּצְאוּ, צְאוּ, צְאוּ מִתּוֹכָהּ. הַתְרָעַת חֵרוּם, כִּי קָרַב הַיּוֹם. אֲסוֹנוֹת כְּבָר שֻׁלְּחוּ עַל פְּנֵי הָאָרֶץ, וְעִם זֹאת סִפַּקְתִּי מִקְלָט וְשָׁלוֹם לְאֵלֶּה הַמְּחַכִּים לִקְרִיאָתִי, וְהַשּׁוֹפָר עוֹמֵד לְהִתָּקַע. לָכֵן, הִשָּׁאֲרוּ בְּבִטְחַת מְקוֹמוֹת הַמִּקְלָט שֶׁהֲכִינוֹתִי לָכֶם, וְהָאָסוֹן שֶׁאֵינוֹ מְיֹעָד לָכֶם יִפְסַח עֲלֵיכֶם כְּדֶרֶךְ שֶׁפָּסַח מַלְאַךְ הַמָּוֶת עַל בְּנֵי יִשְׂרָאֵל בְּמִצְרַיִם. אֶלָּא שֶׁפִּתְאוֹם שׁוּב לֹא תִּהְיוּ עַל הָאָרֶץ, וּבְמַצַּב נָפוּל שׁוּב לֹא תִּהְיוּ לְעוֹלָם, כִּי גְּאֻלַּתְכֶם אֵינָהּ עוֹד מִתְקָרֶבֶת, אֶלָּא נִמְצֵאת עֲלֵיכֶם בִּמְלוֹאָהּ. לָכֵן, הֱיוּ מוּכָנִים.

אֲנִי מְדַבֵּר אֶל הַנֶּאֱמָנִים וּמְבַקֵּשׁ אֶת לִבְּכֶם בְּצִיּוּת מָלֵא לִדְבָרַי.

שׁוּב אֵינֶנִּי נוֹדָע לְפִי הַבָּשָׂר.

אֲנִי אָדָם הָאַחֲרוֹן.

אֲנִי רוּחַ מְחַיָּה.

אֲנִי יֵשׁוּעַ הַמָּשִׁיחַ.

אֲנִי אֲדוֹן הָאֲדוֹנִים וּמֶלֶךְ הַמְּלָכִים.

אֲנִי כֹּהַנְכֶם הַגָּדוֹל לְעוֹלָם וָעֶד עַל-דִּבְרָתִי מַלְכִּי-צֶדֶק.

אֲנִי הוּא אֲשֶׁר שָׁלַח אֶהְיֶה אֲשֶׁר אֶהְיֶה לְהַגְשָׁמַת כָּל הַדְּבָרִים.

שָׁלַחְתִּי דְּבָרִים אֵלֶּה עַל יְדֵי עַבְדִּי, וְעַל יְדֵי עַבְדִּי נִשְׁלְחוּ דְּבָרִים אֵלֶּה.

אֲנִי יוֹעֵץ לְכָל הַשּׁוֹמְעִים דְּבָרִים אֵלֶּה לְהַקְשִׁיב בִּמְלוֹא הַקֶּשֶׁב לְהוֹרָאוֹתַי וּלְהָבִין כִּי עַתָּה הַזְּמַן וּפִתְאֹם תָּבוֹא הַהַגְשָׁמָה בִּמְלוֹאָהּ. אֲנִי דּוֹרֵשׁ אֶת לִבָּם הַשָּׁלֵם שֶׁל נֶאֱמָנַי, לָכֵן הַמְשִׁיכוּ לְהִטָּהֵר בְּדָמִי הַמְּכַפֵּר וְהִתְוַדּוּ עַל חֲטָאֵיכֶם לְפָנַי כְּדֵי שֶׁאוּכַל לְטַהֵר אֶתְכֶם בִּמְלוֹאֲכֶם בְּדָמִי הַמְּכַפֵּר. אַל תִּתְּנוּ לְחוּשֵׁיכֶם לְהַטְעוֹת אֶתְכֶם, כִּי בָּבֶל הַסּוֹדִית עִוֶּרֶת וְחֵרֶשֶׁת וּבוּרָה מִבְּחִירָה בְּנוֹגֵעַ לְמִשְׁפָּטָהּ הַמִּתְקָרֵב, הַמִּתְרַחֵשׁ כְּבָר וְאֵינוֹ נִתָּן לַעֲצִירָה. לָכֵן, אַל תִּשְׁתַּתְּפוּ בְּעִוְרוֹנָהּ פֶּן תִּסְבְּלוּ מִמַּגֵּפוֹתֶיהָ, וְהִבָּדְלוּ מִתּוֹכָם לְמַעַן תִּהְיוּ מוּכָנִים לְגַמְרֵי לְהוֹפָעַת אֲדוֹנְכֶם הַנִּצְחִי. הִנֵּה, כַּלָּתִי! הֶחָתָן מִתְקָרֵב, לָכֵן שִׁמְרוּ מְנוֹרוֹתֵיכֶן מְלֵאוֹת, פְּתִילוֹתֵיכֶן מֵיטָבוֹת וְדוֹלְקוֹת, בְּצִפִּיָּה עֲרֻבָּה לַאֲדוֹנְכֶן הַמִּתְקָרֵב, כִּי עַתָּה עִתּוֹת הָאִסּוּף, וְעִתּוֹת הָאִסּוּף הֵן עַתָּה.
