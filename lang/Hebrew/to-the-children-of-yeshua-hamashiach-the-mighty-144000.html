בְּעֵת כִּנּוּס הַקְּדוֹשִׁים, יֵעָקֵר הַמְעַכֵּב, וְהַמִּשְׁפָּט יָחוּל בִּמְלוֹאוֹ עַל בֵּית הָאֱלֹהִים. וְאֵלֶּה אֲשֶׁר הִרְשׁוּ לִי לְזַקְּקָם בְּאֵשׁ הַמְּצָרֵף יַעַמְדוּ, כִּי טֹהֲרוּ בְּדַם הַשֶּׂה וְיַעַמְדוּ בַּצְּדָקָה אֲשֶׁר הֶעֱנַקְתִּי לָהֶם, כִּי אֲנִי הָאָדוֹן הַצַּדִּיק אֲשֶׁר הוּא רוּחַ מְחַיָּה.

וְכָךְ, תִּתְרַחֵשׁ הַצָּרָה הַגְּדוֹלָה בְּיוֹתֵר, צָרָה אֲשֶׁר כְּמוֹתָהּ לֹא נִרְאֲתָה מֵעוֹלָם עֲלֵי אֲדָמוֹת, וּבֶאֱמֶת, גַּם לֹא תֵּרָאֶה עוֹד. הִנֵּה! אֲנִי, יֵשׁוּעַ הַמָּשִׁיחַ, לֹא אַפְקִיר אֶת עַמִּי לַהֶרֶס פִּתְאוֹמִי, וְעוֹד אוֹשִׁיעַ אֶת יִשְׂרָאֵל, אֶת הַשְּׁאֵרִית הָאַדִּירָה אֲשֶׁר תְּקַבֵּל אוֹתִי כַּמָּשִׁיחַ הָאֲמִתִּי, כִּי לְכָךְ בְּרָאַנִי יהוה. וְכֵן יָבוֹאוּ ה-144,000 שֶׁלִּי לְעֶזְרַת אֵלֶּה אֲשֶׁר בִּלְעֲדֵיהֶם הָיוּ אֲבוּדִים וְקוֹרְאִים בִּשְׁמִי. אֶנְחֶה אוֹתָם צַעַד אַחַר צַעַד, וְה-144,000 שֶׁלִּי יַעֲשׂוּ לְמַעֲנִי מַעֲשִׂים אַדִּירִים רַבִּים.

כִּי אֵלֶּה הֵם הַנִּלְקָחִים בְּכִנּוּס הַקְּדוֹשִׁים אֵלַי, הַמָּשִׁיחַ יֵשׁוּעַ, בָּאֲוִיר, לְקוֹל הַשּׁוֹפָר הָאַחֲרוֹן, וְהַשּׁוֹפָר עוֹמֵד לְהִתָּקַע. אֵלֶּה הֵם נֶאֱמָנַי, אֵלֶּה אֲשֶׁר לֹא נִמְצְאָה בָּהֶם מִרְמָה, כִּי טִהֲרוּ עַצְמָם כִּבְתוּלָה צְנוּעָה בְּדַם הַשֶּׂה, אֲשֶׁר הִקְרַבְתִּי לְעוֹלָם וָעֶד כְּכֹהֲנָם הַגָּדוֹל, כִּי אָנֹכִי הוּא הַכֹּהֵן הַגָּדוֹל לְעוֹלָם עַל דִּבְרָתִי מַלְכִּי-צֶדֶק. אֵלֶּה הֵם כַּלָּתִי, וְהֵם יֵלְכוּ אַחֲרַי לְכָל אֲשֶׁר אֵלֵךְ. הִנֵּה! אֲנִי אֶעֱשֶׂה אוֹתָם עַצְמָתִיִּים, וּדְבַר מָה שֶׁיַּשְׁוֶה לַיְּכֹלֶת אֲשֶׁר אַלְבִּישֵׁם לֹא נִרְאָה מֵעוֹלָם עֲלֵי אֲדָמוֹת. הֵם יִהְיוּ לְבוּשִׁים פִּשְׁתָּן צַח, לָבָן וְנָקִי מִכָּל כֶּתֶם, וִיכֹלְתָם תְּגַלֵּם אֶת הֶשֵּׂגֵיהֶם, וִיכֹלֶת גְּדוֹלָה תִּהְיֶה בְּיָדָם; כִּי נֶאֱמָנִים הֵם, קְרוּאִים וּבְחִירִים, עוֹד מִלִּפְנֵי הִוָּסֵד תֵּבֵל, עַל יְדֵי יהוה לְמַטְּרוֹת אֵלֶּה.

ה-144,000 שֶׁלִּי יֵרְדוּ עַל הַר צִיּוֹן, וַאֲנִי אוֹבִילֵם. הֵם יִקְחוּ אֶת אֵלֶּה אֲשֶׁר אוֹבִיל אוֹתָם אֲלֵיהֶם אֶל הַמִּדְבָּר, לְהִסָּתֵר מִן הָאֲפֵלָה הַמֻּחְלֶטֶת וּמֵרְדִיפוֹת מַמְלֶכֶת הָאוֹיְבִים. הֵם יוּבְלוּ לְמִבְטָחִים מִשֶּׁבַע הַמַּכּוֹת, כִּי בְּמִשְׁפָּטִים אֵלֶּה נוֹעֲדָה חֲמַת זַעַם אֱלֹהִים לְהִשָּׁפֵךְ עַל כֵּס הַחַיָּה וּלְעַנּוֹת אֶת מַמְלֶכֶת הַדְּרָקוֹן, כִּי גָּדוֹל עָנְשָׁם. וְה-144,000 יוֹבִילוּ אוֹתָם לְמִבְטָחִים; הֵם יְרַפְּאוּ אוֹתָם מִפִּצְעֵיהֶם; הֵם יְסַעֲדוּ אוֹתָם וְיַעֲנִיקוּ לָהֶם תִּקְוָה גְּדוֹלָה לִקְרֹא בְּשֵׁם הָאָדוֹן, עַד הַסּוֹף אִם יִהְיֶה צֹרֶךְ. כִּי לוּלֵא קִצַּרְתִּי אֶת הַיָּמִים, לֹא הָיָה נוֹתָר כָּל בָּשָׂר. וְכָךְ יַשְׁלִימוּ אֶת אֲשֶׁר אוֹבִיל אוֹתָם בּוֹ.

עַתָּה אֲדַבֵּר עַל הֶעָתִיד לָבוֹא בְּמַהֲלַךְ שְׁבוּעוֹ הַשִּׁבְעִים שֶׁל דָּנִיֵּאל, וְאֶעֱשֶׂה אֶת נְתִיבוֹתֵיכֶם לִסְלוּלוֹת, וְאֶפְרֹשׂ אֶת חֶסֶד יְשׁוּעָתִי בִּידֵי ה-144,000 שֶׁלִּי. כִּי הֵם אָכֵן בָּאִים לְהוֹשִׁיעַ אֶת עַמִּי מֵהֶרֶס, וְהֵם בָּאִים לְהַסְתִּיר אֶת עַמִּי מִן הַזַּעַם, זַעַם הָאֱלֹהִים הַמְּיֹעָד לַנְּפוּלִים וּלְאֵלֶּה הַדְּבֵקִים בָּהֶם וּבְמַמְלַכְתָּם לְעוֹלָם עַל יְדֵי קַבָּלַת הָאוֹת שֶׁלָּהֶם, הִיא נְשִׁיכַת הַנָּחָשׁ הַחַיָּתִית אֲשֶׁר אֵין לָהּ מַרְפֵּא.

וּבְכָל זֹאת, אֶחְטֹף אֶת כַּלָּתִי אֶל הַשָּׁמַיִם, וְאֶחְתֹּם אֶת ה-144,000 שֶׁלִּי מִן הַשְּׁבָטִים הַבָּאִים, שְׁנֵים-עָשָׂר אֶלֶף מִכָּל אֶחָד מֵאֵלֶּה: יְהוּדָה, רְאוּבֵן, גָּד, אָשֵׁר, נַפְתָּלִי, מְנַשֶּׁה, שִׁמְעוֹן, לֵוִי, יִשָּׂשכָר, זְבוּלוּן, יוֹסֵף וּבִנְיָמִין. לָכֵן, שִׁמְעוּ בְּקוֹלִי!

וְהֵם יִהְיוּ כֹּחַ שֶׁאֵין לַעֲצֹר אוֹתוֹ; כֹּחַ בִּלְתִּי שָׁבִיר יִהְיוּ. כִּי חַיֵּיהֶם נִצְחִיִּים, וְכָל שַׁחַת לֹא תּוּכַל לְשַׁחֲתָם. גּוּפָם חָסִין מִכָּל הֶרֶס, כִּי כָּל הֶרֶס לֹא יוּכַל לֶאֱחֹז בַּגְּאוּלִים; כִּי נִצָּחוֹן עַל הַמָּוֶת וְהַהֶרֶס לָהֶם לְעוֹלְמֵי עוֹלָמִים, לְנֵצַח נְצָחִים. הֵם יִבָּרְאוּ מֵחָדָשׁ כְּבָנִים לָאוֹר, וּבִכְבוֹד עוֹלָם יִתְעַטְּפוּ, וְיִהְיוּ לָהֶם יְכֹלוֹת וְכִשְׁרוֹנוֹת יִחוּדִיִּים הַחֲרוּתִים בְּעֶצֶם הֲוָיָתָם, אֲשֶׁר יְשַׁקְּפוּ אֶת הֶשֵּׂגֵיהֶם הַמַּרְכִּיבִים וּמְגַלְּמִים יְכֹלוֹת שֶׁאֶבְרָא בְּתוֹכָם. כִּי אֲנִי, הַמָּשִׁיחַ יֵשׁוּעַ, הִנְנִי הַמְּצָרֵף הַיּוֹשֵׁב וּמַפְיֵחַ בְּיָדוֹ, מְלַבֶּה אֶת לֶהָבוֹת הַטִּהוּר, לְמַעַן אֶבְרָא אֶתְכֶם כֻּלְּכֶם מֵחָדָשׁ בִּי, הַמָּשִׁיחַ יֵשׁוּעַ, אֲדוֹנְכֶם וּמוֹשִׁיעֲכֶם אֲשֶׁר נִתְעַלָּה; וְכָל הַמִּתְגַּבֵּר, הוּא יִירַשׁ הַכֹּל.

הָבִינוּ, בָּנַי, כִּי כָּל הַנֶּאֱמָנִים בַּמָּשִׁיחַ יֵשׁוּעַ, אֲשֶׁר הֵם קְהִלָּתוֹ, גּוּפוֹ וְכַלָּתוֹ, נִבְחֲרוּ בִּמְיֻחָד בַּמָּשִׁיחַ יֵשׁוּעַ לִפְנֵי הִוָּסֵד תֵּבֵל. בְּעוֹד שֶׁרַבִּים מְפָרְשִׁים מִלִּים אֵלֶּה בְּבִטּוּיִים נְבוּבִים כְּמוֹ "רְצוּיִים מְאֹד וּמְבֹרָכִים בֶּאֱמֶת" כְּדֵי לְנַחֵם אֶת הַבָּשָׂר כַּאֲשֶׁר הֵם נֶאֱמָרִים בְּחֹסֶר אֱמוּנָה וּלְלֹא הֲבָנַת אֲמִתּוֹתַי הַנִּצְחִיּוֹת, אֵין זוֹ הַכַּוָּנָה הַמְּלֵאָה, אַף שֶׁהַדָּבָר נָכוֹן בֶּאֱמֶת לַנֶּאֱמָנִים בַּמָּשִׁיחַ בִּזְכוּת צִיּוּתָם וּתְשׁוּבָתָם.

הָבִינוּ, הַבִּטּוּי "נִבְחֲרוּ בּוֹ לִפְנֵי הִוָּסֵד תֵּבֵל" מְצַיֵּן כִּי אֲדֹנָי-יהוה שֶׁלָּכֶם בָּחַר וְהִנְחָה בִּמְלוֹא עֹז רְצוֹנוֹ וּבְתִכְנוּן מְדֻיָּק וְתַכְלִיתִי, מִבְּלִי לְהָפֵר אֶת רְצוֹנוֹ הַחָפְשִׁי שֶׁל אִישׁ וְאִישׁ כִּמְלֹא הַנִּימָה, עַל יְדֵי יְצִירָה, עֲשִׂיָּה וּבְרִיאָה מְחֻדֶּשֶׁת בִּבְחִירָה, בַּמָּשִׁיחַ יֵשׁוּעַ, כָּל אֶחָד וְאֶחָד מִכֶּם בַּהֲבָנָה מְלֵאָה וּבַחֵרוּת אָמָּנוּתִית וְעִצּוּבִית מְלֵאָה, בִּתְשׂוּמֶת לֵב כֹּה מֻפְלָאָה לִפְרָטִים בְּכָל אֶחָד וְאֶחָד מִכֶּם, שֶׁצָּרְכָה אֶת מְלוֹא הָעִדָּנִים מֵאָדָם וְחַוָּה וְעַד עַתָּה כְּדֵי לְהַשְׁלִימָהּ בְּדַקְדְּקָנוּתִי.

אַתֶּם יְצִירוֹת הַמּוֹפֵת שֶׁלִּי, שֶׁנִּבְרְאוּ בַּמָּשִׁיחַ!

אָמָּנוּת, דִּיּוּק וְעִצּוּב אֵלֶּה זְרוּעִים בַּקּוֹד הַגֶּנֶטִי שֶׁלָּכֶם, שָׁם בָּחַרְתִּי בְּעַצְמִי בְּמֻמְחִיּוּת וּבְאֻמָּנוּת רַבָּה כָּל פְּרָט וּפְרָט בְּכֻלְּכֶם לְמַטְּרוֹתַי הַנִּצְחִיּוֹת שֶׁעֲתִידוֹת לְהִתְגַּלּוֹת בִּמְלוֹאָן בְּאוּלַם כִּסְאִי, מִיָּד וּבְכֹחַ עַל פִּי פְּקֻדָּתִי. לָכֵן, הָבִינוּ כִּי יָצַרְתִּי, עָשִׂיתִי וּבָרָאתִי כָּל אֶחָד מִכֶּם בְּצוּרָה כֹּה סְפֶּצִיפִית וּבְפֵרוּט כֹּה דַּק, שֶׁאֵין בִּיכָלְתְּכֶם לְתָאֵר לְעַצְמְכֶם אֶת הַשִּׁנּוּי שֶׁעֲתִידִים כֻּלְּכֶם לַעֲבֹר עַל פִּי מַטְּרוֹתַי הַיִּחוּדִיּוֹת וְהַנִּבְחָרוֹת לְכָל אֶחָד וְאֶחָד מִכֶּם.

לָכֵן, אַל תִּירְאוּ, אֲדֹנָי-יהוה שֶׁלָּכֶם אֵינוֹ עוֹשֶׂה יֵשׁוּת אַחַת שׁוֹנָה מֵרְעוּתָהּ בִּבְחִירָה, כְּדֵי שֶׁיִּתְרוֹן נִתְפָּס יֻחְגַּג בְּחֵטְא שֶׁל רַהַב, וְאַף לֹא כְּדֵי שֶׁיְּצִירָה אַחַת תִּשְׁלֹט בְּאַחֶרֶת בְּאַלִּימוּת חֲטָאָה, שׂוֹנֵאת וְטִפְּשִׁית, אֶלָּא כְּדֵי שֶׁמַּטְּרוֹתַי הַנִּצְחִיּוֹת תִּתְגַּשֵּׁמְנָה בְּכֻלְּכֶם. וְזוֹ אֱמֶת נִצְחִית: בְּעֵת כְּתִיבַת שׁוּרוֹת אֵלֶּה, לֹא עָלָה עַל לֵב אֱנוֹשׁ מָה תִּהְיוּ בְּהִבָּרַאֲכֶם מֵחָדָשׁ וּלְאֵילוּ מַטָּרוֹת. עַל כֵּן, אֲנִי מְבַקֵּשׁ מִכֶּם, מִתּוֹךְ אַהֲבָתִי אֲלֵיכֶם (בָּנַי הַנִּבְרָאִים וְהַנִּבְחָרִים בִּזְכוּת אֱמוּנַתְכֶם, בָּנַי הַשָּׁבִים בִּתְשׁוּבָה בְּנֶאֱמָנוּת), שֶׁתִּהְיֶה לָכֶם אֱמוּנָה כְּגַרְגֵּר חַרְדָּל וְתָבִינוּ כִּי מֵהַקָּטָן וְעַד הַגָּדוֹל בְּקֶרֶב יִשְׂרָאֵל שֶׁל אֲדֹנָי-יהוה, כֻּלְּכֶם עֲשׂוּיִים וּמְעֻצָּבִים בְּשַׁלְמוּת וּבְפֶלֶא, וַחֲשׁוּלִים בְּעֹז וּבְכֹחַ כֹּה רַב, עַד כִּי יִשְׂרָאֵל שֶׁל אֲדֹנָי-יהוה הוּא שֶׁיִּהְיֶה נַעֲרָץ עַל הַכֹּל.

שָׁלוֹם רַב לָכֶם.

וְזֹאת לְלֹא מַשּׂוֹא פָּנִים, אֶלָּא בְּיִדְעִי הַצַּדִּיק, הַנִּצְחִי וְהַכֹּל-יָכוֹל וּבִשְׁלִיטָתִי הַמְּדֻיֶּקֶת בְּכָל הַדְּבָרִים, שֶׁבָּהֶם בָּחַרְתִּי כָּל אֶחָד מִכֶּם בְּאֹפֶן זֶה לְמַטְּרוֹתַי הַנִּצְחִיּוֹת בַּמָּשִׁיחַ יֵשׁוּעַ. כִּי יָדַעְתִּי בִּמְלוּאִי בְּנִצְחֵי קֶדֶם כִּי כָּל אֶחָד מִכֶּם, בְּאֹפֶן אִישִׁי, יָבוֹא בִּמְלוֹאוֹ אֶל הַמָּשִׁיחַ יֵשׁוּעַ בִּזְמַנּוֹ, וְלָכֵן סִפַּקְתִּי. עַל כֵּן, הָבִינוּ, בְּעוֹד שֶׁמִּסְתּוֹרִין חֲבוּיִים נִגְלִים בְּעֵדוּת וּבְדֻגְמַת שֵׁרוּת בביבלרידיון:144, דְּבָרִים אֵלֶּה נַעֲשִׂים כְּדֵי שֶׁכֻּלְּכֶם תַּתְחִילוּ לְהָבִין אֶת גֹּדֶל נַחֲלַתְכֶם וּמַטְּרוֹתֵיכֶם הַנִּצְחִיּוֹת, שֶׁלִּשְׁמָן בָּרָאתִי, בָּחַרְתִּי וְעָשִׂיתִי אֶתְכֶם כֻּלְּכֶם בְּתִכְנוּן תְּבוּנִי עַד הַפְּרָט הַדַּק בְּיוֹתֵר. גִּילוּ וְשִׂמְחוּ!

כִּי כַּאֲשֶׁר אוֹרִי יְנַפֵּץ אֶת הַחֹשֶׁךְ הֶעָבֶה, שֶׁהֲרֵי הַחֹשֶׁךְ לֹא יוּכַל לְאוֹרִי וְלֹא לְאוֹרוֹת הָעוֹלָם, אֲשֶׁר קִבְּלוּ כֹּחַ מֵהַמָּשִׁיחַ יֵשׁוּעַ וְנִשְׁלְחוּ בְּסַמְכוּת רַבָּה מֵאֲבִי הָאוֹרוֹת, וְעִתּוֹת גְּאֻלַּתְכֶם הַשְּׁלֵמָה הִגִּיעוּ.

לָכֵן, הַבִּיטוּ אֵלַי, הַמָּשִׁיחַ יֵשׁוּעַ, וְהַשְׁאִירוּ עֵינֵיכֶם נְשׂוּאוֹת וּפְקוּחוֹת, מְמֻקָּדוֹת בַּאֲדוֹנְכֶם יֵשׁוּעַ הַמָּשִׁיחַ בְּכָל מוּכָנוּת וְצִיּוּת. סַדְּרוּ אֶת עִנְיְנֵיכֶם, כִּי עַתָּה יִהְיֶה זֶה שֶׁפִּתְאֹם עַל הָאָרֶץ, לֹא תִּהְיוּ עוֹד.

וְעַל כֵּן אֲנִי מְצַוֶּה אֶתְכֶם, אֲהוּבַי, מַלְּאוּ אֶת מְנוֹרוֹתֵיכֶם, וְהֵיטִיבוּ אֶת פְּתִילוֹתֵיכֶם וּקְחוּ אֶת אִשִּׁי וְהַצִּיתוּ אֶת מְנוֹרוֹתֵיכֶם לְהֶאָרָה כְּדֵי לְהָאִיר עַצְמְכֶם בִּדְבָרַי, וְהָבִינוּ כִּי עֲלֵיכֶם לְהַמְתִּין לְצַד הַדֶּרֶךְ, כִּי הֶחָתָן מִתְקָרֵב, לְכַלָּתוֹ וּלְאוֹרְחֵינוּ גַּם יַחַד. סֶלָה.

סֶלָה! אֲהוּבַי, סֶלָה, יוֹנוֹתַי, כִּי אָכֵן אֲנִי אוֹהֵב אֶתְכֶם, וַאֲנִי בָּא וְהַשָּׁעָה הִגִּיעָה.

אָמְנָם, אָמְנָם אֲנִי הוּא, אֲשֶׁר זָרַע וְיָצַר אֶת אִמְרוֹת שִׁבְעַת הָרְעָמִים בְּרוּחוֹ שֶׁל סִפְרִי הַקָּטָן כִּי הוּא הַבִּיבְּלָרִידִיוֹן:144 שֶׁלִּי, שֶׁבְּתוֹכוֹ שִׁבְעַת הָרְעָמִים הַלָּלוּ הַמּוֹשְׁלִים בְּאוֹתוֹ סֵפֶר קָטָן שֶׁפָּתַחְתִּי בְּיָדוֹ, שֶׁהוּא גַּם אוֹתָהּ בְּשׂוֹרַת עוֹלָם, שֶׁבָּהּ קָרָאתִי וְהֵכַנְתִּי אֶת 144,000 שֶׁלִּי, אֲפִלּוּ בִּדְבָרִים אֵלֶּה שֶׁכֻּוְנוּ בְּיָדִי הָאַדִּירָה לִמְצֹא אֶת עֵינֵיהֶם, אָזְנֵיהֶם וְלִבּוֹתֵיהֶם שֶׁל כָּל אֵלֶּה שֶׁאֲלֵיהֶם נִשְׁלְחוּ, לָכֵן שִׁמְעוּ אֶת קוֹלִי הַנִּתָּן בְּדִבְרֵי שְׁלִיחִי וְעַבְדִּי אֲשֶׁר הֶחְבֵּאתִי בְּאַשְׁפָּתִי לְעֵת כָּזֹאת!

לָכֵן הָבִינוּ דַּרְכָּם מַה פֵּשֶׁר הַדָּבָר, אֲהוּבַי, כִּי אֵין עוֹד עִכּוּב, וַאֲנִי, הַמָּשִׁיחַ יֵשׁוּעַ, מַכְרִיז בִּדְמוּתוֹ שֶׁל שְׁלִיחִי וְעַבְדִּי הָאַדִּיר, כִּי לֹא יִהְיֶה עוֹד זְמָן!

כִּי הַזְמָנָתִי הֻכְרְזָה בְּרַחֲבֵי הָאָרֶץ בְּמֶשֶׁךְ עִדַּן חֶסֶד שָׁלֵם, וְרַבִּים נַעֲנוּ, וְרַבִּים בְּהֶתְאֵם יִגָּאֲלוּ בִּמְלֹאוֹ בְּפֶתַע וּבְלִי צֶפִי עַתָּה. וְהַמִּתְגַּבְּרִים שֶׁלִּי יְמַלְּאוּ בִּמְלֹאָם אֶת הַמַּעֲמָדוֹת הָרֵיקִים שֶׁנּוֹתְרוּ פְּתוּחִים בַּשָּׁמַיִם עַל יְדֵי הֵילֵל בֶּן שַׁחַר וְעַל כֵּן אֶגְלֶה אֶת שִׁבְעַת הָרָאשִׁים הַלָּלוּ שֶׁהֵם שִׁבְעַת הַשְּׂטָנִים, בְּהַנְהָגַת הֵילֵל בֶּן שַׁחַר.

אַף עַל פִּי שֶׁבְּמִשְׁפָּטָם הֵם נִשְׁלָלִים מִכָּל אִישִׁיּוּת וּפָשׁוּט נוֹדָעִים כְּהַשְּׂטָנִים, וְשָׂרָפִים רָשָׁעִים אֵלֶּה יִכָּבְלוּ, וְיֻשְׁלְכוּ אֶל לֶהָבוֹת הַתְּכֵלֶת הַחַמּוֹת בְּיוֹתֵר וְעָנְשָׁם הוּא הַחָמוּר מִכֹּל הַנִּדּוֹנִים, אֲשֶׁר יִנָּתְנוּ לְלַהֲבוֹת הָעֹנֶשׁ, כְּבוּלִים לְעוֹלָם לְשִׁלּוּם מָלֵא וּמִשְׁפָּט אֵשׁ לְנֶצַח נְצָחִים שֶׁלֹּא יִמָּלְטוּ מִמֶּנּוּ לְעוֹלָם, וְלֹא תִּהְיֶה לָהֶם מְנוּחָה אַף לֹא לְרֶגַע.

הִכּוֹנוּ, כַּלָּתִי הַלוֹחֶמֶת הָאַדִּירָה, כֵּן, הִכּוֹנוּ כִּי אַתֶּם הֵם 144,000 שֶׁלִּי! הָבִינוּ מִי אַתֶּם, כִּי אַתֶּם כַּלָּתִי הַלּוֹחֶמֶת, וְאֵין דָּבָר שֶׁנִּבְרָא שֶׁיַּשְׁוֶה לָכֶם, כַּלָּתִי הַנֶּאֱמָנָה, שֶׁעִמָּהּ אֶחֱלֹק אֶת כָּל הַדְּבָרִים בְּנַחֲלָתִי הַנִּצְחִית. כִּי אֶחָד מִכֶּם בֶּאֱמֶת יִרְדֹּף אֶלֶף וְאֵין דָּבָר וְאֵין יֵשׁוּת שֶׁתּוּכַל לַעֲמֹד נֶגְדְּכֶם, וְלֹא לַעֲצֹר בַּעַדְכֶם, וְלֹא לִפְגֹּעַ בָּכֶם, וְלֹא לַעֲצֹר אֶתְכֶם כִּמְעַט, כִּי אָמְנָם, אָמְנָם תִּהְיוּ בִּלְתִּי מְנֻצָּחִים לְגַמְרֵי, עַד כְּדֵי כָּךְ שֶׁכֻּלְּכֶם תֵּחָתְמוּ בְּעָצְמָה עַל מִצְחוֹתֵיכֶם שֶׁאֲפִלּוּ זַעַם אֱלֹהִים הַמָּלֵא לֹא יִגַּע בָּכֶם כְּלָל!

הָבִינוּ כִּי תַּצִּילוּ בְּחַיִּים אֶת שְׁאֵרִיתִי הָאַדִּירָה מִיַּעֲקֹב לְפִי הַבָּשָׂר שֶׁתּוֹשִׁיעוּ בְּחַיִּים בַּמִּדְבָּר בִּמְקוֹמוֹת שֶׁהוּכְנוּ מֵרֹאשׁ, מוּכָנִים לְקַבֵּל וּלְטַפֵּחַ אֶת אֵלֶּה שֶׁאוֹשִׁיעַ בְּחַיִּים כְּגוֹאֵל קְרוֹבָם! וְשׁוּם כֹּחַ, וְשׁוּם תַּחְבּוּלָה, וְשׁוּם מַאֲמָץ, וְשׁוּם דָּבָר כָּזֶה שֶׁהָאוֹיֵב יְבַקֵּשׁ לַעֲצֹר לֹא יְשַׁנֶּה כִּמְעַט, אֶלָּא כָּל מַאֲמַצֵּיהֶם יִתְמוֹטְטוּ לְגַמְרֵי עַל רֹאשָׁם הֵם כַּאֲשֶׁר זַעַם אֱלֹהִים הַמָּלֵא יִשָּׁפֵךְ בִּסְגִירַת חֶשְׁבּוֹן מִשְׁפָּטִית סוֹפִית בִּזְמַן זַעַם אֱלֹהִים הַכָּלוּל בְּאוֹתָן שֶׁבַע קְעָרוֹת שֶׁעוֹמְדוֹת לְהִשְׁתַּחְרֵר בִּמְלֹאָן בְּיַד שִׁבְעַת מַלְאָכַי הָאַדִּירִים, וְהֵם לֹא יִמָּלְטוּ!

וְשָׁם, בַּמִּדְבָּר, אֲפִלּוּ חָבוּי, יְטֻפַּל יַעֲקֹב, וְיֵאָכֵל, וְיִטּוּפַח לְגַמְרֵי בִּבְשׂוֹרַת הָעוֹלָם, אֲפִלּוּ אוֹתָהּ בְּשׂוֹרָה שֶׁסִּפְרֵי הַתָּנָ"ךְ הַקְּדוֹשִׁים שֶׁלָּכֶם הָיוּ רַק הַמִּקְדָּמָה שֶׁנִּתְּנָה כְּדֵי לִשְׁמֹר אֶת כֻּלְּכֶם לִגְאֻלָּה שְׁלֵמָה הַשְּׁמוּרָה לְאוֹתוֹ יוֹם קָרֵב שֶׁבָּא עַל הָעוֹלָם כֻּלּוֹ עַתָּה! אָמְנָם, אָמְנָם, יַעֲקֹב יֵאָכֵל וְיִטּוּפַח וְיוּכַן לְגַמְרֵי לְמַטְּרוֹתָיו בְּמַלְכוּת הָאֶלֶף שֶׁלִּי, שֶׁבֶּאֱמֶת שׁוּם דָּבָר רַע לֹא יִכָּנֵס לְתוֹכָהּ לְעוֹלָם.

הִכּוֹנוּ, קְהִלָּתִי, גּוּפִי וְאַף כַּלָּתִי לַשָּׁעָה הַשְּׁלֵמָה וְהַסּוֹפִית שֶׁל גְּאֻלַּתְכֶם הַשְּׁלֵמָה וְהַסּוֹפִית שֶׁהִגִּיעָה!

הִכּוֹנוּ, קְהִלָּתִי, גּוּפִי וְאַף כַּלָּתִי, כִּי כֹּה בְּפֶתַע וּבְלִי צֶפִי עַתָּה תִּמְצְאוּ עַצְמְכֶם עוֹמְדִים גְּאוּלִים לְגַמְרֵי בְּחַדַר הַכֵּס שֶׁל אֲדֹנָי-יהוה עַצְמוֹ, שָׁם אֶעֱמֹד לְהַצִּיג אֶת אוֹרוֹת הָעוֹלָם לִפְנֵי אֲבִי הָאוֹרוֹת, אֲפִלּוּ רֶגַע שֶׁל צִפִּיָּה נִרְגֶּשֶׁת שֶׁבּוֹ אֲדֹנָי-יהוה נִרְגָּשׁ יוֹתֵר מִכֻּלְּכֶם, וַאֲדֹנָי-יהוה חִכָּה יוֹתֵר מִכֻּלְּכֶם יַחַד! לָכֵן, אֲנִי מְצַוֶּה עַל כֻּלְּכֶם לָשֶׁבֶת בְּצִפִּיָּה נִרְגֶּשֶׁת בְּצִיּוּת מָלֵא לִפְקֻדּוֹתַי, וְאִם לֹא, אֶשְׁלֹל גְּמוּלִים מֵהַסַּרְבָנִים!

לָכֵן, אֲנִי מְצַוֶּה אֶתְכֶם, לַעֲמֹד אֵיתָן וְלַעֲבֹר אֶת מִבְחַנְכֶם הָאַחֲרוֹן, אֲהוּבַי, כִּי אֲפִלּוּ עַכְשָׁו אֲנִי עֲדַיִן מְלַבֶּה אֶת לַהֲבוֹת הַטִּהוּר בְּחַיֵּיכֶם וַאֲנִי מְקַפֵּל אֶתְכֶם כְּמַתֶּכֶת בִּידֵי הַנַּפָּח הַמֻּשְׁלָם וְהַצַּדִּיק, כִּי אֲנִי הַנַּפָּח הַמְּלַבֶּה אֶת לַהֲבוֹת הַטִּהוּר, כְּדֵי שֶׁעֲבוֹדוֹת הַיְשׁוּעָה הָאַדִּירוֹת וְהַנִּצְחִיּוֹת שֶׁלִּי יוּכְלוּ לְהַשְׁרִישׁ וּלְהִתְחַזֵּק בְּתוֹכְכֶם בִּמְלֹאָן, לָכֵן, אַל תִּרְפּוּ, אֲהוּבַי!

אֲנִי יֵשׁוּעַ הַמָּשִׁיחַ, אַף הַמָּשִׁיחַ יֵשׁוּעַ, וַאֲנִי שׁוֹלֵחַ דְּבָרִים אֵלֶּה שֶׁל הַכְרָזָה וְצִפִּיָּה לִקְהִלָּתִי, לְגוּפִי, וְאַף לְכַלָּתִי, כְּדֵי שֶׁתָּבִינוּ שֶׁהִתְאַסְּפוּתְכֶם הַצְּפוּיָה לַאֲדוֹנְכֶם הַנִּרְגָּשׁ בָּאֲוִיר הִיא חֵלֶק מִתָּכְנִית יְשׁוּעָה רְחָבָה יוֹתֵר, שֶׁבָּהּ עֲזִיבַתְכֶם תִּתְרַחֵשׁ בְּדִיּוּק כְּפִי שֶׁהֻכְרַז שֶׁתִּקְרֶה בְּיָדִי הָאַדִּירָה וּבִדְבָרִים שֶׁנִּתְּנוּ עַל יְדֵי שְׁלִיחִי וְעַבְדִּי הָאַדִּיר, הַמְּכֻוָּן לְאוֹתָם דְּבָרִים שֶׁזָּרַעְתִּי בְּרוּחוֹ.

כָּךְ שֶׁכַּאֲשֶׁר דְּבָרִים אֵלֶּה מֻכְרָזִים וְנִתָּנִים לָעוֹלָם זֶה בַּפַּעַם הָרִאשׁוֹנָה, אָכֵן הִתְגַּשְׁמוּתָם מַתְחִילָה בְּמִיָּדִי עִם הַכְרָזָתָם וְנְתִינָתָם לָעוֹלָם זֶה. לָכֵן, עַתָּה, אוּלַי תּוּכְלוּ לְהָבִין אֶת אִוַּלְתָּן שֶׁל הָעַלְמוֹת הַכְּסִילוֹת שֶׁאֵינָן מְבִינוֹת אֶת דִּבְרֵי אֲדֹנָי-יהוה! כִּי כַּאֲשֶׁר דְּבָרִים אֵלֶּה נִתָּנִים וְנֶאֱמָרִים לָאָרֶץ בַּפַּעַם הָרִאשׁוֹנָה, הִתְגַּשְׁמוּתָם הַמִּיָּדִית מַתְחִילָה עַל פִּי רְצוֹנוֹ הַנִּצְחִי שֶׁל אֲדֹנָי-יהוה הָעוֹשֶׂה כֹּל כַּעֲצַת רְצוֹנוֹ, וְאֵינוֹ זָקוּק לְיוֹעֵץ.

לָכֵן, הָבִינוּ שֶׁבְּעוֹד שֶׁרֹב הָאֲנָשִׁים טוֹעִים לַחְשֹׁב שֶׁעֲבוֹדָתוֹ שֶׁל שְׁלִיחִי וְעַבְדִּי הִיא פַּסְטוֹרָלִית בְּטִבְעָהּ, בִּיבְּלָרִידִיוֹן:144 הוּא לְמַעֲשֶׂה סוֹכֵן שֶׁל מִשְׁפָּט הַצְהָרָתִי וּכְלִי בָּחִיר לְעַצְמִי שֶׁעֲדַיִן נִסְתָּר לְעֵין כֹּל כְּדֵי לִקְרֹא לַנֶּאֱמָנִים כְּרוֹעֶה שֶׁלִּי שֶׁהִפְקַדְתִּי וְנָתַתִּי לוֹ סַמְכוּת הַצְהָרָתִית וּשְׁפוּטִית מְלֵאָה עַל קְהִלָּתִי, גּוּפִי, וְאַף כַּלָּתִי וְגַם יִשְׂרָאֵל, כִּי שַׂמְתִּי אֶת רַגְלוֹ הַיְמָנִית עַל הַיָּם וְאֶת רַגְלוֹ הַשְּׂמָאלִית עַל הָאָרֶץ בְּכָל סַמְכוּת.

וַאֲנִי הוּא הַמָּשִׁיחַ יֵשׁוּעַ, שֶׁהוּא גַּם יֵשׁוּעַ הַמָּשִׁיחַ, אַף הוּא אֲשֶׁר הִבְדִּיל וְשָׁמַר אֶת שְׁלִיחִי וְעַבְדִּי אֲשֶׁר גִּדַּלְתִּי מִנְּעוּרָיו לְמַטְּרוֹתַי הַנִּצְחִיּוֹת שֶׁעֲתִידוֹת לְהִתְגַּלּוֹת בִּמְלֹאָן בְּאוֹתוֹ הַרְפָּזוֹ אַדִּיר, שָׁם אַצִּיל אֶת הַנֶּאֱמָנִים בַּמָּשִׁיחַ מִשָּׁעָה אַחַת עִם הַחַיָּה, וְאֶשְׁנֶה אֶת כֻּלְּכֶם לְעוֹלָם וּלְעוֹלְמֵי עוֹלָמִים לְמַטְּרוֹתָיו הַנִּצְחִיּוֹת שֶׁל אֲדֹנָי-יהוה, אֲשֶׁר הֵנִיחַ בְּפָנָיו שֶׁל הַמָּשִׁיחַ יֵשׁוּעַ לְהִתְגַּשְׁמוּת מְלֵאָה בְּתוֹךְ כָּל אֶחָד וְאֶחָד מִכֶּם, נֶאֱמָנַי, אֲהוּבַי וְהַנִּכְסָפִים לִי מְאֹד!

וְכֵן, אֲנִי נוֹתֵן דְּבָרִים אֵלֶּה, אוּלַי כְּדֵי שֶׁתָּבִינוּ כַּמָּה טִפְּשִׁים רַבִּים מִכֶּם נִרְאִים, אֲשֶׁר כְּבָר נִמְצָאִים בְּתַהֲלִיךְ שֶׁל הִפָּרְדוּת וּגְאֻלָּה, וְעִם זֹאת צוֹעֲקִים בְּטִפְּשׁוּת וּבְחֹסֶר אֱמוּנָה כָּל הָעֵת, כְּשֶׁהֵם דּוֹרְשִׁים לוּחַ זְמַנִּים מִשֶּׁלָּהֶם, שֶׁהוּא בִּלְתִּי אֶפְשָׁרִי לְהַשָּׂגָה. לָכֵן, חַכּוּ בְּכָל צִפִּיַּת צֶדֶק וְשָׁלוֹם, בִּידִיעָה בְּרוּרָה שֶׁגְּאֻלַּתְכֶם הַשְּׁלֵמָה כְּבָר מִתְרַחֶשֶׁת וְאֵינָהּ יְכוֹלָה לְהֵעָצֵר, אֶלָּא אִם כֵּן תִּכְפְּרוּ בְּקֹצֶר רוּחֲכֶם. וְעִם זֹאת, אֲנִי מְשֻׁכְנָע בְּטוֹב יוֹתֵר לְגַבֵּי נֶאֱמָנַי כִּי אֶגְאַל אֶתְכֶם כְּפִי שֶׁהֻבְטַח, וְאַף מִלָּה אַחַת שֶׁנִּתְּנָה בְּיָדִי, אוֹ בִּידֵי אֲדֹנָי-יהוה, הָאֶהְיֶה אֲשֶׁר אֶהְיֶה הַגָּדוֹל, לֹא תִּפֹּל אַרְצָה רֵיקָם.

אֲנִי יֵשׁוּעַ הַמָּשִׁיחַ, וַאֲנִי בָּא לִקְרַאתְכֶם, אֲהוּבַי, לָכֵן הַמְשִׁיכוּ לְחַכּוֹת, וְטַהֲרוּ עַצְמְכֶם בְּכָל צִפִּיַּת צֶדֶק. מַלְּאוּ אֶת דַּעְתְּכֶם וְלִבְּכֶם בְּדִבְרֵי סִפְרִי הַקָּטָן שֶׁפָּתַחְתִּי בִּידֵי שְׁלִיחִי וְעַבְדִּי, עַד כְּדֵי כָּךְ שֶׁכֻּלְּכֶם תּוּכְלוּ לִסְעֹד בְּשָׁלוֹם וּבְנֶחָמָה רַבָּה עַל דִּבְרֵי בְּשׂוֹרַת הָעוֹלָם שֶׁנִּתְּנוּ כְּדֵי לִשְׁמֹר אֶתְכֶם בַּדֶּרֶךְ, כִּי גְּאֻלַּתְכֶם הַשְּׁלֵמָה הִגִּיעָה. אִם רַק תַּרְשׁוּ לְעַצְמְכֶם לְקַבֵּל אֶת כָּל אֲשֶׁר הֵכַנְתִּי, שֶׁנִּתַּן אֲפִלּוּ מֵאֲדֹנָי-יהוה עַצְמוֹ בְּפָנָיו שֶׁל יֵשׁוּעַ הַמָּשִׁיחַ כִּגְמוּל לַנֶּאֱמָנִים, אֲשֶׁר חִכּוּ בֶּאֱמוּנָה רַבָּה וְעַתָּה יִירְשׁוּ אֶת הַהַבְטָחוֹת שֶׁנִּתְּנוּ לְאַבְרָהָם הַנֶּאֱמָן, שֶׁהוּא אֲבִי הַצַּדִּיקִים הַחַיִּים בֶּאֱמוּנָה.

אֲנִי הוּא אֲדוֹנְכֶם הַנִּצְחִי לַצֶּדֶק, אַף אַרְיֵה מִשֵּׁבֶט יְהוּדָה, וַאֲנִי בָּא לִשְׁפֹּט וְלַעֲשׂוֹת מִלְחָמָה. לָכֵן, הִכּוֹנוּ, לוֹחֲמַי, וְהִכּוֹנוּ אַף עַל פִּי הַהוֹרָאוֹת שֶׁהִשְׁאַרְתִּי בְּסֵפֶר קָטָן וּפָתוּחַ זֶה שֶׁהוּכַן, נִתְקַבֵּל, נִתַּן וְהֻכְרַז עַל יְדֵי עֶדְרִי הַקָּטָן, וּבִמְיֻחָד עַל יְדֵי שְׁלִיחִי וְעַבְדִּי הָאַדִּיר אֲשֶׁר הִפְרַדְתִּי מֵהַקְּהִלָּה לְמַטְּרוֹתַי.

אָמְנָם, אָמְנָם, הֶחְבֵּאתִי אוֹתוֹ בְּאַשְׁפָּתִי לְעֵת כָּזֹאת, אֶלָּא שֶׁעַתָּה הִצַּתִּי אֶת חִצָּיו, וְחִצָּיו קוֹלְעִים לַמַּטָּרָה. כִּי שְׁלִיחִי וְעַבְדִּי הָאַדִּיר עָתִיד לְהִשְׁתַּנּוֹת כֹּה בְּפֶתַע וּבְלִי צֶפִי עַתָּה, כִּי הוּא סוֹכֵן שֶׁל מִשְׁפָּט הַצְהָרָתִי שֶׁיִּפְקֹד עַל הַכֹּחַ הַכָּלוּל בְּשִׁבְעַת הָרְעָמִים שֶׁזָּרַעְתִּי בְּרוּחוֹ, אֲשֶׁר רְעָמִים אֵלֶּה רוֹעֲמִים בּוֹ בִּמְלֹאָם, כִּי שַׂמְתִּי אֶת שַׁאֲגַת וְלֵב אַרְיֵה מִשֵּׁבֶט יְהוּדָה בְּשִׂכְלוֹ וּבְלִבּוֹ הַנִּבְרָאִים מֵחָדָשׁ, סֶלָה.

הִנֵּה! אֲנִי בָּא לִקְרַאתְכֶם, אֲהוּבַי, לָכֵן עִמְדוּ אֵיתָן בְּכָל צִיּוּת לִדְבָרַי שֶׁנּוֹעֲדוּ לִשְׁמֹר וּלְשַׁמֵּר אֶתְכֶם לְצַד הַדֶּרֶךְ בְּעוֹדְכֶם מְחַכִּים בְּצִפִּיָּה מְלֵאָה לְבוֹאִי, שֶׁכְּבָר הִגִּיעַ, אֲהוּבַי.
