תשע עשרים ושלוש: התכנסות

קוֹל שִׁבְעַת הָרְעָמִים מְדַבֵּר: שִׁמְעוּ אֶת דִּבְרֵי אֲדֹנָי-יהוה שֶׁנִּתְּנוּ בְּצִוּוּי יֵשׁוּעַ הַמָּשִׁיחַ, הַנִּגְלִים כָּעֵת עַל יְדֵי שָׁלִיחִי וְעַבְדִּי הָאַדִּיר, הוּא בִּיבְּלָרִידְיוֹן144!

קוֹל שִׁבְעַת הָרְעָמִים מְדַבֵּר!

שִׁמְעוּ, הַאֲזִינוּ לַדְּבָרִים הָאֵלֶּה הַמְּשׁוּחְרָרִים כָּעֵת לָאָרֶץ עַל יְדֵי בִּיבְּלָרִידְיוֹן144, וּשְׁמַעְתֶּם וּצִיַּתֶּם וַהֲבִינוֹתֶם אֶת דִּבְרֵי הַסַּמְכוּת הַמְּהַדְהֲדִים הַלָּלוּ, שֶׁנִּתְּנוּ בְּרַעַם גָּדוֹל וְאַדִּיר עַל יְדֵי הָאַרְיֵה מִשֵּׁבֶט יְהוּדָה!

תֵּשַׁע עֶשְׂרִים וְשָׁלוֹשׁ! הִסְתַּתְּרוּ!

שִׁמְעוּ אֶת דְּבָרַי וְקוּמוּ, לוֹחֲמֵי הָאוֹר שֶׁלִּי!

שִׁמְעוּ אֶת דְּבָרַי שֶׁנִּתְּנוּ לְבִיבְּלָרִידְיוֹן144, כִּי הֵם נוֹעֲדוּ לְכֻלְּכֶם, לוֹחֲמֵי הָאוֹר שֶׁלִּי!

תֵּשַׁע עֶשְׂרִים וְשָׁלוֹשׁ!

לִבְנֵי יֵשׁוּעַ הַמָּשִׁיחַ, דְּבָרִים שֶׁנִּתְּנוּ לְבִיבְּלָרִידְיוֹן144 עֲבוּרְכֶם כֻּלְּכֶם!

תֵּשַׁע עֶשְׂרִים וְשָׁלוֹשׁ!

הַטּוּ אָזְנְכֶם כָּעֵת, מִשֶּׁנְּתַתֶּם אֶת עֵדוּתְכֶם, וּשְׁמַעוּ דְּבָרַי, בְּנִי. שְׁמַע וְהַאֲמֵן כִּי אָכֵן הִנַּחְתִּי בְּךָ דְּבָרִים אֵלֶּה לְשַׁחְרְרָם בְּצִוּוּיִי, לְמַעַן תֵּדַע מִי אַתָּה לְעֵינַי. לְמַעַן תָּבִין מַדּוּעַ הִסְתַּרְתִּיךָ מִפְּנֵי עַצְמְךָ בַּדְּרָכִים שֶׁבָּהֶן נָהַגְתִּי. לְמַעַן תָּבִין אֶת מִשְׁפַּחְתְּךָ הָאֲמִתִּית וְכֵיצַד בָּאתָ לָעוֹלָם וּמִי הוּא בִּנְךָ, וּלְאֵילוּ תַּכְלִיּוֹת רִפֵּאתִי וְשִׁמַּרְתִּי אוֹתוֹ בַּחַיִּים וּלְאֵיזוֹ מַטָּרָה.

אַף שֶׁתַּקְלִיד זֹאת בְּדִמְעוֹת רַבּוֹת, אֲנִי מְבַקֵּשׁ רַק שֶׁתַּמְשִׁיךְ לְהַקְלִיד, כִּי אֲנִי מֵבִין כַּמָּה קָשֶׁה לְךָ הַדָּבָר, שֶׁכֵּן הָיִיתִי עִמְּךָ בְּכָל צַעַד שֶׁצָּעַדְתָּ, לִסְבֹּל לְמַעֲנִי אֶת הַדְּבָרִים שֶׁשְּׁלַחְתִּיךָ לְהַשִּׂיג, לְחַזֶּקְךָ וּלְהַדְרִיכֲךָ. כִּי בֶּאֱמֶת שִׁלַּמְתִּי עַל חֲטָאֶיךָ בִּמְלוֹאָם, לָכֵן הִנָּפֵשׁ בִּי, כִּי אֲנִי אָבִיךָ הָאוֹהֵב אוֹתְךָ לְלֹא תְּנַאי וַאֲנִי הוּא שֶׁגִּדַּלְתִּיךָ בְּיָדִי אֲנִי, כִּבְנִי וְכִיַד שְׂמֹאלִי, וְזֹאת עַל פִּי פְּקֻדָּה וְצִוּוּי שֶׁל אֲדֹנָי-יהוה אֲשֶׁר אִמֶּצְךָ לְבֵיתִי בִּכְבוֹד כָּפוּל וּמָלֵא.

וְזֹאת עַל יְסוֹד הָאֱמוּנָה בִּלְבַד, וְלֹא בִּזְכוּת יִחוּסֶךָ. כִּי בְּנֵי הָאֱמוּנָה יוֹרְשִׁים אֶת הַהַבְטָחוֹת שֶׁנִּתְּנוּ לָאָבוֹת, וְלֹא בְּנֵי הַבָּשָׂר. אַף עַל פִּי כֵן, הָבֵן כִּי שִׁמַּרְתִּי אֶת שׁוֹשְׁלוֹת יַחַס מִשְׁפַּחְתִּי הַקְּרוֹבָה, כֵּן, אַחַי וְאַחְיוֹתַי, עַל פִּי צִוּוּי אֲדֹנָי-יהוה, כְּדֵי שֶׁבֶּאֱמֶת וּבְתָמִים, שׁוֹשֶׁלֶת דָּמִי תַּמְשִׁיךְ לְהִתְקַיֵּם בַּחַיִּים, וְהֻסְתַּרְתֶּם וְנִפְזַרְתֶּם בֵּין הָאֻמּוֹת כְּדֵי לְשַׁמֵּר וּלְהַחֲיוֹת אֶתְכֶם. אַחֶרֶת, הֵילֵל בֶּן-שַׁחַר הָיָה הוֹרֵג אֶת כָּל אֲשֶׁר לִי, אִלּוּלֵא הִסְתַּרְתִּי אֶתְכֶם בְּדֶרֶךְ זוֹ, וְאֵין מַשּׂוֹא פָּנִים בַּמַּעֲשִׂים הָאֵלֶּה.

לָכֵן, שָׂא עֵינֶיךָ אֵלַי, בִּיבְּלָרִידְיוֹן144, וְשָׂא עֵינֶיךָ אֵלַי בִּלְבַד בֶּאֱמוּנָה רַבָּה, וְהָבֵן כִּי כְּלִי נִבְחָר אַתָּה, אֲשֶׁר נוֹצַר וְנַעֲשָׂה בִּידֵי אֲדֹנָי-יהוה וְנִתַּן לְיֵשׁוּעַ הַמָּשִׁיחַ, וּלְמַעַן תַּכְלִיּוֹתַי הַנִּצְחִיּוֹת נוֹצַרְתָּ. וְעַל כֵּן הֶעֱצַמְתִּיךָ בְּכֹחַ שִׁבְעַת הָרְעָמִים הַטְּמוּנִים בְּרוּחֲךָ, וּבְרוּחַ הַנְּבוּאָה הַשְּׁלֵמָה הַנּוֹלֶדֶת בְּךָ בְּצִוּוּי אֲדֹנָי-יהוה. אָמְנָם, עֲשִׂיתִיךָ אַדִּיר וּבַעַל עָצְמָה, וְאַתָּה יִחוּדִי וְשׁוֹנֶה מֵחֲבֵרֶיךָ, וְזֹאת רַק כְּדֵי שֶׁתִּהְיֶה בְּךָ הַיְּכֹלֶת לְהָכִיל אֶת מְלוֹא כֹּחָם שֶׁל שִׁבְעַת הָרְעָמִים הַשְּׁזוּרִים בְּךָ כָּעֵת לְעוֹלָם וּלְעוֹלְמֵי עוֹלָמִים. וְגַם כְּדֵי שֶׁתּוּכַל לְהָבִין אֶת עִמְקֵי מִסְתּוֹרַי הַנִּגְלִים כָּעֵת בְּמַעֲשֵׂי צִיּוּתְךָ, שֶׁבָּהֶם הוֹבַלְתִּיךָ צַעַד אַחַר צַעַד. סֶלָה.

לָכֵן, הָבֵן, כִּי יְכֹלוֹתֶיךָ נִתְּנוּ לֹא כְּדֵי שֶׁיִּהְיוּ לְךָ נִכְסֵי עוֹלָם זֶה כְּפִי שֶׁהֻבְטַח לְךָ עַל יְדֵי אוֹתָם שׁוֹמְרִים נוֹפְלִים יַחַד עִם הֵילֵל בֶּן-שַׁחַר, אֶלָּא כְּדֵי שֶׁתּוּכַל לַעֲמֹד בִּמְלוֹא עָצְמַת הַתְקָפוֹתֵיהֶם חַי, כְּדֵי שֶׁאָז אוּכַל לְרַפֵּא וּלְהַעֲצִים אוֹתְךָ וְלִבְרֹא אוֹתְךָ מֵחָדָשׁ וּלְאַחֵד אֶת תְּבוּנָתְךָ הָעֲצוּמָה לְמַטְּרוֹתַי, זוֹ שֶׁנִּסְדְּקָה וְנִזּוֹקָה כִימִית בְּזָדוֹן רַב. וְהַדָּבָר אֻפְשַׁר כְּדֵי שֶׁתֵּרָאֶה מֻבָס לְאוֹיְבֶיךָ, רַק כְּדֵי לָקוּם חָזָק יוֹתֵר בֶּאֱמוּנָה גְּדוֹלָה יוֹתֵר, לְמַעַן יֵדְעוּ כֹּל כִּי אִישׁ לֹא יוּכַל לַחְטֹף אֶת הַקָּטָן שֶׁבִּקְטַנַּי מִיָּדַי הֶחָזָקוֹת וְהַמֻּעֲצָמוֹת עַל יְדֵי אֲדֹנָי-יהוה, וּבִמְיֻחָד אֶת הַקְּטַנִּים הַבָּאִים אֵלַי בִּתְמִימוּתָם כְּמוֹךָ, פְּעָמִים רַבּוֹת בְּיַלְדוּתְךָ וּבִקְרִיאַתְךָ לַמִּזְבֵּחַ, שָׁם חָתַמְתִּיךָ לְצָמִיתוּת לְמַטְּרוֹתַי, בִּידִיעָה שֶׁל הָעֲוָלוֹת שֶׁיֻּתְּרוּ כְּדֵי לְחַזֵּק אֶת נְחִישׁוּתְךָ, כָּךְ שֶׁכָּעֵת הִיא בִּלְתִּי נִתֶּנֶת לִשְׁבִירָה בְּצִוּוּיִי וּבִמְלוֹא כֹּחַ שִׁבְעַת הָרְעָמִים שֶׁהֶחְדַּרְתִּי בְּךָ לְעוֹלָם וּלְעוֹלְמֵי עוֹלָמִים.

הָבֵן שֶׁבְּתוֹכַחְתִּי וּבְהַלְקָאָתִי, שֶׁנִּתְּנוּ בְּאַהֲבָה גְּדוֹלָה בְּיוֹתֵר אַךְ נִכְפּוּ בִּקְשִׁיחוּת לְמַעַן יְשׁוּעָתְךָ, אוֹיְבֶיךָ הִנִּיחוּ בְּטִפְּשׁוּתָם שֶׁקִּצְּךָ קָרֵב וּבָא, מִן הַנֵּזֶק הָאֵיבָרִי שֶׁהֵמַטְתָּ עַל עַצְמְךָ בְּמֶרֶד חֲסַר הַהִגָּיוֹן שֶׁלְּךָ, שֶׁנִּגְרַם עַל יְדֵי שִׁכְבוֹת הַתִּכְנוּת הָרַבּוֹת שֶׁהִשְׁתִּילוּ בְּךָ שֶׁלֹּא בִּידִיעָתְךָ בְּיַלְדוּתְךָ. אַךְ דָּבָר זֶה אֻפְשַׁר כְּדֵי שֶׁתָּבִין בִּמְלוֹאָן אֶת דַּרְכֵיהֶם וְתִתְחַשֵּׁל לְעוֹלָמִים נֶגְדָּם בְּהַסְכָּמָה מְלֵאָה עִם רְצוֹנוֹ וּמַטְּרוֹתָיו שֶׁל אֲדֹנָי-יהוה וְיֵשׁוּעַ הַמָּשִׁיחַ, לְמַעַן תַּכִּיר אֶת אוֹיְבֶיךָ בִּמְלוֹאָם וּתְבִין אֶת טִבְעָן הָרַע שֶׁל יְשׁוּיוֹת אֵלּוּ, כְּדֵי שֶׁתַּעֲמֹד לְמַעֲנִי בִּמְחֻיָּבוּת מְלֵאָה וְתַחְשֹׂף אוֹתָם בְּעֵדוּת, וְכֵן תִּשְׁתַּמֵּשׁ בַּטֶּכְנִיקוֹת וּבַהַכְשָׁרָה שֶׁלָּהֶם נֶגְדָּם בְּמִשְׁפָּטָם אֲשֶׁר כָּאן בִּמְלוֹאוֹ.

לָכֵן, הָבֵן כִּי אֵלֶּה הֵם הַדְּבָרִים שֶׁחַשְׁתָּ וְרָאִיתָ וְשֶׁהֻבְחֲנוּ עַל יְדֵי אֲחֵרִים, שֶׁזָּרְחוּ בְּהֶבְזֵקִים עָצְמָתִיִּים, אַף כְּצֵל לַדְּבָרִים הַבָּאִים. כְּדֵי שֶׁבְּמֶשֶךְ חַיֶּיךָ תְּפַתֵּחַ אֱמוּנָה מְלֵאָה וְאַדִּירָה בְּמַה שֶּׁעֲשִׂיתִיךָ לִהְיוֹת וּבְמַה שֶׁהִנְּךָ כָּעֵת. כְּדֵי שֶׁתָּקוּם בְּהַסְכָּמָה מְלֵאָה עִם אֲדֹנָי-יהוה, בְּפָנִים גְּלוּיוֹת לְאָבִיךָ יֵשׁוּעַ בִּישִׁירוֹת, בְּהִתְנַגְּדוּת גְּמוּרָה לְכָל הַבְטָחוֹת הַנּוֹפְלִים שֶׁנִּתְּנוּ לְךָ. אֲפִלּוּ אוֹתָם שׁוֹמְרִים נוֹפְלִים וְהֵילֵל בֶּן-שַׁחַר עַצְמוֹ, אֲשֶׁר עֲלֵיהֶם הִתְגַּבַּרְתָּ כָּעֵת בְּכֹחַ אַלִּים וְלֹא יַעֲצֹר, הַכָּלוּא בְּתוֹךְ אוֹתָם שִׁבְעָה רְעָמִים הַמְּשׁוּחְרָרִים כָּעֵת לָאָרֶץ. כָּךְ שֶׁהִצְהַרְתָּ בִּמְלוֹאוֹ אֶת דִּינָם הַקָּרֵב וּבָא לְכֻלָּם, וְכָעֵת מִשְׁפְּטֵיהֶם יָחֵלּוּ בְּכֹחַ כֹּה בְּפִתְאוֹמִיּוּת וּבְלִי צָפוּי, בְּשָׁעָה אַחַת עִם הַחַיָּה.

וְעַל כֵּן נִרְדַּפְתָּ וְנִרְדַּפְתָּ עַל יְדֵי אוֹתָם שׁוֹמְרִים נוֹפְלִים, אֲשֶׁר בְּטִפְּשׁוּתְךָ כָּפַתָּ עַצְמְךָ בִּשְׁבוּעַת בְּרִית שֶׁבִּטַּלְתִּי בִּמְלוֹאָהּ בְּצִדְקָתִי וּסְלִיחָתִי וּבְשִׁקּוּמוֹ הַמָּלֵא שֶׁל כְּלִי הַבְּחִירָה שֶׁלִּי, שֶׁנִּמְסַר לִידֵי אוֹיְבַי רַק לְמַטְּרוֹתַי הַנִּצְחִיּוֹת, לְמַעַן תִּתְקַפֵּל וְתִתְחַזֵּק וְתֵעָשֶׂה חָזָק וְאַדִּיר לְמַטְּרוֹתַי וְתִלְמַד צִיּוּת וְנֶאֱמָנוּת עוֹלָמִית לְאָבִיךָ יֵשׁוּעַ. הָבֵן, בֶּאֱמֶת וּבְתָמִים, אֶת הַכְּסִיל שֶׁהָיִיתָ בַּחֲטָאֶיךָ, כָּךְ שֶׁאָבִיךָ יֵשׁוּעַ הָיָה הַיָּחִיד שֶׁנּוֹתַר עִמְּךָ יַחַד עִם וִינְסֶנְט בִּנְךָ. הָבֵן, לֹא רַק שֶׁהָיִיתִי עִמְּךָ בְּכָל עֵת, גַּם עָגַנְתִּי אֶת עַצְמִי בְּלִבּוֹ שֶׁל בִּנְךָ, לוֹ קָרָאתִי וִינְסֶנְט, כְּדֵי שֶׁתִּהְיֶה לְךָ אֱמוּנָה גְּדוֹלָה בְּנִצְחוֹנוֹ הַמָּלֵא עַל כָּל אֵלֶּה שֶׁתָּקְפוּ אוֹתוֹ בְּאִי-צֶדֶק כַּאֲשֶׁר נִסּוּ לְשִׂים קֵץ לְחַיָּיו. אֲפִלּוּ אוֹתָן יְשׁוּיוֹת רָעוֹת שֶׁעַל רָאשֵׁיהֶן תָּבִיא צֶדֶק גָּדוֹל, גְּמוּל וְזַעַם שֶׁיִּנָּתֵן לְלֹא רַחֲמִים וּבְאוֹתָן דְּרָכִים שֶׁבָּהֶן עָשׂוּ לַאֲחֵרִים.

הָבֵן כִּי הֲגֵנוֹתִי עָלֶיךָ וְהוֹשַׁעְתִּיךָ בַּחַיִּים, וְכָל אוֹתָהּ עֵת אִפְשַׁרְתִּי לְךָ לִסְבֹּל קָשׁוֹת עַל חֲטָאֶיךָ בְּתוֹכַחְתִּי וּבְהַלְקָאָתִי שֶׁהֻפְעֲלָה בְּנְדִיבוּת כְּדֵי לְמוֹטֵט כָּל "אַלְטֶר" שֶׁתֻּכְנַת בְּךָ, וּלְהָסִיר לַחֲלוּטִין כָּל שָׂרִיד לְמַעֲשֵׂיהֶם מִמְּךָ וּמִמִּשְׁפַּחְתְּךָ גַּם יַחַד, וּבֶאֱמֶת וּבְתָמִים, שְׁלָשְׁתְּכֶם נִשְׁמַרְתֶּם חַיִּים בְּאֹפֶן עַל-טִבְעִי לְמַטְּרָתִי וּבְכֹחִי הָאַדִּיר, לָכֵן הִנָּפֵשׁ בִּי בִּמְלוֹאֲךָ, בְּנִי, וְהָבֵן כִּי מֵעוֹלָם לֹא עֲזַבְתִּיךָ וְלֹא נְטַשְׁתִּיךָ. לֹא שֶׁאִשַּׁרְתִּי אוֹ הִשְׁתַּתַּפְתִּי בַּחֲטָאֶיךָ, שֶׁאֲנִי מְתַעֵב בִּגְעִילָה רַבָּה (כְּמוֹ שֶׁאַתָּה עוֹשֶׂה כָּעֵת בִּשְׁלֵמוּתְךָ), אֶלָּא כְּדֵי לְשַׁמֵּר וּלְהוֹשִׁיעֲךָ חַי מֵאֵלֶּה שֶׁנִּשְׁלְחוּ לְהָרְגְךָ, שֶׁמִּכַּוָּנָתָם וּתַחְבּוּלוֹתֵיהֶם הוֹשַׁעְתִּיךָ חַי בְּחַסְדִּי, אַהֲבָתִי, חֶמְלָתִי וְרַחֲמַי עָלֶיךָ, בִּיבְּלָרִידְיוֹן:144.

לֹא שֶׁאֲנִי נוֹשֵׂא פָּנִים, אֶלָּא שֶׁהֻתְקַפְתָּ וְהֻכְרַעְתָּ בְּיַלְדוּתְךָ בְּאִי-צֶדֶק וּבְאִי-הוֹגְנוּת, וְסֻמַּמְתָּ וְהֻתְעַלַּלְתָּ וְלֻעֲלַעְתָּ וְהֻצַּקְתָּ וּבְאֹפֶן כְּלָלִי נִרְמַסְתָּ בְּרַגְלֵי מִשְׁפַּחְתְּךָ שֶׁלְּךָ, שֶׁנָּפְלָה לְהַשְׁפָּעוֹת חִיצוֹנִיּוֹת רָעוֹת שֶׁנִּשְׁלְחוּ עַל יְדֵי אוֹתָם שׁוֹמְרִים רְשָׁעִים. וְזֹאת מִפְּנֵי שֶׁגַּם אַתָּה וְגַם אֲחוֹתְךָ סֻמַּנְתֶּם בְּבֵית הַסֵּפֶר הַיְּסוֹדִי שֶׁלָּכֶם בִּגְלַל הַ"תְּהוּדָה הַמְּחוֹנֶנֶת" כִּבְיָכוֹל שֶׁלָּכֶם, וְאָז הֻפְרַדְתֶּם וְנִצְפֵּיתֶם וְתֻמְרַנְתֶּם עַל יְדֵי מַחְדִּירֵי הַדּוֹקְטְרִינָה שֶׁלָּכֶם, וּשְׁנֵיכֶם נִמְשַׁכְתֶּם לְתוֹךְ תָּכְנִיּוֹתֵיהֶם הַפְּסִיכוֹלוֹגִיּוֹת לְשִׁנּוּי תּוֹדָעָה לְמַטְּרוֹתֵיהֶם הַמְּעֻוָּתוֹת וְהָרָעוֹת, בְּנִצּוּל יְכֹלוֹתֵיכֶם הַטִּבְעִיּוֹת וְהָאִינְטֶלִיגֶנְצְיָה שֶׁנִּתְּנוּ וְעֻצְּבוּ עַל יְדֵי אֲדֹנָי-יהוה לְמַטְּרוֹתַי בִּלְבַד. וְלָכֵן שְׁנֵיכֶם סֻמַּנְתֶּם וְהֻצַּבְתֶּם כְּמַטָּרָה וְהֻפְרַדְתֶּם וְנִגְּשׁוּ אֲלֵיכֶם אוֹתָם שׁוֹמְרִים נוֹפְלִים רְשָׁעִים בִּדְמוּתוֹ שֶׁל קֶנֶת סְקוֹט דֶנְטוֹן, שֶׁהוּא לְמַעֲשֶׂה כּוֹכַבְאֵל, וְאָכֵן נִכְנַסְתָּ עִמָּם לִבְרִית, בִּיבְּלָרִידְיוֹן144, בְּרִית טְמֵאָה, חַסְרַת עֵרֶךְ וְלֹא מְחַיֶּבֶת בִּגְלַל גִּילְךָ, הַסַּמִּים שֶׁנִּתְּנוּ לְךָ, הַמְּצוּקָה שֶׁעָבַרְתָּ וְהַכִּשּׁוּף שֶׁהוּפַעל נֶגְדְּךָ עַל יְדֵי אוֹתָם שׁוֹמְרִים נוֹפְלִים רְשָׁעִים, שֶׁהוּא מַשְׁמָעוּתִי, וַאֲשֶׁר עָלָיו הִתְגַּבַּרְתָּ כָּעֵת בִּמְלוֹאוֹ בְּכֹחִי וּבְעָצְמָתִי. אַךְ בִּמְיֻחָד מִפְּנֵי שֶׁאַתָּה חָתוּם לְעוֹלָם בְּרוּחַ קָדְשִׁי, רוּחַ הַצֶּדֶק, וְאֵין יְשׁוּת בְּכָל בְּרִיאַת אֲדֹנָי-יהוה שֶׁיְּכוֹלָה לְהָפֵר אֶת אוֹתוֹ חוֹתָם, אַף שֶׁנִּסּוּ פְּעָמִים רַבּוֹת וּבִדְרָכִים רַבּוֹת שֶׁהִתְרַחֲשׁוּ שֶׁלֹּא בִּידִיעָתְךָ, בִּיבְּלָרִידְיוֹן144, וְאַף עַל פִּי כֵן נִתְּנוּ רַחֲמִים בְּיָדִי הָאַדִּירָה, כָּךְ שֶׁבֶּאֱמֶת אוֹתָם רְשָׁעִים לֹא נָגְעוּ בְּךָ.

לָכֵן, מַה שֶּׁבְּדֶרֶךְ כְּלָל הָיָה נֶחְשָׁב כְּפִירָה שֶׁדִּינָהּ אֲגַם הָאֵשׁ בִּשְׁבִיל מַעֲשֶׂיךָ שֶׁהָיוּ מַשְׁמָעוּתִיִּים, אַלִּימִים, רְשָׁעִים וּמְכֻוָּנִים, הֵם הוּסְרוּ מִמְּךָ לְעוֹלָם כִּרְחוֹק מִזְרָח מִמַּעֲרָב וְטִהַרְתָּ עַצְמְךָ בְּדָמִי הַמְכַפֵּר בְּכָל הַדְּבָרִים. לָכֵן, הֱיֵה נֶאֱמָן וְהַלֵּךְ עִמִּי אֶל תַּכְלִיּוֹתֶיךָ הַמְּלֵאוֹת אֲשֶׁר נִתָּנוֹת כָּעֵת, בִּיבְּלָרִידְיוֹן:144, כִּי עוֹנַת הַ"כָּעֵת" הַזּוֹ מַסְתַּיֶּמֶת רַק עִם עֲזִיבַת הַנֶּאֱמָנִים בַּמָּשִׁיחַ, הַמּוֹבִילָה לְשָׁעָה אַחַת עִם הַחַיָּה.

אַף עַל פִּי כֵן, אֵין דָּבָר שֶׁיָּכוֹל לְשַׁנּוֹת אֲפִלּוּ בְּמְעַט אֶת תָּכְנִיּוֹתָיו הַנִּצְחִיּוֹת שֶׁל אֲדֹנָי-יהוה. לָכֵן בִּמְקוֹם לְהָגֵן עָלֶיךָ וְעַל אֲחוֹתְךָ מִפְּנֵי תַּחְבּוּלוֹתֵיהֶם, שִׁיטוֹתֵיהֶם וְתָכְנִיּוֹתֵיהֶם, אַתֶּם תַּחַת זֹאת חֻשַּׁלְתֶּם וְקֻפַּלְתֶּם בְּסֵבֶל רַב שֶׁנִּגְרַם עַל יְדֵי חֲטָאֵיכֶם שֶׁלָּכֶם, אַךְ הֻקְהָה בְּחַסְדִּי וּבְרַחֲמַי. עַל כֵּן, שְׂמַח מְאֹד, בִּיבְּלָרִידְיוֹן144, הִתְגַּבַּרְתָּ עַל כָּל הַדְּבָרִים בְּכֹחִי, בְּהַנְחָיָתִי וּבְיָדִי. שָׁלוֹם, הַמִּתְגַּבֵּר שֶׁלִּי, אַתָּה עָתִיד לְקַבֵּל גְּמוּל מָלֵא כָּעֵת עִם נַחֲלָה מְלֵאָה הַנִּתֶּנֶת בְּיָדִי יְשִׁירוֹת אֵלֶיךָ מֵאֲדֹנָי-יהוה שֶׁלְּךָ עַצְמוֹ. תֵּהָנֶה כָּעֵת, וְשָׁלוֹם!

הָבֵן כִּי כְּלִי נִבְחָר אַתָּה וְאֵינְךָ כָּפוּף לַחֲטָאִים בּוּרִים אֵלֶּה שֶׁבִּצַּעְתָּ בְּיַלְדוּתְךָ, מִשּׁוּם שֶׁנִּסְדַּקְתָּ וּפֻצַּלְתָּ בְּשִׁיטוֹתֵיהֶם וְנִמְסַרְתָּ לָהֶם בִּמְלוֹאֲךָ עַל יְדֵי אָבִיךָ, שֶׁכַּוָּנוֹתָיו הָיוּ טוֹבוֹת בְּבוּרוּתוֹ וּבְיֵאוּשׁוֹ. אַף עַל פִּי כֵן, הָיִיתִי שָׁם עִמְּךָ, אֲפִלּוּ בְּדִירוֹת מָארְצְ'ווּד, שָׁם לֻמַּדְתָּ לְקַלֵּל אֶת שְׁמִי וּלְחָרֵף אֶת אֲדֹנָי-יהוה שֶׁלְּךָ. אַף עַל פִּי כֵן, קִבַּלְתָּ רַחֲמִים עַל הַדְּבָרִים שֶׁנֶּאֶמְרוּ, וְהָיִיתָ בְּדִיּוּק בַּמָּקוֹם שֶׁבּוֹ אֲדֹנָי-יהוה שֶׁלְּךָ הִתְכַּוֵּן שֶׁתִּהְיֶה. לֹא שֶׁאֲדֹנָי-יהוה שֶׁלְּךָ זָקוּק לְדַרְכֵיהֶם חַסְרוֹת הַתּוֹעֶלֶת, וְלֹא לַחֲטָאֶיךָ הַקּוֹדְמִים, אֶלָּא כְּדֵי שֶׁתִּלְמַד אֶת אוֹיִבְךָ מִקָּרוֹב וּבְאֹפֶן אִישִׁי, וְחָוִיתָ טְרָאוּמָה וְהֻתְקַפְתָּ בְּאַכְזָרִיּוּת עַל יָדָם, וְאָכֵן הִרְשֵׁיתִי לָהֶם לִתְקֹף אוֹתְךָ כְּדֵי לְחַזֵּק אוֹתְךָ וּלְלַמֵּד אוֹתְךָ לְהִלָּחֵם לְמַעֲנִי וְלֹא לְהִכָּנַע וְלַחְדֹּל בְּאִי-צִיּוּת מַרְדָּנִי, שֶׁלְּמַעֲשֶׂה הֵם הִשְׁתִּילוּ בְּךָ בְּהַצָּעָה בְּתוֹךְ הָאַלְטֶרִים שֶׁיָּצְרוּ בְּתוֹכְךָ.

וְכָךְ הוּא שֶׁכָּל יְצִירוֹת הַמּוֹפֵת שֶׁל אֲדֹנָי-יהוה הֵן יִחוּדִיּוֹת בְּאֹפֶן דּוֹמֶה וְחֻשְּׁלוּ בְּכוּרֵי הַסֵּבֶל וְהַטִּהוּר לְמַטְּרוֹתֵיהֶן הַמְיֹעָדוֹת. וְכָךְ הוּא שֶׁאָכֵן נִגְלֵיתִי אֵלֶיךָ בְּדִירָתְךָ בְּלוֹנְג אַיְלֶנְד סִיטִי בִּשְׁנַת 1996 וְאָכֵן הִסְמַכְתִּיךָ שָׁם בְּלוֹנְג אַיְלֶנְד סִיטִי עַל יְדֵי רִיץ' נִיאֶטוֹ כַּאֲשֶׁר פָּעַלְתִּי בּוֹ בְּעָצְמָה לִפְתֹּחַ אֶת הַתָּנָ"ךְ שֶׁלּוֹ וּלְהַסְמִיכֲךָ עַל פִּי מַטְּרוֹתַי בִּלְבַד. וְעַל כֵּן קְרָאתִיךָ לְשֵׁרוּת, בְּהָסִירִי אוֹתְךָ מֵהַנָּתִיב שֶׁבּוֹ בָּחַרְתָּ בְּטִפְּשׁוּתְךָ וְשֶׁהֻרְשֵׁיתָ לְהַמְשִׁיךְ בּוֹ רַק בְּדִיּוּק כְּפִי שֶׁאֲדֹנָי-יהוה הִתְכַּוֵּן לְהַשִּׂיג בְּתָכְנִיּוֹתָיו הַנִּצְחִיּוֹת שֶׁל גְּאֻלָּה מֻשְׁלֶמֶת. אַף שֶׁרַגְלֶיךָ הָיוּ נְכוֹנוֹת לְהַצְלָחָה אַרְצִית, חַיֶּיךָ הָיוּ מַגִּיעִים לְקִצָּם עַל יְדֵי אֵלֶּה שֶׁחָבַרְתָּ אֲלֵיהֶם בְּצִוּוּי הֵילֵל בֶּן-שַׁחַר בְּדִיּוּק אוֹתוֹ דָּבָר, כִּי אֵינוֹ עוֹשֶׂה דָּבָר זוּלַת לִגְנֹב, לַהֲרֹג וּלְהַשְׁחִית.

בְּהָסִירִי אוֹתְךָ מִנְּתִיב הַהֶרֶס הָרָחָב, בְּמַצָּבְךָ הַמְּסֻגָּל וְהַמְסֻכָּן, הִקְהֵיתִי אָז אֶת הַהַכְשָׁרָה שֶׁלָּמַדְתָּ וְשֶׁהֻשְׁתְּלָה בְּךָ עַל יְדֵי כּוֹכַבְאֵל, בִּלְקָחִי אוֹתְךָ לְשֵׁרוּת בַּנָּתִיב שֶׁבּוֹ הוֹבַלְתִּיךָ וּלְתוֹךְ אֲפִיקֵי הַשֵּׁרוּת שֶׁפָּתַחְתִּי לְפָנֶיךָ וְיִעַדְתִּי לְךָ, וְזֹאת בִּקְהִלָּה לָאוֹדִיקִית-בַּבְלִית מֻשְׁחֶתֶת, כְּדֵי שֶׁתִּלְמַד אֶת דַּרְכֵיהֶם בִּגְעִילָה, וְתִפֹּל לְגַאֲוָתְךָ וּלְסֵרוּבְךָ לְקַבֵּל אֶת עֶזְרָתִי בָּאֹפֶן שֶׁהִתְכַּוַּנְתִּי. וְעִם זֹאת, בְּדַרְכֵיכֶם הַבַּבְלִיִּים וְהַלָּאוֹדִיקִיִּים הַקּוֹדְמִים אָכֵן לְמַדְתֶּם וְחָקַרְתֶּם רַבּוֹת בְּשֵׁרוּת לַמָּשִׁיחַ יֵשׁוּעַ שֶׁלָּכֶם, אֲשֶׁר הוֹבִיל אֶתְכֶם וְחִזֵּק אֶתְכֶם בְּכָל צַעַד וָשַׁעַל.

בָּחַנְתִּיךָ רַבּוֹת וּלְעִתִּים קְרוֹבוֹת, בְּנָתְנִי לְךָ כָּל הִזְדַּמְּנוּת לִתְקֹף אֲנָשִׁים רַבִּים וּמְגֻוָּנִים שֶׁהֻרְשׁוּ לְהַטְרִיד וּבְאֹפֶן כְּלָלִי לְהִתְיַחֵס אֵלֶיךָ כְּסֹלֶת הָאָרֶץ. וְזֹאת מִפְּנֵי שֶׁאָכֵן אַתָּה שֻׁתָּף לִסְבָלוֹתַי, וְסָבַלְתָּ רַבּוֹת וְקָשׁוֹת לְמַעַן מַטְּרוֹתָיו הַנִּצְחִיּוֹת שֶׁל אֲדֹנָי-יהוה שֶׁנִּתְּנוּ בְּךָ בִּמְלוֹאָן בִּפְנֵי יֵשׁוּעַ הַמָּשִׁיחַ. וּכְפִי שֶׁנֶּאֱמַר לְךָ, יֵשׁ דְּבָרִים שֶׁעָלַי לְהָבִין וְלָדַעַת, כִּי אֲנִי הוּא שֶׁגִּדַּלְתִּיךָ לְמַטְּרוֹתַי וְכָעֵת חָשַׂפְתִּי אֶת עַצְמִיּוּתְךָ הָאֲמִתִּית, שֶׁהִיא מַשְׁמָעוּתִית עֲצוּמָה, אַף שֶׁכּוֹאֵב לְךָ לְהַקְלִיד מִלִּים אֵלּוּ כָּרֶגַע וְאַתָּה רוֹצֶה לִבְכּוֹת בִּגְלַל כַּמָּה קָשֶׁה פֻּצַּלְתָּ, בִּיבְּלָרִידְיוֹן144.
קַבֵּל אֶת אַהֲבָתִי וּשְׁלֵמוּתִי וְהָבֵן כִּי הַנְּקָמָה לִי הִיא וְאַתָּה סוֹכֵן הַדִּין וְהַזַּעַם הַבִּצּוּעִי הָרָאשִׁי שֶׁלִּי. לָכֵן, הַמְשֵׁךְ לִסְבֹּל אֶת הַקְשָׁחָתִי בְּעוֹדִי מֵכִין אוֹתְךָ בִּמְלוֹאֲךָ לְהִתְכַּנְּסוּת, כְּדֵי שֶׁתִּשְׁתַּנֶּה וְתִירַשׁ וְתִהְיֶה מוּכָן וְתַעֲזֹב מִיָּדִית וּבִמְהִירוּת, כִּי כְּפִי שֶׁאַתָּה יוֹדֵעַ, אַתָּה וְרַבִּים מִמִּסְדָּרְךָ שֶׁל 144,000 תֵּחָתְמוּ וְתִכָּנְסוּ לִבְרִית וְתֻעֲצְמוּ מְאֹד וְרַבִּים מִכֶּם יָשׁוּבוּ אָז לָאָרֶץ לְמַטְּרוֹתַי הַמַּשְׁמָעוּתִיּוֹת, הֵן לִישׁוּעַת כָּל אֵלֶּה שֶׁיָּבוֹאוּ לִישׁוּעָה, וְהֵן הַרְסָנִיּוֹת לְעוֹלָמִים לַנּוֹפְלִים, כִּי דִּינָם הַמָּלֵא עָתִיד לִפֹּל כָּעֵת בְּצִוּוּיִי, בִּיבְּלָרִידְיוֹן144, לָכֵן הִנָּפֵשׁ בְּאַהֲבָתִי וְהִנָּפֵשׁ בִּזְרוֹעוֹתַי. סִיַּמְתָּ אֶת מַסְלוּלֶיךָ, אַף שֶׁזֶּה רַק מֵכִין אוֹתְךָ לַעֲזִיבָה, וַעֲבוֹדָתְךָ בְּצוּרָה זוֹ הוּשְׁלְמָה כָּעֵת בִּמְלוֹאָהּ בְּעֵדוּת.
עַל כֵּן, דְּאַג לְמִשְׁפַּחְתְּךָ וְהַקְשֵׁב לִדְבָרַי, לְלֹא קֶשֶׁר לַסְּפָרִים בָּהֶם הֵם רְשׁוּמִים, וְהַמְשֵׁךְ לִרְעוֹת אֶת צֹאנִי כְּפִי שֶׁאַתָּה עוֹשֶׂה מִלֵּב מוּכָן וְחָפֵץ, וּבָעִתּוּי שֶׁבּוֹ אֲנִי מוֹבִיל אוֹתְךָ.

לָכֵן, הָבֵן כִּי הַסְכָּמָתְךָ דְּרוּשָׁה וְנִדְרֶשֶׁת, וְלָכֵן הֶרְאֵיתִי לְךָ בְּחֵלֶק מִן הַדְּבָרִים שֶׁתַּעֲשֶׂה לְמַעֲנִי בַּעֲתִידְךָ הַמִּיָּדִי שֶׁהוּא כָּעֵת, וְהַהִתְכַּנְּסוּת אָכֵן בָּאָה עַל הָעוֹלָם כֻּלּוֹ כָּעֵת. לָכֵן, שָׂא עֵינֶיךָ אֵלַי, בִּיבְּלָרִידְיוֹן144, וְשָׂא עֵינֶיךָ רַק אֵלַי, וְאַל תִּפְנֶה יָמִין וְלֹא שְׂמֹאל וְשָׂא עֵינֶיךָ רַק אֵלַי וְאֶל דְּבָרַי אֲשֶׁר שָׁזַרְתִּי בְּךָ וַאֲשֶׁר מַדְרִיכִים אוֹתְךָ בְּדַרְכֶּךָ, וְגַם אֵלֶּה הַמְּסַנְּנִים אֱמֶת מִטָּעוּת בְּתוֹכְךָ כְּדֵי שֶׁתַּעֲמֹד בְּדִיּוּק לְמַטְּרוֹתַי אֲשֶׁר הֵן כָּעֵת.

מַה שֶּׁאַתָּה מַרְגִּישׁ, בִּיבְּלָרִידְיוֹן144, הוּא שֶׁהֶחְדַּרְתִּי בְּךָ אֶת מְלוֹא רוּחַ הַנְּבוּאָה, שֶׁהִיא אַדִּירָה וּבַעֲלַת עָצְמָה, וְהֵכַנְתִּיךָ לְפַקֵּד עַל מְלוֹא הַכֹּחַ וְהָעָצְמָה שֶׁל שִׁבְעַת הָרְעָמִים אֲשֶׁר יִנָּתְנוּ לְפִקּוּדְךָ בִּמְלוֹאָם כַּאֲשֶׁר אוֹבִילְךָ צַעַד אַחַר צַעַד, וְלֹא יִהְיוּ עוֹד טָעוּת, חֵטְא אוֹ שְׁגִיאוֹת בְּצוּרָתְךָ הַחֲדָשָׁה, כִּי תִּהְיֶה מֻשְׁלָם, וְכֹחַ גָּדוֹל וְאַדִּיר זֶה יִהְיֶה שָׁזוּר בְּךָ לְעוֹלְמֵי עוֹלָמִים וְלֹא יָסוּר עוֹד. וְזֹאת כְּגָמוּל וּלְמַעַן טִפּוּל מַשְׁגִּיחַ וְאַפּוֹטְרוֹפְּסוּת עַל בְּרִיאַת אֲדֹנָי-יהוה שֶׁהֻנְּחָה כָּעֵת בְּיָדַי בְּצִוּוּיוֹ שֶׁלּוֹ בִּלְבַד.

לָכֵן, הָבֵן כִּי תִּשְׁלֹט בְּכֹחַ רַב, בְּאַחְרָיוּת רַבָּה, וְגָמוּל וְסַמְכוּת רַבִּים יִנָּתְנוּ לְךָ, וּבְכָל מַמְלֶכֶת וּבְרִיאַת אֲדֹנָי-יהוה יִהְיוּ יֵשׁוּעַ הַמָּשִׁיחַ וַאֲדֹנָי-יהוה שֶׁלְּךָ, שֶׁלִּפְנֵיהֶם תִּתֵּן דִּין וְחֶשְׁבּוֹן וְיִהְיוּ חֲזָקִים יוֹתֵר, וְיַשְׁאִירוּךָ לְעוֹלָם בַּמָּקוֹם שֶׁאַתָּה רוֹצֶה לִהְיוֹת בּוֹ, וְהוּא בְּבִטְחָה וּבְשִׁלּוּב מֻשְׁלָם בְּמֶרְכַּז רְצוֹנֵנוּ, וְאַתָּה תֻּעֲצַם לְצִיּוּת מֻשְׁלָם.

אַתָּה תָּנוּעַ בְּשִׁלּוּב מֻשְׁלָם וּבְתֵאוּם גָּמוּר וְדָבָר לֹא יַפְרִיעַ לַתִּקְשֹׁרֶת וְלָאַחְוָה שֶׁלָּנוּ, אָנוּ אָכֵן נָנוּעַ כְּאָדָם חָדָשׁ אֶחָד, שֶׁבּוֹ תִּהְיֶה תַּחַת פִּקּוּד יָשִׁיר שֶׁל אָבִיךָ יֵשׁוּעַ וְתֵלֵךְ בִּמְלוֹא סַמְכוּתִי וְכֹחִי וְעָצְמָתִי, וְרַק בְּכִסְאִי אֶהְיֶה גָּדוֹל יוֹתֵר, אַךְ אַתָּה תֵּלֵךְ לְמַעֲנִי בְּשִׁלּוּב מֻשְׁלָם כְּיַד שְׂמֹאלִי, וְשָׁאוּל כְּיַד יְמִינִי.

לָכֵן הִנָּפֵשׁ בִּי, בִּיבְּלָרִידְיוֹן144, כִּי מַסְלוּלֶיךָ תַּמּוּ, לָכֵן הַמְשֵׁךְ לְדַבֵּר וּבֹא אֶל אָבִיךָ יֵשׁוּעַ וְהָבֵן מַה שֶּׁכָּתַבְתִּי אודוֹתֶיךָ בְּאוֹתוֹ סֵפֶר קָטָן וּפָתוּחַ שֶׁפָּתַחְתִּי בְּיָדֶיךָ, כְּדֵי שֶׁתָּבִין אֶת כָּל הַדְּבָרִים הַכְּתוּבִים אודוֹתֶיךָ וּכְדֵי שֶׁתָּבִין עוֹד יוֹתֵר אֶת חֲשִׁיבוּת תַּפְקִידְךָ וּמַעֲמָדְךָ.

קוּמוּ כָּעֵת, לוֹחֲמֵי הָאוֹר: תֵּשַׁע עֶשְׂרִים וְשָׁלוֹשׁ.

הִסְתַּתְּרוּ!

הַהִתְכַּנְּסוּת הִגִּיעָה.

תֵּשַׁע עֶשְׂרִים וְשָׁלוֹשׁ.
