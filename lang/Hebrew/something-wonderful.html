עדות כפולה: הקהילה
בכתבי יוחנן המטביל
15 במרץ 2024

דבר מה נפלא

דבר יהוה עליי, הנה דבר יהוה!

הִנֵּה! אָנֹכִי עוֹשֶׂה דָּבָר נִפְלָא לִפְנֵיכֶם, וְרַבִּים מִכֶּם מְמָאֲנִים לִרְאוֹת!

הִנֵּה! אָנֹכִי עוֹשֶׂה מַעֲשֶׂה גָּדוֹל בְּקִרְבְּכֶם אֲשֶׁר קָרוֹב לְהִתְגַּשֵּׁם!

הִנֵּה! אָנֹכִי עוֹשֶׂה מַעֲשֶׂה גָּדוֹל בָּכֶם, שֶׁבְּאֶמְצָעוּתוֹ נוֹשַׁעְתֶּם וְנִרְפֵּאתֶם!

עָשִׂיתִי יוֹתֵר מֵאֲשֶׁר לְאַחוֹת כַּמָּה שְׁבָרִים. עָשִׂיתִי יוֹתֵר מֵאֲשֶׁר לְאַחוֹת אֶתְכֶם שׁוּב יַחַד, בַּמָּקוֹם שֶׁבּוֹ הַתְּפָרִים וְהַצַּלָּקוֹת נִכָּרִים. עָשִׂיתִי אֶת בְּחִירַי וְנֶאֱמָנַי שְׁלֵמִים בִּי, הַמָּשִׁיחַ הַחַי, וּבָרָאתִי אֶתְכֶם מֵחָדָשׁ. דְּעוּ כִּי חַיֵּיכֶם חֲבוּיִים בִּי, וּכְשֶׁאֶגָּלֶה, כֵּן גַּם אַתֶּם תִּגָּלוּ עִמִּי בְּכָבוֹד וְתִהְיוּ עִמִּי לְעוֹלָם וָעֶד. אַתֶּם כַּלָּתִי וְאַתֶּם גּוּפִי; הֲלֹא קָרָאתִי לָכֶם בְּשֵׁם? הֲלֹא סָלַלְתִּי לָכֶם דֶּרֶךְ לָשׁוּב אֶל אֲבִיכֶם, אֲבִי הָאוֹרוֹת?

אֲנִי הוּא אֲשֶׁר הָיָה יָדוּעַ לְאָבִי בְּטֶרֶם הִוָּסֵד תֵּבֵל, אֲשֶׁר נִצְטַוָּה וְנוֹלַד לְמַעַן יְשׁוּעַתְכֶם, אֲשֶׁר נִצְלַב וְנָתַן אֶת טִפַּת דָּמוֹ הָאַחֲרוֹנָה עַל חַטֹּאתֵיכֶם. דָּבָר לֹא הֻשְׁמַט וְדָבָר לֹא נֶעֱזַב בִּלְתִּי גָּמוּר, וּמַה שֶּׁנֶּאֱמַר יִתְקַיֵּם וְיִתְגַּשֵּׁם עַד תֹּם. בָּרָאתִי אֶתְכֶם בְּרוּחַ קָדְשִׁי אֲשֶׁר נָתַן לִי הָאָב לִצֹּק עַל שְׁלִיחַי בְּאוֹתוֹ יוֹם חַג שָׁבוּעוֹת רָחוֹק, כַּאֲשֶׁר דִּבְּרוּ אֶת נִפְלְאוֹת אֱלֹהִים בִּלְשׁוֹנוֹת שֶׁלֹּא יָכְלוּ לְהָבִין בְּעַצְמָם. קָרָאתִי, וְאַף עַל פִּי שֶׁרַבִּים דָּחוּ אוֹתִי, שְׁאֵרִית שֶׁל חֶסֶד וּבְחִירָה נַעֲנְתָה, וְעָשִׂיתִי בָּהֶם מַעֲשֶׂה גָּדוֹל אֲשֶׁר קָרוֹב לְהִתְגַּלּוֹת. אָמְנָם, כַּאֲשֶׁר אַתֶּם מַבִּיטִים בָּרְאִי, אֵינְכֶם רוֹאִים אֶת עַצְמְכֶם הָאֲמִתִּי. עֲשִׂיתִיכֶם גִּבּוֹרִים בְּחֻלְשַׁתְכֶם, וּבְהִתְגַּלּוּתִי כֵּן תִּגָּלוּ גַּם אַתֶּם. תִּהְיוּ כָּמוֹנִי, וְתִהְיוּ עִמִּי לָנֶצַח. אֲנִי בִּכּוּרֵי הַמֵּתִים, וְהַקָּצִיר כִּמְעַט עֲלֵיכֶם. כַּאֲשֶׁר אֶגָּלֶה, תָּקוּמוּ לִקְרָאתִי, וְתִהְיוּ בְּבֵית אָבִי כְּשֶׁאֶחְטֹף אֶתְכֶם אֶל מְעוֹנְכֶם בַּשָּׁמָיִם.

אָמְנָם, אָמְנָם, בִּדְבָרִי הַקָּדוֹשׁ (כִּי אֲנִי הַדָּבָר שֶׁנַּעֲשָׂה בָּשָׂר, הַקָּדוֹשׁ וְהַמִּתְהַלֵּךְ בֵּינֵיכֶם), הֵן, בִּדְבָרִי הַקָּדוֹשׁ, הֲלֹא קְרָאתֶם כִּי הַמִּשְׁפָּט יָחֵל בְּבֵית אֱלֹהִים? אֲנִי עוֹמֵד לְהַשְׁלִים מַעֲשֶׂה בִּבְחִירַי, בְּצַדִּיקַי וּבִנְבְחָרַי, שֶׁבְּאֶמְצָעוּתוֹ תִּבָּרְאוּ בִּי מֵחָדָשׁ לַחֲלוּטִין וְתִהְיוּ כָּמוֹנִי, כִּי נִדְמֵיתֶם לִדְמוּת בְּנוֹ יְחִידוֹ שֶׁל אֱלֹהִים. שְׂכָרִי עִמִּי, וְלֹא אֶשְׁפֹּט אֶתְכֶם עַל חַטֹּאתֵיכֶם; הֲלֹא כְּבָר שִׁלַּמְתִּי עֲלֵיהֶן? בְּקַבָּלַתְכֶם אֶת אֲדֹנוּתִי וּבְהִזְדַּהוּתְכֶם עִמִּי, נִסְלַח לָכֶם וְאַתֶּם צַדִּיקִים בִּי, כִּי שִׁלַּמְתִּי מְחִיר נוֹרָא שֶׁלֹּא הָיָה עָלַי לְשַׁלֵּם כִּי חָטוּא אֵינֶנִּי. לֹא, לָקַחְתִּי מִכֶּם אֶת חַטֹּאתֵיכֶם, עֲוֹנוֹתֵיכֶם, כִּשּׁוּפֵיכֶם, וְאֶת הַזַּעַם וְהָעֹנֶשׁ, לְמַעַן תִּתְאַחֲדוּ עִם הָאָב וְעִם הַבֵּן. לֹא, אֶשְׁפֹּט אֶת מַעֲשֵׂיכֶם שֶׁבֻּצְּעוּ עֲלֵי אֲדָמוֹת.

וְעִם מִשְׁפָּטִי שָׂכָר גָּדוֹל, וּמַעֲשִׂים רַבִּים יִשָּׂרְדוּ, אֲשֶׁר הֵם זָהָב, כֶּסֶף וַאֲבָנִים יְקָרוֹת. אָכֵן, הָעֵץ, הֶחָצִיר וְהַקַּשׁ יִשָּׂרְפוּ כְּמוֹץ, וְתַחְווּ אָבְדָן, אַף כִּי תִּוָּשְׁעוּ כְּמוֹ מִתּוֹךְ אֵשׁ. כִּי אִשִּׁי לָכֶם הִיא אֵשׁ מְצָרֵף, שֶׁבָּהּ טִהַרְתִּי אֶת נַפְשׁוֹתֵיכֶם כְּשֵׁם שֶׁצּוֹרֵף מְטַהֵר אֶת זְהָבוֹ, וּמֵסִיר אֶת הַסִּיגִים. עִמְדוּ אֵיתָן וְדוּמוּ בִּי. מִי הוּא אֲשֶׁר יַטְרִיד אֶתְכֶם, אַף כִּי אָכֵן נִרְדַּפְתֶּם וּסְבַלְתֶּם צָרָה? אַל תִּירְאוּ, אוֹתִי שָׂנְאוּ וְגַם אֶתְכֶם הֵם שׂוֹנְאִים. וְעִם זֹאת, בִּגְאֻלַּתְכֶם תִּהְיוּ נַעֲרָצִים מְאֹד בַּכָּבוֹד שֶׁבּוֹ אַלְבִּישׁ אֶתְכֶם בַּיּוֹם הַקָּרוֹב וְהַמִּתְקָרֵב בִּמְהִירוּת. אָבִי מִשְׁתּוֹקֵק לִרְאוֹתְכֶם לְפָנָיו בְּהֵיכַל כִּסְאוֹ, שָׁם מְחַכִּים לָכֶם מוֹשְׁבוֹתֵיכֶם, כִּי אָבִי הוֹשִׁיב אֶתְכֶם בַּשָּׁמַיִם. הֲלֹא אַתֶּם בְּחִירָיו, שַׁגְרִירָיו, כְּבוֹדוֹ, כִּתְרוֹ וּפְרָסוֹ שֶׁאֵין לוֹ עֵרֶךְ? אָמְנָם, אֵין לָכֶם מְחִיר.

מִשְׁפָּט יִפֹּל עַל בָּבֶל הַגְּדוֹלָה בְּקָרוֹב, וְיִהְיֶה זֶה מִשְׁפָּט נוֹרָא, אַךְ רָאוּי. הוֹי, אָדָם, עַד מָתַי אֶשָּׂאֲךָ? מֵאֻמָּה שֶׁל קְדוֹשִׁים שֶׁשִּׁמְּשָׁה מִגְדַּלּוֹר לָעוֹלָם לָבוֹא אֵלַי, הָפַכְתָּ לְאֻמָּה שֶׁל זוֹנוֹת נוֹאֲפוֹת. הִפְרַדְתִּי אֶת עַמִּי מִכֶּם, וְהַהַפְרָדָה כִּמְעַט מִתְגַּשֶּׁמֶת עִם סִיּוּמוֹ שֶׁל עִדַּן חֶסֶד זֶה. הוֹי, אָדָם, מַדּוּעַ אַתָּה הוֹלֵךְ אַחֲרֵי הָרָשָׁע, וּמְקַבֵּל אֶת שְׁחָדָיו חַסְרֵי הָעֵרֶךְ עַל פָּנַי, הַמָּשִׁיחַ הַנִּצְחִי, בְּכָל הֲדָרִי וְכָבוֹדִי.

אָבִי נָתַן זְמַן לְכָל הָאֻמּוֹת לַחֲזֹר בִּתְשׁוּבָה, וַאֲנָשִׁים רַבִּים, אֲפִלּוּ הָמוֹן גָּדוֹל, קִבְּלוּ אֶת הַצָּעָתִי בְּחֶסֶד יְשׁוּעָתִי שֶׁשָּׁפַכְתִּי עֲלֵיהֶם.

אַךְ אַתָּה, הוֹי אָדָם טִבְעִי הַמְּמָאֵן לְדִבְרֵי הָרוּחַ וּבוֹזֵז לְשְׁכֵנוֹ, תִּסְבֹּל מִשְׁפָּט גָּדוֹל עַל אַדְמָתְךָ, וּבָבֶל הַגְּדוֹלָה לֹא תִּהְיֶה עוֹד. בָּבֶל, הָפַכְתְּ לְאֻמָּה שֶׁל זוֹנוֹת הַמְּשַׂחֲקוֹת אֶת הַתַּפְקִיד עִם שׁוּלֵי שִׂמְלוֹתֵיכֶן מֵעַל רָאשֵׁיכֶן כְּמוֹ אָתוֹן פֶּרֶא בְּיִחוּם הַמְּחַכָּה לְעוֹבֵר הָאוֹרֵחַ הַבָּא. נִטְמֵאתֶן בְּבָשָׂר זָר, הִתְעַנַּגְתֶּן וַאֲהַבְתֶּן אֶת חֲטָאֵיכֶן, וּמִשְׁפַּט עוֹלָם יִפֹּל עַל כָּל קְשֵׁי הַלֵּב הַמַּחְשִׁיבִים אֶת קָרְבָּנִי כְּדָבָר קַל וְחָסֵר עֵרֶךְ לִרְאוֹת. הָפַכְתֶּם לְאֻמָּה שֶׁל עוֹשֵׂי רָעָה וּמְפֵרֵי בְּרִית הַמַּחְלִיפִים נָשִׁים כְּמוֹ בְּגָדִים יְשָׁנִים בַּחֲדָשִׁים עַד שֶׁהֶחָדָשׁ מִתְלַכְלֵךְ. דִּבְרֵיכֶם הַגְּדוֹלִים שֶׁדִּבַּרְתֶּם נֶגֶד אָבִי יִבָּדְקוּ. דִּמְיוֹנוֹתֵיכֶם חַסְרֵי הָעֵרֶךְ יֻנְּחוּ לִפְנֵיכֶם, וְתִתְּנוּ דִּין וְחֶשְׁבּוֹן. אַךְ חֶשְׁבּוֹנְכֶם לְעוֹלָם לֹא יַסְפִּיק כְּדֵי לַעֲמֹד צַדִּיקִים לְפָנַי. עֶצֶם קִיּוּמְכֶם הֻשְׁחַת, וּבְטֵרוּפְכֶם בְּחַרְתֶּם בִּבְשַׂרְכֶם הַנָּמֵק, כִּי כְּמוֹ שֶׁאָמַר אָבִי, מוֹת תָּמוּתוּ. וְעִם זֹאת, עֲדַיִן נִרְאֶה לָכֶם מוּזָר שֶׁבְּמוֹתְכֶם תָּמוּתוּ. אַף עַל פִּי שֶׁתְּמוּגְגוּתְכֶם הַמְּלֵאָה לֹא הָיְתָה מִיָּדִית, אַתֶּם גּוֹוְעִים מֵאָז יוֹם נְשִׁימַתְכֶם הָרִאשׁוֹנָה. מִשְׁפָּט נִגְזַר עַל אֻמַּתְכֶם, וַהֲלֹא הַשָּׁמַיִם זוֹעֲקִים אֶת אוֹתוֹ הַדָּבָר? אַשְׁמִיד אֶת אַרְצְכֶם כָּלִיל, וּמִגְדְּלֵיכֶם הַגְּדוֹלִים יִפְּלוּ, וַחֲשֵׁכָה סְמִיכָה תְּכַסֶּה אֶת הָאָרֶץ. חוֹמוֹתֵיכֶם כְּבָר נִפְרְצוּ, וּמִילְיוֹנִים בָּאוּ אֲלֵיכֶם לָקַחַת מִכֶּם אֶת מַה שֶּׁנָּתַתִּי לָהֶם, כִּי אַפְטִים אֶת יַד אוֹיְבֵיכֶם בְּמִשְׁפַּטְכֶם. זְמַנְּכֶם כִּמְעַט תַּם, וְאַתֶּם אֻמָּה כְּבוּלָה לְפָנַי, הַמְּחַכָּה לְמַלְאָכַי שֶׁיָּבוֹאוּ בְּמִשְׁפָּט. תִּפְּלוּ נְפִילָה גְּדוֹלָה כְּשֶׁהַמַּיִם יָבוֹאוּ עֲלֵיכֶם, כְּשֶׁהָרְעָדָה תָּבוֹא עֲלֵיכֶם, כְּשֶׁהַצְּבָאוֹת יְשֻׁלְּחוּ, וְאַרְצְכֶם תִּבָּקַע לִשְׁנַיִם. רָעָב יִשְׁרֹר בָּעוֹלָם, וּמִלְחָמָה תָּבוֹא עַל כָּל הָעוֹלָם כְּשֶׁהַחַיָּה תַּעֲלֶה מִן הָאֵפֶר בִּמְזִמּוֹתֵיהֶם הַמְּתֻכְנָנוֹת מֵאָז.

זְמַן נִתַּן לִתְשׁוּבָה, וְרַבִּים עָנוּ בֶּאֱמוּנָה, הָרַחֲמִים הַגְּדוֹלִים שֶׁנָּתַתִּי לִפְנֵי שֶׁהַמִּשְׁפָּט יִתְבַּצֵּעַ. אֵין לִי חֵפֶץ בַּחֻרְבָּן הַבָּא, אַךְ גַּם עַכְשָׁו, רַבִּים חַיִּים בְּחֵטְא גָּדוֹל, אֲפִלּוּ טוֹרְפִים אֶת יַלְדֵיכֶם שֶׁלָּכֶם, וְלֹא אִכְפַּת לָכֶם כָּל עוֹד כַּסְפְּכֶם קָרוֹב. אֶקַּח מִכֶּם כָּל עֹשֶׁר, מַעֲדַנִּים, תַּעֲנוּג וְשִׂמְחָה. תִּצְטַמְּקוּ לִפְנֵי הַשָּׁמַיִם, תְּחַפְּשׂוּ פְּרוּסָה שֶׁל מָזוֹן שֶׁרַבִּים לֹא יִמְצְאוּ. אַךְ הַקּוֹרְאִים בְּשֵׁם יהוה יִוָּשֵׁעוּ.

יְלָדַי הַקְּטַנִּים, זוֹהִי הַשָּׁעָה הָאַחֲרוֹנָה שֶׁל הַזְּמַן הָאַחֲרוֹן בְּעִדַּן חֶסֶד זֶה. שָׁלַחְתִּי אֶת נְבִיאַי בְּמַעֲשֵׂהוּ שֶׁל יְדִידִי הַיָּקָר יוֹחָנָן הַמַּטְבִּיל. הֲלֹא הָיוּ לְקוֹל קוֹרֵא בַּמִּדְבָּר? הֲלֹא הֵם רוֹעַי, הַקּוֹרְאִים וּמְחַפְּשִׂים אַחַר אַחֲרוֹנֵי צֹאנִי הָאוֹבֵד? אָמְנָם, יֵשׁ לִי גַּם עֵדֶר אַחֵר, וּזְמַן יְשׁוּעָתָם קָרוֹב, כִּי יַעֲקֹב יִוָּשַׁע וּשְׁאֵרִית יַעֲקֹב תְּקַבֵּל אוֹתִי כַּאֲדוֹנָם, כְּשֶׁכָּל הַבְּחִירִים יִהְיוּ לְאָדָם חָדָשׁ אֶחָד. שָׁלַחְתִּי רַבִּים בְּרַחֲבֵי הָאָרֶץ, הַמַּכְרִיזִים עַל בּוֹאִי.

אַף שֶׁרַבִּים מֵעַמִּי הֻטְעוּ וּפוֹשְׁרִים הֵם, רוּחִי תִּתְעוֹרֵר בְּרַבִּים, וְרַבִּים יָשׁוּבוּ אֵלַי. הִתְגַּלּוּתִי קְרוֹבָה. עִמְדוּ חָזָק וְהַכְרִיזוּ עַל בּוֹאִי וְהַכְרִיזוּ אֶת דִּבְרֵי יְשׁוּעָתִי לַכֹּל. הַזְּמַן כִּמְעַט הִגִּיעַ, וּדְמָמָה עַל פְּנֵי תְּהוֹם כְּשֶׁצִּבְאוֹת הַשָּׁמַיִם מְחַכִּים לִתְקִיעַת הַשּׁוֹפָר הַקְּרוֹבָה יוֹתֵר מִמָּה שֶׁאַתֶּם יְכוֹלִים לְדַמְיֵן. אַךְ עוֹד זְמַן קָצָר מְאֹד נוֹתַר, וְרַבִּים נוֹסָפִים עוֹד יִוָּשְׁעוּ, לָכֵן אַל תִּרְפּוּ יְדֵיכֶם.

אֲנִי הוּא הָרוֹעֶה אֶת קְהִלָּתִי וְהַנּוֹתֵן מַתָּנוֹת לִבְנֵי אָדָם.

אֲנִי יהוה הַגָּדוֹל וְהַגִּבּוֹר, שָׂה טָבוּחַ מִיסוֹד הָעוֹלָם.

אֲנִי הוּא שֶׁנָּתַן אֶת חַיָּיו בְּעַד הַצֹּאן, הַמָּשִׁיחַ הַחַי אֲשֶׁר אֱלֹהִים הֱקִימוֹ מִן הַמֵּתִים.

אֲנִי הַדָּבָר שֶׁנַּעֲשָׂה בָּשָׂר הַשּׁוֹכֵן בֵּינֵיכֶם.

אֲנִי בְּנוֹ יְחִידוֹ שֶׁל אֱלֹהִים שֶׁבָּא בַּבָּשָׂר, הוּא הַמָּשִׁיחַ הַגִּבּוֹר שֶׁל יִשְׂרָאֵל שֶׁל אֱלֹהִים.

אֲנִי קָרוֹב, וְהִתְגַּלּוּתִי מַגִּיעָה.

אֲנִי הַשּׁוֹפֵט הַצַּדִּיק הָרָאוּי לִפְתֹּחַ אֶת הַחוֹתָמוֹת.

אֲנִי שֵׂה הָאֱלֹהִים, הַטָּבוּחַ עַל חַטֹּאת עַמִּי.

אֲנִי הַמּוֹשִׁיעַ חֲסַר הַחֵטְא הַמּוֹשִׁיעַ עַד תֹּם.

אֲנִי אִישׁ מַכְאוֹבוֹת וִידוּעַ חֹלִי, וְעִם זֹאת שִׂמְחָתִי מְלֵאָה תָּמִיד.

אֲנִי הָאָדוֹן הַחַי, הָרוֹעֶה הַצַּדִּיק.

אֲנִי הוּא שֶׁכָּל הַנְּבִיאִים נִבְּאוּ עָלָיו, וּבְיָדִי מַפְתְּחוֹת הַמָּוֶת וְהַשְּׁאוֹל, וְהָאָב נָתַן בְּיָדַי הַגִּבּוֹרוֹת אֶת כָּל אֲשֶׁר בַּשָּׁמַיִם וּבָאָרֶץ.
