עדות כפולה: ישראל
נספח
24 בספטמבר 2025

אל הנאמנים במשיח: היכונו והתכוננו להיאסף, חתנכם קרב ובא!

שִׁמְעוּ, שִׁמְעוּ אֶת דְּבַר יֵשׁוּעַ הַמָּשִׁיחַ וְהַכנִיעוּ אֶת לִבְּכֶם בְּצִיּוּת לְקוֹל דְּבָרַי! וְכֵן הוּא, כִּי רַבִּים מֵעַמִּי פָּנוּ יָמִינָה וְרַבִּים עוֹד פָּנוּ שְׂמֹאלָה, אַף הַרְחֵק מִן הַמָּשִׁיחַ יֵשׁוּעַ אֲדוֹנָם, וְזֹאת בְּעֵת שֶׁאֲנִי בָּא לְקַבֵּץ אֶתְכֶם אֵלַי, אִם תִּשָּׁאֲרוּ נֶאֱמָנִים לְקוֹלִי בְּצִיּוּת מָלֵא לִדְבָרַי. כִּי אָמְנָם, אָמְנָם, אֲנִי הוּא אֲדוֹנְכֶם הָאֲמִתִּי, מַלְכְּכֶם, כֹּהַנְכֶם הַגָּדוֹל וּמְשִׁיחֲכֶם! כֵּן, אֲנִי קוֹרֵא לְכֻלְּכֶם אֶל זְרוֹעוֹתַי הָאוֹהֲבוֹת לְהֵאָסֵף אֵלַי פִּתְאוֹם וּבְלִי צֶפִי כָּעֵת, כָּל אֵלֶּה הַמְּחַכִּים בְּצִיּוּת וּבֶאֱמוּנָה שְׁלֵמָה, וּבִמְיֻחָד כָּל אֵלֶּה שֶׁנָּפְלוּ מִן הַדֶּרֶךְ אֶל הַתַּרְמִית הַזֹּאת שֶׁנּוֹעֲדָה עַל יְדֵי הֵילֵל בֶּן שַׁחַר לְהַטְעוֹת אֶת לֵב הַפְּתָאִים, וּלְרַסֵּק אֶת צִפִּיּוֹתֵיכֶם הַצּוֹדְקוֹת לִגְאוּלַת צֶדֶק וְשָׁלֵם, הִיא גְאוּלַתְכֶם הַמְּלֵאָה אֲשֶׁר תַּשִּׂיג אֶת הַנֶּאֱמָנִים בַּמָּשִׁיחַ, פִּתְאוֹם וּבְלִי צֶפִי כָּעֵת!

אַף דִּבְרֵי שֶׁקֶר שֶׁנִּתְּנוּ בְּמִרְמָה עַל יְדֵי נְבִיאֵי שֶׁקֶר הַקּוֹבְעִים מוֹעֲדִים, הַמְדַבְּרִים שֶׁקֶר בְּשֵׁם יֵשׁוּעַ הַמָּשִׁיחַ, וְאַף טוֹעֲנִים כִּי הֵם מְדַבְּרִים בִּשְׁמִי! אַךְ הֵם אֵינָם מְדַבְּרִים אֶת דְּבָרַי, אַף לֹא עוֹשִׂים אֶת פִּרְיִי, וְלֹא יְדָעוּנִי, וְאוֹמַר לְכָל הַטּוֹעֲנִים כָּזֹאת בַּיּוֹם הַגָּדוֹל הַהוּא: "סוּרוּ מִמֶּנִּי פּוֹעֲלֵי אָוֶן, מֵעוֹלָם לֹא יְדַעְתִּיכֶם!" וְכֵן הוּא, כִּי רַבִּים מִן הַנֶּאֱמָנִים בַּמָּשִׁיחַ, אֲשֶׁר הֵם קְהִלָּתִי, גּוּפִי וְאַף כַּלָּתִי, יוֹשְׁבִים כָּעֵת עֲצוּבִים וּמְאֻכְזָבִים, וְזֹאת בְּעֵת שֶׁחֲתָנָם אָכֵן מִתְקָרֵב כַּאֲשֶׁר הוּבְטַח, הֻכְרַז וְנֶאֱמַר! אַךְ כֹּה רַבִּים מִכֶּם יוֹשְׁבִים כָּעֵת בִּדְמָמָה, כְּאִלּוּ אֶת הִתְאַסְּפוּתָם הַגְּדוֹלָה שֶׁל הַנֶּאֱמָנִים בַּמָּשִׁיחַ יָכוֹל הֵילֵל בֶּן שַׁחַר לְעַכֵּב אוֹ לְבַטֵּל, כִּי אָמְנָם, אָמְנָם, הוּא זֶה שֶׁקָּבַע בְּשֶׁקֶר אֶת מוֹעֵד עֲזִיבַת הַנֶּאֱמָנִים בַּמָּשִׁיחַ וְכֹה רַבִּים מִנֶּאֱמָנַי אָכְluּ אֶת דִּבְרֵי כַּזְבוֹ בִּתְאוֹבָה כְּאִלּוּ דִּבְרֵי שֶׁקֶר אֵלֶּה הֵם מֵאֵת הַמָּשִׁיחַ יֵשׁוּעַ עַצְמוֹ! שִׁמְעוּ אֶת דְּבָרַי, קְהִלָּתִי, גּוּפִי וְאַף כַּלָּתִי! הַאִם אֵינְכֶם מַכִּירִים אֶת קוֹלִי? הַאִם אֵינְכֶם מְזַהִים אֶת דְּבָרַי? כִּי אַף לֹא יוֹד אַחַת, וְאַף לֹא קוֹץ שֶׁל יוֹד יַעֲבֹר עַד אֲשֶׁר יִתְקַיְּמוּ כָּל הַדְּבָרִים בְּדִיּוּק כַּכָּתוּב וְכַמֻּבְטָח.

אַךְ כֵּן הוּא, כִּי רַבִּים מִכֶּם יוֹשְׁבִים כָּעֵת קְדוֹרַנִּית וּשְׁבוּרֵי לֵב בְּאִי־צִיּוּתְכֶם, עַד כִּי אַתֶּם יוֹשְׁבִים כָּעֵת נְפוּלֵי רוּחַ וּכְפוּפִים לְדִבְרֵי לוֹעֲגִים וּמְלַגְלְגִים, אֲשֶׁר מִשְׁפָּטָם יַשִּׂיגֵם פִּתְאוֹם וּבְלִי צֶפִי כָּעֵת וְרַבִּים, רַבִּים מֵאֵלֶּה יִפְּלוּ לְדִין עוֹלָם, וְלֹא יִנָּתֵן לָהֶם הַכָּבוֹד לִקְרֹא בְּשֵׁם יֵשׁוּעַ עַד קִצָּם, כִּי יְשׁוּעָה דֶּרֶךְ מָוֶת עַל קִדּוּשׁ הַשֵּׁם הִיא זְכוּת וְכָבוֹד הַנִּתָּן לְאֵלֶּה שֶׁהֵם יוֹרְשֵׁי חַיֵּי נֶצַח וְחַיֵּי עוֹלָם, אֲשֶׁר יָבוֹאוּ לְשָׂכָר גָּדוֹל עַל עֲמִידָתָם, וַאֲנִי אֲקַבֵּל אוֹתָם אֵלַי לָנֶצַח וּלְעוֹלְמֵי עוֹלָמִים בְּאַהֲבָתִי אֲלֵיהֶם, וְאַךְ כַּלָּתִי הַנֶּאֱמָנָה הִיא שֶׁתִּהְיֶה נַעֲרֶצֶת עַל הַכֹּל בַּיּוֹם הַהוּא! לָכֵן, עִמְדִי אֵיתָן, כַּלָּתִי, וְהַמְלִיחוּ עַצְמְכֶם בִּדְבָרַי, אַף בְּדִבְרֵי בְּשׂוֹרַת הָעוֹלָמִים אֲשֶׁר נָתַתִּי לָאָרֶץ לְהָכִין אֶתְכֶם, אֲהוּבַי, לָכֵן אִגְדוּ אֶת דִּבְרֵי עֵדוּת שְׁלִיחִי וְעַבְדִּי הֵיטֵב בְּלִבְּכֶם, בְּיָדְעֲכֶם כִּי דְּבָרִים טְהוֹרִים אֵלֶּה נִתָּנִים יְשִׁירוֹת מֵאֵת י-ה-ו-ה, בִּפְנֵי הַמָּשִׁיחַ יֵשׁוּעַ, וַאֲנִי מְשַׁחְרֵר אוֹתָם בְּבִינָה עַל פִּי מִצְוָתִי, כַּאֲשֶׁר אֲנִי מְגַלֶּה אוֹתָם בִּשְׁלֵמוּת וּבְדִיּוּק אֶל לֵב שְׁלִיחִי, אַף מֵהַמָּקוֹם בּוֹ זְרַעְתִּים, שְׁמוּרִים לְשִׁחְרוּר עַל פִּי מִצְוָתִי, וְאָמְנָם דְּבָרַי נִתָּנִים בְּכָל אֱמֶת וְדִיּוּק. סֶלָה.

אָמְנָם, אָמְנָם, כֵּן הוּא: הֵילֵל בֶּן שַׁחַר, בְּשִׂנְאָה רַבָּה, מְבַקֵּשׁ בִּפְעֻלָּה לְהָפֵר אֶת אֱמוּנַת כָּל אֵלֶּה שֶׁאֱמוּנָתָם יְכוֹלָה לְהִתְעַרְעֵר עַל יְדֵי דִּבְרֵי כַּזְבוֹ וְתַרְמִיּוֹתָיו הָעַרְמוּמִיּוֹת, כָּל אֵלֶּה שֶׁאֱמוּנָתָם אֵינָהּ נְטוּעָה כֻּלָּהּ עַל סֶלַע הַיְסוֹד הָאֲמִתִּי, הוּא הַמָּשִׁיחַ יֵשׁוּעַ, אֶלָּא מְמֻקֶּדֶת בְּאִי־צִיּוּת בְּדִבְרֵי הַשֶּׁקֶר שֶׁל בְּנֵי אָדָם חוֹטְאִים, אֲשֶׁר אֵינָם מַכִּירִים אֶת הַמָּשִׁיחַ יֵשׁוּעַ וְאַף לֹא מְשָׁרְתִים אוֹתִי כִּמְלֹא נִימָה, אֶלָּא אֶת הֵילֵל בֶּן שַׁחַר הַכַּזְּבָן. וּבְדִבְרֵי כַּזְבָם וּפְרִי עַוְלָתָם תַּכִּירוּם; אַךְ לְרַבִּים מִנֶּאֱמָנַי עֲדַיִן אָזְנַיִם מְגָרְדוֹת, אֲשֶׁר בָּהֶן אַתֶּם חוֹשְׁקִים שֶׁיְּדַגְדְּגוּ אוֹתָן בְּדִבְרֵי שֶׁקֶר וּפִתּוּי כּוֹזְבִים, אַף דִּבְרֵי שֶׁקֶר שֶׁהַמָּשִׁיחַ יֵשׁוּעַ לֹא שָׁלַח, לֹא תָּמַךְ בָּהֶם, וְאַף לֹא אֶשְׁלַח לְעוֹלָם וְלוּ דְּבַר שֶׁקֶר אֶחָד הַסּוֹתֵר אֶת כִּתְבֵי קָדְשִׁי, וְאַף לֹא אֲדַבֵּר דָּבָר אֶחָד הַנּוֹגֵד אֶת הַכְּתוּבִים הַיְצִיבִים שֶׁנָּתַתִּי, כִּי אָמְנָם, אָמְנָם, אֲנִי הוּא דְּבַר י-ה-ו-ה, אֶהְיֶה אֲשֶׁר אֶהְיֶה, שֶׁנַּעֲשָׂה בָּשָׂר, וְנַעֲשֵׂיתִי לְרוּחַ מְחַיָּה, וְלֹא אֶסְתֹּר אֶת עַצְמִי כִּמְלֹא נִימָה, אַף עַל פִּי שֶׁהֵם מְשַׁקְּרִים. לָכֵן, אֲנִי מְצַוֶּה אֶתְכֶם בְּכָל סַמְכוּת לֶאֱחֹז בִּדְבָרַי הֵיטֵב בְּלִבְּכֶם, בִּמְחֻיָּבוּת רַעֲנַנָּה וּמְלוּחָה, לְהִתְעַשֵּׁת מֵחֹסֶר נֶאֱמָנוּתְכֶם, כְּדֵי שֶׁכָּל אֶחָד מִכֶּם שֶׁהֶאֱמִין לְדִבְרֵי שֶׁקֶר אֵלֶּה יוּכַל לַחֲזֹר בִּתְשׁוּבָה וּלְהִתְוַדּוֹת עֲלֵיהֶם לְפָנַי, לְמַעַן אֲטַהֵר אֶתְכֶם כָּלִיל בְּדָמִי הַמְּכַפֵּר וּלְמַעַן תִּהְיוּ מוּכָנִים לְגַמְרֵי לְהוֹפָעָתִי הַקְּרֵבָה אֲלֵיכֶם פִּתְאוֹם וּבְלִי צֶפִי כָּעֵת, בְּעוֹד שְׁאַר הָעוֹלָם נָסוֹג מֵהַמָּשִׁיחַ יֵשׁוּעַ, אֲשֶׁר הָיָה מוֹשִׁיעַ אוֹתָם עַד תַּכְלִית. וְזֹאת בְּעֵת שֶׁהֵם נְסוֹגִים בִּמְחֻיָּבוּת מְלֵאָה לִדְמוּתוֹ שֶׁל קָדוֹשׁ מְעֻנֶּה כּוֹזֵב שֶׁתֻּכְנַן וְנִתַּן כְּדֵי לְסַמֵּן אֶת אֵלֶּה שֶׁיַּשִּׂיגוּ יְשׁוּעָה בְּאֵשׁ הַמְּצָרֵף, כִּי זֹאת תְּחִלַּת הַמִּלְחָמָה עַל הַקְּדוֹשִׁים הָעוֹמֶדֶת לְהִשְׁתַּחְרֵר בִּמְלוֹאָהּ לְטוֹבַת יְשׁוּעָתָם שֶׁל הַסּוֹרְרִים הַנִּמְצָאִים עֲדַיִן בְּשִׁעְבּוּד לַחֵטְא, לְמַעַן יִכְלֶה הַבָּשָׂר, וְהָרוּחַ בֶּאֱמֶת תִּוָּשַׁע, סֶלָה.

אָמְנָם אָמְנָם, אֲנִי הוּא הַמָּשִׁיחַ יֵשׁוּעַ, הוּא יֵשׁוּעַ הַמָּשִׁיחַ, וַאֲנִי בָּא לְרַפֵּא אֶת פִּצְעֵי הַסּוֹרְרִים שֶׁבְּקֶרֶב נֶאֱמָנַי, כָּל אֵלֶּה שֶׁיַּרְשׁוּ לְעַצְמָם לִסְטוֹת מִדְּבָרַי שֶׁבִּבְשׂוֹרַת הָעוֹלָמִים, אַף דְּבָרִים שֶׁנִּתְּנוּ עַל יְדֵי שְׁלִיחַי וּנְבִיאַי בִּתְחִלַּת עִדַּן הַחֶסֶד הַזֶּה, דִּבְרֵי אֱמֶת שֶׁגִּלּוּ אֶת רָזַי בְּעִתּוֹת הִתְגַּלּוּתָם, דְּבָרִים שֶׁהֵם אֲמִתִּיִּים הַיּוֹם כְּשֵׁם שֶׁהָיוּ כַּאֲשֶׁר נִתְּנוּ לָרִאשׁוֹנָה עַל יָדִי לְאָזְנֵי הַנּוֹכְחִים, וּבִמְיֻחָד תַּלְמִידַי, אֲשֶׁר מֵהֶם קָמוּ שְׁלִיחַי: הִזָּהֲרוּ, שִׁקְדוּ וְהִתְפַּלְּלוּ, כִּי אֵינְכֶם יוֹדְעִים מָתַי הָעֵת.

אַךְ כֹּה רַבִּים מִכֶּם מַרְשִׁים לְדִבְרֵי שֶׁקֶר הַסּוֹתְרִים אֶת דִּבְרֵי הַבְטָחָתִי הָאֲמִתִּיִּים, אַף דְּבָרִים שֶׁנּוֹעֲדוּ לְהָפֵר אֶת אֱמוּנַתְכֶם עַל יְדֵי בְּנִיַּת צִפִּיּוֹת שָׁוְא לְמוֹעֵד שֶׁלֹּא נָתַתִּי, צִפִּיּוֹת אֲשֶׁר מְרֻסָּקוֹת לְאַחַר מִכֵּן בְּכַוָּנָה כְּדֵי לְהַטְבִּיעַ אֶתְכֶם בְּצִפִּיּוֹת שֶׁלֹּא הִתְגַּשְּׁמוּ עַל יְדֵי אֵלֶּה הַמְּבַקְּשִׁים לְהַטְרִיד אֶתְכֶם, כָּל אֵלֶּה הַבָּאִים אֲלֵיכֶם בְּחֹסֶר יוֹשֶׁר גָּמוּר וּמַגִּישִׁים שְׁקָרִים הַסּוֹתְרִים אֶת הַכְּתוּבִים, דִּבְרֵי שֶׁקֶר הַנֶּאֱמָרִים עַל יְדֵי אֵלֶּה הַמְדַבְּרִים נֶגֶד דְּבָרַי, אַף אֵלֶּה הַמִּתְנַבְּאִים שֶׁקֶר בִּשְׁמִי.

אַךְ, רַבִּים בְּקֶרֶב קְהִלָּתִי, גּוּפִי וְאַף כַּלָּתִי, מַמְשִׁיכִים לְהַרְשׁוֹת לְרוֹעִים כּוֹזְבִים אֵלֶּה לְהָפֵר אֶת צִפִּיּוֹתֵיהֶם שֶׁחֲתָנָם מִתְקָרֵב בְּיוֹם שֶׁהֵם עַצְמָם מַחְלִיטִים עָלָיו, נְבוּאַת שֶׁקֶר הַסּוֹתֶרֶת אֶת מַה שֶּׁכְּבָר הֻכְרַז עַל יְדֵי י-ה-ו-ה, אֶהְיֶה אֲשֶׁר אֶהְיֶה, וְגַם עַל יְדֵי הַמָּשִׁיחַ יֵשׁוּעַ בְּמִצְוַת אֲדֹנָי-י-ה-ו-ה, כִּי תָּמִיד עוֹשֶׂה אֲנִי אֶת רְצוֹן שׁוֹלְחִי. כִּי אֲנִי הוּא הַדָּבָר שֶׁנַּעֲשָׂה בָּשָׂר, הַשּׁוֹכֵן בֶּאֱמֶת בְּתוֹכְכֶם וְאַף בְּךָ עַל יְדֵי רוּחַ הַקֹּדֶשׁ שֶׁל בָּנִים, הַמְּמַלֵּאת בִּמְלוֹאָהּ אֶת אוֹרוֹת הָעוֹלָם הַלְּבוּשִׁים גְּבוּרָה מִמָּרוֹם! וַעֲדַיִן רַבִּים מִכֶּם בּוֹחֲרִים בְּרָצוֹן לָשֶׁבֶת בַּחֹשֶׁךְ, וּמִתְעַלְּמִים מֵעֶצֶם הַדְּבָרִים שֶׁהָיוּ מְכַבִּים אֶת חִצָּיו הַבּוֹעֲרִים שֶׁל הָרָשָׁע, אֲשֶׁר רַבִּים מִכֶּם מַעֲדִיפִים בְּאִוַּלְתְּכֶם. אָמְנָם, אָמְנָם, אֲנִי הוּא הַיּוֹשֵׁב וּבְיָדוֹ הַמַּפּוּחַ, אַף מְלַבֶּה אֶת לַהֲבוֹת הַטִּהוּר, לְמַעַן בֶּאֱמֶת וּבְתָמִים תִּהְיוּ כֻּלְּכֶם מוּכָנִים לְהוֹפָעַת אֲדוֹנְכֶם, כִּי אֲנִי בָּא לִקְרַאתְכֶם, אֲהוּבַי, וּמִי זֶה יוּכַל לַעֲצֹר בַּעֲדִי? שָׁאַלְתִּי, וְכֵן אֶשְׁאַל שׁוּב, אוֹמֵר אֲנִי, מִי!

אַךְ כֵּן הוּא. שֶׁכַּלָּתִי יוֹשֶׁבֶת כָּעֵת קְדוֹרַנִּית וּנְבוֹכָה, אַף שְׁבוּיָה בְּדִבְרֵי הַמְּצַפְצְפִים וְהַהוֹגִים! וְעַל כֵּן אֲנִי מְצַוֶּה אֶתְכֶם לִדְחוֹת אֶת דִּבְרֵי הַשֶּׁקֶר הַלָּלוּ, אַף כָּל אֵלֶּה שֶׁיָּשִׂימוּ אֶת אֱמוּנָתָם וּבִטְחוֹנָם בְּקּוֹסְמֵי שֶׁקֶר הַמְּצַפְצְפִים וְהַהוֹגִים וְהַמְדַבְּרִים נֶגֶד דִּבְרֵי חַיַּי, כִּי צִיַּנְתִּי בְּבֵרוּר בַּכְּתוּבִים, וְהַכָּתוּב אֵינוֹ יָכוֹל לְהִתְבַּטֵּל: שֶׁאֵין אִישׁ עֲלֵי אֲדָמוֹת יוֹדֵעַ אֶת הַיּוֹם, הָעֵת וְהַשָּׁעָה שֶׁל הִתְאַסְּפוּתָם הַגְּדוֹלָה שֶׁל הַנֶּאֱמָנִים לַמָּשִׁיחַ הָעֲתִידָה לְהִתְרַחֵשׁ פִּתְאוֹם וּבְלִי צֶפִי כָּעֵת. וְצִוִּיתִי עַל כֻּלְּכֶם לְשַׁמֵּר אֶת צִפִּיַּת לִבְּכֶם וְדַעְתְּכֶם מְמֻקֶּדֶת כֻּלָּהּ בַּאֲדוֹן צִדְקַתְכֶם הַנִּצְחִי, אֲשֶׁר בְּוַדַּאי מִתְקָרֵב אֲלֵיכֶם, אֲהוּבַי, כָּל עַלְמוֹתַי הַנְּבוֹנוֹת אֲשֶׁר הֵטַבְתֶּן אֶת פְּתִילוֹתֵיכֶן וּמִלֵּאתֶן אֶת מְנוֹרוֹתֵיכֶן בְּשֶׁמֶן טָהוֹר שֶׁל רוּחִי, לְמַעַן בֶּאֱמֶת וּבְתָמִים תִּהְיוּ מוּכָנוֹת לְהוֹפָעַת חֲתַנְכֶן, וּלְמַעַן תִּהְיוּ מוּכָנוֹת וּמְזֻמָּנוֹת לְגַמְרֵי יַחַד עִם אוֹרְחֵינוּ. כָּל הַקְּרוּאִים, הַבְּחִירִים וְהַמֻּזְמָנִים לְאוֹתָהּ חֲתֻנָּה גְּדוֹלָה שֶׁל הֶחָתָן עִם כַּלָּתוֹ, אוֹתוֹ יוֹם מַמָּשׁ הַדּוֹהֵר לִקְרַאתְכֶם פִּתְאוֹם וּבְלִי צֶפִי כָּעֵת.

אִם כֵּן, מַדּוּעַ כַּלָּתִי יוֹשֶׁבֶת קְדוֹרַנִּית וַעֲצוּבָה, כְּאִלּוּ חֲתָנָהּ זָנַח אוֹתָהּ וְהוֹתִירָהּ נְטוּשָׁה בְּצַד הַדֶּרֶךְ, דְּחוּיָה, זְנוּחָה וּשְׁבוּרַת לֵב בְּצִפִּיּוֹתֶיהָ? לֹא, נֶאֱמָנַי, דְּבָרִים אֵלֶּה אֵינָם עֲבוּרְכֶם, אֶלָּא יִנָּתְנוּ בְּדִין לְחַסְרֵי הָאֱמוּנָה, וּלְכָל אֵלֶּה הַמְּסָרְבִים לְהַזְמָנַת אֲדוֹן צִדְקָתָם הַנִּצְחִי לְהִתְכּוֹנֵן עַל פִּי מִצְוֹתַי, בְּצִיּוּת אוֹהֵב לְקוֹל וּלְהוֹרָאוֹת אֲדוֹנָם!

לָכֵן, אֲנִי בָּא לְהַזְכִּיר לָכֶם שׁוּב אֶת הַדְּבָרִים הַלָּלוּ שֶׁנִּתְּנוּ בִּימֵי קֶדֶם לִבְנֵי הַהַבְטָחָה, הוּא יִשְׂרָאֵל לְפִי הַבָּשָׂר, בְּעִתּוֹת הִתְגַּלּוּתָם, וְאַף דְּבָרִים הָעוֹמְדִים עַד הַיּוֹם הַזֶּה, אֲהוּבַי, לָכֵן שִׁמְעוּ אֶת דִּבְרֵי אֲדוֹנְכֶם וְקַבְּלוּ אֶת מוּסָרִי בְּכָל עֲנָוָה, צְנִיעוּת, שִׁפְלוּת רוּחַ וְנָכוֹנוּת דַּעַת. שִׁמְעוּ וְצַיְּתוּ: וְכִי תֹאמַר בִּלְבָבֶךָ, אֵיכָה נֵדַע אֶת הַדָּבָר אֲשֶׁר לֹא דִבְּרוֹ יְהוָה? אֲשֶׁר יְדַבֵּר הַנָּבִיא בְּשֵׁם יְהוָה, וְלֹא יִהְיֶה הַדָּבָר וְלֹא יָבוֹא, הוּא הַדָּבָר אֲשֶׁר לֹא דִבְּרוֹ יְהוָה, בְּזָדוֹן דִּבְּרוֹ הַנָּבִיא, לֹא תָגוּר מִמֶּנּוּ.

אַךְ, כֹּה רַבִּים בְּקִרְבְּכֶם שָׁכְחוּ אֶת הַדְּבָרִים הַלָּלוּ הַנִּתָּנִים כִּגְבוּלוֹת וּכְמִגְנָנִים מִפְּנֵי זְאֵבִים טוֹרְפִים, אֲשֶׁר דִּינָם בְּוַדַּאי לֹא יִתְמַהְמַהּ. לָכֵן, הַשְׁווּ בִּשְׁקִידָה אֶת דִּבְרֵי כָּל הַטּוֹעֵן לְדַבֵּר בְּשֵׁם הַמָּשִׁיחַ יֵשׁוּעַ, הוּא יֵשׁוּעַ הַמָּשִׁיחַ, לִדְבָרַי הַיְצִיבִים שֶׁנִּתְּנוּ וְנִשְׁמְרוּ בְּכִתְבֵי הַקֹּדֶשׁ, אַף דִּבְרֵי קֹדֶשׁ שֶׁנִּתְּנוּ בִּימֵי קֶדֶם עַל יְדֵי אֲדֹנָי-י-ה-ו-ה, אֶהְיֶה אֲשֶׁר אֶהְיֶה, וְאַף דִּבְרֵי קֹדֶשׁ שֶׁנִּתָּנִים בִּזְמַן הַזֶּה עַל יְדֵי אֲדֹנָי-י-ה-ו-ה וְהַמָּשִׁיחַ יֵשׁוּעַ, הוּא יֵשׁוּעַ הַמָּשִׁיחַ.

לָכֵן, אֲהוּבַי, שִׁמְעוּ אֶת דִּבְרֵי אֲדוֹן צִדְקַתְכֶם הַנִּצְחִי, אַף דְּבָרִים הַנִּתָּנִים בְּיָדַי כְּשֶׁאֲנִי חוֹשֵׂף אוֹתָם אֶל לֵב שְׁלִיחִי וְעַבְדִּי הָאַדִּיר, וּבְיָדַי וּבִרְצוֹנִי נֶחְשָׂפִים דְּבָרִים אֵלֶּה בְּהַבְנָתָם מִלֵּב שְׁלִיחִי וְעַבְדִּי הָאַדִּיר, שָׁם אֲנִי, הַמָּשִׁיחַ יֵשׁוּעַ, נוֹתֵן אֶת דִּבְרֵי הַנֶּחָמָה הַלָּלוּ, אַף דְּבָרִים מַרְגִּיעִים לְנַחֵם אֶת הַנֶּאֱמָנִים בַּמָּשִׁיחַ, עַלְמוֹתַי הַנְּבוֹנוֹת אֲשֶׁר הֵכִינוּ עַצְמָן עַל יְדֵי דְּבָרִים אֵלֶּה הַנִּתָּנִים לְשִׁחְרוּר אֶל הָאָרֶץ עֲבוּר הַנֶּאֱמָנִים בַּמָּשִׁיחַ בְּכָל צִיּוּת, כְּדֵי שֶׁכֻּלְּכֶם תִּהְיוּ מְזֻמָּנִים וּמוּכָנִים, מַמְתִּינִים בְּצַד הַדֶּרֶךְ עִם מְנוֹרוֹתֵיכֶם הַמְּלֵאוֹת וּפְתִילוֹתֵיכֶם הַמֻּטָּבוֹת, כָּל אֵלֶּה מִכֶּם שֶׁלָּקְחוּ אֶת אִשִּׁי וְהִצִּיתוּ אֶת מְנוֹרוֹתֵיהֶם בְּצִיּוּת צַדִּיק לִדְבָרַי, כִּי אֲנִי מִתְקָרֵב אֲלֵיכֶם, אֲהוּבַי.

וְכֵן הוּא, הֶחָתָן מִתְקָרֵב אֶל כַּלָּתוֹ וְהָאָרֶץ פִּתְאוֹם תִּהְיֶה אֲפוּפָה חֹשֶׁךְ עָמֹק, כִּי לֹא אֶתֵּן אֶת כְּבוֹדִי לְאַחֵר, וּכְשֶׁאֲפֵלַת חֹשֶׁךְ תְּכַסֶּה אֶת הָאָרֶץ, כְּבוֹד וְזֹהַר הוֹפָעָתִי יָאִירוּ אֶת הָאָרֶץ כְּבָרָק הַיּוֹצֵא מִמִּזְרָח וְהוֹלֵךְ עַד מַעֲרָב, וְכָךְ פִּתְאוֹם וּבְלִי צֶפִי כָּעֵת יִמְצְאוּ עַצְמָם קְהִלָּתִי, גּוּפִי וְאַף כַּלָּתִי עוֹמְדִים בִּשְׁמֵי הַשָּׁמַיִם, אַף עוֹמְדִים בְּהֵיכַל כֵּס י-ה-ו-ה, אֶהְיֶה אֲשֶׁר אֶהְיֶה, רֶגַע בַּזְּמַן שֶׁאֲדֹנָי-י-ה-ו-ה חִכָּה לוֹ מִנִּצְחֵי עַד, עֵת כֹּה רְחוֹקָה שֶׁאֵינְכֶם יְכוֹלִים לִתְפֹּס בְּעֵת הַזֹּאת. וְכֻלְּכֶם הָיִיתֶם חֲבוּיִים בִּגְבוּרָתוֹ וּבְלֵב אַהֲבָתוֹ וְאוֹרוֹ הַנִּפְלָא, שָׁם שָׁמַר עֲלֵיכֶם מוּגָנִים בְּרָז הַגָּדוֹל בְּיוֹתֵר שֶׁלּוֹ, רָזָא רַבָּא דְּי-ה-ו-ה, אֶהְיֶה אֲשֶׁר אֶהְיֶה, וְאָמְנָם אָמְנָם הַשָּׁמַיִם מוּכָנִים וּכְבָר שְׂמֵחִים לִקְרַאת בּוֹאֲכֶם הַפִּתְאוֹמִי כָּעֵת, אַף לִפְנֵי שֶׁהִגַּעְתֶּם. כִּי גַּם הַמַּלְאָכִים הַקְּדוֹשִׁים וְהַיְצוּרִים הַמַּרְהִיבִים שֶׁבַּשָּׁמַיִם מְצַפִּים בְּהִתְרַגְּשׁוּת לְהוֹפָעַתְכֶם וּלְנוֹכְחוּתְכֶם, אֲהוּבַי.

לָכֵן, הָנִיחוּ לְדַעְתְּכֶם וּלְלִבְּכֶם לָנוּחַ כָּלִיל בְּאַהֲבָתִי אֲלֵיכֶם, יְלָדַי, וְאַל תַּרְשׁוּ לְדִבְרֵי שֶׁקֶר שֶׁל נְבִיאֵי שֶׁקֶר הַמְּצַפְצְפִים וְהַהוֹגִים וְקוֹבְעִים מוֹעֲדִים לַחֲתֻנָּתִי הַגְּלִילִית לְהָפֵר אֶת אֱמוּנַתְכֶם, וְאַף לֹא לְהַעֲצִיב אֶת לִבְּכֶם כְּאִלּוּ אֶזְנַח אֶת כַּלָּתִי הַנָּאָה אֲשֶׁר הֵכִינָה עַצְמָהּ בְּצִיּוּת אוֹהֵב לְקוֹל חֲתָנָהּ. וְעַל כֵּן אֲנִי מְצַוֶּה אֶתְכֶם, אֲהוּבַי, לְמַקֵּד אֶת כָּל לְבַבְכֶם, נַפְשְׁכֶם, דַּעְתְּכֶם וּמְאֹדְכֶם בְּדִבְרֵי אֲדוֹן צִדְקַתְכֶם הַנִּצְחִי, אַף דִּבְרֵי הַקֹּדֶשׁ שֶׁבַּכְּתוּבִים שֶׁגִּלִּיתִי בְּאוֹתוֹ סֵפֶר קָטָן וּפָתוּחַ שֶׁפָּתַחְתִּי בְּיָדֵי שְׁלִיחִי וְעַבְדִּי, כִּי הַסֵּפֶר הַקָּטָן שֶׁבְּהִתְגַּלּוּת פֶּרֶק י' מֵכִיל אֶת דִּבְרֵי שִׁבְעַת הָרְעָמִים וְכֵן אֶת בְּשׂוֹרַת הָעוֹלָמִים בְּחֶלְקָהּ, וְאֵלֶּה הֵם דְּבָרִים שֶׁנִּתְּנוּ לְהַנְחוֹת אֶת עַמִּי דֶּרֶךְ קְצִירֵי צָרַת יַעֲקֹב, הִיא הַשָּׁבוּעַ הַשִּׁבְעִים שֶׁל דָּנִיֵּאל. דְּבָרִים שֶׁנֶּחְתְּמוּ וְנִשְׁמְרוּ בַּסֵּתֶר בְּמִצְוַת י-ה-ו-ה, אֶהְיֶה אֲשֶׁר אֶהְיֶה, כִּי אָמְנָם, אָמְנָם, דִּבְרֵי וְקוֹל שִׁבְעַת הָרְעָמִים אָכֵן נִשְׁמָעִים, וְשִׁבְעַת רְעָמִים אֵלֶּה הֵם הַפְעָלַת סֵפֶר הַהִתְגַּלּוּת, הִיא הַשָּׁבוּעַ הַשִּׁבְעִים שֶׁל דָּנִיֵּאל, אֲשֶׁר תּוֹבִיל לְהַשְׁמָדָה מְלֵאָה וּמוּחְלֶטֶת שֶׁל מַלְכוּת הֵילֵל בֶּן שַׁחַר לְעוֹלְמֵי עוֹלָמִים, וְהָאָרֶץ תִּמָּשֵׁל בְּצֶדֶק, צֶדֶק אֲשֶׁר בּוֹ שׁוּם דָּבָר רַע לֹא יִכָּנֵס לְמַלְכוּת אַלְפֵי הַשָּׁנִים שֶׁלִּי, לָכֵן נוּחוּ בְּתִקְוָה גְּדוֹלָה בְּדִבְרֵי תִּקְוָה וְהַבְטָחָה נִצְחִית אֵלֶּה, אֲהוּבַי.

אֲנִי הוּא הַמָּשִׁיחַ יֵשׁוּעַ, וַאֲנִי בָּא לְהָבִיא אֶת דִּבְרֵי הַתּוֹכֵחָה, הַנְּזִיפָה וְהָאַהֲבָה הַלָּלוּ לַצַּיְתָנִים וְלַסּוֹרְרִים, לְמַעַן בֶּאֱמֶת וּבְתָמִים לֹא תִּפְּלוּ לְצִדָּהּ הַסּוֹרֵר שֶׁל חֶרֶב הָרוּחַ שֶׁלִּי, כְּדֵי שֶׁתֵּחָשְׁבוּ רְאוּיִים לְהֵאָסֵף עִם הַנֶּאֱמָנִים פִּתְאוֹם וּבְלִי צֶפִי כָּעֵת. כִּי גַּם בְּדַקּוֹת אַחֲרוֹנוֹת אֵלֶּה, לִפְנֵי שֶׁאוֹתָהּ תְּרוּעָה גְּדוֹלָה תְּהַדְהֵד, וְלִפְנֵי שֶׁאוֹתָהּ תְּקִיעַת שׁוֹפָר אַדִּירָה תִּשָּׁמַע בְּעָצְמָה וּבְלִי צֶפִי כָּעֵת, עֲדַיִן אֲנִי מַרְעִיד וְגוֹרֵם לִרְעִידָה גְּדוֹלָה לָבוֹא עַל עַמִּי, כְּדֵי שֶׁאוּכַל לְנַעֵר אֶת חַסְרֵי הָאֱמוּנָה לְמַעַן יִהְיוּ לְנֶאֱמָנִים בְּצִיּוּת צַדִּיק, אוֹ יִפְּלוּ אֶל אֵשׁ הַמְּצָרֵף בְּאִי־צִיּוּתָם, מַמָּשׁ לִפְנֵי אוֹתוֹ רֶגַע קַירוֹס שֶׁיִּצְטָלֵב עִם הַזְּמַן הַכְרוֹנוֹלוֹגִי הָעוֹמֵד לְהַשִּׂיג אֶתְכֶם פִּתְאוֹם וּבְלִי צֶפִי כָּעֵת. וְזֹאת בְּעוֹד הָעוֹלָם מִתְאַבֵּל עַל מוֹשִׁיעוֹ הַכּוֹזֵב, קָדוֹשׁ מְעֻנֶּה כּוֹזֵב שֶׁלֹּא אֲנִי שְׁלַחְתִּיו, אֶלָּא סֻפַּק עַל יְדֵי בַּר־אַבָּא תַּחְתָּיו, וְאַף עַל פִּי כֵן כֹּה רַבִּים הוֹלְכִים אַחֲרֵי קָדוֹשׁ מְעֻנֶּה כּוֹזֵב זֶה, אַף מֻטְעִים מִקּוֹלוֹ שֶׁל הַמָּשִׁיחַ יֵשׁוּעַ, הוּא יֵשׁוּעַ הַמָּשִׁיחַ. סֶלָה.

וְכֵן הוּא, שֶׁאֲנִי מֵכִין אֶת אֵלֶּה שֶׁיִּוָּתְרוּ בִּמְלוֹאָם לְאֵשׁ הַמְּצָרֵף, כָּל אֵלֶּה שֶׁתִּנָּתֵן לָהֶם הַהִזְדַּמְּנוּת לִקְרֹא בְּשֵׁם יֵשׁוּעַ הַמָּשִׁיחַ עַד קִצָּם, כִּי כָּל הָאוֹהֲבִים אֶת חַיֵּיהֶם עַד כְּדֵי כָּךְ יִמְצְאוּ יְשׁוּעַת עוֹלָמִים, אֲשֶׁר בֶּאֱמֶת וּבְתָמִים אֵינָהּ בַּת הַשְׁוָאָה לַסֵּבֶל שֶׁיִּסְבְּלוּ לְמַעַן שְׁמִי.

הִכּוֹנוּ, קְהִלָּתִי, גּוּפִי וְאַף כַּלָּתִי, הִכּוֹנוּ וְהִתְכּוֹנְנוּ עַל פִּי דְּבָרַי הַקְּדוֹשִׁים וְכִתְבֵי בְּשׂוֹרַת הָעוֹלָמִים, וְחַכּוּ לִי, אֲהוּבַי, כִּי אֲנִי בָּא לִקְרַאתְכֶם וְלֹא תִּתְאַכְזְבוּ בְּצִפִּיּוֹתֵיכֶם. לָכֵן, קַבְּלוּ אֶת נְזִיפָתִי וְתוֹכַחְתִּי וְהֱיוּ צַיְתָנִים לִדְבָרַי וּלְאַזְהָרוֹתַי, כִּי אֲנִי בָּא לִקְרַאתְכֶם, אֲהוּבַי, וְלֹא אֶעֱזֹב לְעוֹלָם אֶת הַנֶּאֱמָנִים בַּמָּשִׁיחַ לְאֵשׁ הַמְּצָרֵף, וְאַף לֹא לְמִשְׁפַּט הַחוֹטְאִים שֶׁאֵינוֹ מְיֹעָד לָכֶם.

הִתְכּוֹנְנוּ, אֲהוּבַי, בְּכָל צִיּוּת וְצִפִּיָּה, וְהִתְעַמְּקוּ בְּדִבְרֵי בְּשׂוֹרַת הָעוֹלָמִים, וְהַחֲזִיקוּ בָּהֶם בְּדַעְתְּכֶם וּבְלִבְּכֶם וְהָבִינוּ אֶת אוֹצַר הַכְּתוּבִים שֶׁנָּתַתִּי לָאָרֶץ לִשְׁמֹר עֲלֵיכֶם בְּעִתּוֹת הִתְגַּלּוּת אֵלֶּה, הִתְגַּלּוּת שֶׁהֵחֵלָּה בְּשׁוֹפָרוֹת מוּכָנִים רַבִּים בְּעֶדְרִי הַקָּטָן שֶׁתָּקְעוּ בֶּאֱמוּנָה, שׁוֹפָרוֹת אֲשֶׁר הֵכִינוּ אֶת הַדֶּרֶךְ לִשְׁלִיחִי וּלְעַבְדִּי, שֶׁהוּא כְּלִי בָּחוּר לַאֲדוֹנוֹ הַנִּצְחִי, הוּא הַמָּשִׁיחַ יֵשׁוּעַ, הוּא יֵשׁוּעַ הַמָּשִׁיחַ, שָׁם אֲנִי מְגַלֶּה אֶת דְּבָרַי לְתוֹךְ כְּלִי קָדְשִׁי, הַמָּצוּי בְּהַסְכָּמָה וּבִמְחֻיָּבוּת מְלֵאָה לַמְּשִׂימוֹת שֶׁהִפְקַדְתִּי בְּיָדוֹ, וּבְמִצְוָתִי וּבְכֹחִי מוֹצְאוֹת אֶצְבְּעוֹתָיו אֶת הַמַּקָּשִׁים אַף לְגַלּוֹת אֶת דִּבְרֵי הַקֹּדֶשׁ הַלָּלוּ שֶׁנָּתַתִּי בְּתוֹכוֹ לְשַׁחְרֵר אֶל הָאָרֶץ לָרִאשׁוֹנָה, דְּבָרִים הַמְּכֻוָּנִים לִמְצֹא אֶת עֵינֵי, אָזְנֵי וְלֵב כָּל אֵלֶּה שֶׁאֲלֵיהֶם הֵם נִשְׁלָחִים. סֶלָה.

לָכֵן, שִׁמְעוּ אֶת קוֹלִי בְּהַכְרָזַת קֹדֶשׁ זוֹ הַנִּתֶּנֶת לִמְשֹׁךְ אֶת הַנֶּאֱמָנִים לְמִצְוָתִי, וּלְכַוֵּן אֶתְכֶם בְּצִיּוּת לְקוֹל אוֹתוֹ שׁוֹפָר, לְמַעַן תֵּאָסְפוּ אֵלַי עַל פִּי מִצְוָתִי.

הִנֵּה! אֲנִי חוֹתֵם עַל הַכְרָזָה וּמִכְתָּב זֶה בְּיַד שְׁלִיחִי וְעַבְדִּי, אַךְ חָתוּם בִּשְׁמִי שֶׁלִּי מַמָּשׁ בְּכָל אַהֲבָה וְסַמְכוּת,

יֵשׁוּעַ הַמָּשִׁיחַ, אֲדוֹן הַצֶּדֶק הַנִּצְחִי.
